34template <class T, class... Ts>
35std::string concat(const T& first, const Ts&... args) noexcept {
36 std::ostringstream ss;
37 ss << first << '\n' << concat(std::forward<const Ts>(args)...);
38 return ss.str();
39}

41template <class... Ts>
42inline void do_assert(bool expr, const common::internal::source_location& loc, const char* expression,
43 const Ts&... ts) noexcept {
44 if (!expr) {
45 std::ostringstream ss;
46 ss << "[ERROR] " << loc << '\n' << expression << '\n' << concat(ts...) << "\n";
47 std::cerr << ss.str();
48 std::terminate();
49 }
50}
