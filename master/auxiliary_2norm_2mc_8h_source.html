23 // Compute max norm of the lower triangular part of the distributed matrix
24 // https://en.wikipedia.org/wiki/Matrix_norm#Max_norm
25 //
26 // It just addresses tiles with elements belonging to the lower triangular part of the matrix
27 //
28 // Thanks to the nature of the max norm, it is valid for:
29 // - sy/he lower
30 // - tr lower non-unit
31 template <class T>
32 dlaf::BaseType<T> Norm<Backend::MC, Device::CPU, T>::max_L(comm::CommunicatorGrid comm_grid,
33  comm::Index2D rank,
34  Matrix<const T, Device::CPU>& matrix) {
35  using namespace dlaf::matrix;
36  namespace ex = pika::execution::experimental;
37  using pika::this_thread::experimental::sync_wait;
38 
39  using dlaf::common::make_data;
40  using dlaf::common::internal::vector;
41  using pika::unwrapping;
42 
43  using dlaf::tile::internal::lange;
44  using dlaf::tile::internal::lantr;
45 
46  using NormT = dlaf::BaseType<T>;
47 
48  const auto& distribution = matrix.distribution();
49 
50  DLAF_ASSERT(square_size(matrix), matrix);
51  DLAF_ASSERT(square_blocksize(matrix), matrix);
52 
53  vector<ex::unique_any_sender<NormT>> tiles_max;
54  tiles_max.reserve(distribution.localNrTiles().rows() * distribution.localNrTiles().cols());
55 
56  // for each local tile in the (global) lower triangular matrix, create a task that finds the max element in the tile
57  for (auto tile_wrt_local : iterate_range2d(distribution.localNrTiles())) {
58  auto tile_wrt_global = distribution.globalTileIndex(tile_wrt_local);
59 
60  if (tile_wrt_global.row() < tile_wrt_global.col())
61  continue;
62 
63  bool is_diag = tile_wrt_global.row() == tile_wrt_global.col();
64  auto norm_max_f = [is_diag](const matrix::Tile<const T, Device::CPU>& tile) noexcept -> NormT {
65  if (is_diag)
66  return lantr(lapack::Norm::Max, blas::Uplo::Lower, blas::Diag::NonUnit, tile);
67  else
68  return lange(lapack::Norm::Max, tile);
69  };
70  auto current_tile_max =
71  matrix.read(tile_wrt_local) |
72  dlaf::internal::transform(dlaf::internal::Policy<Backend::MC>(), std::move(norm_max_f));
73 
74  tiles_max.push_back(std::move(current_tile_max));
75  }
76 
77  // then it is necessary to reduce max values from all ranks into a single max value for the matrix
78 
79  auto max_element = [](std::vector<NormT>&& values) {
80  DLAF_ASSERT(!values.empty(), "");
81  return *std::max_element(values.begin(), values.end());
82  };
83  NormT local_max_value =
84  tiles_max.empty() ? NormT{0}
85  : sync_wait(ex::when_all_vector(std::move(tiles_max)) |
86  dlaf::internal::transform(dlaf::internal::Policy<Backend::MC>(),
87  std::move(max_element)));
88  NormT max_value;
89  dlaf::comm::sync::reduce(comm_grid.rankFullCommunicator(rank), comm_grid.fullCommunicator(), MPI_MAX,
90  make_data(&local_max_value, 1), make_data(&max_value, 1));
91 
92  return max_value;
93 }
