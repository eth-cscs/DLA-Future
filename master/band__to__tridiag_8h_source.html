70 template <Backend B, Device D, class T>
71 TridiagResult<T, Device::CPU> bandToTridiag(blas::Uplo uplo, SizeType band_size,
72  Matrix<const T, D>& mat_a) {
73  DLAF_ASSERT(matrix::square_size(mat_a), mat_a);
74  DLAF_ASSERT(matrix::square_blocksize(mat_a), mat_a);
75  DLAF_ASSERT(mat_a.blockSize().rows() % band_size == 0, mat_a.blockSize().rows(), band_size);
76  DLAF_ASSERT(matrix::local_matrix(mat_a), mat_a);
77  DLAF_ASSERT(band_size >= 2, band_size);
78 
79  switch (uplo) {
80  case blas::Uplo::Lower:
81  return internal::BandToTridiag<B, D, T>::call_L(band_size, mat_a);
82  break;
83  case blas::Uplo::Upper:
84  DLAF_UNIMPLEMENTED(uplo);
85  break;
86  case blas::Uplo::General:
87  DLAF_UNIMPLEMENTED(uplo);
88  break;
89  }
90 
91  return DLAF_UNREACHABLE(TridiagResult<T, Device::CPU>);
92 }

141 template <Backend backend, Device device, class T>
142 TridiagResult<T, Device::CPU> bandToTridiag(comm::CommunicatorGrid grid, blas::Uplo uplo,
143  SizeType band_size, Matrix<const T, device>& mat_a) {
144  DLAF_ASSERT(matrix::square_size(mat_a), mat_a);
145  DLAF_ASSERT(matrix::square_blocksize(mat_a), mat_a);
146  DLAF_ASSERT(matrix::equal_process_grid(mat_a, grid), mat_a, grid);
147  DLAF_ASSERT(band_size >= 2, band_size);
148 
149  // If the grid contains only one rank force local implementation.
150  if (grid.size() == comm::Size2D(1, 1))
151  return bandToTridiag<backend, device, T>(uplo, band_size, mat_a);
152 
153  switch (uplo) {
154  case blas::Uplo::Lower:
155  return internal::BandToTridiag<backend, device, T>::call_L(grid, band_size, mat_a);
156  break;
157  case blas::Uplo::Upper:
158  DLAF_UNIMPLEMENTED(uplo);
159  break;
160  case blas::Uplo::General:
161  DLAF_UNIMPLEMENTED(uplo);
162  break;
163  }
164 
165  return DLAF_UNREACHABLE(TridiagResult<T, Device::CPU>);
166 }
