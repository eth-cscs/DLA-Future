59 #define DLAF_DEFINE_GPUBLAS_OP(Name, Type, f) \
60  template <> \
61  struct Name<Type> { \
62  template <typename... Args> \
63  static void call(cublasHandle_t handle, Args&&... args) { \
64  auto workspace = DLAF_GET_ROCBLAS_WORKSPACE(f); \
65  DLAF_GPUBLAS_CHECK_ERROR(rocblas_set_workspace(static_cast<rocblas_handle>(handle), workspace(), \
66  to_sizet(workspace.size()))); \
67  DLAF_GPUBLAS_CHECK_ERROR(rocblas_##f(handle, std::forward<Args>(args)...)); \
68  DLAF_GPUBLAS_CHECK_ERROR(rocblas_set_workspace(static_cast<rocblas_handle>(handle), nullptr, 0)); \
69  ::dlaf::tile::internal::extendROCBlasWorkspace(handle, std::move(workspace)); \
70  } \
71  }
