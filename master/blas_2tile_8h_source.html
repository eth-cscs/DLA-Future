35 #define DLAF_GET_ROCBLAS_WORKSPACE(f) \
36  [&]() { \
37  std::size_t workspace_size; \
38  DLAF_GPUBLAS_CHECK_ERROR( \
39  rocblas_start_device_memory_size_query(static_cast<rocblas_handle>(handle))); \
40  DLAF_ROCBLAS_WORKSPACE_CHECK_ERROR(rocblas_##f(handle, std::forward<Args>(args)...)); \
41  DLAF_GPUBLAS_CHECK_ERROR( \
42  rocblas_stop_device_memory_size_query(static_cast<rocblas_handle>(handle), &workspace_size)); \
43  return ::dlaf::memory::MemoryView<std::byte, Device::GPU>(to_int(workspace_size)); \
44  }();

56 #define DLAF_DEFINE_GPUBLAS_OP(Name, Type, f) \
57  template <> \
58  struct Name<Type> { \
59  template <typename... Args> \
60  static void call(cublasHandle_t handle, Args&&... args) { \
61  auto workspace = DLAF_GET_ROCBLAS_WORKSPACE(f); \
62  DLAF_GPUBLAS_CHECK_ERROR(rocblas_set_workspace(static_cast<rocblas_handle>(handle), workspace(), \
63  to_sizet(workspace.size()))); \
64  DLAF_GPUBLAS_CHECK_ERROR(rocblas_##f(handle, std::forward<Args>(args)...)); \
65  DLAF_GPUBLAS_CHECK_ERROR(rocblas_set_workspace(static_cast<rocblas_handle>(handle), nullptr, 0)); \
66  ::dlaf::tile::internal::extendROCBlasWorkspace(handle, std::move(workspace)); \
67  } \
68  }
