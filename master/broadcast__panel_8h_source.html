34 // helper function that identifies the owner of a transposed coordinate,
35 // it returns both the component of the rank in the transposed dimension and
36 // its global cross coordinate (i.e. row == col in the global frame of reference)
37 template <Coord dst_coord>
38 std::pair<SizeType, comm::IndexT_MPI> transposedOwner(const matrix::Distribution& dist,
39  const LocalTileIndex idx) {
40  const auto idx_cross = dist.template globalTileFromLocalTile<dst_coord>(idx.get(dst_coord));
41  const auto rank_owner = dist.template rankGlobalTile<orthogonal(dst_coord)>(idx_cross);
42  return std::make_pair(idx_cross, rank_owner);
43 }

61  comm::CommunicatorPipeline<coord_to_communicator_type(orthogonal(axis))>& serial_comm) {
62  constexpr auto comm_coord = axis;
63 
64  // do not schedule communication tasks if there is no reason to do so...
65  if (panel.parentDistribution().commGridSize().get(comm_coord) <= 1)
66  return;
67 
68  const auto rank = panel.parentDistribution().rankIndex().get(comm_coord);
69 
70  namespace ex = pika::execution::experimental;
71  for (const auto& index : panel.iteratorLocal()) {
72  if (rank == rank_root)
73  ex::start_detached(scheduleSendBcast(serial_comm.exclusive(), panel.read(index)));
74  else
75  ex::start_detached(scheduleRecvBcast(serial_comm.exclusive(), rank_root, panel.readwrite(index)));
76  }

121 template <class T, Device D, Coord axis, matrix::StoreTransposed storage,
122  matrix::StoreTransposed storageT, class = std::enable_if_t<!std::is_const_v<T>>>
123 void broadcast(comm::IndexT_MPI rank_root, matrix::Panel<axis, T, D, storage>& panel,
124  matrix::Panel<orthogonal(axis), T, D, storageT>& panelT,
125  comm::CommunicatorPipeline<comm::CommunicatorType::Row>& row_task_chain,
126  comm::CommunicatorPipeline<comm::CommunicatorType::Col>& col_task_chain) {
127  constexpr Coord axisT = orthogonal(axis);
128 
129  constexpr Coord coord = std::decay_t<decltype(panel)>::coord;
130  constexpr Coord coordT = std::decay_t<decltype(panelT)>::coord;
131 
132  // Note:
133  // Given a source panel, this communication pattern makes every rank access tiles of both the
134  // source panel and it's tranposed variant (just tile coordinates, data is not transposed).
135  // In particular, it does not give access to all the tiles, but just the ones of interest for
136  // each rank, i.e. the rows and columns of a distributed matrix that the ranks stores locally.
137  //
138  // This happens in two steps (for the sake of example, let's consider a column -> row broadcast,
139  // the opposite is dual):
140  //
141  // 1. broadcast the source panel to panel with the same shape on other ranks
142  // 2. populate the transposed destination panel
143  //
144  // Once the source panel is share by all ranks, the transposed panel can be easily populated,
145  // because each destination tile can be populated with data from the rank owning the diagonal one,
146  // the point of contact between the row and the column (row == col).
147  //
148  // If it is already available locally, the tile is not copied and it just gets "linked", in order
149  // to easily access it via panel coordinates with minimal (to null) overhead.
150  // For this reason, the destination panel will depend on the source panel (as the source panel
151  // may already depend on the matrix).
152 
153  DLAF_ASSERT(panel.parentDistribution() == panelT.parentDistribution(),
154  "they must refer to the same matrix");
155 
156  const auto& dist = panel.parentDistribution();
157 
158  DLAF_ASSERT(square_size(dist), dist.size());
159  DLAF_ASSERT(square_blocksize(dist), dist.blockSize());
160 
161  // Note:
162  // This algorithm allow to broadcast panel to panelT using as mirror the parent matrix main diagonal.
163  // This means that it is possible to broadcast panels with different axes just if their global offset
164  // lie on the diagonal.
165  DLAF_ASSERT(panel.rangeStart() == panelT.rangeStart(), panel.rangeStart(), panelT.rangeStart());
166  DLAF_ASSERT_MODERATE(panel.rangeEnd() == panelT.rangeEnd(), panel.rangeEnd(), panelT.rangeEnd());
167 
168  // if no panel tiles, just skip it
169  if (panel.rangeStart() == panel.rangeEnd())
170  return;
171 
172  // STEP 1
173  constexpr auto comm_dir_step1 = orthogonal(axis);
174  auto& chain_step1 = internal::get_taskchain<comm_dir_step1>(row_task_chain, col_task_chain);
175 
176  broadcast(rank_root, panel, chain_step1);
177 
178  // STEP 2
179  constexpr auto comm_dir_step2 = orthogonal(axisT);
180  constexpr auto comm_coord_step2 = axisT;
181 
182  auto& chain_step2 = internal::get_taskchain<comm_dir_step2>(row_task_chain, col_task_chain);
183 
184  const SizeType last_tile = std::max(panelT.rangeStart(), panelT.rangeEnd() - 1);
185  const auto owner = dist.template rankGlobalTile<coordT>(last_tile);
186  const auto range = dist.rankIndex().get(coordT) == owner
187  ? common::iterate_range2d(*panelT.iteratorLocal().begin(),
188  LocalTileIndex(coordT, panelT.rangeEndLocal() - 1, 1))
189  : panelT.iteratorLocal();
190 
191  for (const auto& indexT : range) {
192  auto [index_diag, owner_diag] = internal::transposedOwner<coordT>(dist, indexT);
193 
194  namespace ex = pika::execution::experimental;
195  if (dist.rankIndex().get(coord) == owner_diag) {
196  const auto index_diag_local = dist.template localTileFromGlobalTile<coord>(index_diag);
197  panelT.setTile(indexT, panel.read({coord, index_diag_local}));
198 
199  if (dist.commGridSize().get(comm_coord_step2) > 1)
200  ex::start_detached(scheduleSendBcast(chain_step2.exclusive(), panelT.read(indexT)));
201  }
202  else {
203  if (dist.commGridSize().get(comm_coord_step2) > 1)
204  ex::start_detached(scheduleRecvBcast(chain_step2.exclusive(), owner_diag,
205  panelT.readwrite(indexT)));
206  }
207  }
208 }
209 
210 }
