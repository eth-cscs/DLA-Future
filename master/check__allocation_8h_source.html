43template <class MatrixLike>
44bool is_allocated_as_col_major(MatrixLike& mat) {
45 if (mat.allocation() != MatrixAllocation::ColMajor)
46 return false;
47
48 Distribution dist = mat.distribution();
49
50 if (dist.local_nr_tiles().isEmpty())
51 return true;
52
53 auto* ptr_00 = internal::sync_tile_ptr(mat, {0, 0});
54 SizeType ld_00 = internal::sync_tile_ld(mat, {0, 0});
55
56 for (auto& ij : iterate_range2d(dist.local_nr_tiles())) {
57 SizeType i_el_local = dist.local_element_from_local_tile_and_tile_element<Coord::Row>(ij.row(), 0);
58 SizeType j_el_local = dist.local_element_from_local_tile_and_tile_element<Coord::Col>(ij.col(), 0);
59 SizeType offset = i_el_local + ld_00 * j_el_local;
60 auto* ptr_ij = internal::sync_tile_ptr(mat, ij);
61 SizeType ld_ij = internal::sync_tile_ld(mat, ij);
62 if (ptr_ij - ptr_00 != offset || ld_ij != ld_00)
63 return false;
64 }
65 return true;
66}

68template <class MatrixLike>
69bool is_allocated_as_blocks(MatrixLike& mat) {
70 if (mat.allocation() != MatrixAllocation::Blocks)
71 return false;
72
73 const Distribution& dist = mat.distribution();
74 Distribution helper_dist = dlaf::matrix::internal::create_single_tile_per_block_distribution(dist);
75 LocalBlockSize local_nr_blocks = helper_dist.local_nr_blocks();
76
77 if (dist.local_nr_tiles().isEmpty())
78 return true;
79
80 for (SizeType j_bl = 0, j = 0; j_bl < local_nr_blocks.cols(); ++j_bl) {
81 // When computing the number of tiles in a block, special care is needed when only a single block is
82 // present, as it can contain two incomplete tiles.
83 SizeType nb = helper_dist.local_tile_size_of<Coord::Col>(j_bl);
84 SizeType tiles_in_block_cols = nb / dist.tile_size().cols();
85 if (local_nr_blocks.cols() == 1)
86 tiles_in_block_cols = dist.local_nr_tiles().cols();
87
88 for (SizeType i_bl = 0, i = 0; i_bl < local_nr_blocks.rows(); ++i_bl) {
89 // When computing the number of tiles in a block, special care is needed when only a single block
90 // is present, as it can contain two incomplete tiles.
91 SizeType mb = helper_dist.local_tile_size_of<Coord::Row>(i_bl);
92 SizeType tiles_in_block_rows = mb / dist.tile_size().rows();
93 if (local_nr_blocks.rows() == 1)
94 tiles_in_block_rows = dist.local_nr_tiles().rows();
95
96 // pointer and ld of the first element of the block.
97 auto* ptr_bl = internal::sync_tile_ptr(mat, {i, j});
98 SizeType ld_bl = internal::sync_tile_ld(mat, {i, j});
99 SizeType i_el_local_bl = dist.local_element_from_local_tile_and_tile_element<Coord::Row>(i, 0);
100 SizeType j_el_local_bl = dist.local_element_from_local_tile_and_tile_element<Coord::Col>(j, 0);
101
102 for (SizeType j_tl_bl = 0; j_tl_bl < tiles_in_block_cols; ++j_tl_bl) {
103 for (SizeType i_tl_bl = 0; i_tl_bl < tiles_in_block_rows; ++i_tl_bl) {
104 LocalTileIndex ij{i + i_tl_bl, j + j_tl_bl};
105 SizeType i_el_local =
106 dist.local_element_from_local_tile_and_tile_element<Coord::Row>(ij.row(), 0);
107 SizeType j_el_local =
108 dist.local_element_from_local_tile_and_tile_element<Coord::Col>(ij.col(), 0);
109 SizeType offset = (i_el_local - i_el_local_bl) + ld_bl * (j_el_local - j_el_local_bl);
110 auto* ptr_ij = internal::sync_tile_ptr(mat, ij);
111 SizeType ld_ij = internal::sync_tile_ld(mat, ij);
112
113 if (ptr_ij - ptr_bl != offset || ld_ij != ld_bl)
114 return false;
115 }
116 }
117 i += tiles_in_block_rows;
118 }
119 j += tiles_in_block_cols;
120 }
121
122 return true;
123}

125template <class MatrixLike>
126bool is_allocated_as_tiles(MatrixLike& mat) {
127 return mat.allocation() == MatrixAllocation::Tiles;
128}

130template <class MatrixLike>
131bool is_allocated_as(MatrixLike& mat, MatrixAllocation alloc) {
132 switch (alloc) {
133 case MatrixAllocation::ColMajor:
134 return is_allocated_as_col_major(mat);
135 case MatrixAllocation::Blocks:
136 return is_allocated_as_blocks(mat);
137 case MatrixAllocation::Tiles:
138 return is_allocated_as_tiles(mat);
139 }
140 return false;
141}
