38template <class T, Device Source, Device Destination>
39void copy(LocalTileSize sz, LocalTileIndex idx_source_begin, Matrix<const T, Source>& source,
40 LocalTileIndex idx_dest_begin, Matrix<T, Destination>& dest) {
41 // If @p source and @p dest is the same matrix and the iteration range fully overlaps, return.
42 if constexpr (Source == Destination) {
43 if (idx_source_begin == idx_dest_begin && &source == &dest)
44 return;
45 }
46
47 // Given that `sz` is the same for both `source` and `dest` it is sufficient to only check if the local
48 // length of the copied region is the same.
49 DLAF_ASSERT(
50 source.distribution().localElementDistanceFromLocalTile(idx_source_begin, idx_source_begin + sz) ==
51 dest.distribution().localElementDistanceFromLocalTile(idx_dest_begin, idx_dest_begin + sz),
52 source, dest);
53
54 namespace ex = pika::execution::experimental;
55 for (auto idx_dest : common::iterate_range2d(idx_dest_begin, sz)) {
56 LocalTileIndex idx_source = idx_source_begin + (idx_dest - idx_dest_begin);
57 ex::start_detached(ex::when_all(source.read(idx_source), dest.readwrite(idx_dest)) |
58 copy(dlaf::internal::Policy<internal::CopyBackend_v<Source, Destination>>{}));
59 }
60}

82template <class T, Device Source, Device Destination>
83void copy(MatrixRef<const T, Source>& src, MatrixRef<T, Destination>& dst) {
84 namespace ex = pika::execution::experimental;
85
86 if constexpr (Source == Destination) {
87 if (&src == &dst)
88 return;
89 }
90
91 DLAF_ASSERT(src.size() == dst.size(), src.size(), dst.size());
92 DLAF_ASSERT(src.tile_size() == dst.tile_size(), src.tile_size(), dst.tile_size());
93
94 DLAF_ASSERT(dlaf::matrix::same_process_grid(src, dst), src, dst);
95
96 if (!local_matrix(src)) {
97 DLAF_ASSERT(src.block_size() == dst.block_size(), src.block_size(), dst.block_size());
98 DLAF_ASSERT(src.source_rank_index() == dst.source_rank_index(), src.source_rank_index(),
99 dst.source_rank_index());
100 }
101
102 if (src.nr_tiles().isEmpty())
103 return;
104
105 [[maybe_unused]] GlobalTileIndex i_00{0, 0};
106 DLAF_ASSERT(src.distribution().tile_size_of(i_00) == dst.distribution().tile_size_of(i_00),
107 src.distribution().tile_size_of(i_00), dst.distribution().tile_size_of(i_00));
108
109 const dlaf::internal::Policy<matrix::internal::CopyBackend_v<Source, Destination>> policy;
110 for (SizeType j = 0; j < src.distribution().local_nr_tiles().cols(); ++j) {
111 for (SizeType i = 0; i < src.distribution().local_nr_tiles().rows(); ++i) {
112 ex::start_detached(ex::when_all(src.read(LocalTileIndex{i, j}),
113 dst.readwrite(LocalTileIndex{i, j})) |
114 matrix::copy(policy));
115 }
116 }
117}

139template <class T, Device Source, Device Destination>
140void copy(Matrix<const T, Source>& src, Matrix<T, Destination>& dst, comm::CommunicatorGrid& grid) {
141 namespace ex = pika::execution::experimental;
142
143 DLAF_ASSERT_MODERATE(equal_size(src, dst), src.size(), dst.size());
144 DLAF_ASSERT_MODERATE(equal_process_grid(src, grid), src.commGridSize(), grid.size());
145 DLAF_ASSERT_MODERATE(equal_process_grid(dst, grid), dst.commGridSize(), grid.size());
146
147 DLAF_ASSERT_MODERATE(single_tile_per_block(src), src);
148 DLAF_ASSERT_MODERATE(single_tile_per_block(src), dst);
149
150 // Note:
151 // From an algorithmic point of view it would be better to reason in terms of block instead of tiles,
152 // with the aim of reducing the number of communications.
153 // Current implementation reasons in terms of tiles due to a limitation for retiled matrices, which
154 // cannot access the original block of the original matrix, but just their specific tile size (i.e.
155 // tiles cannot be upscaled upto block). Dealing with tiles leads to a sub-optimal solution: smaller
156 // chunks are communicated, leading to a potentially higher number of communications.
157 const TileElementSize tile_size_src = src.baseTileSize();
158 const TileElementSize tile_size_dst = dst.baseTileSize();
159
160 const SizeType mb = std::min<SizeType>(tile_size_src.rows(), tile_size_dst.rows());
161 const SizeType nb = std::min<SizeType>(tile_size_src.cols(), tile_size_dst.cols());
162
163 DLAF_ASSERT_MODERATE(tile_size_src.rows() % mb == 0, tile_size_src.rows(), mb);
164 DLAF_ASSERT_MODERATE(tile_size_dst.rows() % mb == 0, tile_size_dst.rows(), mb);
165 DLAF_ASSERT_MODERATE(tile_size_src.cols() % nb == 0, tile_size_src.cols(), nb);
166 DLAF_ASSERT_MODERATE(tile_size_dst.cols() % nb == 0, tile_size_dst.cols(), nb);
167
168 const LocalTileSize scale_factor_src{tile_size_src.rows() / mb, tile_size_src.cols() / nb};
169 const LocalTileSize scale_factor_dst{tile_size_dst.rows() / mb, tile_size_dst.cols() / nb};
170
171 Matrix<const T, Source> src_retiled = src.retiledSubPipelineConst(scale_factor_src);
172 Matrix<T, Destination> dst_retiled = dst.retiledSubPipeline(scale_factor_dst);
173
174 const comm::Index2D rank = grid.rank();
175 auto mpi_chain = grid.full_communicator_pipeline();
176
177 auto tag = [dist = src_retiled.distribution()](GlobalTileIndex ij) -> comm::IndexT_MPI {
178 // Note:
179 // Source distribution is used as reference for both sending and receiving side.
180 // The tag is computed as a function of local tile index, in order to keep its value as small as
181 // possible. Moreover, since the local size of the matrix might change on different ranks, linear
182 // local tile index is computed using a rank-independent ld.
183 const auto size = dist.commGridSize();
184 const LocalTileIndex source_ij_lc{ij.row() / size.rows(), ij.col() / size.cols()};
185 const SizeType ld = dlaf::util::ceilDiv(dist.nrTiles().rows(), to_SizeType(size.rows()));
186 return to_int(ij.row() + ij.col() * ld);
187 };
188
189 for (const LocalTileIndex ij_lc : common::iterate_range2d(src_retiled.distribution().localNrTiles())) {
190 const GlobalTileIndex ij = src_retiled.distribution().globalTileIndex(ij_lc);
191 const comm::Index2D src_rank = src_retiled.distribution().rankGlobalTile(ij);
192 const comm::Index2D dst_rank = dst_retiled.distribution().rankGlobalTile(ij);
193
194 const bool src_is_mine = rank == src_rank;
195 const bool dst_is_mine = rank == dst_rank;
196
197 if (src_is_mine != dst_is_mine) {
198 ex::start_detached(comm::schedule_send(mpi_chain.shared(), grid.rankFullCommunicator(dst_rank),
199 tag(ij), src_retiled.read(ij_lc)));
200 }
201 }
202
203 const dlaf::internal::Policy<matrix::internal::CopyBackend_v<Source, Destination>> policy;
204 for (const LocalTileIndex ij_lc : common::iterate_range2d(dst_retiled.distribution().localNrTiles())) {
205 const GlobalTileIndex ij = dst_retiled.distribution().globalTileIndex(ij_lc);
206 const comm::Index2D src_rank = src_retiled.distribution().rankGlobalTile(ij);
207 const comm::Index2D dst_rank = dst_retiled.distribution().rankGlobalTile(ij);
208
209 const bool src_is_mine = rank == src_rank;
210 const bool dst_is_mine = rank == dst_rank;
211
212 if (src_is_mine == dst_is_mine) {
213 ex::start_detached(ex::when_all(src_retiled.read(ij), dst_retiled.readwrite(ij_lc)) |
214 matrix::copy(policy));
215 }
216 else {
217 ex::start_detached(comm::schedule_recv(mpi_chain.shared(), grid.rankFullCommunicator(src_rank),
218 tag(ij), dst_retiled.readwrite(ij_lc)));
219 }
220 }
221}
