136template <class DataIn, class DataOut>
137void copy(const DataIn& src, const DataOut& dest) {
138 static_assert(not std::is_const_v<typename data_traits<DataOut>::element_t>,
139 "Cannot copy to a const Data");
140
141 if (data_iscontiguous(dest) && data_iscontiguous(src)) {
142 DLAF_ASSERT_HEAVY(data_blocksize(src) == data_blocksize(dest), data_blocksize(src),
143 data_blocksize(dest));
144
145 std::copy(data_pointer(src), data_pointer(src) + data_blocksize(src), data_pointer(dest));
146 }
147 else {
148 if (data_iscontiguous(dest)) {
149 DLAF_ASSERT_HEAVY(data_nblocks(src) * data_blocksize(src) == data_blocksize(dest),
150 data_nblocks(src), data_blocksize(src), data_blocksize(dest));
151
152 for (SizeType i_block = 0; i_block < data_nblocks(src); ++i_block) {
153 auto ptr_block_start = data_pointer(src) + i_block * data_stride(src);
154 auto dest_block_start = data_pointer(dest) + i_block * data_blocksize(src);
155 std::copy(ptr_block_start, ptr_block_start + data_blocksize(src), dest_block_start);
156 }
157 }
158 else if (data_iscontiguous(src)) {
159 DLAF_ASSERT_HEAVY(data_blocksize(src) == data_nblocks(dest) * data_blocksize(dest),
160 data_blocksize(src), data_nblocks(dest), data_blocksize(dest));
161
162 for (SizeType i_block = 0; i_block < data_nblocks(dest); ++i_block) {
163 auto ptr_block_start = data_pointer(src) + i_block * data_blocksize(dest);
164 auto dest_block_start = data_pointer(dest) + i_block * data_stride(dest);
165 std::copy(ptr_block_start, ptr_block_start + data_blocksize(dest), dest_block_start);
166 }
167 }
168 else {
169 DLAF_ASSERT_HEAVY(data_count(src) == data_count(dest), data_count(src), data_count(dest));
170
171 for (SizeType i = 0; i < data_count(src); ++i) {
172 auto i_src = i / data_blocksize(src) * data_stride(src) + i % data_blocksize(src);
173 auto i_dest = i / data_blocksize(dest) * data_stride(dest) + i % data_blocksize(dest);
174
175 data_pointer(dest)[i_dest] = data_pointer(src)[i_src];
176 }
177 }
178 }
179}
