64 DataDescriptor(T* ptr, SizeType num_blocks, SizeType blocksize, SizeType stride) noexcept
65 : data_(ptr), nblocks_(num_blocks), blocksize_(blocksize), stride_(num_blocks == 1 ? 0 : stride) {
66 DLAF_ASSERT(num_blocks >= 0, num_blocks);
67 DLAF_ASSERT(blocksize >= 0, blocksize);
68 DLAF_ASSERT(stride >= 0, stride);
69
70 if (blocksize_ == stride_) {
71 blocksize_ = num_blocks * blocksize_;
72 nblocks_ = 1;
73 stride_ = 0;

211auto make_contiguous(DataIn input, Buffer& support_buffer) {
212 using DataT = typename common::data_traits<DataIn>::element_t;
213
214 if (!input.is_contiguous()) {
