585 template <Coord rc>
586 SizeType tile_size_of(SizeType global_tile) const noexcept {
587 DLAF_ASSERT_HEAVY(0 <= global_tile && global_tile < nr_tiles_.get<rc>(), global_tile,
588 nr_tiles_.get<rc>());
589 SizeType n = size_.get<rc>();
590 SizeType nb = tile_size_.get<rc>();
591 if (global_tile == 0) {
592 return std::min(nb - global_tile_element_offset<rc>(), n);
593 }
594 return std::min(nb, n + global_tile_element_offset<rc>() - global_tile * nb);
595 }
