583  template <Coord rc>
584  SizeType tile_size_of(SizeType global_tile) const noexcept {
585  DLAF_ASSERT_HEAVY(0 <= global_tile && global_tile < nr_tiles_.get<rc>(), global_tile,
586  nr_tiles_.get<rc>());
587  SizeType n = size_.get<rc>();
588  SizeType nb = tile_size_.get<rc>();
589  if (global_tile == 0) {
590  return std::min(nb - global_tile_element_offset<rc>(), n);
591  }
592  return std::min(nb, n + global_tile_element_offset<rc>() - global_tile * nb);
593  }
