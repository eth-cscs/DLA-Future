386  template <Coord rc>
387  SizeType localElementDistanceFromGlobalTile(SizeType i_begin, SizeType i_end) const noexcept {
388  DLAF_ASSERT_HEAVY(i_begin <= i_end, i_begin, i_end);
389  DLAF_ASSERT_HEAVY(0 <= i_begin && i_end <= global_nr_tiles_.get<rc>(), i_begin, i_end,
390  global_nr_tiles_.get<rc>());
391  DLAF_ASSERT(tilesPerBlock<rc>() == 1, "Multi Tile distribution block is not supported yet");
392 
393  // Note the second assert is already done by the following calls.
394  SizeType i_loc_begin = nextLocalTileFromGlobalTile<rc>(i_begin);
395  SizeType i_loc_last = nextLocalTileFromGlobalTile<rc>(i_end) - 1;
396  if (i_loc_begin > i_loc_last)
397  return 0;
398  SizeType l = local_size_.get<rc>();
399  SizeType nb = tile_size_.get<rc>();
400  SizeType nbr = std::min(nb, l - i_loc_last * nb); // size of last local tile along `rc`
401  return (i_loc_last - i_loc_begin) * nb + nbr;
402  }
