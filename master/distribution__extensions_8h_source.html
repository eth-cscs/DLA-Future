115template <Coord rc>
116SizeType global_tile_from_local_tile_on_rank(const Distribution& dist, comm::IndexT_MPI rank,
117 SizeType local_tile) noexcept {
118 using util::matrix::global_tile_from_local_tile;
119 // Assertion 0 <= local_tile is performed by global_tile_from_local_tile
120 const SizeType tiles_per_block = dist.block_size().get<rc>() / dist.tile_size().get<rc>();
121 const SizeType global_tile =
122 global_tile_from_local_tile(local_tile, tiles_per_block, dist.grid_size().get<rc>(), rank,
123 dist.source_rank_index().get<rc>(), global_tile_offset<rc>(dist));
124 // Assert on the result to avoid to compute the local number of tiles on the given rank
125 DLAF_ASSERT_HEAVY(0 <= global_tile && global_tile < dist.nr_tiles().get<rc>(), global_tile,
126 dist.nr_tiles().get<rc>());
127 return global_tile;
128}

169inline bool is_compatible_sub_distribution(const Distribution& distribution,
170 const GlobalElementIndex& sub_offset,
171 const Distribution& sub_distribution) noexcept {
172 const bool compatible_grid = distribution.block_size() == sub_distribution.block_size() &&
173 distribution.tile_size() == sub_distribution.tile_size() &&
174 distribution.rank_index() == sub_distribution.rank_index() &&
175 distribution.grid_size() == sub_distribution.grid_size();
176 const bool compatible_source_rank_index =
177 distribution.rank_global_tile(distribution.global_tile_index(sub_offset)) ==
178 sub_distribution.source_rank_index();
179 const bool compatible_size =
180 sub_offset.row() + sub_distribution.size().rows() <= distribution.size().rows() &&
181 sub_offset.col() + sub_distribution.size().cols() <= distribution.size().cols();
182 return compatible_grid && compatible_source_rank_index && compatible_size;
183}

187inline GlobalTileIndex global_tile_index_from_sub_distribution(
188 const Distribution& distribution, const GlobalElementIndex& sub_offset,
189 [[maybe_unused]] const Distribution& sub_distribution, const GlobalTileIndex& sub_index) noexcept {
190 DLAF_ASSERT_HEAVY(sub_index.isIn(sub_distribution.nr_tiles()), sub_index, sub_distribution.nr_tiles());
191 DLAF_ASSERT_HEAVY(is_compatible_sub_distribution(distribution, sub_offset, sub_distribution), "");
192 const GlobalTileIndex tile_offset = distribution.global_tile_index(sub_offset);
193 return tile_offset + common::sizeFromOrigin(sub_index);
194}

207 : 0,

238template <Coord rc>
239SizeType global_tile_element_distance(const Distribution& dist, SizeType i_begin,
240 SizeType i_end) noexcept {
241 DLAF_ASSERT_HEAVY(i_begin <= i_end, i_begin, i_end);
242
243 const SizeType el_begin = dist.global_element_from_global_tile_and_tile_element<rc>(i_begin, 0);
244
245 if (i_end == dist.nr_tiles().get<rc>())
246 return dist.size().get<rc>() - el_begin;
247
248 return dist.global_element_from_global_tile_and_tile_element<rc>(i_end, 0) - el_begin;
249}

253template <Coord rc>
254SizeType local_element_distance_from_local_tile(const Distribution& dist, SizeType i_loc_begin,
255 SizeType i_loc_end) noexcept {
256 DLAF_ASSERT_HEAVY(i_loc_begin <= i_loc_end, i_loc_begin, i_loc_end);
257 DLAF_ASSERT_HEAVY(0 <= i_loc_begin && i_loc_end <= dist.local_nr_tiles().get<rc>(), i_loc_begin,
258 i_loc_end, dist.local_nr_tiles().get<rc>());
259
260 SizeType i_loc_last = i_loc_end - 1;
261 if (i_loc_begin > i_loc_last)
262 return 0;
263
264 SizeType i_el_begin = dist.local_element_from_local_tile_and_tile_element<rc>(i_loc_begin, 0);
265 SizeType i_el_end =
266 dist.local_element_from_local_tile_and_tile_element<rc>(
267 i_loc_last, dist.tile_size_of<rc>(dist.global_tile_from_local_tile<rc>(i_loc_last)) - 1) +
268 1;
269 return i_el_end - i_el_begin;
270}

274template <Coord rc>
275SizeType local_element_distance_from_global_tile(const Distribution& dist, SizeType i_begin,
276 SizeType i_end) noexcept {
277 DLAF_ASSERT_HEAVY(i_begin <= i_end, i_begin, i_end);
278 DLAF_ASSERT_HEAVY(0 <= i_begin && i_end <= dist.nr_tiles().get<rc>(), i_begin, i_end,
279 dist.nr_tiles().get<rc>());
280
281 SizeType i_loc_begin = dist.next_local_tile_from_global_tile<rc>(i_begin);
282 SizeType i_loc_end = dist.next_local_tile_from_global_tile<rc>(i_end);
283 return local_element_distance_from_local_tile<rc>(dist, i_loc_begin, i_loc_end);
284}
