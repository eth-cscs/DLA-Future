26template <class T>
27SizeType nrSweeps(SizeType size) noexcept {
28 // Complex needs an extra sweep to have a real tridiagonal matrix.
29 return isComplex_v<T> ? size - 1 : size - 2;
30}
31
32inline SizeType nrStepsForSweep(SizeType sweep, SizeType size, SizeType band) noexcept {
33 // Sweep size-2 should be handled differently
34 // as it is executed only for complex types (see nrSweeps)
35 return sweep == size - 2 ? 1 : util::ceilDiv(size - sweep - 2, band);
36}
