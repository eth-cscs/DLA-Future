56 template <class T>
57 matrix::Tile<T, Device::CPU> setupVWellFormed(const SizeType b,
58  const matrix::Tile<const T, Device::CPU>& tile_hh,
59  matrix::Tile<T, Device::CPU> tile_v) {
60  using lapack::lacpy;
61  using lapack::laset;
62 
63  common::internal::SingleThreadedBlasScope single;
64 
65  // Note: the size of of tile_hh and tile_v embeds a relevant information about the number of
66  // reflecotrs and their max size. This will be exploited to correctly setup the well formed
67  // tile with reflectors in place as they will be applied.
68  const auto k = tile_v.size().cols();
69 
70  // copy from compact representation reflector values (the first component set to 1 is not there)
71  for (SizeType j = 0; j < k; ++j) {
72  const auto compact_refl_size =
73  std::min<SizeType>(tile_v.size().rows() - (1 + j), tile_hh.size().rows() - 1);
74 
75  // this is needed because of complex last reflector (i.e. just 1 element long)
76  if (compact_refl_size == 0)
77  continue;
78 
79  lacpy(blas::Uplo::General, compact_refl_size, 1, tile_hh.ptr({1, j}), tile_hh.ld(),
80  tile_v.ptr({1 + j, j}), tile_v.ld());
81  }
82 
83  // Note:
84  // In addition to setting the diagonal to 1 for missing first components, here it zeros out
85  // both the upper and the lower part. Indeed due to the skewed shape, reflectors do not occupy
86  // the full tile height, and V should be fully well-formed because the next triangular
87  // multiplication, i.e. `V . T`, and the gemm `V* . E`, will use V as a general matrix.
88  laset(blas::Uplo::Upper, tile_v.size().rows(), k, T(0), T(1), tile_v.ptr({0, 0}), tile_v.ld());
89 
90  if (tile_v.size().rows() > b)
91  laset(blas::Uplo::Lower, tile_v.size().rows() - b, k - 1, T(0), T(0), tile_v.ptr({b, 0}),
92  tile_v.ld());
93 
94  return tile_v;
95 }

97 template <class T>
98 void computeTFactor(const matrix::Tile<const T, Device::CPU>& tile_taus,
99  const matrix::Tile<const T, Device::CPU>& tile_v,
100  const matrix::Tile<T, Device::CPU>& tile_t) {
101  using namespace lapack;
102 
103  common::internal::SingleThreadedBlasScope single;
104 
105  // taus have to be extracted from the compact form (i.e. first row of the input tile)
106  std::vector<T> taus;
107  taus.resize(to_sizet(tile_v.size().cols()));
108  for (SizeType i = 0; i < to_SizeType(taus.size()); ++i)
109  taus[to_sizet(i)] = tile_taus({0, i});
110 
111  const auto n = tile_v.size().rows();
112  const auto k = tile_v.size().cols();
113  larft(Direction::Forward, StoreV::Columnwise, n, k, tile_v.ptr(), tile_v.ld(), taus.data(),
114  tile_t.ptr(), tile_t.ld());
115 }

117 template <class T>
118 std::tuple<matrix::Tile<T, Device::CPU>, matrix::Tile<T, Device::CPU>> computeVT(
119  const SizeType b, const matrix::Tile<const T, Device::CPU>& tile_hh, const SizeType hhr_nb,
120  matrix::Tile<T, Device::CPU> tile_v, matrix::Tile<T, Device::CPU> tile_t) {
121  auto tile_v2 = setupVWellFormed(b, tile_hh, std::move(tile_v));
122  for (SizeType j = 0; j < tile_v2.size().cols(); j += hhr_nb) {
123  const SizeType jb = std::min(hhr_nb, tile_v2.size().cols() - j);
124  const SizeType ib = std::min(jb + b - 1, tile_v2.size().rows() - j);
125  auto subtile_t = tile_t.subTileReference({{j, j}, {jb, jb}});
126  auto subtile_hh = tile_hh.subTileReference({{0, j}, {1, jb}});
127  auto subtile_v = tile_v2.subTileReference({{j, j}, {ib, jb}});
128  computeTFactor(subtile_hh, subtile_v, subtile_t);
129  }
130  return std::make_tuple(std::move(tile_v2), std::move(tile_t));
131 }

133 template <class T>
134 std::tuple<matrix::Tile<T, Device::CPU>, matrix::Tile<T, Device::CPU>> computeVW(
135  const SizeType b, const matrix::Tile<const T, Device::CPU>& tile_hh, const SizeType hhr_nb,
136  matrix::Tile<T, Device::CPU> tile_v, matrix::Tile<T, Device::CPU> tile_t,
137  matrix::Tile<T, Device::CPU> tile_w) {
138  using namespace blas;
139 
140  auto [tile_v2, tile_t2] = computeVT(b, tile_hh, hhr_nb, std::move(tile_v), std::move(tile_t));
141 
142  for (SizeType j = 0; j < tile_v2.size().cols(); j += hhr_nb) {
143  const SizeType jb = std::min(hhr_nb, tile_v2.size().cols() - j);
144  const SizeType ib = std::min(jb + b - 1, tile_v2.size().rows() - j);
145  auto subtile_t = tile_t2.subTileReference({{j, j}, {jb, jb}});
146  auto subtile_v = tile_v2.subTileReference({{j, j}, {ib, jb}});
147  auto subtile_w = tile_w.subTileReference({{j, j}, {ib, jb}});
148 
149  dlaf::tile::internal::trmm3(Side::Right, Uplo::Upper, Op::NoTrans, Diag::NonUnit, T(1), subtile_t,
150  subtile_v, subtile_w);
151  }
152  return std::make_tuple(std::move(tile_v2), std::move(tile_w));
153 }

155 template <class Tile, class CTile>
156 std::tuple<CTile, CTile, Tile, Tile> applyHHToSingleTileRowSubtileHelper( //
157  const SizeType j, const SizeType jb, const CTile& tile_v, const CTile& tile_w, const Tile& tile_w2,
158  const Tile& tile_e) {
159  DLAF_ASSERT_HEAVY(tile_v.size() == tile_w.size(), tile_v, tile_w);
160  DLAF_ASSERT_HEAVY(tile_e.size().rows() - 1 == tile_v.size().rows(), tile_e, tile_v);
161  DLAF_ASSERT_HEAVY(tile_e.size().cols() == tile_w2.size().cols(), tile_e, tile_w2);
162 
163  const SizeType ib = tile_v.size().rows() - j;
164  auto subtile_v = tile_v.subTileReference({{j, j}, {ib, jb}});
165  auto subtile_w = tile_w.subTileReference({{j, j}, {ib, jb}});
166  auto subtile_w2 = tile_w2.subTileReference({{0, 0}, {jb, tile_w2.size().cols()}});
167  auto subtile_e = tile_e.subTileReference({{j + 1, 0}, tile_e.size() - TileElementSize{j + 1, 0}});
168 
169  return {std::move(subtile_v), std::move(subtile_w), std::move(subtile_w2), std::move(subtile_e)};
170 }

175 template <class T>
176 struct ApplyHHToSingleTileRow<Backend::MC, T> {
177  void operator()(const SizeType hhr_nb, const matrix::Tile<const T, Device::CPU>& tile_v,
178  const matrix::Tile<const T, Device::CPU>& tile_w,
179  const matrix::Tile<T, Device::CPU>& tile_w2,
180  const matrix::Tile<T, Device::CPU>& tile_e) {
181  using namespace blas;
182  using tile::internal::gemm;
183 
184  common::internal::SingleThreadedBlasScope single;
185 
186  for (SizeType j = (util::ceilDiv(tile_v.size().cols(), hhr_nb) - 1) * hhr_nb; j >= 0; j -= hhr_nb) {
187  const SizeType jb = std::min(hhr_nb, tile_v.size().cols() - j);
188  auto [subtile_v, subtile_w, subtile_w2, subtile_e] =
189  applyHHToSingleTileRowSubtileHelper(j, jb, tile_v, tile_w, tile_w2, tile_e);
190 
191  // W2 = V* . E
192  gemm(Op::ConjTrans, Op::NoTrans, T(1), subtile_v, subtile_e, T(0), subtile_w2);
193  // E -= W . W2
194  gemm(Op::NoTrans, Op::NoTrans, T(-1), subtile_w, subtile_w2, T(1), subtile_e);
195  }
196  }
197 };

226 template <class Tile, class CTile>
227 std::tuple<CTile, CTile, CTile, CTile, Tile, Tile, Tile> applyHHToDoubleTileRowSubtileHelper( //
228  const SizeType j, const SizeType jb, const CTile& tile_v, const CTile& tile_w, const Tile& tile_w2,
229  const Tile& tile_e_top, const Tile& tile_e_bottom) {
230  DLAF_ASSERT_HEAVY(tile_v.size() == tile_w.size(), tile_v, tile_w);
231  DLAF_ASSERT_HEAVY(tile_e_top.size().rows() + tile_e_bottom.size().rows() - 1 == tile_v.size().rows(),
232  tile_e_top, tile_e_bottom, tile_v);
233  DLAF_ASSERT_HEAVY(tile_e_top.size().cols() == tile_w2.size().cols(), tile_e_top, tile_w2);
234  DLAF_ASSERT_HEAVY(tile_e_bottom.size().cols() == tile_w2.size().cols(), tile_e_bottom, tile_w2);
235 
236  auto subtile_e_top =
237  tile_e_top.subTileReference({{j + 1, 0}, tile_e_top.size() - TileElementSize{j + 1, 0}});
238  auto subtile_e_bottom = tile_e_bottom.subTileReference(
239  {{0, 0},
240  TileElementSize{std::min(tile_e_bottom.size().rows(), j + jb), tile_e_bottom.size().cols()}});
241 
242  matrix::SubTileSpec spec_top{{j, j}, {subtile_e_top.size().rows(), jb}};
243  matrix::SubTileSpec spec_bottom{{tile_e_top.size().rows() - 1, j},
244  {subtile_e_bottom.size().rows(), jb}};
245 
246  auto subtile_v_top = tile_v.subTileReference(spec_top);
247  auto subtile_v_bottom = tile_v.subTileReference(spec_bottom);
248  auto subtile_w_top = tile_w.subTileReference(spec_top);
249  auto subtile_w_bottom = tile_w.subTileReference(spec_bottom);
250  auto subtile_w2 = tile_w2.subTileReference({{0, 0}, {jb, tile_w2.size().cols()}});
251 
252  return {std::move(subtile_v_top), std::move(subtile_v_bottom), std::move(subtile_w_top),
253  std::move(subtile_w_bottom), std::move(subtile_w2), std::move(subtile_e_top),
254  std::move(subtile_e_bottom)};
255 }

321 struct TileAccessHelper {
322  TileAccessHelper(const SizeType b, const SizeType nrefls, const matrix::Distribution& dist_hh,
323  const matrix::Distribution& dist_e, const GlobalElementIndex offset)
324  : nrefls_(nrefls), input_spec_{dist_hh.tileElementIndex(offset),
325  {std::min(b, dist_hh.size().rows() - offset.row()),
326  std::min(b, dist_hh.size().cols() - offset.col())}},
327  index_e_(dist_e.globalTileFromGlobalElement<Coord::Row>(offset.row())) {
328  // Note:
329  // Next logic is about detecting the available application space for reflectors, which once
330  // extracted they expands to a matrix with 2 * b - 1 height.
331  //
332  // There are two main scenarios:
333  // - reflectors involves rows of a single tile;
334  // - reflectors involves rows across two different tiles.
335 
336  rows_v_ = std::min(2 * b, dist_e.size().rows() - offset.row()) - 1;
337  rows_v_top_ = std::min(rows_v_, b - 1);
338 
339  // Note:
340  // In general, keep in mind that the first row of the top tile_e should not be touched.
341  // The structs ApplyHHTo... take care of it.
342  //
343  // e.g. b = 4
344  // reflectors matrix
345  // X X X X
346  // 1 0 0 0 X X X X
347  // a 1 0 0 X X X X
348  // a b 1 0 X X X X
349  // -------
350  // a b c 1 Y Y Y Y
351  // 0 b c d Y Y Y Y
352  // 0 0 c d Y Y Y Y
353  // 0 0 0 d Y Y Y Y
354  //
355  // From the drawing above, it is possible to see the dashed tile separation between X and Y,
356  // and how the reflectors on the left are going to be applied. In particular, the first row of
357  // the upper tile is not affected.
358 
359  // Note:
360  // A single tile is involved only if it is the last row tile, so by construction reflectors will be
361  // applied to a single tile;
362  across_tiles_ = !(index_e_ == dist_e.nrTiles().rows() - 1);
363 
364  across_ranks_ = across_tiles_ && (dist_e.rankGlobalTile<Coord::Row>(index_e_) !=
365  dist_e.rankGlobalTile<Coord::Row>(index_e_ + 1));
366  }
367 
368  // Return true if the application of Householder reflectors involves multiple tiles
369  bool affectsMultipleTiles() const noexcept {
370  return across_tiles_;
371  }
372 
373  // Return true if the application of Householder reflectors involves multiple ranks
374  bool affectsMultipleRanks() const noexcept {
375  return across_ranks_;
376  }
377 
378  // Return SubTileSpec to use for accessing Householder reflectors in compact form
379  //
380  // SubTileSpec returned points to the sub-block in the full tile containing the HH data in compact
381  // form. If @p reset_origin is true, then the origin component of the SubTileSpec is resetted and
382  // it will just describe the size of the sub-block containing the Householder reflectors (useful
383  // for panel access which might not have full-tiles).
384  matrix::SubTileSpec specHHCompact(const bool reset_origin = false) const noexcept {
385  if (reset_origin)
386  return {{0, 0}, input_spec_.size};
387  return input_spec_;
388  }
389 
390  // Return SubTileSpec to use for accessing Householder reflectors in well formed form
391  matrix::SubTileSpec specHH() const noexcept {
392  return {{0, 0}, {rows_v_, nrefls_}};
393  }
394 
395  // Return SubTileSpec to use for accessing Householder reflectors in well formed form
396  // for the distributed case.
397  matrix::SubTileSpec specHH(bool is_top_rank) const noexcept {
398  if (is_top_rank)
399  return {{0, 0}, {rows_v_top_, nrefls_}};
400  DLAF_ASSERT_MODERATE(affectsMultipleTiles(), affectsMultipleTiles());
401  return {{rows_v_top_, 0}, {rows_v_ - rows_v_top_, nrefls_}};
402  }
403 
404  // Return SubTileSpec to use for accessing T factor
405  matrix::SubTileSpec specT() const noexcept {
406  return {{0, 0}, {nrefls_, nrefls_}};
407  }
408 
409  matrix::SubTileSpec specW2(const SizeType cols) const noexcept {
410  return {{0, 0}, {nrefls_, cols}};
411  }
412 
413  GlobalTileIndex topIndexE(const SizeType j) const noexcept {
414  return {index_e_, j};
415  }
416 
417  GlobalTileIndex bottomIndexE(const SizeType j) const noexcept {
418  DLAF_ASSERT_MODERATE(affectsMultipleTiles(), affectsMultipleTiles());
419  return {index_e_ + 1, j};
420  }
421 
422 private:
423  SizeType nrefls_;
424  matrix::SubTileSpec input_spec_;
425  SizeType index_e_;
426  SizeType rows_v_;
427  SizeType rows_v_top_;
428 
429  bool across_tiles_;
430  bool across_ranks_;
431 };

445 struct DistIndexing {
446  DistIndexing(const TileAccessHelper& helper, const matrix::Distribution& dist_hh, const SizeType b,
447  const GlobalTileIndex& ij, const SizeType& ij_b_row)
448  : dist_hh(dist_hh), b(b), mb(dist_hh.blockSize().rows()), helper(helper), ij(ij),
449  ij_b_row(ij_b_row) {
450  rank = dist_hh.rankIndex();
451  rankHH = dist_hh.rankGlobalTile(ij);
452  n_ws_per_block = to_SizeType(static_cast<size_t>(std::ceil(mb / b / 2.0f)) + 1);
453  }
454 
455  comm::IndexT_MPI rankRowPartner() const {
456  return (rankHH.row() + 1) % dist_hh.commGridSize().rows();
457  }
458 
459  bool isInvolved() const {
460  const bool isSameRow = rank.row() == rankHH.row();
461  const bool isPartnerRow = rank.row() == rankRowPartner();
462  return isSameRow || (isPartnerRow && helper.affectsMultipleRanks());
463  }
464 
465  LocalTileIndex wsIndexHH() const {
466  const SizeType row = [&]() -> SizeType {
467  if (rank.row() == rankHH.row()) {
468  // Note: index starts at 1 (0 is the extra workspace), moreover max half blocks will run in parallel
469  const SizeType intra_idx = 1 + (ij_b_row % (mb / b)) / 2;
470  DLAF_ASSERT_HEAVY(intra_idx < n_ws_per_block, intra_idx, n_ws_per_block);
471  return dist_hh.localTileFromGlobalTile<Coord::Row>(ij.row()) * n_ws_per_block + intra_idx;
472  }
473  else {
474  DLAF_ASSERT_HEAVY(helper.affectsMultipleRanks() && (rank.row() == rankRowPartner()),
475  helper.affectsMultipleRanks(), rank.row(), rankRowPartner());
476  return dist_hh.localNrTiles().isEmpty()
477  ? 0

479  }

604 template <Backend B, Device D, class T>
605 void BackTransformationT2B<B, D, T>::call(const SizeType band_size, Matrix<T, D>& mat_e,
606  Matrix<const T, Device::CPU>& mat_hh) {
607  using pika::execution::thread_priority;
608  using pika::execution::thread_stacksize;
609  namespace ex = pika::execution::experimental;
610 
611  using common::iterate_range2d;
612  using common::RoundRobin;
613  using matrix::Panel;
614  using namespace bt_tridiag;
615 
616  if (mat_hh.size().isEmpty() || mat_e.size().isEmpty())
617  return;
618 
619  // Note: if no householder reflectors are going to be applied (in case of trivial matrix)
620  if (mat_hh.size().rows() <= (dlaf::isComplex_v<T> ? 1 : 2))
621  return;
622 
623  const SizeType b = band_size;
624  const SizeType group_size = getTuneParameters().bt_band_to_tridiag_hh_apply_group_size;
625  const SizeType nsweeps = nrSweeps<T>(mat_hh.size().cols());
626 
627  const LocalTileSize tiles_per_block(mat_e.blockSize().rows() / b, 1);
628  Matrix<T, D> mat_e_rt = mat_e.retiledSubPipeline(tiles_per_block);
629 
630  const auto& dist_hh = mat_hh.distribution();
631  const auto& dist_e_rt = mat_e_rt.distribution();
632 
633  // Note: w_tile_sz can store reflectors as they are actually applied, opposed to how they are
634  // stored in compact form.
635  //
636  // e.g. Given b = 4
637  //
638  // compact w_tile_sz
639  // 1 1 1 1 1 0 0 0
640  // a b c d a 1 0 0
641  // a b c d a b 1 0
642  // a b c d a b c 1
643  // 0 b c d
644  // 0 0 c d
645  // 0 0 0 d
646  const TileElementSize w_tile_sz(2 * b - 1, b);
647 
648  const SizeType dist_w_rows = mat_e_rt.nrTiles().rows() * w_tile_sz.rows();
649  const matrix::Distribution dist_w({dist_w_rows, b}, w_tile_sz);
650  const matrix::Distribution dist_t({mat_hh.size().rows(), b}, {b, b});
651  const matrix::Distribution dist_w2({b, mat_e_rt.size().cols()}, {b, mat_e_rt.blockSize().cols()});
652 
653  constexpr std::size_t n_workspaces = 2;
654  RoundRobin<Panel<Coord::Col, T, D>> t_panels(n_workspaces, dist_t);
655  RoundRobin<Panel<Coord::Col, T, D>> v_panels(n_workspaces, dist_w);
656  RoundRobin<Panel<Coord::Col, T, D>> w_panels(n_workspaces, dist_w);
657  RoundRobin<Panel<Coord::Row, T, D>> w2_panels(n_workspaces, dist_w2);
658 
659  HHManager<B, D, T> helperBackend(b, n_workspaces, dist_t, dist_w);
660 
661  // Note: sweep are on diagonals, steps are on verticals
662  const SizeType j_last_sweep = (nsweeps - 1) / b;
663  for (SizeType j = j_last_sweep; j >= 0; --j) {
664  auto& mat_t = t_panels.nextResource();
665  auto& mat_v = v_panels.nextResource();
666  auto& mat_w = w_panels.nextResource();
667  auto& mat_w2 = w2_panels.nextResource();
668 
669  // Note: apply the entire column (steps)
670  const SizeType steps = nrStepsForSweep(j * b, mat_hh.size().cols(), b);
671  for (SizeType step = 0; step < steps; ++step) {
672  const SizeType i = j + step;
673 
674  const GlobalElementIndex ij_el(i * b, j * b);
675  const LocalTileIndex ij(dist_hh.localTileIndex(dist_hh.globalTileIndex(ij_el)));
676 
677  // Note: reflector with size = 1 must be ignored, except for the last step of the last sweep
678  // with complex type
679  const SizeType nrefls = [&]() {
680  const bool allowSize1 = isComplex_v<T> && j == j_last_sweep && step == steps - 1;
681  const GlobalElementSize delta(dist_hh.size().rows() - ij_el.row() - 1,
682  std::min(b, dist_hh.size().cols() - ij_el.col()));
683  return std::min(b, std::min(delta.rows() - (allowSize1 ? 0 : 1), delta.cols()));
684  }();
685 
686  const TileAccessHelper helper(b, nrefls, dist_hh, dist_e_rt, ij_el);
687 
688  if (nrefls < b) {
689  mat_t.setWidth(nrefls);
690  mat_v.setWidth(nrefls);
691  mat_w.setWidth(nrefls);
692  mat_w2.setHeight(nrefls);
693  }
694 
695  auto [tile_v_unshared, tile_w_unshared] =
696  helperBackend.computeVW(group_size, ij, helper,
697  splitTile(mat_hh.read(ij), helper.specHHCompact()), mat_v, mat_t,
698  mat_w);
699  auto tile_v = matrix::shareReadWriteTile(ex::make_unique_any_sender(std::move(tile_v_unshared)));
700  auto tile_w = matrix::shareReadWriteTile(ex::make_unique_any_sender(std::move(tile_w_unshared)));
701 
702  for (SizeType j_e = 0; j_e < dist_e_rt.nrTiles().cols(); ++j_e) {
703  const auto idx_e = helper.topIndexE(j_e);
704 
705  if (!helper.affectsMultipleTiles()) {
706  ex::start_detached(
707  ex::when_all(ex::just(group_size), tile_v, tile_w,
708  mat_w2.readwrite(LocalTileIndex(0, j_e)), mat_e_rt.readwrite(idx_e)) |
709  dlaf::internal::transform<dlaf::internal::TransformDispatchType::Blas>(
710  dlaf::internal::Policy<B>(thread_priority::normal, thread_stacksize::nostack),
711  ApplyHHToSingleTileRow<B, T>{}));
712  }
713  else {
714  ex::start_detached(
715  ex::when_all(ex::just(group_size), tile_v, tile_w,
716  mat_w2.readwrite(LocalTileIndex(0, j_e)), mat_e_rt.readwrite(idx_e),
717  mat_e_rt.readwrite(helper.bottomIndexE(j_e))) |
718  dlaf::internal::transform<dlaf::internal::TransformDispatchType::Blas>(
719  dlaf::internal::Policy<B>(thread_priority::normal, thread_stacksize::nostack),
720  ApplyHHToDoubleTileRow<B, T>{}));
721  }
722  }
723 
724  mat_t.reset();
725  mat_v.reset();
726  mat_w.reset();
727  mat_w2.reset();
728  }
729  }
730 }

732 template <Backend B, Device D, class T>
733 void BackTransformationT2B<B, D, T>::call(comm::CommunicatorGrid grid, const SizeType band_size,
734  Matrix<T, D>& mat_e, Matrix<const T, Device::CPU>& mat_hh) {
735  using pika::execution::thread_priority;
736  using pika::execution::thread_stacksize;
737  namespace ex = pika::execution::experimental;
738 
739  using common::iterate_range2d;
740  using common::RoundRobin;
741  using matrix::Panel;
742  using namespace bt_tridiag;
743 
744  if (mat_hh.size().isEmpty() || mat_e.size().isEmpty())
745  return;
746 
747  // Note: if no householder reflectors are going to be applied (in case of trivial matrix)
748  if (nrSweeps<T>(mat_hh.size().rows()) == 0)
749  return;
750 
751  const SizeType b = band_size;
752  const SizeType mb = mat_hh.blockSize().rows();
753  const SizeType group_size = getTuneParameters().bt_band_to_tridiag_hh_apply_group_size;
754 
755  const LocalTileSize tiles_per_block(mat_e.blockSize().rows() / b, 1);
756  Matrix<T, D> mat_e_rt = mat_e.retiledSubPipeline(tiles_per_block);
757 
758  const auto& dist_hh = mat_hh.distribution();
759  const auto& dist_e_rt = mat_e_rt.distribution();
760 
761  // Note: w_tile_sz can store reflectors as they are actually applied, opposed to how they are
762  // stored in compact form.
763  //
764  // e.g. Given b = 4
765  //
766  // compact w_tile_sz
767  // 1 1 1 1 1 0 0 0
768  // a b c d a 1 0 0
769  // a b c d a b 1 0
770  // a b c d a b c 1
771  // 0 b c d
772  // 0 0 c d
773  // 0 0 0 d
774  const TileElementSize w_tile_sz(2 * b - 1, b);
775 
776  const SizeType nlocal_ws =
777  std::max<SizeType>(1, dist_hh.localNrTiles().rows() * (util::ceilDiv<SizeType>(mb / b, 2) + 1));
778  const matrix::Distribution dist_ws_hh({nlocal_ws * b, b}, {b, b});
779  const matrix::Distribution dist_ws_v({nlocal_ws * w_tile_sz.rows(), w_tile_sz.cols()}, w_tile_sz);
780  const matrix::Distribution dist_ws_w2({nlocal_ws * b, mat_e_rt.size().cols()},
781  {b, mat_e_rt.blockSize().cols()});
782 
783  constexpr std::size_t n_workspaces = 2;
784 
785  RoundRobin<Panel<Coord::Col, T, D>> t_panels(n_workspaces, dist_ws_hh);
786  RoundRobin<Panel<Coord::Col, T, Device::CPU>> hh_panels(n_workspaces, dist_ws_hh);
787 
788  RoundRobin<Panel<Coord::Col, T, D>> v_panels(n_workspaces, dist_ws_v);
789  RoundRobin<Panel<Coord::Col, T, D>> w_panels(n_workspaces, dist_ws_v);
790 
791  RoundRobin<Panel<Coord::Row, T, D>> w2_panels(n_workspaces, dist_ws_w2);
792  RoundRobin<Panel<Coord::Row, T, D>> w2tmp_panels(n_workspaces, dist_ws_w2);
793 
794  HHManager<B, D, T> helperBackend(b, n_workspaces, dist_ws_hh, dist_ws_v);
795 
796  // Note: This distributed algorithm encompass two communication categories:
797  // 1. exchange of HH: broadcast + send p2p
798  // 2. reduction for computing W2: all reduce p2p
799  // P2P communication can happen out of order since they can be matched via tags, but this is not
800  // possible for collective operations such as the broadcast.
801  //
802  // For this reason, communications of the phase 1 will be ordered with a pipeline. Instead, for the
803  // second part, with the aim to not over constrain execution of the update, no order will be
804  // enforced by relying solely on tags.
805  common::Pipeline<comm::Communicator> mpi_chain_row(grid.rowCommunicator().clone());
806  common::Pipeline<comm::Communicator> mpi_chain_col(grid.colCommunicator().clone());
807  const auto mpi_col_comm = ex::just(grid.colCommunicator().clone());
808 
809  const SizeType idx_last_sweep_b = (nrSweeps<T>(mat_hh.size().cols()) - 1) / b;
810  const SizeType maxsteps_b = nrStepsForSweep(0, mat_hh.size().rows(), b);
811 
812  // Note: Next two nested `for`s describe a special order loop over the matrix, which allow to
813  // better schedule communications considering the structure of the algorithm.
814  //
815  // Each element depends on:
816  // - top
817  // - bottom-right
818  // - right
819  //
820  // This basic rule for dependencies can be described collectively as a mechanism where elements are
821  // "unlocked" in different epochs, which forms a pattern like if the matrix get scanned not
822  // perpendicularly to their main axis, but instead it gets scanned by a slightly skewed line that goes
823  // from top right to bottom left.
824  //
825  // 5 x x x x
826  // 6 4 x x x
827  // 7 5 3 x x
828  // 8 6 4 2 x
829  // 9 7 5 3 1
830  //
831  // Elements of the same epoch are somehow "independent" and so they can potentially run in parallel,
832  // given that previous epoch has been completed. Since scheduling happens sequentially, elements
833  // of the same epoch will be ordered starting from top-most one, resulting in
834  //
835  // 7 x x x x
836  // 10 5 x x x
837  // 12 8 3 x x
838  // 14 11 6 2 x
839  // 15 13 9 4 1
840  for (SizeType k = idx_last_sweep_b; k > -maxsteps_b; --k) {
841  auto& mat_t = t_panels.nextResource();
842  auto& panel_hh = hh_panels.nextResource();
843  auto& mat_v = v_panels.nextResource();
844  auto& mat_w = w_panels.nextResource();
845  auto& mat_w2 = w2_panels.nextResource();
846  auto& mat_w2tmp = w2tmp_panels.nextResource();
847 
848  for (SizeType i_b = std::abs<SizeType>(k), j_b = std::max<SizeType>(0, k);
849  i_b < j_b + nrStepsForSweep(j_b * b, mat_hh.size().cols(), b); i_b += 2, ++j_b) {
850  const SizeType step_b = i_b - j_b;
851  const GlobalElementIndex ij_el(i_b * b, j_b * b);
852  const GlobalTileIndex ij_g(dist_hh.globalTileIndex(ij_el));
853 
854  const comm::Index2D rank = dist_hh.rankIndex();
855  const comm::Index2D rankHH = dist_hh.rankGlobalTile(ij_g);
856 
857  // Note: reflector with size = 1 must be ignored, except for the last step of the last sweep
858  // with complex type
859  const SizeType nrefls = [&]() {
860  const bool allowSize1 = isComplex_v<T> && j_b == idx_last_sweep_b &&
861  step_b == nrStepsForSweep(j_b * b, mat_hh.size().cols(), b) - 1;
862  const GlobalElementSize delta(dist_hh.size().rows() - ij_el.row() - 1,
863  std::min(b, dist_hh.size().cols() - ij_el.col()));
864  return std::min(b, std::min(delta.rows() - (allowSize1 ? 0 : 1), delta.cols()));
865  }();
866 
867  const TileAccessHelper helper(b, nrefls, dist_hh, dist_e_rt, ij_el);
868  const DistIndexing indexing_helper(helper, dist_hh, b, ij_g, i_b);
869 
870  if (!indexing_helper.isInvolved())
871  continue;
872 
873  if (nrefls < b) {
874  mat_t.setWidth(nrefls);
875  mat_v.setWidth(nrefls);
876  mat_w.setWidth(nrefls);
877  mat_w2.setHeight(nrefls);
878  mat_w2tmp.setHeight(nrefls);
879  }
880 
881  // Note:
882  // From HH it is possible to extract V that is needed for computing W and W2, both required
883  // for updating E.
884 
885  // Send HH to all involved ranks: broadcast on row + send p2p on col
886  const LocalTileIndex ij_hh_panel = indexing_helper.wsIndexHH();
887 
888  // Broadcast on ROW
889  if (grid.size().cols() > 1 && rank.row() == rankHH.row()) {
890  if (rank.col() == rankHH.col()) {
891  ex::start_detached(comm::scheduleSendBcast(
892  mpi_chain_row(), splitTile(mat_hh.read(ij_g), helper.specHHCompact())));
893  }
894  else {
895  ex::start_detached(comm::scheduleRecvBcast(mpi_chain_row(), rankHH.col(),
896  splitTile(panel_hh.readwrite(ij_hh_panel),
897  helper.specHHCompact(true))));
898  }
899  }
900 
901  // Jump to the next loop iteration if the local part of E is not affected by the update.
902  const SizeType ncols_local = dist_e_rt.localNrTiles().cols();
903  if (ncols_local == 0)
904  continue;
905 
906  // Send P2P on col
907  if (helper.affectsMultipleRanks()) {
908  const comm::IndexT_MPI rank_src = rankHH.row();
909  const comm::IndexT_MPI rank_dst = indexing_helper.rankRowPartner();
910 
911  if (rank.row() == rank_src) {
912  auto tile_hh = rank.col() == rankHH.col()
913  ? splitTile(mat_hh.read(ij_g), helper.specHHCompact())
914  : splitTile(panel_hh.read(ij_hh_panel), helper.specHHCompact(true));
915  ex::start_detached(comm::scheduleSend(mpi_chain_col(), rank_dst, 0, std::move(tile_hh)));
916  }
917  else if (rank.row() == rank_dst) {
918  ex::start_detached(comm::scheduleRecv(mpi_chain_col(), rank_src, 0,
919  panel_hh.readwrite(ij_hh_panel)));
920  }
921  }
922 
923  // Batched application increases the communication messages and volume,
924  // therefore we use it only for local applications.
925  const SizeType current_group_size = helper.affectsMultipleRanks() ? b : group_size;
926 
927  // COMPUTE V and W from HH and T
928  auto tile_hh = (rankHH == rank)
929  ? splitTile(mat_hh.read(ij_g), helper.specHHCompact())
930  : splitTile(panel_hh.read(ij_hh_panel), helper.specHHCompact(true));
931  auto [tile_v_unshared, tile_w_unshared] =
932  helperBackend.computeVW(current_group_size, indexing_helper.wsIndexHH(), helper,
933  std::move(tile_hh), mat_v, mat_t, mat_w);
934  auto tile_v = matrix::shareReadWriteTile(ex::make_unique_any_sender(std::move(tile_v_unshared)));
935  auto tile_w = matrix::shareReadWriteTile(ex::make_unique_any_sender(std::move(tile_w_unshared)));
936 
937  // UPDATE E
938  for (SizeType j_e = 0; j_e < ncols_local; ++j_e) {
939  const SizeType j_e_g = dist_e_rt.template globalTileFromLocalTile<Coord::Col>(j_e);
940  const LocalTileIndex idx_w2(indexing_helper.wsIndexHH().row(), j_e);
941 
942  const GlobalTileIndex idx_e_top = helper.topIndexE(j_e_g);
943  const auto nb = mat_e_rt.tileSize(idx_e_top).cols();
944 
945  // SINGLE ROW UPDATE
946  if (!helper.affectsMultipleTiles()) {
947  ex::start_detached(ex::when_all(ex::just(current_group_size), tile_v, tile_w,
948  splitTile(mat_w2.readwrite(idx_w2), helper.specW2(nb)),
949  mat_e_rt.readwrite(idx_e_top)) |
950  dlaf::internal::transform<dlaf::internal::TransformDispatchType::Blas>(
951  dlaf::internal::Policy<B>(thread_priority::normal,
952  thread_stacksize::nostack),
953  ApplyHHToSingleTileRow<B, T>{}));
954  }
955  // TWO ROWs
956  else {
957  const GlobalTileIndex idx_e_bottom = helper.bottomIndexE(j_e_g);
958 
959  // TWO ROWs (same RANK)
960  if (!helper.affectsMultipleRanks()) {
961  ex::start_detached(
962  ex::when_all(ex::just(current_group_size), tile_v, tile_w,
963  splitTile(mat_w2.readwrite(idx_w2), helper.specW2(nb)),
964  mat_e_rt.readwrite(idx_e_top), mat_e_rt.readwrite(idx_e_bottom)) |
965  dlaf::internal::transform<dlaf::internal::TransformDispatchType::Blas>(
