128 template <Backend backend, Device device, class T>
129 void BackTransformationReductionToBand<backend, device, T>::call(
130  const SizeType b, Matrix<T, device>& mat_c, Matrix<const T, device>& mat_v,
131  Matrix<const T, Device::CPU>& mat_taus) {
132  using namespace bt_red_band;
133 
134  auto hp = pika::execution::thread_priority::high;
135  auto np = pika::execution::thread_priority::normal;
136 
137  const SizeType m = mat_c.nrTiles().rows();
138  const SizeType n = mat_c.nrTiles().cols();
139  const SizeType mb = mat_v.blockSize().rows();
140 
141  if (m == 0 || n == 0)
142  return;
143 
144  // Note: "-1" added to deal with size 1 reflector.
145  const SizeType total_nr_reflector = mat_v.size().rows() - b - 1;
146 
147  if (total_nr_reflector <= 0)
148  return;
149 
150  const auto dist_v = mat_v.distribution();
151  const auto dist_c = mat_c.distribution();
152 
153  constexpr std::size_t n_workspaces = 2;
154  common::RoundRobin<matrix::Panel<Coord::Col, T, device>> panelsV(n_workspaces, dist_v);
155  common::RoundRobin<matrix::Panel<Coord::Col, T, device>> panelsW(n_workspaces, dist_v);
156  common::RoundRobin<matrix::Panel<Coord::Row, T, device>> panelsW2(n_workspaces, dist_c);
157 
158  dlaf::matrix::Distribution dist_t({mb, total_nr_reflector}, {mb, mb});
159  matrix::Panel<Coord::Row, T, device> panelT(dist_t);
160 
161  const SizeType nr_reflector_blocks = dist_t.nrTiles().cols();
162 
163  for (SizeType k = nr_reflector_blocks - 1; k >= 0; --k) {
164  bool is_last = (k == nr_reflector_blocks - 1);
165  const SizeType nr_reflectors = dist_t.tileSize({0, k}).cols();
166 
167  const GlobalElementIndex v_offset(k * mb + b, k * mb);
168  const GlobalElementIndex c_offset(k * mb + b, 0);
169 
170  const matrix::SubPanelView panel_view(dist_v, v_offset, nr_reflectors);
171  const matrix::SubMatrixView mat_c_view(dist_c, c_offset);
172 
173  auto& panelV = panelsV.nextResource();
174  auto& panelW = panelsW.nextResource();
175  auto& panelW2 = panelsW2.nextResource();
176 
177  panelV.setRangeStart(v_offset);
178  panelW.setRangeStart(v_offset);
179 
180  if (is_last) {
181  panelT.setHeight(nr_reflectors);
182  panelW2.setHeight(nr_reflectors);
183  panelW.setWidth(nr_reflectors);
184  panelV.setWidth(nr_reflectors);
185  }
186 
187  for (const auto& i : panel_view.iteratorLocal()) {
188  // Column index of the HH reflector which starts in the first row of this tile.
189  const SizeType j_diag =
190  std::max<SizeType>(0, i.row() * mat_v.blockSize().rows() - panel_view.offsetElement().row());
191 
192  if (j_diag < nr_reflectors) {
193  auto tile_v = splitTile(mat_v.read(i), panel_view(i));
194  copyAndSetHHUpperTiles<backend>(j_diag, std::move(tile_v), panelV.readwrite(i));
195  }
196  else if (j_diag < mb) {
197  panelV.setTile(i, splitTile(mat_v.read(i), panel_view(i)));
198  }
199  else {
200  panelV.setTile(i, mat_v.read(i));
201  }
202  }
203 
204  const LocalTileIndex taus_index{Coord::Row, k};
205  const LocalTileIndex t_index{Coord::Col, k};
206  dlaf::factorization::internal::computeTFactor<backend>(panelV, mat_taus.read(taus_index),
207  panelT.readwrite(t_index));
208 
209  // W = V T
210  auto tile_t = panelT.read(t_index);
211  for (const auto& idx : panelW.iteratorLocal()) {
212  trmmPanel<backend>(np, tile_t, panelV.read(idx), panelW.readwrite(idx));
213  }
214 
215  // W2 = W C
216  matrix::util::set0<backend>(hp, panelW2);
217  for (const auto& ij : mat_c_view.iteratorLocal()) {
218  gemmUpdateW2<backend>(np, panelW.read(ij), splitTile(mat_c.read(ij), mat_c_view(ij)),
219  panelW2.readwrite(ij));
220  }
221 
222  // Update trailing matrix: C = C - V W2
223  for (const auto& ij : mat_c_view.iteratorLocal()) {
224  gemmTrailingMatrix<backend>(np, panelV.read(ij), panelW2.read(ij),
225  splitTile(mat_c.readwrite(ij), mat_c_view(ij)));
226  }
227 
228  panelV.reset();
229  panelW.reset();
230  panelW2.reset();
231  panelT.reset();
232  }
233 }

235 template <Backend B, Device D, class T>
236 void BackTransformationReductionToBand<B, D, T>::call(comm::CommunicatorGrid& grid, const SizeType b,
237  Matrix<T, D>& mat_c, Matrix<const T, D>& mat_v,
238  Matrix<const T, Device::CPU>& mat_taus) {
239  namespace ex = pika::execution::experimental;
240  using namespace bt_red_band;
241 
242  auto hp = pika::execution::thread_priority::high;
243  auto np = pika::execution::thread_priority::normal;
244 
245  // Set up MPI
246  auto mpi_col_task_chain = grid.col_communicator_pipeline();
247  auto mpi_row_task_chain = grid.row_communicator_pipeline();
248 
249  auto dist_v = mat_v.distribution();
250  auto dist_c = mat_c.distribution();
251 
252  const SizeType m = mat_c.nrTiles().rows();
253  const SizeType n = mat_c.nrTiles().cols();
254  const SizeType mb = mat_v.blockSize().rows();
255 
256  const comm::Index2D this_rank = grid.rank();
257 
258  if (m == 0 || n == 0)
259  return;
260 
261  // Note: "-1" added to deal with size 1 reflector.
262  const SizeType total_nr_reflector = mat_v.size().cols() - b - 1;
263 
264  if (total_nr_reflector <= 0)
265  return;
266 
267  constexpr std::size_t n_workspaces = 2;
268  common::RoundRobin<matrix::Panel<Coord::Col, T, D>> panelsV(n_workspaces, dist_v);
269  common::RoundRobin<matrix::Panel<Coord::Col, T, D>> panelsW(n_workspaces, dist_v);
270  common::RoundRobin<matrix::Panel<Coord::Row, T, D>> panelsW2(n_workspaces, dist_c);
271 
272  dlaf::matrix::Distribution dist_t({mb, total_nr_reflector}, {mb, mb}, grid.size(), this_rank,
273  dist_v.sourceRankIndex());
274  matrix::Panel<Coord::Row, T, D> panelT(dist_t);
275 
276  const SizeType nr_reflector_blocks = dist_t.nrTiles().cols();
277 
278  for (SizeType k = nr_reflector_blocks - 1; k >= 0; --k) {
279  const bool is_last = (k == nr_reflector_blocks - 1);
280  const SizeType nr_reflectors = dist_t.tileSize({0, k}).cols();
281 
282  const GlobalElementIndex v_offset(k * mb + b, k * mb);
283  const GlobalElementIndex c_offset(k * mb + b, 0);
284 
285  const matrix::SubPanelView panel_view(dist_v, v_offset, nr_reflectors);
286  const matrix::SubMatrixView mat_c_view(dist_c, c_offset);
287 
288  auto& panelV = panelsV.nextResource();
289  auto& panelW = panelsW.nextResource();
290  auto& panelW2 = panelsW2.nextResource();
291 
292  panelV.setRangeStart(v_offset);
293  panelW.setRangeStart(v_offset);
294  panelT.setRange(GlobalTileIndex(Coord::Col, k), GlobalTileIndex(Coord::Col, k + 1));
295 
296  if (is_last) {
297  panelT.setHeight(nr_reflectors);
298  panelW2.setHeight(nr_reflectors);
299  panelW.setWidth(nr_reflectors);
300  panelV.setWidth(nr_reflectors);
301  }
302 
303  const comm::IndexT_MPI k_rank_col = dist_v.template rankGlobalTile<Coord::Col>(k);
304 
305  if (this_rank.col() == k_rank_col) {
306  for (const auto& ik : panel_view.iteratorLocal()) {
307  const SizeType i_row_g = dist_v.template globalTileFromLocalTile<Coord::Row>(ik.row());
308 
309  // Column index of the HH reflector which starts in the first row of this tile.
310  const SizeType j_diag =
311  std::max<SizeType>(0, i_row_g * mat_v.blockSize().rows() - panel_view.offsetElement().row());
312 
313  if (j_diag < nr_reflectors) {
314  auto tile_v = splitTile(mat_v.read(ik), panel_view(ik));
315  copyAndSetHHUpperTiles<B>(j_diag, tile_v, panelV.readwrite(ik));
316  }
317  else if (j_diag < mb) {
318  panelV.setTile(ik, splitTile(mat_v.read(ik), panel_view(ik)));
319  }
320  else {
321  panelV.setTile(ik, mat_v.read(ik));
322  }
323  }
324 
325  const GlobalTileIndex taus_index{Coord::Row, k};
326  const SizeType k_local = dist_t.template localTileFromGlobalTile<Coord::Col>(k);
327  const LocalTileIndex t_index{Coord::Col, k_local};
328  dlaf::factorization::internal::computeTFactor<B>(panelV, mat_taus.read(taus_index),
329  panelT.readwrite(t_index), mpi_col_task_chain);
330 
331  // WH = V T
332  for (const auto& idx : panel_view.iteratorLocal()) {
333  trmmPanel<B>(np, panelT.read(t_index), panelV.read(idx), panelW.readwrite(idx));
334  }
335  }
336 
337  matrix::util::set0<B>(hp, panelW2);
338 
339  broadcast(k_rank_col, panelW, mpi_row_task_chain);
340 
341  // W2 = W C
342  for (const auto& ij : mat_c_view.iteratorLocal()) {
343  gemmUpdateW2<B>(np, panelW.readwrite(ij), splitTile(mat_c.read(ij), mat_c_view(ij)),
344  panelW2.readwrite(ij));
345  }
346 
347  if (mpi_col_task_chain.size() > 1) {
348  for (const auto& kj_panel : panelW2.iteratorLocal())
349  ex::start_detached(dlaf::comm::scheduleAllReduceInPlace(mpi_col_task_chain.exclusive(), MPI_SUM,
350  panelW2.readwrite(kj_panel)));
351  }
352 
353  broadcast(k_rank_col, panelV, mpi_row_task_chain);
354 
355  // C = C - V W2
356  for (const auto& ij : mat_c_view.iteratorLocal()) {
357  gemmTrailingMatrix<B>(np, panelV.read(ij), panelW2.read(ij),
358  splitTile(mat_c.readwrite(ij), mat_c_view(ij)));
359  }
360 
361  panelV.reset();
362  panelW.reset();
363  panelW2.reset();
364  panelT.reset();
365  }
366 }
