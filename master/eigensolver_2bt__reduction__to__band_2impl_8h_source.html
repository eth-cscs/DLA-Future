53 template <class T>
54 static void copyAndSetHHUpperTiles(SizeType j_diag, const matrix::Tile<const T, Device::CPU>& src,
55 const matrix::Tile<T, Device::CPU>& dst) {
56 matrix::internal::copy_o(src, dst);
57 common::internal::SingleThreadedBlasScope single;
58 lapack::laset(blas::Uplo::Upper, dst.size().rows(), dst.size().cols() - j_diag, T{0.}, T{1.},
59 dst.ptr({0, j_diag}), dst.ld());
60 }
61};

132template <Backend backend, Device device, class T>
133void BackTransformationReductionToBand<backend, device, T>::call(const SizeType b,
134 MatrixRef<T, device>& mat_c,
135 Matrix<const T, device>& mat_v,
136 Matrix<const T, device>& mat_taus) {
137 using namespace bt_red_band;
138 using dlaf::factorization::internal::computeTFactor;
139
140 auto hp = pika::execution::thread_priority::high;
141 auto np = pika::execution::thread_priority::normal;
142
143 const SizeType m = mat_c.nrTiles().rows();
144 const SizeType n = mat_c.nrTiles().cols();
145 const SizeType mb = mat_v.blockSize().rows();
146
147 if (m == 0 || n == 0)
148 return;
149
150 // Note: "-1" added to deal with size 1 reflector.
151 const SizeType total_nr_reflector = mat_v.size().rows() - b - 1;
152
153 if (total_nr_reflector <= 0)
154 return;
155
156 const auto dist_v = mat_v.distribution();
157 const auto dist_c = mat_c.distribution();
158
159 constexpr std::size_t n_workspaces = 2;
160 common::RoundRobin<matrix::Panel<Coord::Col, T, device>> panelsV(n_workspaces, dist_v);
161 common::RoundRobin<matrix::Panel<Coord::Col, T, device>> panelsW(n_workspaces, dist_v);
162 common::RoundRobin<matrix::Panel<Coord::Row, T, device>> panelsW2(n_workspaces, dist_c);
163
164 dlaf::matrix::Distribution dist_t({mb, total_nr_reflector}, {mb, mb});
165 matrix::Panel<Coord::Row, T, device> panelT(dist_t);
166
167 const auto dist_ws = [&]() {
168 using dlaf::factorization::internal::get_tfactor_num_workers;
169 const SizeType nworkspaces =
170 to_SizeType(std::max<std::size_t>(0, get_tfactor_num_workers<backend>() - 1));
171 const SizeType nrefls_step = dist_v.tile_size().cols();
172 return matrix::Distribution{{nworkspaces * nrefls_step, nrefls_step}, {nrefls_step, nrefls_step}};
173 }();
174 common::RoundRobin<matrix::Panel<Coord::Col, T, device>> panelsWS(n_workspaces, dist_ws);
175
176 const SizeType nr_reflector_blocks = dist_t.nrTiles().cols();
177
178 for (SizeType k = nr_reflector_blocks - 1; k >= 0; --k) {
179 const bool is_last = (k == nr_reflector_blocks - 1);
180 const SizeType nr_reflectors = dist_t.tileSize({0, k}).cols();
181
182 const GlobalElementIndex v_offset(k * mb + b, k * mb);
183 const GlobalElementIndex c_offset(k * mb + b, 0);
184
185 const matrix::SubPanelView panel_view(dist_v, v_offset, nr_reflectors);
186 const matrix::SubMatrixView mat_c_view(dist_c, c_offset);
187
188 auto& panelWS = panelsWS.nextResource();
189 auto& panelV = panelsV.nextResource();
190 auto& panelW = panelsW.nextResource();
191 auto& panelW2 = panelsW2.nextResource();
192
193 panelV.setRangeStart(v_offset);
194 panelW.setRangeStart(v_offset);
195
196 if (is_last) {
197 panelWS.setWidth(nr_reflectors);
198 panelT.setHeight(nr_reflectors);
199 panelW2.setHeight(nr_reflectors);
200 panelW.setWidth(nr_reflectors);
201 panelV.setWidth(nr_reflectors);
202 }
203
204 for (const auto& i : panel_view.iteratorLocal()) {
205 // Column index of the HH reflector which starts in the first row of this tile.
206 const SizeType j_diag =
207 std::max<SizeType>(0, i.row() * mat_v.blockSize().rows() - panel_view.offsetElement().row());
208
209 if (j_diag < nr_reflectors) {
210 auto tile_v = splitTile(mat_v.read(i), panel_view(i));
211 copyAndSetHHUpperTiles<backend>(j_diag, std::move(tile_v), panelV.readwrite(i));
212 }
213 else if (j_diag < mb) {
214 panelV.setTile(i, splitTile(mat_v.read(i), panel_view(i)));
215 }
216 else {
217 panelV.setTile(i, mat_v.read(i));
218 }
219 }
220
221 const LocalTileIndex taus_index{Coord::Row, k};
222 const LocalTileIndex t_index{Coord::Col, k};
223
224 computeTFactor<backend>(panelV, mat_taus.read(taus_index), panelT.readwrite(t_index), panelWS);
225
226 // W = V T
227 auto tile_t = panelT.read(t_index);
228 for (const auto& idx : panelW.iteratorLocal()) {
229 trmmPanel<backend>(np, tile_t, panelV.read(idx), panelW.readwrite(idx));
230 }
231
232 // W2 = W C
233 matrix::util::set0<backend>(hp, panelW2);
234 for (const auto& ij : mat_c_view.iteratorLocal()) {
235 gemmUpdateW2<backend>(np, panelW.read(ij), splitTile(mat_c.read(ij), mat_c_view(ij)),
236 panelW2.readwrite(ij));
237 }
238
239 // Update trailing matrix: C = C - V W2
240 for (const auto& ij : mat_c_view.iteratorLocal()) {
241 gemmTrailingMatrix<backend>(np, panelV.read(ij), panelW2.read(ij),
242 splitTile(mat_c.readwrite(ij), mat_c_view(ij)));
243 }
244
245 panelV.reset();
246 panelW.reset();
247 panelW2.reset();
248 panelT.reset();
249 panelWS.reset();
250 }

253template <Backend B, Device D, class T>
254void BackTransformationReductionToBand<B, D, T>::call(comm::CommunicatorGrid& grid, const SizeType b,
255 MatrixRef<T, D>& mat_c, Matrix<const T, D>& mat_v,
256 Matrix<const T, D>& mat_taus) {
257 namespace ex = pika::execution::experimental;
258 using namespace bt_red_band;
259 using dlaf::factorization::internal::computeTFactor;
260
261 auto hp = pika::execution::thread_priority::high;
262 auto np = pika::execution::thread_priority::normal;
263
264 // Set up MPI
265 auto mpi_col_task_chain = grid.col_communicator_pipeline();
266 auto mpi_row_task_chain = grid.row_communicator_pipeline();
267
268 auto dist_v = mat_v.distribution();
269 auto dist_c = mat_c.distribution();
270
271 const SizeType m = mat_c.nrTiles().rows();
272 const SizeType n = mat_c.nrTiles().cols();
273 const SizeType mb = mat_v.blockSize().rows();
274
275 const comm::Index2D this_rank = grid.rank();
276
277 if (m == 0 || n == 0)
278 return;
279
280 // Note: "-1" added to deal with size 1 reflector.
281 const SizeType total_nr_reflector = mat_v.size().cols() - b - 1;
282
283 if (total_nr_reflector <= 0)
284 return;
285
286 constexpr std::size_t n_workspaces = 2;
287 common::RoundRobin<matrix::Panel<Coord::Col, T, D>> panelsV(n_workspaces, dist_v);
288 common::RoundRobin<matrix::Panel<Coord::Col, T, D>> panelsW(n_workspaces, dist_v);
289 common::RoundRobin<matrix::Panel<Coord::Row, T, D>> panelsW2(n_workspaces, dist_c);
290
291 dlaf::matrix::Distribution dist_t({mb, total_nr_reflector}, {mb, mb}, grid.size(), this_rank,
292 dist_v.sourceRankIndex());
293 matrix::Panel<Coord::Row, T, D> panelT(dist_t);
294
295 const auto dist_ws = [&]() {
296 using dlaf::factorization::internal::get_tfactor_num_workers;
297 const SizeType nworkspaces = to_SizeType(std::max<std::size_t>(0, get_tfactor_num_workers<B>() - 1));
298 const SizeType nrefls_step = dist_v.tile_size().cols();
299 return matrix::Distribution{{nworkspaces * nrefls_step, nrefls_step}, {nrefls_step, nrefls_step}};
300 }();
301 common::RoundRobin<matrix::Panel<Coord::Col, T, D>> panelsWS(n_workspaces, dist_ws);
302
303 const SizeType nr_reflector_blocks = dist_t.nrTiles().cols();
304
305 for (SizeType k = nr_reflector_blocks - 1; k >= 0; --k) {
306 const bool is_last = (k == nr_reflector_blocks - 1);
307 const SizeType nr_reflectors = dist_t.tileSize({0, k}).cols();
308
309 const GlobalElementIndex v_offset(k * mb + b, k * mb);
310 const GlobalElementIndex c_offset(k * mb + b, 0);
311
312 const matrix::SubPanelView panel_view(dist_v, v_offset, nr_reflectors);
313 const matrix::SubMatrixView mat_c_view(dist_c, c_offset);
314
315 auto& panelWS = panelsWS.nextResource();
316 auto& panelV = panelsV.nextResource();
317 auto& panelW = panelsW.nextResource();
318 auto& panelW2 = panelsW2.nextResource();
319
320 panelV.setRangeStart(v_offset);
321 panelW.setRangeStart(v_offset);
322 panelT.setRange(GlobalTileIndex(Coord::Col, k), GlobalTileIndex(Coord::Col, k + 1));
323
324 if (is_last) {
325 panelWS.setWidth(nr_reflectors);
326 panelT.setHeight(nr_reflectors);
327 panelW2.setHeight(nr_reflectors);
328 panelW.setWidth(nr_reflectors);
329 panelV.setWidth(nr_reflectors);
330 }
331
332 const comm::IndexT_MPI k_rank_col = dist_v.template rankGlobalTile<Coord::Col>(k);
333
334 if (this_rank.col() == k_rank_col) {
335 for (const auto& ik : panel_view.iteratorLocal()) {
336 const SizeType i_row_g = dist_v.template globalTileFromLocalTile<Coord::Row>(ik.row());
337
338 // Column index of the HH reflector which starts in the first row of this tile.
339 const SizeType j_diag =
340 std::max<SizeType>(0, i_row_g * mat_v.blockSize().rows() - panel_view.offsetElement().row());
341
342 if (j_diag < nr_reflectors) {
343 auto tile_v = splitTile(mat_v.read(ik), panel_view(ik));
344 copyAndSetHHUpperTiles<B>(j_diag, tile_v, panelV.readwrite(ik));
345 }
346 else if (j_diag < mb) {
347 panelV.setTile(ik, splitTile(mat_v.read(ik), panel_view(ik)));
348 }
349 else {
350 panelV.setTile(ik, mat_v.read(ik));
351 }
352 }
353
354 const GlobalTileIndex taus_index{Coord::Row, k};
355 const SizeType k_local = dist_t.template localTileFromGlobalTile<Coord::Col>(k);
356 const LocalTileIndex t_index{Coord::Col, k_local};
357
358 computeTFactor<B>(panelV, mat_taus.read(taus_index), panelT.readwrite(t_index), panelWS,
359 mpi_col_task_chain);
360
361 // WH = V T
362 for (const auto& idx : panel_view.iteratorLocal()) {
363 trmmPanel<B>(np, panelT.read(t_index), panelV.read(idx), panelW.readwrite(idx));
364 }
365 }
366
367 matrix::util::set0<B>(hp, panelW2);
368
369 broadcast(k_rank_col, panelW, mpi_row_task_chain);
370
371 // W2 = W C
372 for (const auto& ij : mat_c_view.iteratorLocal()) {
373 gemmUpdateW2<B>(np, panelW.readwrite(ij), splitTile(mat_c.read(ij), mat_c_view(ij)),
374 panelW2.readwrite(ij));
375 }
376
377 if (mpi_col_task_chain.size() > 1) {
378 for (const auto& kj_panel : panelW2.iteratorLocal()) {
379 ex::start_detached(dlaf::comm::schedule_all_reduce_in_place(
380 mpi_col_task_chain.exclusive(), MPI_SUM, panelW2.readwrite(kj_panel)));
381 }
382 }
383
384 broadcast(k_rank_col, panelV, mpi_row_task_chain);
385
386 // C = C - V W2
387 for (const auto& ij : mat_c_view.iteratorLocal()) {
388 gemmTrailingMatrix<B>(np, panelV.read(ij), panelW2.read(ij),
389 splitTile(mat_c.readwrite(ij), mat_c_view(ij)));
390 }
391
392 panelV.reset();
393 panelW.reset();
394 panelW2.reset();
395 panelT.reset();
396 panelWS.reset();
397 }
