37template <Backend B, Device D, class T>
38void Eigensolver<B, D, T>::call(blas::Uplo uplo, Matrix<T, D>& mat_a, Matrix<BaseType<T>, D>& evals,
39 Matrix<T, D>& mat_e, const SizeType eigenvalues_index_begin,
40 const SizeType eigenvalues_index_end) {
41 const SizeType band_size = getBandSize(mat_a.blockSize().rows());
42
43 // need uplo check as reduction to band doesn't have the uplo argument yet.
44 if (uplo != blas::Uplo::Lower)
45 DLAF_UNIMPLEMENTED(uplo);
46
47 auto mat_taus = reduction_to_band<B>(mat_a, band_size);
48 auto ret = band_to_tridiagonal<Backend::MC>(uplo, band_size, mat_a);
49
50 tridiagonal_eigensolver<B>(ret.tridiagonal, evals, mat_e);
51
52 auto spec = matrix::util::internal::sub_matrix_spec_slice_cols(mat_e, eigenvalues_index_begin,
53 eigenvalues_index_end);
54
55 matrix::internal::MatrixRef mat_e_ref(mat_e, spec);
56 bt_band_to_tridiagonal<B>(band_size, mat_e_ref, ret.hh_reflectors);
57 bt_reduction_to_band<B>(band_size, mat_e_ref, mat_a, mat_taus);
58}

60template <Backend B, Device D, class T>
61void Eigensolver<B, D, T>::call(comm::CommunicatorGrid& grid, blas::Uplo uplo, Matrix<T, D>& mat_a,
62 Matrix<BaseType<T>, D>& evals, Matrix<T, D>& mat_e,
63 const SizeType eigenvalues_index_begin,
64 const SizeType eigenvalues_index_end) {
65 const SizeType band_size = getBandSize(mat_a.blockSize().rows());
66
67 // need uplo check as reduction to band doesn't have the uplo argument yet.
68 if (uplo != blas::Uplo::Lower)
69 DLAF_UNIMPLEMENTED(uplo);
70
71#ifdef DLAF_WITH_HDF5
72 static std::atomic<size_t> num_eigensolver_calls = 0;
73 std::stringstream fname;
74 fname << "eigensolver-" << matrix::internal::TypeToString_v<T> << "-"
75 << std::to_string(num_eigensolver_calls) << ".h5";
76 std::optional<matrix::internal::FileHDF5> file;
77
78 if (getTuneParameters().debug_dump_eigensolver_data) {
79 file = matrix::internal::FileHDF5(grid.fullCommunicator(), fname.str());
80 file->write(mat_a, "/input");
81 }
82#endif
83
84 auto mat_taus = reduction_to_band<B>(grid, mat_a, band_size);
85
86 auto ret = band_to_tridiagonal<Backend::MC>(grid, uplo, band_size, mat_a);
87
88 tridiagonal_eigensolver<B>(grid, ret.tridiagonal, evals, mat_e);
89
90 auto spec = matrix::util::internal::sub_matrix_spec_slice_cols(mat_e, eigenvalues_index_begin,
91 eigenvalues_index_end);
92 matrix::internal::MatrixRef mat_e_ref(mat_e, spec);
93
94 bt_band_to_tridiagonal<B>(grid, band_size, mat_e_ref, ret.hh_reflectors);
95 bt_reduction_to_band<B>(grid, band_size, mat_e_ref, mat_a, mat_taus);
96
97#ifdef DLAF_WITH_HDF5
98 if (getTuneParameters().debug_dump_eigensolver_data) {
99 file->write(evals, "/evals");
100 file->write(mat_e, "/evecs");
101 }
102
103 num_eigensolver_calls++;
104#endif
105}
