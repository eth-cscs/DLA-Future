38 template <Backend B, Device D, class T>
39 void Eigensolver<B, D, T>::call(blas::Uplo uplo, Matrix<T, D>& mat_a, Matrix<BaseType<T>, D>& evals,
40  Matrix<T, D>& mat_e) {
41  const SizeType band_size = getBandSize(mat_a.blockSize().rows());
42 
43  // need uplo check as reduction to band doesn't have the uplo argument yet.
44  if (uplo != blas::Uplo::Lower)
45  DLAF_UNIMPLEMENTED(uplo);
46 
47  auto mat_taus = reduction_to_band<B>(mat_a, band_size);
48  auto ret = band_to_tridiagonal<Backend::MC>(uplo, band_size, mat_a);
49 
50  tridiagonal_eigensolver<B>(ret.tridiagonal, evals, mat_e);
51 
52  bt_band_to_tridiagonal<B>(band_size, mat_e, ret.hh_reflectors);
53  bt_reduction_to_band<B>(band_size, mat_e, mat_a, mat_taus);
54 }

56 template <Backend B, Device D, class T>
57 void Eigensolver<B, D, T>::call(comm::CommunicatorGrid& grid, blas::Uplo uplo, Matrix<T, D>& mat_a,
58  Matrix<BaseType<T>, D>& evals, Matrix<T, D>& mat_e) {
59  const SizeType band_size = getBandSize(mat_a.blockSize().rows());
60 
61  // need uplo check as reduction to band doesn't have the uplo argument yet.
62  if (uplo != blas::Uplo::Lower)
63  DLAF_UNIMPLEMENTED(uplo);
64 
65 #ifdef DLAF_WITH_HDF5
66  static std::atomic<size_t> num_eigensolver_calls = 0;
67  std::stringstream fname;
68  fname << "eigensolver-" << matrix::internal::TypeToString_v<T> << "-"
69  << std::to_string(num_eigensolver_calls) << ".h5";
70  std::optional<matrix::internal::FileHDF5> file;
71 
72  if (getTuneParameters().debug_dump_eigensolver_data) {
73  file = matrix::internal::FileHDF5(grid.fullCommunicator(), fname.str());
74  file->write(mat_a, "/input");
75  }
76 #endif
77 
78  auto mat_taus = reduction_to_band<B>(grid, mat_a, band_size);
79 
80  auto ret = band_to_tridiagonal<Backend::MC>(grid, uplo, band_size, mat_a);
81 
82  tridiagonal_eigensolver<B>(grid, ret.tridiagonal, evals, mat_e);
83 
84  bt_band_to_tridiagonal<B>(grid, band_size, mat_e, ret.hh_reflectors);
85  bt_reduction_to_band<B>(grid, band_size, mat_e, mat_a, mat_taus);
86 
87 #ifdef DLAF_WITH_HDF5
88  if (getTuneParameters().debug_dump_eigensolver_data) {
89  file->write(evals, "/evals");
90  file->write(mat_e, "/evecs");
91  }
92 
93  num_eigensolver_calls++;
94 #endif
95 }
