30template <Backend B, Device D, class T>
31void GenEigensolver<B, D, T>::call(blas::Uplo uplo, Matrix<T, D>& mat_a, Matrix<T, D>& mat_b,
32 Matrix<BaseType<T>, D>& eigenvalues, Matrix<T, D>& eigenvectors,
33 const Factorization factorization,
34 const SizeType eigenvalues_index_begin,
35 const SizeType eigenvalues_index_end) {
36 if (factorization == Factorization::do_factorization) {
37 cholesky_factorization<B>(uplo, mat_b);
38 }
39 generalized_to_standard<B>(uplo, mat_a, mat_b);
40
41 hermitian_eigensolver<B>(uplo, mat_a, eigenvalues, eigenvectors, eigenvalues_index_begin,
42 eigenvalues_index_end);
43
44 auto spec = matrix::util::internal::sub_matrix_spec_slice_cols(eigenvectors, eigenvalues_index_begin,
45 eigenvalues_index_end);
46 matrix::internal::MatrixRef eigenvectors_ref(eigenvectors, spec);
47 solver::internal::triangular_solver<B>(blas::Side::Left, uplo, blas::Op::ConjTrans,
48 blas::Diag::NonUnit, T(1), mat_b, eigenvectors_ref);
49}

51template <Backend B, Device D, class T>
52void GenEigensolver<B, D, T>::call(comm::CommunicatorGrid& grid, blas::Uplo uplo, Matrix<T, D>& mat_a,
53 Matrix<T, D>& mat_b, Matrix<BaseType<T>, D>& eigenvalues,
54 Matrix<T, D>& eigenvectors, const Factorization factorization,
55 const SizeType eigenvalues_index_begin,
56 const SizeType eigenvalues_index_end) {
57#ifdef DLAF_WITH_HDF5
58 static std::atomic<size_t> num_gen_eigensolver_calls = 0;
59 std::stringstream fname;
60 fname << "generalized-eigensolver-";
61 if (factorization == Factorization::already_factorized) {
62 fname << "factorized-";
63 }
64 fname << matrix::internal::TypeToString_v<T> << "-" << std::to_string(num_gen_eigensolver_calls)
65 << ".h5";
66
67 std::optional<matrix::internal::FileHDF5> file;
68
69 if (getTuneParameters().debug_dump_generalized_eigensolver_data) {
70 file = matrix::internal::FileHDF5(grid.fullCommunicator(), fname.str());
71 file->write(mat_a, "/input-a");
72 if (factorization == Factorization::do_factorization) {
73 file->write(mat_b, "/input-b");
74 }
75 else { // Already factorized
76 file->write(mat_b, "/input-b-factorized");
77 }
78 }
79#endif
80
81 if (factorization == Factorization::do_factorization) {
82 cholesky_factorization<B>(grid, uplo, mat_b);
83 }
84
85 generalized_to_standard<B>(grid, uplo, mat_a, mat_b);
86
87 hermitian_eigensolver<B>(grid, uplo, mat_a, eigenvalues, eigenvectors, eigenvalues_index_begin,
88 eigenvalues_index_end);
89
90 auto spec = matrix::util::internal::sub_matrix_spec_slice_cols(eigenvectors, eigenvalues_index_begin,
91 eigenvalues_index_end);
92 matrix::internal::MatrixRef eigenvectors_ref(eigenvectors, spec);
93 solver::internal::triangular_solver<B>(grid, blas::Side::Left, uplo, blas::Op::ConjTrans,
94 blas::Diag::NonUnit, T(1), mat_b, eigenvectors_ref);
95
96#ifdef DLAF_WITH_HDF5
97 if (getTuneParameters().debug_dump_generalized_eigensolver_data) {
98 file->write(eigenvalues, "/evals");
99 file->write(eigenvectors, "/evecs");
100 }
101
102 num_gen_eigensolver_calls++;
103#endif
104}
