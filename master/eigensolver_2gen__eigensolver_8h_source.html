60template <Backend B, Device D, class T>
61EigensolverResult<T, D> hermitian_generalized_eigensolver_helper(blas::Uplo uplo, Matrix<T, D>& mat_a,
62 Matrix<T, D>& mat_b,
63 const Factorization factorization) {
64 DLAF_ASSERT(matrix::local_matrix(mat_a), mat_a);
65 DLAF_ASSERT(matrix::local_matrix(mat_b), mat_b);
66 DLAF_ASSERT(matrix::square_size(mat_a), mat_a);
67 DLAF_ASSERT(matrix::square_blocksize(mat_a), mat_a);
68 DLAF_ASSERT(matrix::square_size(mat_b), mat_b);
69 DLAF_ASSERT(matrix::square_blocksize(mat_b), mat_b);
70 DLAF_ASSERT(mat_a.size() == mat_b.size(), mat_a, mat_b);
71 DLAF_ASSERT(mat_a.blockSize() == mat_b.blockSize(), mat_a, mat_b);
72
73 const SizeType size = mat_a.size().rows();
74
75 matrix::Matrix<BaseType<T>, D> eigenvalues(LocalElementSize(size, 1),
76 TileElementSize(mat_a.blockSize().rows(), 1));
77 matrix::Matrix<T, D> eigenvectors(LocalElementSize(size, size), mat_a.blockSize());
78
79 hermitian_generalized_eigensolver_helper<B, D, T>(uplo, mat_a, mat_b, eigenvalues, eigenvectors,
80 factorization);
81
82 return {std::move(eigenvalues), std::move(eigenvectors)};
83}
