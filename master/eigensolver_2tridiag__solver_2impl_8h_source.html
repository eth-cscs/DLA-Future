36 // Splits [i_begin, i_end) in the middle and waits for all splits on [i_begin, i_split] and [i_split,
37 // i_end) before saving the triad <i_begin, i_split, i_end> into `indices`.
38 //
39 // The recursive calls span a binary tree which is traversed in depth first left-right-root order. That
40 // is also the order of triads in `indices`.
41 //
42 inline void splitIntervalInTheMiddleRecursively(
43  const SizeType i_begin, const SizeType i_end,
44  std::vector<std::tuple<SizeType, SizeType, SizeType>>& indices) {
45  if (i_begin + 1 == i_end)
46  return;
47  const SizeType i_split = util::ceilDiv<SizeType>(i_begin + i_end, 2);
48  splitIntervalInTheMiddleRecursively(i_begin, i_split, indices);
49  splitIntervalInTheMiddleRecursively(i_split, i_end, indices);
50  indices.emplace_back(i_begin, i_split, i_end);
51 }

53 // Generates an array of triad indices. Each triad is composed of begin <= middle < end indices and
54 // represents two intervals [begin, middle] and [middle + 1, end]. The two intervals are the subproblems
55 // that have to be merged to arrive at [begin, end].
56 //
57 // Note: the intervals are all closed!
58 //
59 inline std::vector<std::tuple<SizeType, SizeType, SizeType>> generateSubproblemIndices(const SizeType n) {
60  if (n == 0)
61  return {};
62 
63  std::vector<std::tuple<SizeType, SizeType, SizeType>> indices;
64  indices.reserve(to_sizet(n));
65  splitIntervalInTheMiddleRecursively(0, n, indices);
66  return indices;
67 }

69 template <class T>
70 auto cuppensDecomposition(Matrix<T, Device::CPU>& tridiag) {
71  namespace ex = pika::execution::experimental;
72  using sender_type = decltype(ex::split(
73  ex::ensure_started(cuppensDecompAsync<T>(tridiag.readwrite(std::declval<LocalTileIndex>()),
74  tridiag.readwrite(std::declval<LocalTileIndex>())))));
75  using vector_type = std::vector<sender_type>;
76 
77  if (tridiag.nrTiles().rows() == 0)
78  return vector_type{};
79 
80  const SizeType i_end = tridiag.nrTiles().rows();
81  vector_type offdiag_vals;
82  offdiag_vals.reserve(to_sizet(i_end - 1));
83 
84  for (SizeType i_split = 1; i_split < i_end; ++i_split) {
85  offdiag_vals.push_back(ex::split(
86  ex::ensure_started(cuppensDecompAsync<T>(tridiag.readwrite(LocalTileIndex(i_split - 1, 0)),
87  tridiag.readwrite(LocalTileIndex(i_split, 0))))));
88  }
89  return offdiag_vals;
90 }

132 // Notation:
133 //
134 // nb - the block/tile size of all matrices and vectors
135 // n1 - the size of the top subproblem
136 // n2 - the size of the bottom subproblem
137 // Q1 - (n1 x n1) the orthogonal matrix of the top subproblem
138 // Q2 - (n2 x n2) the orthogonal matrix of the bottom subproblem
139 // n := n1 + n2, the size of the merged problem
140 //
141 // ┌───┬───┐
142 // │Q1 │ │
143 // Q := ├───┼───┤ , (n x n) orthogonal matrix composed of the top and bottom subproblems
144 // │ │Q2 │
145 // └───┴───┘
146 // D := diag(Q), (n x 1) the diagonal of Q
147 // z := (n x 1) rank 1 update vector
148 // rho := rank 1 update scaling factor
149 // D + rho*z*z^T := rank 1 update problem
150 // U := (n x n) matrix of eigenvectors of the rank 1 update problem:
151 //
152 // k := the size of the deflated rank 1 update problem (k <= n)
153 // D' := (k x 1), deflated D
154 // z' := (k x 1), deflated z
155 // D' + rho*z'*z'^T := deflated rank 1 update problem
156 // U' := (k x k) matrix of eigenvectors of the deflated rank 1 update problem
157 //
158 // l1 := number of columns of the top subproblem after deflation, l1 <= n1
159 // l2 := number of columns of the bottom subproblem after deflation, l2 <= n2
160 // Q1' := (n1 x l1) the non-deflated part of Q1 (l1 <= n1)
161 // Q2' := (n2 x l2) the non-deflated part of Q2 (l2 <= n2)
162 // Qd := (n-k x n) the deflated parts of Q1 and Q2
163 // U1' := (l1 x k) is the first l1 rows of U'
164 // U2' := (l2 x k) is the last l2 rows of U'
165 // I := (n-k x n-k) identity matrix
166 // P := (n x n) permutation matrix used to bring Q and U into multiplication form
167 //
168 // Q-U multiplication form to arrive at the eigenvectors of the merged problem:
169 //
170 // ┌────────┬──────┬────┐ ┌───────────────┬────┐ ┌───────────────┬────┐
171 // │ │ │ │ │ │ │ │ │ │
172 // │ Q1' │ │ │ │ │ │ │ Q1'xU1' │ │
173 // │ │ │ │ │ U1' │ │ │ │ │
174 // │ │ │ │ │ ──────┤ │ │ │ │
175 // ├──────┬─┴──────┤ Qd │ ├─────── │ │ ├───────────────┤ Qd │
176 // │ │ │ │ X │ U2' │ │ = │ │ │
177 // │ │ │ │ │ │ │ │ Q2'xU2' │ │
178 // │ │ Q2' │ │ ├───────────────┼────┤ │ │ │
179 // │ │ │ │ │ │ I │ │ │ │
180 // │ │ │ │ │ │ │ │ │ │
181 // └──────┴────────┴────┘ └───────────────┴────┘ └───────────────┴────┘
182 //
183 // Note:
184 // 1. U1' and U2' may overlap (in practice they almost always do)
185 // 2. The overlap between U1' and U2' matches the number of shared columns between Q1' and Q2'
186 // 3. The overlap region is due to deflation via Givens rotations of a column vector from Q1 with a
187 // column vector of Q2.
188 //
189 template <Backend B, Device D, class T>
190 void TridiagSolver<B, D, T>::call(Matrix<T, Device::CPU>& tridiag, Matrix<T, D>& evals,
191  Matrix<T, D>& evecs) {
192  using pika::execution::thread_priority;
193 
194  // Quick return for empty matrix
195  if (evecs.size().isEmpty())
196  return;
197 
198  // If the matrix is composed by a single tile simply call stedc.
199  if (evecs.nrTiles().linear_size() == 1) {
200  if constexpr (D == Device::CPU) {
201  solveLeaf(tridiag, evecs);
202  }
203  else {
204  Matrix<T, Device::CPU> h_evecs{evecs.distribution()};
205  solveLeaf(tridiag, evecs, h_evecs);
206  }
207  offloadDiagonal(tridiag, evals);
208  return;
209  }
210 
211  // Auxiliary matrix used for the D&C algorithm
212  const matrix::Distribution& distr = evecs.distribution();
213  const LocalElementSize vec_size(distr.size().rows(), 1);
214  const TileElementSize vec_tile_size(distr.blockSize().rows(), 1);
215  WorkSpace<T, D> ws{Matrix<T, D>(distr), // e0
216  Matrix<T, D>(distr), // e1
217  evecs, // e2
218  evals, // d1
219  Matrix<T, D>(vec_size, vec_tile_size), // z0
220  Matrix<T, D>(vec_size, vec_tile_size), // z1
221  Matrix<SizeType, D>(vec_size, vec_tile_size), // i2
222  Matrix<SizeType, D>(vec_size, vec_tile_size), // i5
223  Matrix<SizeType, D>(vec_size, vec_tile_size)}; // i6
224 
225  WorkSpaceHost<T> ws_h{Matrix<T, Device::CPU>(vec_size, vec_tile_size), // d0
226  Matrix<ColType, Device::CPU>(vec_size, vec_tile_size), // c
227  Matrix<SizeType, Device::CPU>(vec_size, vec_tile_size), // i1
228  Matrix<SizeType, Device::CPU>(vec_size, vec_tile_size), // i3
229  Matrix<SizeType, Device::CPU>(vec_size, vec_tile_size)}; // i4
230 
231  // Mirror workspace on host memory for CPU-only kernels
232  WorkSpaceHostMirror<T, D> ws_hm{initMirrorMatrix(ws.e2), initMirrorMatrix(ws.d1),
233  initMirrorMatrix(ws.z0), initMirrorMatrix(ws.z1),
234  initMirrorMatrix(ws.i2), initMirrorMatrix(ws.i5)};
235 
236  // Set `ws.e0` to `zero` (needed for Given's rotation to make sure no random values are picked up)
237  matrix::util::set0<B, T, D>(thread_priority::normal, ws.e0);
238 
239  // Cuppen's decomposition
240  auto offdiag_vals = cuppensDecomposition(tridiag);
241 
242  // Solve with stedc for each tile of `tridiag` (nb x 2) and save eigenvectors in diagonal tiles of
243  // `evecs` (nb x nb)
244  if constexpr (D == Device::CPU) {
245  solveLeaf(tridiag, ws.e0);
246  }
247  else {
248  solveLeaf(tridiag, ws.e0, ws_hm.e2);
249  }
250 
251  // Offload the diagonal from `tridiag` to `d0`
252  offloadDiagonal(tridiag, ws_h.d0);
253 
254  // Each triad represents two subproblems to be merged
255  for (auto [i_begin, i_split, i_end] : generateSubproblemIndices(distr.nrTiles().rows())) {
256  mergeSubproblems<B>(i_begin, i_split, i_end, offdiag_vals[to_sizet(i_split - 1)], ws, ws_h, ws_hm);
257  }
258 
259  const SizeType n = evecs.nrTiles().rows();
260  copy(ws_hm.i2, ws.i2);
261 
262  // Note: ws_hm.d1 is the mirror of ws.d1 which is evals
263  applyIndex(0, n, ws_hm.i2, ws_h.d0, ws_hm.d1);
264  copy(ws_hm.d1, evals);
265 
266  dlaf::permutations::permute<B, D, T, Coord::Col>(0, n, ws.i2, ws.e0, evecs);
267 }

283 // Solve for each tile of the local matrix @p tridiag (n x 2) with `stedc()` and save the result in the
284 // corresponding diagonal tile of the distribtued matrix @p evecs (n x n)
285 //
286 // @p tridiag is a local matrix of size (n x 2)
287 // @p evecs is a distributed matrix of size (n x n)
288 //
289 template <class T>
290 void solveDistLeaf(comm::CommunicatorGrid grid, common::Pipeline<comm::Communicator>& full_task_chain,
291  Matrix<T, Device::CPU>& tridiag, Matrix<T, Device::CPU>& evecs) {
292  const matrix::Distribution& dist = evecs.distribution();
293  namespace ex = pika::execution::experimental;
294 
295  const comm::Index2D this_rank = dist.rankIndex();
296  const SizeType ntiles = dist.nrTiles().rows();
297  for (SizeType i = 0; i < ntiles; ++i) {
298  const GlobalTileIndex ii_tile(i, i);
299  const comm::Index2D ii_rank = dist.rankGlobalTile(ii_tile);
300  const GlobalTileIndex id_tr(i, 0);
301  if (ii_rank == this_rank) {
302  stedcAsync(tridiag.readwrite(id_tr), evecs.readwrite(ii_tile));
303  ex::start_detached(comm::scheduleSendBcast(full_task_chain(), tridiag.read(id_tr)));
304  }
305  else {
306  const comm::IndexT_MPI root_rank = grid.rankFullCommunicator(ii_rank);
307  ex::start_detached(comm::scheduleRecvBcast(full_task_chain(), root_rank,
308  tridiag.readwrite(id_tr)));
309  }
310  }
311 }

350 void TridiagSolver<B, D, T>::call(comm::CommunicatorGrid grid, Matrix<T, Device::CPU>& tridiag,
351  Matrix<T, D>& evals, Matrix<T, D>& evecs) {
352  using pika::execution::thread_priority;
353 
354  common::Pipeline<comm::Communicator> full_task_chain(grid.fullCommunicator().clone());
355 
356  // Quick return for empty matrix
357  if (evecs.size().isEmpty())
358  return;
359 
360  // If the matrix is composed by a single tile simply call stedc.
361  if (evecs.nrTiles().linear_size() == 1) {
362  if constexpr (D == Device::CPU) {
363  solveDistLeaf(grid, full_task_chain, tridiag, evecs);
364  }
365  else {
366  Matrix<T, Device::CPU> h_evecs{evecs.distribution()};
367  solveDistLeaf(grid, full_task_chain, tridiag, evecs, h_evecs);
368  }
369  offloadDiagonal(tridiag, evals);
370  return;
371  }
372 
373  // Auxiliary matrix used for the D&C algorithm
374  const matrix::Distribution& dist_evecs = evecs.distribution();
375  const matrix::Distribution& dist_evals = evals.distribution();
376 
377  WorkSpace<T, D> ws{Matrix<T, D>(dist_evecs), // e0
378  Matrix<T, D>(dist_evecs), // e1
379  evecs, // e2
380  evals, // d1
381  Matrix<T, D>(dist_evals), // z0
382  Matrix<T, D>(dist_evals), // z1
383  Matrix<SizeType, D>(dist_evals), // i2
384  Matrix<SizeType, D>(dist_evals), // i5
385  Matrix<SizeType, D>(dist_evals)}; // i6
386 
387  WorkSpaceHost<T> ws_h{Matrix<T, Device::CPU>(dist_evals), // d0
388  Matrix<ColType, Device::CPU>(dist_evals), // c
389  Matrix<SizeType, Device::CPU>(dist_evals), // i1
390  Matrix<SizeType, Device::CPU>(dist_evals), // i3
391  Matrix<SizeType, Device::CPU>(dist_evals)}; // i4
392 
393  // Mirror workspace on host memory for CPU-only kernels
394  DistWorkSpaceHostMirror<T, D> ws_hm{initMirrorMatrix(ws.e0), initMirrorMatrix(ws.e2),
395  initMirrorMatrix(ws.d1), initMirrorMatrix(ws.z0),
396  initMirrorMatrix(ws.z1), initMirrorMatrix(ws.i2),
397  initMirrorMatrix(ws.i5), initMirrorMatrix(ws.i6)};
398 
399  // Set `ws.e0` to `zero` (needed for Given's rotation to make sure no random values are picked up)
400  matrix::util::set0<B, T, D>(thread_priority::normal, ws.e0);
401 
402  // Cuppen's decomposition
403  auto offdiag_vals = cuppensDecomposition(tridiag);
404 
405  common::Pipeline<comm::Communicator> row_task_chain(grid.rowCommunicator().clone());
406  common::Pipeline<comm::Communicator> col_task_chain(grid.colCommunicator().clone());
407 
408  // Solve with stedc for each tile of `tridiag` (nb x 2) and save eigenvectors in diagonal tiles of
409  // `evecs` (nb x nb)
410  if constexpr (D == Device::CPU) {
411  solveDistLeaf(grid, full_task_chain, tridiag, ws.e0);
412  }
413  else {
414  solveDistLeaf(grid, full_task_chain, tridiag, ws.e0, ws_hm.e0);
415  }
416 
417  // Offload the diagonal from `tridiag` to `evals`
418  offloadDiagonal(tridiag, ws_h.d0);
419 
420  // Each triad represents two subproblems to be merged
421  SizeType nrtiles = dist_evecs.nrTiles().rows();
422  for (auto [i_begin, i_split, i_end] : generateSubproblemIndices(nrtiles)) {
423  mergeDistSubproblems<B>(grid, full_task_chain, row_task_chain, col_task_chain, i_begin, i_split,
424  i_end, offdiag_vals[to_sizet(i_split - 1)], ws, ws_h, ws_hm);
425  }
426 
427  const SizeType n = evecs.nrTiles().rows();
428  copy(ws.e0, ws_hm.e0);
429 
430  // Note: ws_hm.d1 is the mirror of ws.d1 which is evals
431  applyIndex(0, n, ws_h.i1, ws_h.d0, ws_hm.d1);
432  copy(ws_hm.d1, evals);
433 
434  // Note: ws_hm.e2 is the mirror of ws.e2 which is evecs
435  dlaf::permutations::permute<Backend::MC, Device::CPU, T, Coord::Col>(grid, row_task_chain, 0, n,
436  ws_h.i1, ws_hm.e0, ws_hm.e2);
437  copy(ws_hm.e2, evecs);
438 }
