53// Cuppen's decomposition
54//
55// Substracts the offdiagonal element at the split from the top and bottom diagonal elements and
56// returns the offdiagonal element. The split is between the last row of the top tile and the first
57// row of the bottom tile.
58//
59template <class T>
60T cuppensDecomp(const matrix::Tile<T, Device::CPU>& top, const matrix::Tile<T, Device::CPU>& bottom) {
61 TileElementIndex offdiag_idx{top.size().rows() - 1, 1};
62 TileElementIndex top_idx{top.size().rows() - 1, 0};
63 TileElementIndex bottom_idx{0, 0};
64 const T offdiag_val = top(offdiag_idx);
65
66 // Refence: Lapack working notes: LAWN 69, Serial Cuppen algorithm, Chapter 3
67 //
68 top(top_idx) -= std::abs(offdiag_val);
69 bottom(bottom_idx) -= std::abs(offdiag_val);
70 return offdiag_val;
71}

73DLAF_MAKE_CALLABLE_OBJECT(cuppensDecomp);

75template <class T>
76void copyDiagonalFromCompactTridiagonal(const matrix::Tile<const T, Device::CPU>& tridiag_tile,
77 const matrix::Tile<T, Device::CPU>& diag_tile) {
78 const SizeType m = tridiag_tile.size().rows();
79 DLAF_ASSERT_HEAVY(m == diag_tile.size().rows(), m, diag_tile.size().rows());
80 for (SizeType i = 0; i < m; ++i) {
81 diag_tile(TileElementIndex(i, 0)) = tridiag_tile(TileElementIndex(i, 0));
82 }
83}

100template <class T>
101void assembleRank1UpdateVectorTile(bool is_top_tile, T rho,
102 const matrix::Tile<const T, Device::CPU>& evecs_tile,
103 const matrix::Tile<T, Device::CPU>& rank1_tile) {
104 const SizeType m = evecs_tile.size().rows();
105 const SizeType n = evecs_tile.size().cols();
106 DLAF_ASSERT_HEAVY(n == rank1_tile.size().rows(), n, rank1_tile.size().rows());
107
108 // Copy the bottom row of the top tile or the top row of the bottom tile
109 SizeType row = (is_top_tile) ? m - 1 : 0;
110
111 // Negate the last row of the top eigenvector subproblem matrix (Q1) if rho < 0
112 //
113 // lapack 3.10.0, dlaed2.f, line 280 and 281
114 int sign = (is_top_tile && rho < 0) ? -1 : 1;
115
116 for (SizeType i = 0; i < n; ++i) {
117 rank1_tile(TileElementIndex(i, 0)) = sign * evecs_tile(TileElementIndex(row, i)) / T(std::sqrt(2));
118 }
119}
