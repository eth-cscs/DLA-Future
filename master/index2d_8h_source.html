287 template <class IndexT, class Tag>
288 Index2D<IndexT, Tag> computeCoordsRowMajor(SizeType linear_index,
289  const Size2D<IndexT, Tag>& dims) noexcept {
290  using dlaf::util::ptrdiff_t::mul;
291 
292  // `linear_index` is wrapped with `std::to_string` because uint8_t or int8_t are interpreted as chars,
293  // so the equivalent ASCII mapping is printed instead of the numeric value.
294  DLAF_ASSERT_MODERATE(linear_index >= 0, std::to_string(linear_index));
295  DLAF_ASSERT_MODERATE(linear_index < mul(dims.rows(), dims.cols()), std::to_string(linear_index), dims);
296 
297  SizeType leading_size = dims.cols();
298  return {to_signed<IndexT>(linear_index / leading_size),
299  to_signed<IndexT>(linear_index % leading_size)};
300 }

352 template <class LinearIndexT, class IndexT, class Tag>
353 LinearIndexT computeLinearIndexRowMajor(const Index2D<IndexT, Tag>& index,
354  const Size2D<IndexT, Tag>& dims) noexcept {
355  using dlaf::util::ptrdiff_t::mul;
356  using dlaf::util::ptrdiff_t::sum;
357 
358  static_assert(std::is_integral_v<LinearIndexT>, "LinearIndexT must be an integral type");
359 
360  DLAF_ASSERT_MODERATE(index.isIn(dims), index, dims);
361 
362  SizeType linear_index = sum(mul(index.row(), dims.cols()), index.col());
363  return integral_cast<LinearIndexT>(linear_index);
