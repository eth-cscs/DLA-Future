299 template <class IndexT, class Tag>
300 Index2D<IndexT, Tag> computeCoordsRowMajor(SizeType linear_index,
301  const Size2D<IndexT, Tag>& dims) noexcept {
302  using dlaf::util::ptrdiff_t::mul;
303 
304  // `linear_index` is wrapped with `std::to_string` because uint8_t or int8_t are interpreted as chars,
305  // so the equivalent ASCII mapping is printed instead of the numeric value.
306  DLAF_ASSERT_MODERATE(linear_index >= 0, std::to_string(linear_index));
307  DLAF_ASSERT_MODERATE(linear_index < mul(dims.rows(), dims.cols()), std::to_string(linear_index), dims);
308 
309  SizeType leading_size = dims.cols();
310  return {to_signed<IndexT>(linear_index / leading_size),
311  to_signed<IndexT>(linear_index % leading_size)};
312 }

364 template <class LinearIndexT, class IndexT, class Tag>
365 LinearIndexT computeLinearIndexRowMajor(const Index2D<IndexT, Tag>& index,
366  const Size2D<IndexT, Tag>& dims) noexcept {
367  using dlaf::util::ptrdiff_t::mul;
368  using dlaf::util::ptrdiff_t::sum;
369 
370  static_assert(std::is_integral_v<LinearIndexT>, "LinearIndexT must be an integral type");
371 
372  DLAF_ASSERT_MODERATE(index.isIn(dims), index, dims);
373 
374  SizeType linear_index = sum(mul(index.row(), dims.cols()), index.col());
375  return integral_cast<LinearIndexT>(linear_index);
