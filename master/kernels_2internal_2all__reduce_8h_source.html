32 template <class T, Device DIn, Device DOut>
33 auto allReduce(const Communicator& comm, MPI_Op reduce_op, const matrix::Tile<const T, DIn>& tile_in,
34  const matrix::Tile<T, DOut>& tile_out, MPI_Request* req) {
35 #if !defined(DLAF_WITH_MPI_GPU_AWARE)
36  static_assert(DIn == Device::CPU, "DLAF_WITH_MPI_GPU_AWARE=off, MPI accepts only CPU memory.");
37  static_assert(DOut == Device::CPU, "DLAF_WITH_MPI_GPU_AWARE=off, MPI accepts only CPU memory.");
38 #endif
39  DLAF_ASSERT(tile_in.is_contiguous(), "");
40  DLAF_ASSERT(tile_out.is_contiguous(), "");
41 
42  auto msg_in = comm::make_message(common::make_data(tile_in));
43  auto msg_out = comm::make_message(common::make_data(tile_out));
44  DLAF_MPI_CHECK_ERROR(MPI_Iallreduce(msg_in.data(), msg_out.data(), msg_in.count(), msg_in.mpi_type(),
45  reduce_op, comm, req));
46 }

50 template <Device DCommIn, Device DCommOut, class T, Device DIn, Device DOut>
51 [[nodiscard]] auto scheduleAllReduce(
52  pika::execution::experimental::unique_any_sender<CommunicatorPipelineExclusiveWrapper> pcomm,
53  MPI_Op reduce_op, dlaf::matrix::ReadOnlyTileSender<T, DIn> tile_in,
54  dlaf::matrix::ReadWriteTileSender<T, DOut> tile_out) {
55  using dlaf::comm::CommunicationDevice_v;
56  using dlaf::comm::internal::allReduce_o;
57  using dlaf::comm::internal::transformMPI;
58  using dlaf::internal::CopyFromDestination;
59  using dlaf::internal::CopyToDestination;
60  using dlaf::internal::RequireContiguous;
61  using dlaf::internal::SenderSingleValueType;
62  using dlaf::internal::whenAllLift;
63  using dlaf::internal::withTemporaryTile;
64 
65  // We create two nested scopes for the input and output tiles with
66  // withTemporaryTile. The output tile is in the outer scope as the output tile
67  // will be returned by the returned sender.
68  auto all_reduce_final = [reduce_op, pcomm = std::move(pcomm),
69  tile_in = std::move(tile_in)](const auto& tile_out_comm) mutable {
70  auto all_reduce = [reduce_op, pcomm = std::move(pcomm),
71  &tile_out_comm](auto const& tile_in_comm) mutable {
72  return whenAllLift(std::move(pcomm), reduce_op, std::cref(tile_in_comm),
73  std::cref(tile_out_comm)) |
74  transformMPI(allReduce_o);
75  };
76  // The input tile must be copied to the temporary tile used for the
77  // reduction, but the temporary tile does not need to be copied back to the
78  // input since the data is not changed by the reduction (the result is
79  // written into the output tile instead). The reduction is explicitly done
80  // on CPU memory so that we can manage potential asynchronous copies between
81  // CPU and GPU. A reduction requires contiguous memory.
82  return withTemporaryTile<DCommIn, CopyToDestination::Yes, CopyFromDestination::No,
83  RequireContiguous::Yes>(std::move(tile_in), std::move(all_reduce));
84  };
85 
86 #if !defined(DLAF_WITH_MPI_GPU_AWARE)
87  static_assert(DCommIn == Device::CPU, "DLAF_WITH_MPI_GPU_AWARE=off, MPI accepts only CPU memory.");
88  static_assert(DCommOut == Device::CPU, "DLAF_WITH_MPI_GPU_AWARE=off, MPI accepts only CPU memory.");
89 #endif
90 
91  // The output tile does not need to be copied to the temporary tile since it
92  // is only written to. The written data is copied back from the temporary tile
93  // to the output tile. The reduction is explicitly done on CPU memory so that
94  // we can manage potential asynchronous copies between CPU and GPU. A
95  // reduction requires contiguous memory.
96  return withTemporaryTile<DCommOut, CopyToDestination::No, CopyFromDestination::Yes,
97  RequireContiguous::Yes>(std::move(tile_out), std::move(all_reduce_final));
98 }

100 template <class T, Device D>
101 auto allReduceInPlace(const Communicator& comm, MPI_Op reduce_op, const matrix::Tile<T, D>& tile,
102  MPI_Request* req) {
103 #if !defined(DLAF_WITH_MPI_GPU_AWARE)
104  static_assert(D == Device::CPU, "DLAF_WITH_MPI_GPU_AWARE=off, MPI accepts only CPU memory.");
105 #endif
106  DLAF_ASSERT(tile.is_contiguous(), "");
107 
108  auto msg = comm::make_message(common::make_data(tile));
109  DLAF_MPI_CHECK_ERROR(MPI_Iallreduce(MPI_IN_PLACE, msg.data(), msg.count(), msg.mpi_type(), reduce_op,
110  comm, req));
111 }

115 template <Device DComm, class T, Device D>
116 [[nodiscard]] dlaf::matrix::ReadWriteTileSender<T, D> schedule_all_reduce_in_place(
117  pika::execution::experimental::unique_any_sender<CommunicatorPipelineExclusiveWrapper> pcomm,
118  MPI_Op reduce_op, dlaf::matrix::ReadWriteTileSender<T, D> tile) {
119  using dlaf::comm::CommunicationDevice_v;
120  using dlaf::comm::internal::allReduceInPlace_o;
121  using dlaf::comm::internal::transformMPI;
122  using dlaf::internal::CopyFromDestination;
123  using dlaf::internal::CopyToDestination;
124  using dlaf::internal::RequireContiguous;
125  using dlaf::internal::whenAllLift;
126  using dlaf::internal::withTemporaryTile;
127 
128 #if defined(__GNUC__)
129 #pragma GCC diagnostic push
130 #pragma GCC diagnostic ignored "-Warray-bounds"
131 #endif
132  auto all_reduce_in_place = [reduce_op, pcomm = std::move(pcomm)](const auto& tile_comm) mutable {
133  return whenAllLift(std::move(pcomm), reduce_op, std::cref(tile_comm)) |
134  transformMPI(allReduceInPlace_o);
135  };
136 #if defined(__GNUC__)
137 #pragma GCC diagnostic pop
138 #endif
139 
140 #if !defined(DLAF_WITH_MPI_GPU_AWARE)
141  static_assert(DComm == Device::CPU, "DLAF_WITH_MPI_GPU_AWARE=off, MPI accepts only CPU memory.");
142 #endif
143 
144  // The tile has to be copied both to and from the temporary tile since the
145  // reduction is done in-place. The reduction is explicitly done on CPU memory
146  // so that we can manage potential asynchronous copies between CPU and GPU. A
147  // reduction requires contiguous memory.
148  return withTemporaryTile<DComm, CopyToDestination::Yes, CopyFromDestination::Yes,
149  RequireContiguous::Yes>(std::move(tile), std::move(all_reduce_in_place));
150 }
