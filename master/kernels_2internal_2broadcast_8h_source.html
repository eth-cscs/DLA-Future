35template <class T, Device D>
36void sendBcast(const Communicator& comm, const matrix::Tile<const T, D>& tile, MPI_Request* req) {
37#if !defined(DLAF_WITH_MPI_GPU_AWARE)
38 static_assert(D == Device::CPU, "DLAF_WITH_MPI_GPU_AWARE=off, MPI accepts only CPU memory.");
39#endif
40
41 auto msg = comm::make_message(common::make_data(tile));
42 DLAF_MPI_CHECK_ERROR(MPI_Ibcast(const_cast<T*>(msg.data()), msg.count(), msg.mpi_type(), comm.rank(),
43 comm, req));
44}

48template <Device DComm, dlaf::internal::RequireContiguous require_contiguous, class T, Device D,
49 class Comm>
50[[nodiscard]] auto schedule_bcast_send(pika::execution::experimental::unique_any_sender<Comm> pcomm,
51 dlaf::matrix::ReadOnlyTileSender<T, D> tile) {
52 using dlaf::internal::CopyFromDestination;
53 using dlaf::internal::CopyToDestination;
54 using dlaf::internal::SenderSingleValueType;
55 using dlaf::internal::whenAllLift;
56 using dlaf::internal::withTemporaryTile;
57
58 auto send = [pcomm = std::move(pcomm)](const auto& tile_comm) mutable {
59 return whenAllLift(std::move(pcomm), std::cref(tile_comm)) | transformMPI(sendBcast_o);
60 };
61
62#if !defined(DLAF_WITH_MPI_GPU_AWARE)
63 static_assert(DComm == Device::CPU, "DLAF_WITH_MPI_GPU_AWARE=off, MPI accepts only CPU memory.");
64#endif
65
66 // The input tile must be copied to the temporary tile used for the send, but
67 // the temporary tile does not need to be copied back to the input since the
68 // data is not changed by the send. A send does not require contiguous memory.
69 return withTemporaryTile<DComm, CopyToDestination::Yes, CopyFromDestination::No, require_contiguous>(
70 std::move(tile), std::move(send));
71}

86template <Device DComm, dlaf::internal::RequireContiguous require_contiguous, class T, Device D,
87 class Comm>
88[[nodiscard]] dlaf::matrix::ReadWriteTileSender<T, D> schedule_bcast_recv(
89 pika::execution::experimental::unique_any_sender<Comm> pcomm, comm::IndexT_MPI root_rank,
90 dlaf::matrix::ReadWriteTileSender<T, D> tile) {
91 using dlaf::comm::internal::recvBcast_o;
92 using dlaf::internal::CopyFromDestination;
93 using dlaf::internal::CopyToDestination;
94 using dlaf::internal::SenderSingleValueType;
95 using dlaf::internal::whenAllLift;
96 using dlaf::internal::withTemporaryTile;
97
98#if defined(__GNUC__)
99#pragma GCC diagnostic push
100#pragma GCC diagnostic ignored "-Warray-bounds"
101#endif
102 auto recv = [root_rank, pcomm = std::move(pcomm)](const auto& tile_comm) mutable {
103 return whenAllLift(std::move(pcomm), root_rank, std::cref(tile_comm)) | transformMPI(recvBcast_o);
104 };
105#if defined(__GNUC__)
106#pragma GCC diagnostic pop
107#endif
108
109#if !defined(DLAF_WITH_MPI_GPU_AWARE)
110 static_assert(DComm == Device::CPU, "DLAF_WITH_MPI_GPU_AWARE=off, MPI accepts only CPU memory.");
111#endif
112
113 // Since this is a receive we don't need to copy the input to the temporary
114 // tile (the input tile may be uninitialized). The received data is copied
115 // back from the temporary tile to the input. A receive does not require
116 // contiguous memory.
117 return withTemporaryTile<DComm, CopyToDestination::No, CopyFromDestination::Yes, require_contiguous>(
118 std::move(tile), std::move(recv));
119}
