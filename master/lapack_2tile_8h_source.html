314 template <class T>
315 dlaf::BaseType<T> lantr(const lapack::Norm norm, const blas::Uplo uplo, const blas::Diag diag,
316  const Tile<T, Device::CPU>& a) noexcept {
317  switch (uplo) {
318  case blas::Uplo::Lower:
319  DLAF_ASSERT(a.size().rows() >= a.size().cols(), a);
320  break;
321  case blas::Uplo::Upper:
322  DLAF_ASSERT(a.size().rows() <= a.size().cols(), a);
323  break;
324  case blas::Uplo::General:
325  DLAF_ASSERT(blas::Uplo::General == uplo, uplo);
326  break;
327  }
328  common::internal::SingleThreadedBlasScope single;
329  return lapack::lantr(norm, uplo, diag, a.size().rows(), a.size().cols(), a.ptr(), a.ld());

449 inline void extendROCSolverWorkspace(cusolverDnHandle_t handle,
450  ::dlaf::memory::MemoryView<std::byte, Device::GPU>&& workspace) {
451  whip::stream_t stream;
452  DLAF_GPUBLAS_CHECK_ERROR(cublasGetStream(handle, &stream));
453  auto f = [workspace = std::move(workspace)](whip::error_t status) { whip::check_error(status); };

459  struct Cusolver##Name<Type> { \

461  static void call(cusolverDnHandle_t handle, Args&&... args) { \

467  extendROCSolverWorkspace(handle, std::move(workspace)); \
468  } \
469  }

549 void hegst(cusolverDnHandle_t handle, const int itype, const blas::Uplo uplo,
550  const matrix::Tile<T, Device::GPU>& a, const matrix::Tile<T, Device::GPU>& b) {
551  DLAF_ASSERT(square_size(a), a);
552  DLAF_ASSERT(square_size(b), b);
553  DLAF_ASSERT(a.size() == b.size(), a, b);
554  const auto n = a.size().rows();
555 
556 #ifdef DLAF_WITH_CUDA
557  int workspace_size;
558  internal::CusolverHegst<T>::callBufferSize(handle, itype, util::blasToCublas(uplo), to_int(n),
559  util::blasToCublasCast(a.ptr()), to_int(a.ld()),
560  util::blasToCublasCast(b.ptr()), to_int(b.ld()),
561  &workspace_size);
562  internal::CusolverInfo<T> info{std::max(1, workspace_size)};
563  internal::CusolverHegst<T>::call(handle, itype, util::blasToCublas(uplo), to_int(n),
564  util::blasToCublasCast(a.ptr()), to_int(a.ld()),
565  util::blasToCublasCast(b.ptr()), to_int(b.ld()),
566  util::blasToCublasCast(info.workspace()), info.info());
567 
568  assertExtendInfo(dlaf::gpulapack::internal::assertInfoHegst, handle, std::move(info));
569 #elif defined(DLAF_WITH_HIP)
570  internal::CusolverHegst<T>::call(handle, util::blasToRocblas(itype), util::blasToRocblas(uplo),
571  to_int(n), util::blasToRocblasCast(a.ptr()), to_int(a.ld()),
572  util::blasToRocblasCast(b.ptr()), to_int(b.ld()));
573 #endif
