103template <class T>
104void lauum_workspace(const blas::Uplo uplo, const Tile<T, Device::CPU>& a,
105 const Tile<T, Device::CPU>& ws) {
106 DLAF_ASSERT(square_size(a), a);
107
108 const auto ws_sub = ws.subTileReference({{0, 0}, a.size()});
109 const blas::Side side = uplo == blas::Uplo::Lower ? blas::Side::Left : blas::Side::Right;
110
111 common::internal::SingleThreadedBlasScope single;
112 set0(ws_sub);
113 lapack::lacpy(uplo, a.size().rows(), a.size().cols(), a.ptr(), a.ld(), ws_sub.ptr(), ws_sub.ld());
114 trmm(side, uplo, blas::Op::ConjTrans, blas::Diag::NonUnit, static_cast<T>(1.), a, ws_sub);
115 lapack::lacpy(uplo, a.size().rows(), a.size().cols(), ws_sub.ptr(), ws_sub.ld(), a.ptr(), a.ld());
116}

118template <class T>
119void trtri_workspace(const blas::Uplo uplo, const blas::Diag diag, const Tile<T, Device::CPU>& a,
120 const Tile<T, Device::CPU>& ws) noexcept {
121 DLAF_ASSERT(square_size(a), a);
122
123 if (a.size().rows() == 0 || (a.size().rows() == 1 && diag == blas::Diag::Unit))
124 return;
125
126 const auto ws_sub = ws.subTileReference({{0, 0}, a.size()});
127
128 common::internal::SingleThreadedBlasScope single;
129 laset(blas::Uplo::General, static_cast<T>(0.0), static_cast<T>(1.0), ws_sub);
130 trsm(blas::Side::Left, uplo, blas::Op::NoTrans, diag, static_cast<T>(1.), a, ws_sub);
131 if (diag == blas::Diag::Unit) {
132 TileElementIndex offset =
133 uplo == blas::Uplo::Lower ? TileElementIndex{1, 0} : TileElementIndex{0, 1};
134 lapack::lacpy(uplo, a.size().rows() - 1, a.size().cols() - 1, ws_sub.ptr(offset), ws_sub.ld(),
135 a.ptr(offset), a.ld());
136 }
137 else
138 lapack::lacpy(uplo, a.size().rows(), a.size().cols(), ws_sub.ptr(), ws_sub.ld(), a.ptr(), a.ld());
139}
