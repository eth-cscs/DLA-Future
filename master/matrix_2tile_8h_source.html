115 inline bool subTileSpecsOverlap(const SubTileSpec& spec1, const SubTileSpec& spec2) {
116  // no overlap if either of the sizes is empty.
117  if (spec1.size.isEmpty() || spec2.size.isEmpty()) {
118  return false;
119  }
120 
121  const auto& start1 = spec1.origin;
122  const auto end1 = start1 + spec1.size;
123  const auto& start2 = spec2.origin;
124  const auto end2 = start2 + spec2.size;
125 
126  // no overlap if rows do not overlap.
127  if (end1.row() <= start2.row() || end2.row() <= start1.row()) {
128  return false;
129  }
130  // no overlap if cols do not overlap.
131  if (end1.col() <= start2.col() || end2.col() <= start1.col()) {
132  return false;
133  }
134 
135  return true;
136 }

419  void prepareDisjointTile() {
420  // We only expect read-write tiles for disjoint access. That means that the
421  // dependency tracker holds anything but a read-only wrapper.
422  DLAF_ASSERT((!std::holds_alternative<internal::TileAsyncRwMutexReadOnlyWrapper<T, D>>(dep_tracker_)),
423  "");
424 
425  // If a tile is untracked (std::monostate), or already a disjoint subtile
426  // (std::shared_ptr<internal::TileAsyncRwMutexReadWriteWrapper<T, D>) we don't do
427  // anything. If a tile is in read-write mode we upgrade it to allow shared,
428  // but disjoint, access.
429  if (std::holds_alternative<internal::TileAsyncRwMutexReadWriteWrapper<T, D>>(dep_tracker_)) {
430  dep_tracker_ = std::make_shared<internal::TileAsyncRwMutexReadWriteWrapper<T, D>>(
431  std::get<internal::TileAsyncRwMutexReadWriteWrapper<T, D>>(std::move(dep_tracker_)));
432  }
433  }

435  Tile createDisjointSubTile(const SubTileSpec& spec) const& {
436  // We only expect read-write tiles for disjoint access. They should be
437  // either untracked or disjoint tracked read-write access.
438  DLAF_ASSERT((!std::holds_alternative<internal::TileAsyncRwMutexReadOnlyWrapper<T, D>>(dep_tracker_)),
439  "");
440  DLAF_ASSERT(
441  (!std::holds_alternative<internal::TileAsyncRwMutexReadWriteWrapper<T, D>>(dep_tracker_)), "");
442 
443  Tile subtile(spec.size, ConstTileType::createMemoryViewForSubtile(*this, spec), this->ld());
444  // Not all possible states of the dependency tracker are copyable. This
445  // means that we can't copy the variant as a whole, but only copy the states
446  // that are copyable, if they are active.
447  if (std::holds_alternative<std::shared_ptr<internal::TileAsyncRwMutexReadWriteWrapper<T, D>>>(
448  dep_tracker_)) {
449  subtile.dep_tracker_ =
450  std::get<std::shared_ptr<internal::TileAsyncRwMutexReadWriteWrapper<T, D>>>(dep_tracker_);
451  }
452 
453  return subtile;
454  }

569 template <class T, Device D>
570 std::vector<ReadWriteTileSender<T, D>> splitTileDisjoint(ReadWriteTileSender<T, D>&& tile,
571  const std::vector<SubTileSpec>& specs) {
572  // If there are no specs we still consume the tile by starting it.
573  if (specs.empty()) {
574  pika::execution::experimental::start_detached(std::move(tile));
575  return {};
576  }
577 
578  // If there is only one spec we can avoid calling execution::split on the
579  // input tile (and thus avoid a heap allocation) and directly call the
580  // single-spec splitTile instead.
581  if (specs.size() == 1) {
582  auto subtiles = std::vector<ReadWriteTileSender<T, D>>();
583  subtiles.push_back(splitTile(std::move(tile), specs[0]));
584  return subtiles;
585  }
586 
587 #ifdef DLAF_ASSERT_MODERATE_ENABLE
588  for (auto it1 = specs.cbegin(); it1 < specs.cend(); ++it1) {
589  for (auto it2 = specs.cbegin(); it2 < it1; ++it2) {
590  DLAF_ASSERT_MODERATE(!internal::subTileSpecsOverlap(*it1, *it2), it1->origin, it1->size,
591  it2->origin, it2->size);
592  }
593  }
594 #endif
595 
596  std::vector<ReadWriteTileSender<T, D>> senders;
597  senders.reserve(specs.size());
598 
599  // We first need mutable access to extract a dependency manager, if one exists
600  auto prepared_tile = std::move(tile) |
601  pika::execution::experimental::then(&internal::prepareDisjointTile<T, D>) |
602  pika::execution::experimental::split();
603 
604  for (const auto& spec : specs) {
605  // Once the tile has been prepared for disjoint access, we can then actually
606  // extract the subtiles
607  auto disjoint_tile =
608  prepared_tile | pika::execution::experimental::then([spec](const Tile<T, D>& tile) {
609  return internal::createDisjointSubTile(tile, spec);
610  });
611  senders.push_back(std::move(disjoint_tile));
612  }
613 
614  return senders;
615 }
