38 template <class Func>
39 auto selectGeneric(Func&& f, common::IterableRange2D<SizeType, LocalTile_TAG> range) {
40  using RetT = decltype(f(LocalTileIndex{}));
41 
42  std::vector<RetT> tiles;
43  tiles.reserve(to_sizet(std::distance(range.begin(), range.end())));
44  std::transform(range.begin(), range.end(), std::back_inserter(tiles),
45  [&](auto idx) { return f(idx); });
46  return tiles;
47 }
