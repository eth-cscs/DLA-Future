36 template <class Func>
37 auto selectGeneric(Func&& f, common::IterableRange2D<SizeType, LocalTile_TAG> range) {
38  using RetT = decltype(f(LocalTileIndex{}));
39 
40  std::vector<RetT> tiles;
41  tiles.reserve(to_sizet(std::distance(range.begin(), range.end())));
42  std::transform(range.begin(), range.end(), std::back_inserter(tiles),
43  [&](auto idx) { return f(idx); });
44  return tiles;
45 }
