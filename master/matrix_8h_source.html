37 template <class Func>
38 auto selectGeneric(Func&& f, common::IterableRange2D<SizeType, LocalTile_TAG> range) {
39  using RetT = decltype(f(LocalTileIndex{}));
40 
41  std::vector<RetT> tiles;
42  tiles.reserve(to_sizet(std::distance(range.begin(), range.end())));
43  std::transform(range.begin(), range.end(), std::back_inserter(tiles),
44  [&](auto idx) { return f(idx); });
45  return tiles;
46 }
