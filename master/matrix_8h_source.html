38template <class Func>
39auto selectGeneric(Func&& f, common::IterableRange2D<SizeType, LocalTile_TAG> range) {
40 using RetT = decltype(f(LocalTileIndex{}));
41
42 std::vector<RetT> tiles;
43 tiles.reserve(to_sizet(std::distance(range.begin(), range.end())));
44 std::transform(range.begin(), range.end(), std::back_inserter(tiles),
45 [&](auto idx) { return f(idx); });
46 return tiles;
47}

319template <class T, Device D>
320void Matrix<const T, D>::waitLocalTiles() noexcept {
321 // Note:
322 // Using a readwrite access to the tile ensures that the access is exclusive and not shared
323 // among multiple tasks.
324
325 const auto range_local = common::iterate_range2d(distribution().local_nr_tiles());
326
327 auto s = pika::execution::experimental::when_all_vector(internal::selectGeneric(
328 [this](const LocalTileIndex& index) {
329 return this->tile_managers_[tileLinearIndex(index)].readwrite();
330 },
331 range_local)) |
332 pika::execution::experimental::drop_value();
333 pika::this_thread::experimental::sync_wait(std::move(s));
334}

336template <class T, Device D>
337void Matrix<const T, D>::setUpTiles(const memory::MemoryView<ElementType, D>& mem,
338 const LayoutInfo& layout) noexcept {
339 const auto& nr_tiles = layout.nrTiles();
340
341 DLAF_ASSERT(tile_managers_.empty(), "");
342 tile_managers_.reserve(to_sizet(nr_tiles.linear_size()));
343
344 using MemView = memory::MemoryView<T, D>;
345
346 for (SizeType j = 0; j < nr_tiles.cols(); ++j) {
347 for (SizeType i = 0; i < nr_tiles.rows(); ++i) {
348 LocalTileIndex ind(i, j);
349 TileElementSize tile_size = layout.tileSize(ind);
350 tile_managers_.emplace_back(
351 TileDataType(tile_size, MemView(mem, layout.tileOffset(ind), layout.minTileMemSize(tile_size)),
352 layout.ldTile()));
353 }
354 }

357template <class T, Device D>
358void Matrix<const T, D>::setUpSubPipelines(Matrix<const T, D>& mat) noexcept {
359 namespace ex = pika::execution::experimental;
360
361 // TODO: Optimize read-after-read. This is currently forced to access the base
362 // matrix in readwrite mode so that we can move the tile into the
363 // sub-pipeline. This is semantically not required and should eventually be
364 // optimized.
365 tile_managers_.reserve(mat.tile_managers_.size());
366 for (auto& tm : mat.tile_managers_) {
367 tile_managers_.emplace_back(Tile<T, D>());
368 auto s = ex::when_all(tile_managers_.back().readwrite_with_wrapper(), tm.readwrite()) |
369 ex::then([](internal::TileAsyncRwMutexReadWriteWrapper<T, D> empty_tile_wrapper,
370 Tile<T, D> tile) { empty_tile_wrapper.get() = std::move(tile); });
371 ex::start_detached(std::move(s));
372 }

375template <class T, Device D>
376void Matrix<const T, D>::setUpRetiledSubPipelines(Matrix<const T, D>& mat,
377 const LocalTileSize& tiles_per_block) noexcept {
378 DLAF_ASSERT(mat.blockSize() == mat.tile_size(), mat.blockSize(), mat.tile_size());
379
380 using common::internal::vector;
381 namespace ex = pika::execution::experimental;
382
383 const auto n = to_sizet(distribution().local_nr_tiles().linear_size());
384 tile_managers_.reserve(n);
385 for (std::size_t i = 0; i < n; ++i) {
386 tile_managers_.emplace_back(Tile<T, D>());
387 }
388
389 const auto tile_size = distribution().tile_size();
390 vector<SubTileSpec> specs;
391 vector<LocalTileIndex> indices;
392 specs.reserve(tiles_per_block.linear_size());
393 indices.reserve(tiles_per_block.linear_size());
394
395 // TODO: Optimize read-after-read. This is currently forced to access the base matrix in readwrite
396 // mode so that we can move the tile into the sub-pipeline. This is semantically not required and
397 // should eventually be optimized.
398 for (const auto& orig_tile_index : common::iterate_range2d(mat.distribution().local_nr_tiles())) {
399 const auto original_tile_size = mat.tileSize(mat.distribution().global_tile_index(orig_tile_index));
400
401 for (SizeType j = 0; j < original_tile_size.cols(); j += tile_size.cols())
402 for (SizeType i = 0; i < original_tile_size.rows(); i += tile_size.rows()) {
403 indices.emplace_back(
404 LocalTileIndex{orig_tile_index.row() * tiles_per_block.rows() + i / tile_size.rows(),
405 orig_tile_index.col() * tiles_per_block.cols() + j / tile_size.cols()});
406 specs.emplace_back(SubTileSpec{{i, j},
407 tileSize(distribution().global_tile_index(indices.back()))});
408 }
409
410 auto sub_tiles =
411 splitTileDisjoint(mat.tile_managers_[mat.tileLinearIndex(orig_tile_index)].readwrite(), specs);
412
413 DLAF_ASSERT_HEAVY(specs.size() == indices.size(), specs.size(), indices.size());
414 for (SizeType j = 0; j < specs.size(); ++j) {
415 const auto i = tileLinearIndex(indices[j]);
416
417 // Move subtile to be managed by the tile manager of RetiledMatrix. We
418 // use readwrite_with_wrapper to get access to the original tile managed
419 // by the underlying async_rw_mutex.
420 auto s =
421 ex::when_all(tile_managers_[i].readwrite_with_wrapper(), std::move(sub_tiles[to_sizet(j)])) |
422 ex::then([](internal::TileAsyncRwMutexReadWriteWrapper<T, D> empty_tile_wrapper,
423 Tile<T, D> sub_tile) { empty_tile_wrapper.get() = std::move(sub_tile); });
424 ex::start_detached(std::move(s));
425 }
426
427 specs.clear();
428 indices.clear();
429 }
