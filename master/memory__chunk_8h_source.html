45template <class T, Device D>
46class MemoryChunk {
47public:
48 using ElementType = T;
49
51 MemoryChunk() : MemoryChunk(0) {}
52
58 MemoryChunk(SizeType size) : size_(size), ptr_(nullptr), allocated_(true) {
59 DLAF_ASSERT(size >= 0, size);
60
61 if (size == 0)
62 return;
63
64 std::size_t mem_size = static_cast<std::size_t>(size_) * sizeof(T);
65#ifdef DLAF_WITH_GPU
66 if (D == Device::CPU) {
67 ptr_ = static_cast<T*>(internal::getUmpireHostAllocator().allocate(mem_size));
68 }
69 else {
70 ptr_ = static_cast<T*>(internal::getUmpireDeviceAllocator().allocate(mem_size));
71 }
72#else
73 if (D == Device::CPU) {
74 ptr_ = static_cast<T*>(internal::getUmpireHostAllocator().allocate(mem_size));
75 }
76 else {
77 std::cout
78 << "[ERROR] GPU memory was requested but the `DLAF_WITH_CUDA` or `DLAF_WITH_HIP` flags were not passed!";
79 std::terminate();
80 }
81#endif
82 }
83
89 MemoryChunk(T* ptr, SizeType size) : size_(size), ptr_(size > 0 ? ptr : nullptr), allocated_(false) {
90 DLAF_ASSERT_HEAVY(size == 0 ? ptr_ == nullptr : ptr_ != nullptr, size);
91
92 using dlaf::memory::internal::get_memory_type;
93 using dlaf::memory::internal::MemoryType;
94 auto memory_type = get_memory_type(ptr);
95 switch (memory_type) {
96 case MemoryType::Host:
97 DLAF_ASSERT(D == Device::CPU, "Using a host pointer to construct a device MemoryChunk");
98 break;
99 case MemoryType::Device:
100 DLAF_ASSERT(D == Device::GPU, "Using a device pointer to construct a host MemoryChunk");
101 break;
102 case MemoryType::Managed:
103 break;
104 case MemoryType::Unified:
105 break;
106 }
107 }
