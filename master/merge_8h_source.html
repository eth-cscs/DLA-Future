46 // Auxiliary matrix and vectors used for the D&C algorithm
47 //
48 // - e0: (matrix)
49 // In: Holds the eigenvectors of the two subproblems (same order as d0).
50 // Out: Holds the eigenvectors of the merged problem (same order as d0).
51 // - e1: (matrix)
52 // Holds the deflated eigenvectors.
53 // - e2: (matrix)
54 // (The original evecs matrix used as workspace) Holds the rank-1 eigenvectors.
55 //
56 // - d0: (vector)
57 // In: Holds the eigenvalues of the two subproblems (in ascending order if permuted with i1).
58 // Out: Holds the eigenvalues of the merged problem (in ascending order if permuted with i1).
59 // - d1: (vector)
60 // Holds the values of the deflated diagonal sorted in ascending order
61 // - z0: (vector)
62 // Holds the values of Cuppen's rank-1 vector
63 // Also used as temporary workspace
64 // - z1: (vector)
65 // Holds the values of the rank-1 update vector (same order as d1)
66 //
67 // - c:
68 // Assigns a type to each column of Q which is used to calculate the permutation indices for Q and U
69 // that bring them in matrix multiplication form.
70 //
71 // - i1, i2, i3: (vectors of indices)
72 // Hold the permutation indices.
73 // In: i1 contains the permutation to sort d0 of the two subproblems in ascending order.
74 // If the subproblem involves a single tile the values of i1 are replaced with an identity permutation.
75 // Out: i1 and i2(Device::CPU) contain the permutation to sort d0 of the merged problem in ascending
76 // order.
77 //
78 // Steps:
79 //
80 // 1. Sort index based on diagonal values in ascending order. This creates a map sorted to inital indices
81 //
82 // initial <--- pre_sorted
83 //
84 // 2. Sort index based on column types such that all deflated entries are at the end
85 //
86 // pre_sorted <--- deflated
87 //
88 // 3. Sort index based on updated diagonal values in ascending order. The diagonal conatins eigenvalues
89 // of the deflated problem and deflated entreis from the initial diagonal
90 //
91 // deflated <--- post_sorted
92 //
93 // 4. Sort index based on column types such that matrices `Q` and `U` are in matrix multiplication form.
94 //
95 // post_sorted <--- matmul
96 //

190  Matrix<const T, D>& evecs, Matrix<T, D>& z) {
191  // Iterate over tiles of Q1 and Q2 around the split row `i_split`.
192  for (SizeType i = i_begin; i < i_end; ++i) {
193  // True if tile is in Q1
194  const bool top_tile = i < i_split;
195  // Move to the row below `i_split` for `Q2`
196  const SizeType evecs_row = i_split - ((top_tile) ? 1 : 0);
197  const GlobalTileIndex idx_evecs(evecs_row, i);
198  // Take the last row of a `Q1` tile or the first row of a `Q2` tile
199  const GlobalTileIndex z_idx(i, 0);
200 
201  // Copy the row into the column vector `z`
202  assembleRank1UpdateVectorTileAsync<T, D>(top_tile, rho, evecs.read(idx_evecs), z.readwrite(z_idx));
203  }

262 // The index array `out_ptr` holds the indices of elements of `c_ptr` that order it such that
263 // ColType::Deflated entries are moved to the end. The `c_ptr` array is implicitly ordered according to
264 // `in_ptr` on entry.
265 //
266 inline SizeType stablePartitionIndexForDeflationArrays(const SizeType n, const ColType* c_ptr,
267  const SizeType* in_ptr, SizeType* out_ptr) {
268  // Get the number of non-deflated entries
269  SizeType k = 0;
270  for (SizeType i = 0; i < n; ++i) {
271  if (c_ptr[i] != ColType::Deflated)
272  ++k;
273  }
274 
275  SizeType i1 = 0; // index of non-deflated values in out
276  SizeType i2 = k; // index of deflated values
277  for (SizeType i = 0; i < n; ++i) {
278  const SizeType ii = in_ptr[i];
279  SizeType& io = (c_ptr[ii] != ColType::Deflated) ? i1 : i2;
280  out_ptr[io] = ii;
281  ++io;
282  }
283  return k;
284 }

344 // Assumption 1: The algorithm assumes that the arrays `d_ptr`, `z_ptr` and `c_ptr` are of equal length
345 // `len` and are sorted in ascending order of `d_ptr` elements with `i_ptr`.
346 //
347 // Note: parallelizing this algorithm is non-trivial because the deflation regions due to Givens
348 // rotations can cross over tiles and are of unknown length. However such algorithm is unlikely to
349 // benefit much from parallelization anyway as it is quite light on flops and it appears memory bound.
350 //
351 // Returns an array of Given's rotations used to update the colunmns of the eigenvector matrix Q
352 template <class T>
353 std::vector<GivensRotation<T>> applyDeflationToArrays(T rho, T tol, const SizeType len,
354  const SizeType* i_ptr, T* d_ptr, T* z_ptr,
355  ColType* c_ptr) {
356  std::vector<GivensRotation<T>> rots;
357  rots.reserve(to_sizet(len));
358 
359  SizeType i1 = 0; // index of 1st element in the Givens rotation
360  // Iterate over the indices of the sorted elements in pair (i1, i2) where i1 < i2 for every iteration
361  for (SizeType i2 = 1; i2 < len; ++i2) {
362  const SizeType i1s = i_ptr[i1];
363  const SizeType i2s = i_ptr[i2];
364  T& d1 = d_ptr[i1s];
365  T& d2 = d_ptr[i2s];
366  T& z1 = z_ptr[i1s];
367  T& z2 = z_ptr[i2s];
368  ColType& c1 = c_ptr[i1s];
369  ColType& c2 = c_ptr[i2s];
370 
371  // if z1 nearly zero deflate the element and move i1 forward to i2
372  if (std::abs(rho * z1) <= tol) {
373  c1 = ColType::Deflated;
374  i1 = i2;
375  continue;
376  }
377 
378  // Deflate the second element if z2 nearly zero
379  if (std::abs(rho * z2) <= tol) {
380  c2 = ColType::Deflated;
381  continue;
382  }
383 
384  // Given's deflation condition is the same as the one used in LAPACK's stedc implementation [1].
385  // However, here the second entry is deflated instead of the first (z2/d2 instead of z1/d1), thus
386  // `s` is not negated.
387  //
388  // [1] LAPACK 3.10.0, file dlaed2.f, line 393
389  T r = std::sqrt(z1 * z1 + z2 * z2);
390  T c = z1 / r;
391  T s = z2 / r;
392 
393  // If d1 is not nearly equal to d2, move i1 forward to i2
394  if (std::abs(c * s * (d2 - d1)) > tol) {
395  i1 = i2;
396  continue;
397  }
398 
399  // When d1 is nearly equal to d2 apply Givens rotation
400  z1 = r;
401  z2 = 0;
402  T tmp = d1 * s * s + d2 * c * c;
403  d1 = d1 * c * c + d2 * s * s;
404  d2 = tmp;
405 
406  rots.push_back(GivensRotation<T>{i1s, i2s, c, s});
407  // Set the the `i1` column as "Dense" if the `i2` column has opposite non-zero structure (i.e if
408  // one comes from Q1 and the other from Q2 or vice-versa)
409  if ((c1 == ColType::UpperHalf && c2 == ColType::LowerHalf) ||
410  (c1 == ColType::LowerHalf && c2 == ColType::UpperHalf)) {
411  c1 = ColType::Dense;
412  }
413  c2 = ColType::Deflated;
414  }
415 
416  return rots;
417 }

453 template <class T, class KSender, class RhoSender>
454 void solveRank1Problem(const SizeType i_begin, const SizeType i_end, KSender&& k, RhoSender&& rho,
455  Matrix<const T, Device::CPU>& d, Matrix<T, Device::CPU>& z,
456  Matrix<T, Device::CPU>& evals, Matrix<T, Device::CPU>& evecs) {

459 
460  const SizeType n = problemSize(i_begin, i_end, evals.distribution());
461  const SizeType nb = evals.distribution().blockSize().rows();
462 
463  TileCollector tc{i_begin, i_end};
464 
465  const std::size_t nthreads = getTridiagRank1NWorkers();
466  ex::start_detached(
467  ex::when_all(ex::just(std::make_shared<pika::barrier<>>(nthreads)), std::forward<KSender>(k),
468  std::forward<RhoSender>(rho), ex::when_all_vector(tc.read(d)),
469  ex::when_all_vector(tc.readwrite(z)), ex::when_all_vector(tc.readwrite(evals)),
470  ex::when_all_vector(tc.readwrite(evecs)),
471  ex::just(std::vector<memory::MemoryView<T, Device::CPU>>())) |
472  ex::transfer(di::getBackendScheduler<Backend::MC>(pika::execution::thread_priority::high)) |
473  ex::bulk(nthreads, [nthreads, n, nb](std::size_t thread_idx, auto& barrier_ptr, auto& k, auto& rho,
474  auto& d_tiles_futs, auto& z_tiles, auto& eval_tiles,
475  auto& evec_tiles, auto& ws_vecs) {
476  const matrix::Distribution distr(LocalElementSize(n, n), TileElementSize(nb, nb));
477 
478  const std::size_t batch_size = util::ceilDiv(to_sizet(k), nthreads);
479  const std::size_t begin = thread_idx * batch_size;
480  const std::size_t end = std::min(thread_idx * batch_size + batch_size, to_sizet(k));
481 
482  // STEP 0: Initialize workspaces (single-thread)
483  if (thread_idx == 0) {
484  ws_vecs.reserve(nthreads);
485  for (std::size_t i = 0; i < nthreads; ++i)
486  ws_vecs.emplace_back(to_sizet(k));
487  }
488 
489  barrier_ptr->arrive_and_wait();
490 
491  // STEP 1: LAED4 (multi-thread)
492  const T* d_ptr = d_tiles_futs[0].get().ptr();
493  const T* z_ptr = z_tiles[0].ptr();
494 
495  {
496  common::internal::SingleThreadedBlasScope single;
497 
498  T* eval_ptr = eval_tiles[0].ptr();
499 
500  for (std::size_t i = begin; i < end; ++i) {
501  T& eigenval = eval_ptr[i];
502 
503  const SizeType i_tile = distr.globalTileLinearIndex(GlobalElementIndex(0, to_SizeType(i)));
504  const SizeType i_col = distr.tileElementFromGlobalElement<Coord::Col>(to_SizeType(i));
505  T* delta = evec_tiles[to_sizet(i_tile)].ptr(TileElementIndex(0, i_col));
506 
507  lapack::laed4(to_int(k), to_int(i), d_ptr, z_ptr, delta, rho, &eigenval);
508  }
509  // Note for in-place row permutation implementation: The rows should be permuted for the k=2 case as well.
510  if (k <= 2)
511  return;
512  }
513 
514  barrier_ptr->arrive_and_wait();
515 
516  // STEP 2a Compute weights (multi-thread)
517  auto& q = evec_tiles;
518  T* w = ws_vecs[thread_idx]();
519 
520  // - copy diagonal from q -> w (or just initialize with 1)
521  if (thread_idx == 0) {
522  for (auto i = 0; i < k; ++i) {
523  const GlobalElementIndex kk(i, i);
524  const auto diag_tile = distr.globalTileLinearIndex(kk);
525  const auto diag_element = distr.tileElementIndex(kk);
526 
527  w[i] = q[to_sizet(diag_tile)](diag_element);
528  }
529  }
530  else {
531  std::fill_n(w, k, T(1));
532  }
533 
534  // - compute productorial
535  auto compute_w = [&](const GlobalElementIndex ij) {
536  const auto q_tile = distr.globalTileLinearIndex(ij);
537  const auto q_ij = distr.tileElementIndex(ij);
538 
539  const SizeType i = ij.row();
540  const SizeType j = ij.col();
541 
542  w[i] *= q[to_sizet(q_tile)](q_ij) / (d_ptr[to_sizet(i)] - d_ptr[to_sizet(j)]);
543  };
544 
545  for (auto j = to_SizeType(begin); j < to_SizeType(end); ++j) {
546  for (auto i = 0; i < j; ++i)
547  compute_w({i, j});
548

550  compute_w({i, j});
551  }
552

554 
555  // STEP 2B: reduce, then finalize computation with sign and square root (single-thread)
556  if (thread_idx == 0) {
557  for (int i = 0; i < k; ++i) {
558  for (std::size_t tidx = 1; tidx < nthreads; ++tidx) {
559  const T* w_partial = ws_vecs[tidx]();
560  w[i] *= w_partial[i];
561  }
562  z_tiles[0].ptr()[i] = std::copysign(std::sqrt(-w[i]), z_ptr[to_sizet(i)]);
563  }
564  }
565 
566  barrier_ptr->arrive_and_wait();
567 
568  // STEP 3: Compute eigenvectors of the modified rank-1 modification (normalize) (multi-thread)
569  {
570  common::internal::SingleThreadedBlasScope single;
571 
572  const T* w = z_ptr;
573  T* s = ws_vecs[thread_idx]();
574 
575  for (auto j = to_SizeType(begin); j < to_SizeType(end); ++j) {
576  for (int i = 0; i < k; ++i) {
577  const auto q_tile = distr.globalTileLinearIndex({i, j});
578  const auto q_ij = distr.tileElementIndex({i, j});
579 
580  s[i] = w[i] / q[to_sizet(q_tile)](q_ij);
581  }
582 
583  const T vec_norm = blas::nrm2(k, s, 1);
584

586  const auto q_tile = distr.globalTileLinearIndex({i, j});
587  const auto q_ij = distr.tileElementIndex({i, j});
588 
589  q[to_sizet(q_tile)](q_ij) = s[i] / vec_norm;
590  }
591  }
592  }
593  }));
