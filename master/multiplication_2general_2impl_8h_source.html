33template <Backend B, Device D, class T>
34void General<B, D, T>::callNN(const T alpha, MatrixRef<const T, D>& mat_a, MatrixRef<const T, D>& mat_b,
35 const T beta, MatrixRef<T, D>& mat_c) {
36 namespace ex = pika::execution::experimental;
37
38 if (mat_a.nr_tiles().cols() == 0) {
39 // Note: if beta == 1, we optimize by not even scheduling anything
40 if (beta != T(1)) {
41 for (SizeType j = 0; j < mat_c.distribution().local_nr_tiles().cols(); ++j)
42 for (SizeType i = 0; i < mat_c.distribution().local_nr_tiles().rows(); ++i)
43 ex::start_detached(dlaf::internal::whenAllLift(beta, mat_c.readwrite(LocalTileIndex(i, j))) |
44 tile::scal(dlaf::internal::Policy<B>()));
45 }
46 return;
47 }
48
49 for (SizeType j = 0; j < mat_c.distribution().local_nr_tiles().cols(); ++j) {
50 for (SizeType i = 0; i < mat_c.distribution().local_nr_tiles().rows(); ++i) {
51 for (SizeType k = 0; k < mat_a.distribution().local_nr_tiles().cols(); ++k) {
52 ex::start_detached(dlaf::internal::whenAllLift(blas::Op::NoTrans, blas::Op::NoTrans, alpha,
53 mat_a.read(LocalTileIndex(i, k)),
54 mat_b.read(LocalTileIndex(k, j)),
55 k == 0 ? beta : T(1),
56 mat_c.readwrite(LocalTileIndex(i, j))) |
57 tile::gemm(dlaf::internal::Policy<B>()));
58 }
59 }
60 }

63template <Backend B, Device D, class T>
64void General<B, D, T>::callNN(comm::CommunicatorPipeline<comm::CommunicatorType::Row>& row_task_chain,
65 comm::CommunicatorPipeline<comm::CommunicatorType::Col>& col_task_chain,
66 const T alpha, MatrixRef<const T, D>& mat_a, MatrixRef<const T, D>& mat_b,
67 const T beta, MatrixRef<T, D>& mat_c) {
68 namespace ex = pika::execution::experimental;
69
70 if (mat_c.size().isEmpty())
71 return;
72
73 const matrix::Distribution& dist_a = mat_a.distribution();
74 const matrix::Distribution& dist_b = mat_b.distribution();
75 const matrix::Distribution& dist_c = mat_c.distribution();
76 const auto rank = dist_c.rank_index();
77
78 if (mat_a.nr_tiles().cols() == 0) {
79 // Note: if beta == 1, we optimize by not even scheduling anything
80 if (beta != T(1)) {
81 for (SizeType j = 0; j < mat_c.distribution().local_nr_tiles().cols(); ++j)
82 for (SizeType i = 0; i < mat_c.distribution().local_nr_tiles().rows(); ++i)
83 ex::start_detached(dlaf::internal::whenAllLift(beta, mat_c.readwrite(LocalTileIndex(i, j))) |
84 tile::scal(dlaf::internal::Policy<B>()));
85 }
86 return;
87 }
88
89 constexpr std::size_t n_workspaces = 2;
90 common::RoundRobin<matrix::Panel<Coord::Col, T, D>> panelsA(n_workspaces, dist_a);
91 common::RoundRobin<matrix::Panel<Coord::Row, T, D>> panelsB(n_workspaces, dist_b);
92
93 DLAF_ASSERT_HEAVY(mat_a.nr_tiles().cols() == mat_b.nr_tiles().rows(), mat_a.nr_tiles(),
94 mat_b.nr_tiles());
95
96 // This loops over the global indices for k, because every rank has to participate in communication
97 for (SizeType k = 0; k < mat_a.nr_tiles().cols(); ++k) {
98 auto& panelA = panelsA.nextResource();
99 auto& panelB = panelsB.nextResource();
100
101 if (k == 0 || k == mat_a.nr_tiles().cols() - 1) {
102 DLAF_ASSERT_HEAVY(dist_a.tile_size_of<Coord::Col>(k) == dist_b.tile_size_of<Coord::Row>(k),
103 dist_a.tile_size_of<Coord::Col>(k), dist_b.tile_size_of<Coord::Row>(k));
104 const SizeType kSize = dist_a.tile_size_of<Coord::Col>(k);
105 panelA.setWidth(kSize);
106 panelB.setHeight(kSize);
107 }
108
109 // Setup the column workspace for the root ranks, i.e. the ones in the current col
110 const auto rank_k_col = dist_a.rank_global_tile<Coord::Col>(k);
111 if (rank_k_col == rank.col()) {
112 const auto k_local = dist_a.local_tile_from_global_tile<Coord::Col>(k);
113 for (SizeType i = 0; i < dist_c.local_nr_tiles().rows(); ++i) {
114 const LocalTileIndex ik(i, k_local);
115 panelA.setTile(ik, mat_a.read(ik));
116 }
117 }
118 // Setup the row workspace for the root ranks, i.e. the ones in the current row
119 const auto rank_k_row = dist_b.rank_global_tile<Coord::Row>(k);
120 if (rank_k_row == rank.row()) {
121 const auto k_local = dist_b.local_tile_from_global_tile<Coord::Row>(k);
122 for (SizeType j = 0; j < dist_c.local_nr_tiles().cols(); ++j) {
123 const LocalTileIndex kj(k_local, j);
124 panelB.setTile(kj, mat_b.read(kj));
125 }
126 }
127
128 // Broadcast both column and row panel from root to others (row-wise and col-wise, respectively)
129 broadcast(rank_k_col, panelA, row_task_chain);
130 broadcast(rank_k_row, panelB, col_task_chain);
131
132 // This is the core loop where the k step performs the update over the entire local matrix using
133 // the col and row workspaces.
134 // Everything needed for the update is available locally thanks to previous broadcasts.
135 for (SizeType i = 0; i < dist_c.local_nr_tiles().rows(); ++i) {
136 for (SizeType j = 0; j < dist_c.local_nr_tiles().cols(); ++j) {
137 const LocalTileIndex ij(i, j);
138
139 ex::start_detached(dlaf::internal::whenAllLift(blas::Op::NoTrans, blas::Op::NoTrans, alpha,
140 panelA.read(ij), panelB.read(ij),
141 k == 0 ? beta : T(1), mat_c.readwrite(ij)) |
142 tile::gemm(dlaf::internal::Policy<B>()));
143 }
144 }
145
146 panelA.reset();
147 panelB.reset();
148 }

152void GeneralSub<B, D, T>::callNN(const SizeType idx_begin, const SizeType idx_end, const blas::Op opA,
153 const blas::Op opB, const T alpha, Matrix<const T, D>& mat_a,
154 Matrix<const T, D>& mat_b, const T beta, Matrix<T, D>& mat_c) {
155 namespace ex = pika::execution::experimental;
156
157 for (SizeType j = idx_begin; j < idx_end; ++j) {
158 for (SizeType i = idx_begin; i < idx_end; ++i) {
159 for (SizeType k = idx_begin; k < idx_end; ++k) {
160 ex::start_detached(
161 dlaf::internal::whenAllLift(opA, opB, alpha, mat_a.read(GlobalTileIndex(i, k)),
162 mat_b.read(GlobalTileIndex(k, j)), k == idx_begin ? beta : T(1),
163 mat_c.readwrite(GlobalTileIndex(i, j))) |
164 tile::gemm(dlaf::internal::Policy<B>()));
165 }

170// This implementation is based on
171//
172// Van De Geijn, Robert A., and Jerrell Watts.
173// SUMMA: Scalable universal matrix multiplication algorithm.
174// Concurrency: Practice and Experience 9.4 (1997): 255-274
175template <Backend B, Device D, class T>
176void GeneralSub<B, D, T>::callNN(comm::CommunicatorPipeline<comm::CommunicatorType::Row>& row_task_chain,
177 comm::CommunicatorPipeline<comm::CommunicatorType::Col>& col_task_chain,
178 const SizeType idx_begin, const SizeType idx_end, const T alpha,
179 Matrix<const T, D>& mat_a, Matrix<const T, D>& mat_b, const T beta,
180 Matrix<T, D>& mat_c) {
181 namespace ex = pika::execution::experimental;
182
183 if (idx_begin == idx_end)
184 return;
185
186 const auto& dist_a = mat_a.distribution();
187 const auto rank = dist_a.rankIndex();
188
189 // which rank has the last tile involved
190 const bool rankHasLastRow = rank.row() == dist_a.template rankGlobalTile<Coord::Row>(idx_end - 1);
191 const bool rankHasLastCol = rank.col() == dist_a.template rankGlobalTile<Coord::Col>(idx_end - 1);
192
193 // translate from global to local indices
194 const SizeType i_beg = dist_a.template nextLocalTileFromGlobalTile<Coord::Row>(idx_begin);
195 const SizeType i_end = dist_a.template nextLocalTileFromGlobalTile<Coord::Row>(idx_end);
196
197 const SizeType j_beg = dist_a.template nextLocalTileFromGlobalTile<Coord::Col>(idx_begin);
198 const SizeType j_end = dist_a.template nextLocalTileFromGlobalTile<Coord::Col>(idx_end);
199
200 const SizeType mb = dist_a.blockSize().rows();
201 const SizeType lastTileElement = std::min(idx_end * mb - 1, dist_a.size().rows() - 1);
202 const SizeType nrefls = lastTileElement - idx_begin * mb + 1;
203
204 // Note: if last tile is incomplete, compute the size of it
205 const bool isEndRangePartial = nrefls % mb != 0;
206 const SizeType partialSize = (nrefls % mb);
207
208 // Note:
209 // Workspace needed is limited to the range [i_begin:i_end). Its allocation is obtained by creating an
210 // ad hoc distribution that starts in the origin of the matrix and with a size covering all needed
211 // elements. This would lead to a [0:i_end) range, but by using panel offset at initialization, the
212 // part before the range will be left out from allocation, actually getting [i_begin:i_end)
213 const GlobalTileIndex panel_offset(idx_begin, idx_begin);
214 const matrix::Distribution dist_panel({lastTileElement + 1, lastTileElement + 1}, dist_a.blockSize(),
215 dist_a.commGridSize(), dist_a.rankIndex(),
216 dist_a.sourceRankIndex());
217
218 constexpr std::size_t n_workspaces = 2;
219 common::RoundRobin<matrix::Panel<Coord::Col, T, D>> panelsA(n_workspaces, dist_panel, panel_offset);
220 common::RoundRobin<matrix::Panel<Coord::Row, T, D>> panelsB(n_workspaces, dist_panel, panel_offset);
221
222 // This loops over the global indices for k, because every rank have to participate in communication
223 for (SizeType k = idx_begin; k < idx_end; ++k) {
224 auto& panelA = panelsA.nextResource();
225 auto& panelB = panelsB.nextResource();
226
227 const bool isKPartial = k == idx_end - 1 && isEndRangePartial;
228 const SizeType kSize = isKPartial ? partialSize : mb;
229 if (isKPartial) {
230 panelA.setWidth(kSize);
231 panelB.setHeight(kSize);
232 }
233
234 const auto rank_k = dist_a.rankGlobalTile({k, k});
235
236 // Setup the column workspace for the root ranks, i.e. the ones in the current col
237 if (rank_k.col() == rank.col()) {
238 const auto k_local = dist_a.template localTileFromGlobalTile<Coord::Col>(k);
239 for (SizeType i = i_beg; i < i_end; ++i) {
240 const LocalTileIndex ik(i, k_local);
241 const bool isRowPartial = (i == i_end - 1 && isEndRangePartial && rankHasLastRow);
242 const SizeType nrows = isRowPartial ? partialSize : mb;
243 panelA.setTile(ik, (isRowPartial || isKPartial)
244 ? splitTile(mat_a.read(ik), {{0, 0}, {nrows, kSize}})
245 : mat_a.read(ik));
246 }
247 }
248 // Setup the row workspace for the root ranks, i.e. the ones in the current row
249 if (rank_k.row() == rank.row()) {
250 const auto k_local = dist_a.template localTileFromGlobalTile<Coord::Row>(k);
251 for (SizeType j = j_beg; j < j_end; ++j) {
252 const LocalTileIndex kj(k_local, j);
253 const bool isColPartial = (j == j_end - 1 && isEndRangePartial && rankHasLastCol);
254 const SizeType ncols = isColPartial ? partialSize : mb;
255 panelB.setTile(kj, (isKPartial || isColPartial)
256 ? splitTile(mat_b.read(kj), {{0, 0}, {kSize, ncols}})
257 : mat_b.read(kj));
258 }
259 }
260
261 // Broadcast both column and row panel from root to others (row-wise and col-wise, respectively)
262 broadcast(rank_k.col(), panelA, row_task_chain);
263 broadcast(rank_k.row(), panelB, col_task_chain);
264
265 // This is the core loop where the k step performs the update over the entire local matrix using
266 // the col and row workspaces.
267 // Everything needed for the update is available locally thanks to previous broadcasts.
268 for (SizeType i = i_beg; i < i_end; ++i) {
269 const bool isRowPartial = (i == i_end - 1 && isEndRangePartial && rankHasLastRow);
270 const SizeType nrows = isRowPartial ? partialSize : mb;
271
272 for (SizeType j = j_beg; j < j_end; ++j) {
273 const LocalTileIndex ij(i, j);
274
275 const bool isColPartial = (j == j_end - 1 && isEndRangePartial && rankHasLastCol);
276 const SizeType ncols = isColPartial ? partialSize : mb;
277
278 ex::start_detached(
279 dlaf::internal::whenAllLift(blas::Op::NoTrans, blas::Op::NoTrans, alpha, panelA.read(ij),
280 panelB.read(ij), k == idx_begin ? beta : T(1),
281 (isRowPartial || isColPartial)
282 ? splitTile(mat_c.readwrite(ij), {{0, 0}, {nrows, ncols}})
283 : mat_c.readwrite(ij)) |
284 tile::gemm(dlaf::internal::Policy<B>()));
285 }
286 }
287
288 panelA.reset();
289 panelB.reset();
290 }
291}
