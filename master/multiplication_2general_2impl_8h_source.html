34template <Backend B, Device D, class T>
35void General<B, D, T>::callNN(const T alpha, MatrixRef<const T, D>& mat_a, MatrixRef<const T, D>& mat_b,
36 const T beta, MatrixRef<T, D>& mat_c) {
37 namespace ex = pika::execution::experimental;
38
39 if (mat_a.nr_tiles().cols() == 0) {
40 // Note: if beta == 1, we optimize by not even scheduling anything
41 if (beta != T(1)) {
42 for (SizeType j = 0; j < mat_c.distribution().local_nr_tiles().cols(); ++j)
43 for (SizeType i = 0; i < mat_c.distribution().local_nr_tiles().rows(); ++i)
44 ex::start_detached(dlaf::internal::whenAllLift(beta, mat_c.readwrite(LocalTileIndex(i, j))) |
45 tile::scal(dlaf::internal::Policy<B>()));
46 }
47 return;
48 }
49
50 for (SizeType j = 0; j < mat_c.distribution().local_nr_tiles().cols(); ++j) {
51 for (SizeType i = 0; i < mat_c.distribution().local_nr_tiles().rows(); ++i) {
52 for (SizeType k = 0; k < mat_a.distribution().local_nr_tiles().cols(); ++k) {
53 ex::start_detached(dlaf::internal::whenAllLift(blas::Op::NoTrans, blas::Op::NoTrans, alpha,
54 mat_a.read(LocalTileIndex(i, k)),
55 mat_b.read(LocalTileIndex(k, j)),
56 k == 0 ? beta : T(1),
57 mat_c.readwrite(LocalTileIndex(i, j))) |
58 tile::gemm(dlaf::internal::Policy<B>()));
59 }
60 }
61 }

64template <Backend B, Device D, class T>
65void General<B, D, T>::callNN(comm::CommunicatorPipeline<comm::CommunicatorType::Row>& row_task_chain,
66 comm::CommunicatorPipeline<comm::CommunicatorType::Col>& col_task_chain,
67 const T alpha, MatrixRef<const T, D>& mat_a, MatrixRef<const T, D>& mat_b,
68 const T beta, MatrixRef<T, D>& mat_c) {
69 namespace ex = pika::execution::experimental;
70
71 if (mat_c.size().isEmpty())
72 return;
73
74 const matrix::Distribution& dist_a = mat_a.distribution();
75 const matrix::Distribution& dist_b = mat_b.distribution();
76 const matrix::Distribution& dist_c = mat_c.distribution();
77 const auto rank = dist_c.rank_index();
78
79 if (mat_a.nr_tiles().cols() == 0) {
80 // Note: if beta == 1, we optimize by not even scheduling anything
81 if (beta != T(1)) {
82 for (SizeType j = 0; j < mat_c.distribution().local_nr_tiles().cols(); ++j)
83 for (SizeType i = 0; i < mat_c.distribution().local_nr_tiles().rows(); ++i)
84 ex::start_detached(dlaf::internal::whenAllLift(beta, mat_c.readwrite(LocalTileIndex(i, j))) |
85 tile::scal(dlaf::internal::Policy<B>()));
86 }
87 return;
88 }
89
90 constexpr std::size_t n_workspaces = 2;
91 common::RoundRobin<matrix::Panel<Coord::Col, T, D>> panelsA(n_workspaces, dist_a);
92 common::RoundRobin<matrix::Panel<Coord::Row, T, D>> panelsB(n_workspaces, dist_b);
93
94 DLAF_ASSERT_HEAVY(mat_a.nr_tiles().cols() == mat_b.nr_tiles().rows(), mat_a.nr_tiles(),
95 mat_b.nr_tiles());
96
97 // This loops over the global indices for k, because every rank has to participate in communication
98 for (SizeType k = 0; k < mat_a.nr_tiles().cols(); ++k) {
99 auto& panelA = panelsA.nextResource();
100 auto& panelB = panelsB.nextResource();
101
102 if (k == 0 || k == mat_a.nr_tiles().cols() - 1) {
103 DLAF_ASSERT_HEAVY(dist_a.global_tile_size_of<Coord::Col>(k) ==
104 dist_b.global_tile_size_of<Coord::Row>(k),
105 dist_a.global_tile_size_of<Coord::Col>(k),
106 dist_b.global_tile_size_of<Coord::Row>(k));
107 const SizeType kSize = dist_a.global_tile_size_of<Coord::Col>(k);
108 panelA.setWidth(kSize);
109 panelB.setHeight(kSize);
110 }
111
112 // Setup the column workspace for the root ranks, i.e. the ones in the current col
113 const auto rank_k_col = dist_a.rank_global_tile<Coord::Col>(k);
114 if (rank_k_col == rank.col()) {
115 const auto k_local = dist_a.local_tile_from_global_tile<Coord::Col>(k);
116 for (SizeType i = 0; i < dist_c.local_nr_tiles().rows(); ++i) {
117 const LocalTileIndex ik(i, k_local);
118 panelA.setTile(ik, mat_a.read(ik));
119 }
120 }
121 // Setup the row workspace for the root ranks, i.e. the ones in the current row
122 const auto rank_k_row = dist_b.rank_global_tile<Coord::Row>(k);
123 if (rank_k_row == rank.row()) {
124 const auto k_local = dist_b.local_tile_from_global_tile<Coord::Row>(k);
125 for (SizeType j = 0; j < dist_c.local_nr_tiles().cols(); ++j) {
126 const LocalTileIndex kj(k_local, j);
127 panelB.setTile(kj, mat_b.read(kj));
128 }
129 }
130
131 // Broadcast both column and row panel from root to others (row-wise and col-wise, respectively)
132 broadcast(rank_k_col, panelA, row_task_chain);
133 broadcast(rank_k_row, panelB, col_task_chain);
134
135 // This is the core loop where the k step performs the update over the entire local matrix using
136 // the col and row workspaces.
137 // Everything needed for the update is available locally thanks to previous broadcasts.
138 for (SizeType i = 0; i < dist_c.local_nr_tiles().rows(); ++i) {
139 for (SizeType j = 0; j < dist_c.local_nr_tiles().cols(); ++j) {
140 const LocalTileIndex ij(i, j);
141
142 ex::start_detached(dlaf::internal::whenAllLift(blas::Op::NoTrans, blas::Op::NoTrans, alpha,
143 panelA.read(ij), panelB.read(ij),
144 k == 0 ? beta : T(1), mat_c.readwrite(ij)) |
145 tile::gemm(dlaf::internal::Policy<B>()));
146 }
147 }
148
149 panelA.reset();
150 panelB.reset();
151 }
152}
