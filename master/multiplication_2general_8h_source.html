130template <Backend B, Device D, class T>
131void generalSubMatrix(const SizeType a, const SizeType b, const blas::Op opA, const blas::Op opB,
132 const T alpha, Matrix<const T, D>& mat_a, Matrix<const T, D>& mat_b, const T beta,
133 Matrix<T, D>& mat_c) {
134 DLAF_ASSERT(dlaf::matrix::square_blocksize(mat_a), mat_a);
135 DLAF_ASSERT(dlaf::matrix::square_blocksize(mat_b), mat_b);
136 DLAF_ASSERT(dlaf::matrix::square_blocksize(mat_c), mat_c);
137
138 DLAF_ASSERT(matrix::single_tile_per_block(mat_a), mat_a);
139 DLAF_ASSERT(matrix::single_tile_per_block(mat_b), mat_b);
140 DLAF_ASSERT(matrix::single_tile_per_block(mat_c), mat_c);
141
142 DLAF_ASSERT(matrix::local_matrix(mat_a), mat_a);
143 DLAF_ASSERT(matrix::local_matrix(mat_b), mat_b);
144 DLAF_ASSERT(matrix::local_matrix(mat_c), mat_c);
145
146 // Note:
147 // This is an over-constraint, since the algorithm just cares about the sub-matrix size.
148 // It simplifies next check about [a,b) range validity, that otherwise would require it to be
149 // validated against every single sub-matrix in a, b and c that might have different element sizes.
150 //
151 // At the moment, we don't have this use-case, so let's keep it simple.
152 DLAF_ASSERT(dlaf::matrix::square_size(mat_a), mat_a);
153 DLAF_ASSERT(equal_size(mat_a, mat_b), mat_a, mat_b);
154 DLAF_ASSERT(equal_size(mat_a, mat_c), mat_a, mat_c);
155
156 [[maybe_unused]] const SizeType m = mat_a.nrTiles().rows();
157 DLAF_ASSERT(a <= b, a, b);
158 DLAF_ASSERT(a >= 0 && a <= m, a, m);
159 DLAF_ASSERT(b >= 0 && b <= m, b, m);
160
161 using namespace blas;
162
163 if (opA == Op::NoTrans && opB == Op::NoTrans)
164 internal::GeneralSub<B, D, T>::callNN(a, b, opA, opB, alpha, mat_a, mat_b, beta, mat_c);
165 else
166 DLAF_UNIMPLEMENTED(opA, opB);
167}
