46 void generalSubMatrix(const SizeType a, const SizeType b, const blas::Op opA, const blas::Op opB,
47  const T alpha, Matrix<const T, D>& mat_a, Matrix<const T, D>& mat_b, const T beta,
48  Matrix<T, D>& mat_c) {
49  DLAF_ASSERT(dlaf::matrix::square_blocksize(mat_a), mat_a);
50  DLAF_ASSERT(dlaf::matrix::square_blocksize(mat_b), mat_b);
51  DLAF_ASSERT(dlaf::matrix::square_blocksize(mat_c), mat_c);
52 
53  DLAF_ASSERT(matrix::local_matrix(mat_a), mat_a);
54  DLAF_ASSERT(matrix::local_matrix(mat_b), mat_b);
55  DLAF_ASSERT(matrix::local_matrix(mat_c), mat_c);
56 
57  // Note:
58  // This is an over-constraint, since the algorithm just cares about the sub-matrix size.
59  // It simplifies next check about [a,b) range validity, that otherwise would require it to be
60  // validated against every single sub-matrix in a, b and c that might have different element sizes.
61  //
62  // At the moment, we don't have this use-case, so let's keep it simple.
63  DLAF_ASSERT(dlaf::matrix::square_size(mat_a), mat_a);
64  DLAF_ASSERT(equal_size(mat_a, mat_b), mat_a, mat_b);
65  DLAF_ASSERT(equal_size(mat_a, mat_c), mat_a, mat_c);
66 
67  [[maybe_unused]] const SizeType m = mat_a.nrTiles().rows();
68  DLAF_ASSERT(a <= b, a, b);
69  DLAF_ASSERT(a >= 0 && a <= m, a, m);
70  DLAF_ASSERT(b >= 0 && b <= m, b, m);
71 
72  using namespace blas;
73 
74  if (opA == Op::NoTrans && opB == Op::NoTrans)
75  internal::GeneralSub<B, D, T>::callNN(a, b, opA, opB, alpha, mat_a, mat_b, beta, mat_c);
76  else
77  DLAF_UNIMPLEMENTED(opA, opB);
78 }
