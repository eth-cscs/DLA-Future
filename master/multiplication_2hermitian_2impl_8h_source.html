63 template <Backend B, Device D, class T>
64 void Hermitian<B, D, T>::call_LL(const T alpha, Matrix<const T, D>& mat_a, Matrix<const T, D>& mat_b,
65  const T beta, Matrix<T, D>& mat_c) {
66  using namespace hermitian_ll;
67  const SizeType k = mat_a.distribution().localNrTiles().cols();
68 
69  for (const auto ij : common::iterate_range2d(mat_c.distribution().localNrTiles())) {
70  T beta_ij = beta;
71  for (SizeType l = 0; l < ij.row(); ++l) {
72  auto il = LocalTileIndex{ij.row(), l};
73  auto lj = LocalTileIndex{l, ij.col()};
74  gemmN<B>(alpha, mat_a.read(il), mat_b.read(lj), beta_ij, mat_c.readwrite(ij));
75  beta_ij = 1;
76  }
77 
78  {
79  auto ii = LocalTileIndex{ij.row(), ij.row()};
80  hemm<B>(alpha, mat_a.read(ii), mat_b.read(ij), beta_ij, mat_c.readwrite(ij));
81  beta_ij = 1;
82  }
83 
84  for (SizeType l = ij.row() + 1; l < k; ++l) {
85  auto li = LocalTileIndex{l, ij.row()};
86  auto lj = LocalTileIndex{l, ij.col()};
87  gemmC<B>(alpha, mat_a.read(li), mat_b.read(lj), beta_ij, mat_c.readwrite(ij));
88  beta_ij = 1;
89  }
90  }
91 }

93 template <Backend B, Device D, class T>
94 void Hermitian<B, D, T>::call_LL(comm::CommunicatorGrid& grid, const T alpha, Matrix<const T, D>& mat_a,
95  Matrix<const T, D>& mat_b, const T beta, Matrix<T, D>& mat_c) {
96  using namespace hermitian_ll;
97  namespace ex = pika::execution::experimental;
98  using pika::execution::thread_priority;
99 
100  const comm::Index2D this_rank = grid.rank();
101 
102  const matrix::Distribution& distr_a = mat_a.distribution();
103  const matrix::Distribution& distr_b = mat_b.distribution();
104  const matrix::Distribution& distr_c = mat_c.distribution();
105 
106  if (mat_b.size().isEmpty())
107  return;
108 
109  auto mpi_row_task_chain = grid.row_communicator_pipeline();
110  auto mpi_col_task_chain = grid.col_communicator_pipeline();
111 
112  constexpr std::size_t n_workspaces = 2;
113  common::RoundRobin<matrix::Panel<Coord::Col, T, D>> a_panels(n_workspaces, distr_a);
114  common::RoundRobin<matrix::Panel<Coord::Row, T, D>> b_panels(n_workspaces, distr_b);
115  common::RoundRobin<matrix::Panel<Coord::Row, T, D>> c_panels(n_workspaces, distr_c);
116 
117  const SizeType m_loc = distr_c.localNrTiles().rows();
118  const SizeType n_loc = distr_c.localNrTiles().cols();
119  const SizeType k = distr_a.nrTiles().cols();
120  T beta_ = beta;
121 
122  for (SizeType l = 0; l < k; ++l) {
123  const GlobalTileIndex ll{l, l};
124  const LocalTileIndex ll_offset{distr_a.nextLocalTileFromGlobalTile<Coord::Row>(ll.row()),
125  distr_a.nextLocalTileFromGlobalTile<Coord::Col>(ll.col())};
126 
127  const SizeType l_loc = distr_c.nextLocalTileFromGlobalTile<Coord::Row>(ll.row());
128  const SizeType l1_loc = distr_c.nextLocalTileFromGlobalTile<Coord::Row>(ll.row() + 1);
129  const LocalTileIndex diag_offset(l_loc, 0);
130  const LocalTileIndex diag_end_offset(l1_loc, n_loc);
131  const LocalTileIndex lower_offset(l1_loc, 0);
132  const LocalTileIndex lower_end_offset(m_loc, n_loc);
133 
134  auto& a_panel = a_panels.nextResource();
135  auto& b_panel = b_panels.nextResource();
136  auto& c_panel = c_panels.nextResource();
137 
138  a_panel.setRangeStart(ll);
139 
140  if (l == mat_a.nrTiles().rows() - 1) {
141  a_panel.setWidth(mat_a.tileSize(ll).cols());
142  b_panel.setHeight(mat_a.tileSize(ll).cols());
143  c_panel.setHeight(mat_a.tileSize(ll).cols());
144  }
145 
146  const auto rank_ll = distr_a.rankGlobalTile(ll);
147  if (this_rank.col() == rank_ll.col()) {
148  for (SizeType i_loc = ll_offset.row(); i_loc < distr_a.localNrTiles().rows(); ++i_loc) {
149  const LocalTileIndex il{i_loc, ll_offset.col()};
150  a_panel.setTile(il, mat_a.read(il));
151  }
152  }
153  comm::broadcast(rank_ll.col(), a_panel, mpi_row_task_chain);
154 
155  if (this_rank.row() == rank_ll.row()) {
156  for (SizeType j_loc = 0; j_loc < distr_b.localNrTiles().cols(); ++j_loc) {
157  const LocalTileIndex lj{diag_offset.row(), j_loc};
158  b_panel.setTile(lj, mat_b.read(lj));
159  }
160  }
161  comm::broadcast(rank_ll.row(), b_panel, mpi_col_task_chain);
162 
163  for (const auto ij : common::iterate_range2d(diag_offset, diag_end_offset)) {
164  hemm<B>(alpha, a_panel.read(ij), b_panel.read(ij), beta_, mat_c.readwrite(ij));
165  }
166 
167  if (l < k - 1) {
168  matrix::util::set0<B>(thread_priority::normal, c_panel);
169  // Note: As A is square, B and C have the same size and blocksize
170  for (const auto ij : common::iterate_range2d(lower_offset, lower_end_offset)) {
171  // No Transpose part
172  // C_ij += A_ik * B_kj
173  gemmN<B>(alpha, a_panel.read(ij), b_panel.read(ij), beta_, mat_c.readwrite(ij));
174 
175  // ConjTranspose part
176  // C_kj += (A_ik)^T * B_ij
177  gemmC<B>(alpha, a_panel.read(ij), mat_b.read(ij), T{1}, c_panel.readwrite(ij));
178  }
179 
180  if (grid.colCommunicator().size() != 1) {
181  for (const auto& idx : c_panel.iteratorLocal()) {
182  if (this_rank.row() == rank_ll.row()) {
183  ex::start_detached(comm::schedule_reduce_recv_in_place(mpi_col_task_chain.exclusive(),
184  MPI_SUM, c_panel.readwrite(idx)));
185  }
186  else {
187  ex::start_detached(comm::schedule_reduce_send(mpi_col_task_chain.exclusive(), rank_ll.row(),
188  MPI_SUM, c_panel.read(idx)));
189  }
190  }
191  }
192  for (const auto lj : common::iterate_range2d(diag_offset, diag_end_offset)) {
193  pika::execution::experimental::start_detached(
194  dlaf::internal::whenAllLift(T{1}, c_panel.read(lj), mat_c.readwrite(lj)) |
195  tile::add(dlaf::internal::Policy<B>(thread_priority::high)));
196  }
197  }
198 
199  // First iteration scales all the C tiles.
200  beta_ = T{1};
201 
202  a_panel.reset();
203  b_panel.reset();
204  c_panel.reset();
205  }
206 }
