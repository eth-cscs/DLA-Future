68template <Backend B, Device D, class T>
69void Hermitian<B, D, T>::call_LL(const T alpha, Matrix<const T, D>& mat_a, Matrix<const T, D>& mat_b,
70 const T beta, Matrix<T, D>& mat_c) {
71 using namespace hermitian_ll;
72 const SizeType k = mat_a.distribution().localNrTiles().cols();
73
74 for (const auto ij : common::iterate_range2d(mat_c.distribution().localNrTiles())) {
75 T beta_ij = beta;
76 for (SizeType l = 0; l < ij.row(); ++l) {
77 auto il = LocalTileIndex{ij.row(), l};
78 auto lj = LocalTileIndex{l, ij.col()};
79 gemmN<B>(alpha, mat_a.read(il), mat_b.read(lj), beta_ij, mat_c.readwrite(ij));
80 beta_ij = 1;
81 }
82
83 {
84 auto ii = LocalTileIndex{ij.row(), ij.row()};
85 hemm<B>(alpha, mat_a.read(ii), mat_b.read(ij), beta_ij, mat_c.readwrite(ij));
86 beta_ij = 1;
87 }
88
89 for (SizeType l = ij.row() + 1; l < k; ++l) {
90 auto li = LocalTileIndex{l, ij.row()};
91 auto lj = LocalTileIndex{l, ij.col()};
92 gemmC<B>(alpha, mat_a.read(li), mat_b.read(lj), beta_ij, mat_c.readwrite(ij));
93 beta_ij = 1;
94 }
95 }

98template <Backend B, Device D, class T>
99void Hermitian<B, D, T>::call_LL(comm::CommunicatorGrid& grid, const T alpha, Matrix<const T, D>& mat_a,
100 MatrixRef<const T, D>& mat_b, const T beta, Matrix<T, D>& mat_c) {
101 using namespace hermitian_ll;
102 namespace ex = pika::execution::experimental;
103 using pika::execution::thread_priority;
104
105 const comm::Index2D this_rank = grid.rank();
106
107 const matrix::Distribution& distr_a = mat_a.distribution();
108 const matrix::Distribution& distr_b = mat_b.distribution();
109 const matrix::Distribution& distr_c = mat_c.distribution();
110
111 if (mat_b.size().isEmpty())
112 return;
113
114 auto mpi_row_task_chain = grid.row_communicator_pipeline();
115 auto mpi_col_task_chain = grid.col_communicator_pipeline();
116
117 constexpr std::size_t n_workspaces = 2;
118 common::RoundRobin<matrix::Panel<Coord::Col, T, D>> a_panels(n_workspaces, distr_a);
119 common::RoundRobin<matrix::Panel<Coord::Row, T, D>> b_panels(n_workspaces, distr_b);
120 common::RoundRobin<matrix::Panel<Coord::Row, T, D>> c_panels(n_workspaces, distr_c);
121
122 const SizeType m_loc = distr_c.localNrTiles().rows();
123 const SizeType n_loc = distr_c.localNrTiles().cols();
124 const SizeType k = distr_a.nrTiles().cols();
125 T beta_ = beta;
126
127 for (SizeType l = 0; l < k; ++l) {
128 const GlobalTileIndex ll{l, l};
129 const LocalTileIndex ll_offset{distr_a.nextLocalTileFromGlobalTile<Coord::Row>(ll.row()),
130 distr_a.nextLocalTileFromGlobalTile<Coord::Col>(ll.col())};
131
132 const SizeType l_loc = distr_c.nextLocalTileFromGlobalTile<Coord::Row>(ll.row());
133 const SizeType l1_loc = distr_c.nextLocalTileFromGlobalTile<Coord::Row>(ll.row() + 1);
134 const LocalTileIndex diag_offset(l_loc, 0);
135 const LocalTileIndex diag_end_offset(l1_loc, n_loc);
136 const LocalTileIndex lower_offset(l1_loc, 0);
137 const LocalTileIndex lower_end_offset(m_loc, n_loc);
138
139 auto& a_panel = a_panels.nextResource();
140 auto& b_panel = b_panels.nextResource();
141 auto& c_panel = c_panels.nextResource();
142
143 a_panel.setRangeStart(ll);
144
145 if (l == mat_a.nrTiles().rows() - 1) {
146 a_panel.setWidth(mat_a.tileSize(ll).cols());
147 b_panel.setHeight(mat_a.tileSize(ll).cols());
148 c_panel.setHeight(mat_a.tileSize(ll).cols());
149 }
150
151 const auto rank_ll = distr_a.rankGlobalTile(ll);
152 if (this_rank.col() == rank_ll.col()) {
153 for (SizeType i_loc = ll_offset.row(); i_loc < distr_a.localNrTiles().rows(); ++i_loc) {
154 const LocalTileIndex il{i_loc, ll_offset.col()};
155 a_panel.setTile(il, mat_a.read(il));
156 }
157 }
158 comm::broadcast(rank_ll.col(), a_panel, mpi_row_task_chain);
159
160 if (this_rank.row() == rank_ll.row()) {
161 for (SizeType j_loc = 0; j_loc < distr_b.localNrTiles().cols(); ++j_loc) {
162 const LocalTileIndex lj{diag_offset.row(), j_loc};
163 b_panel.setTile(lj, mat_b.read(lj));
164 }
165 }
166 comm::broadcast(rank_ll.row(), b_panel, mpi_col_task_chain);
167
168 for (const auto ij : common::iterate_range2d(diag_offset, diag_end_offset)) {
169 hemm<B>(alpha, a_panel.read(ij), b_panel.read(ij), beta_, mat_c.readwrite(ij));
170 }
171
172 if (l < k - 1) {
173 matrix::util::set0<B>(thread_priority::normal, c_panel);
174 // Note: As A is square, B and C have the same size and blocksize
175 for (const auto ij : common::iterate_range2d(lower_offset, lower_end_offset)) {
176 // No Transpose part
177 // C_ij += A_ik * B_kj
178 gemmN<B>(alpha, a_panel.read(ij), b_panel.read(ij), beta_, mat_c.readwrite(ij));
179
180 // ConjTranspose part
181 // C_kj += (A_ik)^T * B_ij
182 gemmC<B>(alpha, a_panel.read(ij), mat_b.read(ij), T{1}, c_panel.readwrite(ij));
183 }
184
185 if (grid.colCommunicator().size() != 1) {
186 for (const auto& idx : c_panel.iteratorLocal()) {
187 if (this_rank.row() == rank_ll.row()) {
188 ex::start_detached(comm::schedule_reduce_recv_in_place(mpi_col_task_chain.exclusive(),
189 MPI_SUM, c_panel.readwrite(idx)));
190 }
191 else {
192 ex::start_detached(comm::schedule_reduce_send(mpi_col_task_chain.exclusive(), rank_ll.row(),
193 MPI_SUM, c_panel.read(idx)));
194 }
195 }
196 }
197 for (const auto lj : common::iterate_range2d(diag_offset, diag_end_offset)) {
198 pika::execution::experimental::start_detached(
199 dlaf::internal::whenAllLift(T{1}, c_panel.read(lj), mat_c.readwrite(lj)) |
200 tile::add(dlaf::internal::Policy<B>(thread_priority::high)));
201 }
202 }
203
204 // First iteration scales all the C tiles.
205 beta_ = T{1};
206
207 a_panel.reset();
208 b_panel.reset();
209 c_panel.reset();
210 }
211}
