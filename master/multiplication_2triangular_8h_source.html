39 template <Backend backend, Device device, class T>
40 void triangular(blas::Side side, blas::Uplo uplo, blas::Op op, blas::Diag diag, T alpha,
41  Matrix<const T, device>& mat_a, Matrix<T, device>& mat_b) {
42  DLAF_ASSERT(matrix::square_size(mat_a), mat_a);
43  DLAF_ASSERT(matrix::square_blocksize(mat_a), mat_a);
44  DLAF_ASSERT(matrix::local_matrix(mat_a), mat_a);
45  DLAF_ASSERT(matrix::local_matrix(mat_b), mat_b);
46 
47  if (side == blas::Side::Left) {
48  DLAF_ASSERT(matrix::multipliable(mat_a, mat_b, mat_b, op, blas::Op::NoTrans), mat_a, mat_b, op);
49 
50  if (uplo == blas::Uplo::Lower) {
51  if (op == blas::Op::NoTrans) {
52  internal::Triangular<backend, device, T>::call_LLN(diag, alpha, mat_a, mat_b);
53  }
54  else {
55  internal::Triangular<backend, device, T>::call_LLT(op, diag, alpha, mat_a, mat_b);
56  }
57  }
58  else {
59  if (op == blas::Op::NoTrans) {
60  internal::Triangular<backend, device, T>::call_LUN(diag, alpha, mat_a, mat_b);
61  }
62  else {
63  internal::Triangular<backend, device, T>::call_LUT(op, diag, alpha, mat_a, mat_b);
64  }
65  }
66  }
67  else {
68  DLAF_ASSERT(matrix::multipliable(mat_b, mat_a, mat_b, blas::Op::NoTrans, op), mat_a, mat_b, op);
69 
70  if (uplo == blas::Uplo::Lower) {
71  if (op == blas::Op::NoTrans) {
72  internal::Triangular<backend, device, T>::call_RLN(diag, alpha, mat_a, mat_b);
73  }
74  else {
75  internal::Triangular<backend, device, T>::call_RLT(op, diag, alpha, mat_a, mat_b);
76  }
77  }
78  else {
79  if (op == blas::Op::NoTrans) {
80  internal::Triangular<backend, device, T>::call_RUN(diag, alpha, mat_a, mat_b);
81  }
82  else {
83  internal::Triangular<backend, device, T>::call_RUT(op, diag, alpha, mat_a, mat_b);
84  }
85  }
86  }
87 }

106 template <Backend backend, Device device, class T>
107 void triangular(comm::CommunicatorGrid grid, blas::Side side, blas::Uplo uplo, blas::Op op,
108  blas::Diag diag, T alpha, Matrix<const T, device>& mat_a, Matrix<T, device>& mat_b) {
109  DLAF_ASSERT(matrix::square_size(mat_a), mat_a);
110  DLAF_ASSERT(matrix::square_blocksize(mat_a), mat_a);
111  DLAF_ASSERT(matrix::equal_process_grid(mat_a, grid), mat_a, grid);
112  DLAF_ASSERT(matrix::equal_process_grid(mat_b, grid), mat_b, grid);
113 
114  if (side == blas::Side::Left) {
115  DLAF_ASSERT(matrix::multipliable(mat_a, mat_b, mat_b, op, blas::Op::NoTrans), mat_a, mat_b, op);
116 
117  if (uplo == blas::Uplo::Lower) {
118  if (op == blas::Op::NoTrans) {
119  internal::Triangular<backend, device, T>::call_LLN(grid, diag, alpha, mat_a, mat_b);
120  }
121  else {
122  // Left Lower Trans/ConjTrans
123  DLAF_UNIMPLEMENTED(side, uplo, op, diag);
124  }
125  }
126  else {
127  if (op == blas::Op::NoTrans) {
128  internal::Triangular<backend, device, T>::call_LUN(grid, diag, alpha, mat_a, mat_b);
129  }
130  else {
131  // Left Upper Trans/ConjTrans
132  DLAF_UNIMPLEMENTED(side, uplo, op, diag);
133  }
134  }
135  }
136  else {
137  DLAF_ASSERT(matrix::multipliable(mat_b, mat_a, mat_b, blas::Op::NoTrans, op), mat_a, mat_b, op);
138 
139  if (uplo == blas::Uplo::Lower) {
140  if (op == blas::Op::NoTrans) {
141  internal::Triangular<backend, device, T>::call_RLN(grid, diag, alpha, mat_a, mat_b);
142  }
143  else {
144  // Right Lower Trans/ConjTrans
145  DLAF_UNIMPLEMENTED(side, uplo, op, diag);
146  }
147  }
148  else {
149  if (op == blas::Op::NoTrans) {
150  internal::Triangular<backend, device, T>::call_RUN(grid, diag, alpha, mat_a, mat_b);
151  }
152  else {
153  // Right Upper Trans/ConjTrans
154  DLAF_UNIMPLEMENTED(side, uplo, op, diag);
155  }
156  }
157  }
158 }
