34template <Backend backend, Device device, class T>
35dlaf::BaseType<T> max_norm(comm::CommunicatorGrid& grid, comm::Index2D rank, blas::Uplo uplo,
36 Matrix<const T, device>& A) {
37 using dlaf::matrix::equal_process_grid;
38 using dlaf::matrix::single_tile_per_block;
39
40 DLAF_ASSERT(equal_process_grid(A, grid), A, grid);
41 DLAF_ASSERT(single_tile_per_block(A), A);
42
43 // LAPACK documentation specify that if any dimension is 0, the result is 0
44 if (A.size().isEmpty())
45 return {0};
46
47 switch (uplo) {
48 case blas::Uplo::Lower:
49 return internal::Norm<backend, device, T>::max_L(grid, rank, A);
50 case blas::Uplo::Upper:
51 DLAF_UNIMPLEMENTED(uplo);
52 return {};
53 case blas::Uplo::General:
54 return internal::Norm<backend, device, T>::max_G(grid, rank, A);
55 default:
56 return {};
57 }
58}
