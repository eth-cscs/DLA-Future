40 template <Backend backend, Device device, class T>
41 dlaf::BaseType<T> norm(comm::CommunicatorGrid grid, comm::Index2D rank, lapack::Norm norm_type,
42  blas::Uplo uplo, Matrix<const T, device>& A) {
43  using dlaf::matrix::equal_process_grid;
44 
45  DLAF_ASSERT(equal_process_grid(A, grid), A, grid);
46 
47  // LAPACK documentation specify that if any dimension is 0, the result is 0
48  if (A.size().isEmpty())
49  return {0};
50 
51  switch (norm_type) {
52  case lapack::Norm::One:
53  case lapack::Norm::Two:
54  case lapack::Norm::Inf:
55  case lapack::Norm::Fro:
56  DLAF_UNIMPLEMENTED(norm_type);
57  return {};
58  case lapack::Norm::Max:
59  switch (uplo) {
60  case blas::Uplo::Lower:
61  return internal::Norm<backend, device, T>::max_L(grid, rank, A);
62  case blas::Uplo::Upper:
63  DLAF_UNIMPLEMENTED(norm_type, uplo);
64  return {};
65  case blas::Uplo::General:
66  return internal::Norm<backend, device, T>::max_G(grid, rank, A);
67  default:
68  return {};
69  }
70  default:
71  return {};
72  }
