316  TileElementSize tileSize(const LocalTileIndex& index) const {
317  // Transform to global panel index.
318  const auto panel_coord = dist_matrix_.globalTileFromLocalTile<coord>(index.get<coord>());
319  const GlobalTileIndex panel_index(coord, panel_coord);
320 
321  const bool is_first_global_tile = isFirstGlobalTile(index);
322  const auto size_coord = dist_matrix_.tileSize(panel_index).template get<coord>() -
323  (is_first_global_tile ? start_offset_ : 0);
324  const auto size_axis = dim_ < 0 ? dist_matrix_.blockSize().template get<axis>() : dim_;
325 
326  return {axis, size_axis, size_coord};

329  static LocalElementSize computePanelSize(LocalElementSize size, TileElementSize blocksize,
330  LocalTileIndex start) {
331  const auto mb = blocksize.rows();
332  const auto nb = blocksize.cols();
333 
334  const auto mat_size = size.get(coord);
335  const auto i_tile = start.get(coord);
336 
337  switch (axis) {
338  case Coord::Col:
339  return {mat_size - i_tile * mb, nb};
340  case Coord::Row:
341  return {mb, mat_size - i_tile * nb};
342  }
343  }

349  static Matrix<T, D> setupInternalMatrix(const Distribution& dist, const GlobalTileIndex start) {
350  constexpr auto CT = coord;
351 
352  const LocalTileIndex start_loc(CT, dist.template nextLocalTileFromGlobalTile<CT>(start.get(CT)));
353  const auto panel_size = computePanelSize(dist.localSize(), dist.blockSize(), start_loc);
354 
355  Distribution dist_internal{panel_size, dist.blockSize()};
356  auto layout = tileLayout(dist_internal);
357  return {std::move(dist_internal), layout};
