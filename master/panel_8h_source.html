348  static LocalElementSize computePanelSize(LocalElementSize size, TileElementSize blocksize,
349  LocalTileIndex start) {
350  const auto mb = blocksize.rows();
351  const auto nb = blocksize.cols();
352 
353  const auto mat_size = size.get(coord);
354  const auto i_tile = start.get(coord);
355 
356  switch (axis) {
357  case Coord::Col:
358  return {mat_size - i_tile * mb, nb};
359  case Coord::Row:
360  return {mb, mat_size - i_tile * nb};
361  }
362  }

368  static Matrix<T, D> setupInternalMatrix(const Distribution& dist, const GlobalTileIndex start) {
369  DLAF_ASSERT((start == GlobalTileIndex{0, 0} || dist.offset() == GlobalElementIndex{0, 0}),
370  "not supported.", start, dist.offset());
371 
372  constexpr auto CT = coord;
373 
374  const LocalTileIndex start_loc(CT, dist.template nextLocalTileFromGlobalTile<CT>(start.get(CT)));
375 
376  auto panel_size = computePanelSize(dist.localSize(), dist.blockSize(), start_loc);
377 
378  // Note:
379  // Account for the offset during allocation, otherwise tiles would be "shifted" wrt to the
380  // matrix which this panel is "bound" to.
381  const bool isFirst = dist.source_rank_index().get<CT>() == dist.rank_index().get<CT>();
382  if (isFirst)
383  panel_size =
384  LocalElementSize(CT, panel_size.get<CT>() + dist.offset().get<CT>(), panel_size.get<axis>());
385 
386  Distribution dist_internal{panel_size, dist.blockSize()};
387  auto layout = tileLayout(dist_internal);
388  return {std::move(dist_internal), layout};
389  }
