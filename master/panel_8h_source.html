338 TileElementSize tileSize(const LocalTileIndex& index) const {
339 // Transform to global panel index.
340 const auto panel_coord = dist_matrix_.globalTileFromLocalTile<coord>(index.get<coord>());
341 const GlobalTileIndex panel_index(coord, panel_coord);
342
343 const bool is_first_global_tile = isFirstGlobalTile(index);
344 const auto size_coord = dist_matrix_.tileSize(panel_index).template get<coord>() -
345 (is_first_global_tile ? start_offset_ : 0);
346 const auto size_axis = dim_ < 0 ? dist_matrix_.blockSize().template get<axis>() : dim_;
347
348 return {axis, size_axis, size_coord};
349 }

351 static LocalElementSize computePanelSize(LocalElementSize size, TileElementSize blocksize,
352 LocalTileIndex start) {
353 const auto mb = blocksize.rows();
354 const auto nb = blocksize.cols();
355
356 const auto mat_size = size.get(coord);
357 const auto i_tile = start.get(coord);
358
359 switch (axis) {
360 case Coord::Col:
361 return {mat_size - i_tile * mb, nb};
362 case Coord::Row:
363 return {mb, mat_size - i_tile * nb};
364 default:
365 return DLAF_UNREACHABLE(LocalElementSize);
366 }
367 }

373 static Matrix<T, D> setupInternalMatrix(const Distribution& dist, const GlobalTileIndex start) {
374 DLAF_ASSERT((start == GlobalTileIndex{0, 0} || dist.offset() == GlobalElementIndex{0, 0}),
375 "not supported.", start, dist.offset());
376
377 constexpr auto CT = coord;
378
379 const LocalTileIndex start_loc(CT, dist.template nextLocalTileFromGlobalTile<CT>(start.get(CT)));
380
381 auto panel_size = computePanelSize(dist.localSize(), dist.blockSize(), start_loc);
382
383 // Note:
384 // Account for the offset during allocation, otherwise tiles would be "shifted" wrt to the
385 // matrix which this panel is "bound" to.
386 const bool isFirst = dist.source_rank_index().get<CT>() == dist.rank_index().get<CT>();
387 if (isFirst)
388 panel_size =
389 LocalElementSize(CT, panel_size.get<CT>() + dist.offset().get<CT>(), panel_size.get<axis>());
390
391 Distribution dist_internal{panel_size, dist.blockSize()};
392 auto layout = tileLayout(dist_internal);
393 return {std::move(dist_internal), layout};
394 }
