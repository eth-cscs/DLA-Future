43 // Applies the permutation index `perm_arr` to a portion of the columns/rows(depends on coord) [1] of an
44 // input submatrix [2] and saves the result into a subregion [3] of an output submatrix [4].
45 //
46 // Example column permutations with `perm_arr = [8, 2, 5]`:
47 //
48 // 2 5 8 out_begin
49 // ┌────────────────────┐ ┌─────┬─────────────┐
50 // │ in_offset in │ │ │ out │
51 // │ │ │ │ └─►┌─┬─┬─┐ │
52 // │◄┘ ┌─┐ ┌─┐ ┌─┐ │ │ │c│a│b│ │
53 // │ │a│ │b│ │c│ │ │ ┌──►│ │ │ │ │
54 // │ │ │ ┌►│ │ │ │◄─┼────┼──┐ │ └─┴─┴─┘ │
55 // │ └─┘ │ └─┘ └─┘ │ │ │ │ ▲ │
56 // │ │ │ │sz.rows() │ │
57 // │ sz.rows() │ │ └─ │
58 // │ │ │ sz.cols()│
59 // └────────────────────┘ └───────────────────┘
60 //
61 // Example row permutations with `perm_arr = [3, 1]`:
62 //
63 // ┌─── in_offset
64 // │ out_begin
65 // ┌───────▼────────────┐ ┌──┬────────────────┐
66 // │ in │ │ │ out │
67 // │ ┌─────┐ │ │ └►┌─────┐ │
68 // 1 │ │ a │ │ │ │ b │ │
69 // │ └─────┘ │ │ ├─────┤ ◄─┐ │
70 // │ │ │ │ a │ │ │
71 // │ ┌─────┐ │ │ └──▲──┘ │ │
72 // 3 │ │ b │ │ │ │ │ │
73 // │ └──▲──┘ │ │ sz.cols() │ │
74 // │ │ │ │ sz.rows()│
75 // └──────────┼─────────┘ └───────────────────┘
76 // │
77 // sz.cols()
78 //
79 //
80 // [1]: The portion of each input column or row is defined by the interval [in_offset, in_offset +
81 // sz.col()) or the interval [in_offset, in_offset + sz.row()) respectively.
82 // [2]: The input submatrix is defined by `begin_tiles`, `ld_tiles`, `distr` and `in_tiles`
83 // [3]: The subregion is defined by `begin` and `sz`
84 // [4]: The output submatrix is defined by `begin_tiles`, `ld_tiles`, `distr` and `out_tiles`
85 template <class T, Device D, Coord coord, class... Args>
86 void applyPermutations(
87  const GlobalElementIndex out_begin, const GlobalElementSize sz, const SizeType in_offset,
88  const matrix::Distribution& distr, const SizeType* perm_arr,
89  const std::vector<matrix::internal::TileAsyncRwMutexReadOnlyWrapper<T, D>>& in_tiles,
90  const std::vector<matrix::Tile<T, D>>& out_tiles, [[maybe_unused]] Args&&... args) {
91  if constexpr (D == Device::CPU) {
92  constexpr Coord orth_coord = orthogonal(coord);
93  std::vector<SizeType> splits =
94  util::interleaveSplits(sz.get<orth_coord>(), distr.blockSize().get<orth_coord>(),
95  distr.distanceToAdjacentTile<orth_coord>(in_offset),
96  distr.distanceToAdjacentTile<orth_coord>(out_begin.get<orth_coord>()));
97 
98  // Parallelized over the number of permuted columns or rows
99  const SizeType nperms = sz.get<coord>();
100  pika::for_loop(pika::execution::par, to_sizet(0), to_sizet(nperms), [&](SizeType i_perm) {
101  DLAF_ASSERT_HEAVY(i_perm >= 0 && i_perm < nperms, i_perm, nperms);
102  DLAF_ASSERT_HEAVY(perm_arr[i_perm] >= 0 && perm_arr[i_perm] < nperms, i_perm, nperms);
103 
104  for (std::size_t i_split = 0; i_split < splits.size() - 1; ++i_split) {
105  const SizeType split = splits[i_split];
106 
107  GlobalElementIndex i_split_gl_in(split + in_offset, perm_arr[i_perm]);
108  GlobalElementIndex i_split_gl_out(split + out_begin.get<orth_coord>(),
109  out_begin.get<coord>() + i_perm);
110  TileElementSize region(splits[i_split + 1] - split, 1);
111  if constexpr (coord == Coord::Row) {
112  region.transpose();
113  i_split_gl_in.transpose();
114  i_split_gl_out.transpose();
115  }
116 
117  const TileElementIndex i_subtile_in = distr.tileElementIndex(i_split_gl_in);
118  const auto& tile_in = in_tiles[to_sizet(distr.globalTileLinearIndex(i_split_gl_in))].get();
119  const TileElementIndex i_subtile_out = distr.tileElementIndex(i_split_gl_out);
120  auto& tile_out = out_tiles[to_sizet(distr.globalTileLinearIndex(i_split_gl_out))];
121 
122  dlaf::tile::lacpy<T>(region, i_subtile_in, tile_in, i_subtile_out, tile_out);
123  }
124  });
125  }
126  else if constexpr (D == Device::GPU) {
127 #if defined(DLAF_WITH_GPU)
128  applyPermutationsOnDevice<T, coord>(out_begin, sz, in_offset, distr, perm_arr, in_tiles, out_tiles,
129  args...);
130 #endif
131  }
132 }

134 // FilterFunc is a function with signature bool(*)(SizeType)
135 template <class T, Device D, Coord C, class FilterFunc>
136 void applyPermutationsFiltered(
137  const GlobalElementIndex out_begin, const GlobalElementSize sz, const SizeType in_offset,
138  const matrix::Distribution& subm_dist, const SizeType* perm_arr,
139  const std::vector<matrix::internal::TileAsyncRwMutexReadOnlyWrapper<T, D>>& in_tiles_fut,
140  const std::vector<matrix::Tile<T, D>>& out_tiles, FilterFunc&& filter) {
141  constexpr auto OC = orthogonal(C);
142  std::vector<SizeType> splits =
143  dlaf::util::interleaveSplits(sz.get<OC>(), subm_dist.blockSize().get<OC>(),
144  subm_dist.distanceToAdjacentTile<OC>(in_offset),
145  subm_dist.distanceToAdjacentTile<OC>(out_begin.get<OC>()));
146 
147  const SizeType nperms = subm_dist.size().get<C>();
148 
149  // Parallelized over the number of permutations
150  pika::for_loop(pika::execution::par, to_sizet(0), to_sizet(nperms), [&](SizeType i_perm) {
151  if (!filter(perm_arr[i_perm]))
152  return;
153 
154  for (std::size_t i_split = 0; i_split < splits.size() - 1; ++i_split) {
155  const SizeType split = splits[i_split];
156 
157  GlobalElementIndex i_split_gl_in(split + in_offset, perm_arr[i_perm]);
158  GlobalElementIndex i_split_gl_out(split + out_begin.get<OC>(), out_begin.get<C>() + i_perm);
159  TileElementSize region(splits[i_split + 1] - split, 1);
160 
161  if constexpr (C == Coord::Row) {
162  region.transpose();
163  i_split_gl_in.transpose();
164  i_split_gl_out.transpose();
165  }
166 
167  const TileElementIndex i_subtile_in = subm_dist.tileElementIndex(i_split_gl_in);
168  const auto& tile_in = in_tiles_fut[to_sizet(subm_dist.globalTileLinearIndex(i_split_gl_in))].get();
169  const TileElementIndex i_subtile_out = subm_dist.tileElementIndex(i_split_gl_out);
170  auto& tile_out = out_tiles[to_sizet(subm_dist.globalTileLinearIndex(i_split_gl_out))];
171 
172  dlaf::tile::lacpy<T>(region, i_subtile_in, tile_in, i_subtile_out, tile_out);
173  }
174  });
175 }

177 template <Backend B, Device D, class T, Coord C>
178 void Permutations<B, D, T, C>::call(const SizeType i_begin, const SizeType i_end,
179  Matrix<const SizeType, D>& perms, Matrix<const T, D>& mat_in,
180  Matrix<T, D>& mat_out) {
181  namespace ut = matrix::util;
182  namespace ex = pika::execution::experimental;
183 
184  if (i_begin == i_end)
185  return;
186 
187  const matrix::Distribution& distr = mat_in.distribution();
188  const SizeType m = distr.globalTileElementDistance<Coord::Row>(i_begin, i_end);
189  const SizeType n = distr.globalTileElementDistance<Coord::Col>(i_begin, i_end);
190  matrix::Distribution subm_distr(LocalElementSize(m, n), distr.blockSize());
191  const SizeType ntiles = i_end - i_begin;
192 
193  auto perms_range = common::iterate_range2d(LocalTileIndex(i_begin, 0), LocalTileSize(ntiles, 1));
194  auto mat_range =
195  common::iterate_range2d(LocalTileIndex(i_begin, i_begin), LocalTileSize(ntiles, ntiles));
196  auto sender = ex::when_all(ex::when_all_vector(matrix::selectRead(perms, std::move(perms_range))),
197  ex::when_all_vector(matrix::selectRead(mat_in, mat_range)),
198  ex::when_all_vector(matrix::select(mat_out, mat_range)));
199 
200  auto permute_fn = [subm_distr](const auto& index_tile_futs, const auto& mat_in_tiles,
201  const auto& mat_out_tiles, auto&&... ts) {
202  TileElementIndex zero(0, 0);
203  const SizeType* i_ptr = index_tile_futs[0].get().ptr(zero);
204  applyPermutations<T, D, C>(GlobalElementIndex(0, 0), subm_distr.size(), 0, subm_distr, i_ptr,
205  mat_in_tiles, mat_out_tiles, std::forward<decltype(ts)>(ts)...);
206  };
207  ex::start_detached(
208  dlaf::internal::transform(dlaf::internal::Policy<B>(), std::move(permute_fn), std::move(sender)));
209 }

245 template <class T, Device D, Coord C, class SendCountsSender, class RecvCountsSender>
246 void all2allData(common::Pipeline<comm::Communicator>& sub_task_chain, int nranks,
247  LocalElementSize sz_loc, SendCountsSender&& send_counts_sender,
248  Matrix<const T, D>& send_mat, RecvCountsSender&& recv_counts_sender,
249  Matrix<T, D>& recv_mat) {
250  namespace ex = pika::execution::experimental;
251 
252  using dlaf::common::DataDescriptor;
253 
254  const SizeType vec_size = sz_loc.get<orthogonal(C)>();
255  auto sendrecv_f = [vec_size](comm::Communicator& comm, std::vector<int> send_counts,
256  std::vector<int> send_displs,
257  const std::vector<matrix::internal::TileAsyncRwMutexReadOnlyWrapper<T, D>>&
258  send_tiles_fut,
259  std::vector<int> recv_counts, std::vector<int> recv_displs,
260  const std::vector<matrix::Tile<T, D>>& recv_tiles) {
261  // Note: both guaranteed to be column-major on allocation
262  const T* send_ptr = send_tiles_fut[0].get().ptr();
263  T* recv_ptr = recv_tiles[0].ptr();
264 
265  const SizeType send_ld = send_tiles_fut[0].get().ld();
266  const SizeType recv_ld = recv_tiles[0].ld();
267 
268  const SizeType send_perm_stride = (C == Coord::Col) ? send_ld : 1;
269  const SizeType recv_perm_stride = (C == Coord::Col) ? recv_ld : 1;
270 
271  // cumulative sum for computing rank data displacements in packed vectors
272  std::exclusive_scan(send_counts.begin(), send_counts.end(), send_displs.begin(), 0);
273  std::exclusive_scan(recv_counts.begin(), recv_counts.end(), recv_displs.begin(), 0);
274 
275  std::vector<ex::unique_any_sender<>> all_comms;
276  all_comms.reserve(to_sizet(comm.size() - 1) * 2);
277  const comm::IndexT_MPI rank = comm.rank();
278  for (comm::IndexT_MPI rank_partner = 0; rank_partner < comm.size(); ++rank_partner) {
279  if (rank == rank_partner)
280  continue;
281 
282  const auto rank_partner_index = to_sizet(rank_partner);
283 
284  if (send_counts[rank_partner_index])
285  all_comms.push_back(
286  ex::just() | dlaf::comm::internal::transformMPI([=](MPI_Request* req) {
287  const SizeType nperms = send_counts[rank_partner_index];
288  auto message = dlaf::comm::make_message(
289  DataDescriptor<const T>(send_ptr + send_displs[rank_partner_index] * send_perm_stride,
290  C == Coord::Col ? nperms : vec_size,
291  C == Coord::Col ? vec_size : nperms, send_ld));
292 
293  DLAF_MPI_CHECK_ERROR(MPI_Isend(message.data(), message.count(), message.mpi_type(),
294  rank_partner, 0, comm, req));
295  }));
296  if (recv_counts[rank_partner_index])
297  all_comms.push_back(
298  ex::just() | dlaf::comm::internal::transformMPI([=](MPI_Request* req) {
299  const SizeType nperms = recv_counts[rank_partner_index];
300  auto message = dlaf::comm::make_message(
301  DataDescriptor<T>(recv_ptr + recv_displs[rank_partner_index] * recv_perm_stride,
302  C == Coord::Col ? nperms : vec_size,
303  C == Coord::Col ? vec_size : nperms, recv_ld));
304 
305  DLAF_MPI_CHECK_ERROR(MPI_Irecv(message.data(), message.count(), message.mpi_type(),
306  rank_partner, 0, comm, req));
307  }));
308  }
309 
310  pika::this_thread::experimental::sync_wait(ex::when_all_vector(std::move(all_comms)));
311  };
312 
313  ex::when_all(sub_task_chain(), std::forward<SendCountsSender>(send_counts_sender),
314  ex::just(std::vector<int>(to_sizet(nranks))), whenAllReadOnlyTilesArray(send_mat),
315  std::forward<RecvCountsSender>(recv_counts_sender),
316  ex::just(std::vector<int>(to_sizet(nranks))), whenAllReadWriteTilesArray(recv_mat)) |
317  dlaf::internal::transformDetach(dlaf::internal::Policy<Backend::MC>(), sendrecv_f);
318 }

320 // @param nranks number of ranks
321 // @param offset_sub where the sub-matrix to permute starts in the global matrix
322 // @param loc2sub_index a column matrix that represents a map from local to sub indices
323 // @param packing_index a column matrix that represents a map from packed indices to local indices
324 //
325 // Note: the order of the packed rows or columns on the send side must match the expected order at
326 // unpacking on the receive side
327 template <Coord C, bool BackwardMapping = false>
328 auto initPackingIndex(comm::IndexT_MPI nranks, SizeType offset_sub, const matrix::Distribution& dist,
329  Matrix<const SizeType, Device::CPU>& loc2sub_index,
330  Matrix<SizeType, Device::CPU>& packing_index) {
331  namespace ex = pika::execution::experimental;
332  namespace di = dlaf::internal;
333 
334  auto counts_fn = [nranks, offset_sub, dist,
335  nperms = packing_index.size().rows()](const auto& loc2gl_index_tiles,
336  const auto& packing_index_tiles) {
337  const SizeType* loc2sub = loc2gl_index_tiles[0].get().ptr();
338  SizeType* out = packing_index_tiles[0].ptr();
339 
340  std::vector<int> counts(to_sizet(nranks));
341 
342  for (int rank = 0, rank_displacement = 0; rank < nranks; ++rank) {
343  int& nperms_local = counts[to_sizet(rank)] = 0;
344 
345  for (SizeType perm_index_local = 0; perm_index_local < nperms; ++perm_index_local) {
346  const SizeType perm_index_global = offset_sub + loc2sub[perm_index_local];
347  DLAF_ASSERT_HEAVY(perm_index_local >= 0 && perm_index_local < nperms, perm_index_local, nperms);
348  DLAF_ASSERT_HEAVY(perm_index_global >= 0 && perm_index_global < dist.size().get<C>(),
349  perm_index_global, dist.size());
350  if (dist.rankGlobalElement<C>(perm_index_global) == rank) {
351  const SizeType perm_index_packed = rank_displacement + nperms_local;
352 
353  if constexpr (BackwardMapping)
354  out[perm_index_packed] = perm_index_local;
355  else
356  out[perm_index_local] = perm_index_packed;
357 
358  ++nperms_local;
359  }
360  }
361 
362  // Note:
363  // This sorting actually "inverts" the mapping.
364  if constexpr (BackwardMapping)
365  std::sort(out + rank_displacement, out + rank_displacement + nperms_local,
366  [loc2sub](SizeType i1, SizeType i2) { return loc2sub[i1] < loc2sub[i2]; });
367 
368  rank_displacement += nperms_local;
369  }
370 
371  return counts;
372  };
373 
374  return ex::ensure_started(
375  ex::when_all(whenAllReadOnlyTilesArray(loc2sub_index), whenAllReadWriteTilesArray(packing_index)) |
376  di::transform(di::Policy<Backend::MC>{}, std::move(counts_fn)));
377 }

414 template <class T, Coord C>
415 void permuteOnCPU(common::Pipeline<comm::Communicator>& sub_task_chain, SizeType i_begin, SizeType i_end,
416  Matrix<const SizeType, Device::CPU>& perms, Matrix<const T, Device::CPU>& mat_in,
417  Matrix<T, Device::CPU>& mat_out) {
418  constexpr Device D = Device::CPU;
419 
420  using namespace dlaf::matrix;
421 
422  namespace ex = pika::execution::experimental;
423  namespace di = dlaf::internal;
424 
425  if (i_begin == i_end)
426  return;
427 
428  const Distribution& dist = mat_in.distribution();
429  const comm::IndexT_MPI nranks = to_int(dist.commGridSize().get<C>());
430 
431  // Local size and index of subproblem [i_begin, i_end)
432  const SizeType offset_sub = dist.globalElementFromGlobalTileAndTileElement<C>(i_begin, 0);
433  const TileElementSize blk = dist.blockSize();
434 
435  const LocalTileIndex i_loc_begin{dist.nextLocalTileFromGlobalTile<Coord::Row>(i_begin),
436  dist.nextLocalTileFromGlobalTile<Coord::Col>(i_begin)};
437  const LocalTileIndex i_loc_end{dist.nextLocalTileFromGlobalTile<Coord::Row>(i_end),
438  dist.nextLocalTileFromGlobalTile<Coord::Col>(i_end)};
439  // Note: the local shape of the permutation region may not be square if the process grid is not square
440  const LocalElementSize sz_loc{dist.localElementDistanceFromGlobalTile<Coord::Row>(i_begin, i_end),
441  dist.localElementDistanceFromGlobalTile<Coord::Col>(i_begin, i_end)};
442 
443  // If there are no tiles in this rank, nothing to do here
444  if (sz_loc.isEmpty())
445  return;
446 
447  // Create a map from send indices to receive indices (inverse of perms)
448  Matrix<SizeType, D> inverse_perms(perms.distribution());
449  eigensolver::internal::invertIndex(i_begin, i_end, perms, inverse_perms);
450 
451  // Local distribution used for packing and unpacking
452  const Distribution subm_dist(sz_loc, blk);
453 
454  // Local single tile column matrices representing index maps used for packing and unpacking of
455  // communication data
456  const SizeType nvecs = sz_loc.get<C>();
457  const Distribution index_dist(LocalElementSize(nvecs, 1), TileElementSize(blk.get<C>(), 1));
458  Matrix<SizeType, D> local2global_index(index_dist);
459  Matrix<SizeType, D> packing_index(index_dist);
460  Matrix<SizeType, D> unpacking_index(index_dist);
461 
462  // Local matrices used for packing data for communication. Both matrices are in column-major order.
463  // The particular constructor is used on purpose to guarantee that columns are stored contiguosly,

499  const auto& recv_counts,

503  const size_t rank_index = to_sizet(rank);
504 
505  const SizeType* perm_arr = index_tile_futs[0].get().ptr();
506  const GlobalElementSize sz = subm_dist.size();

508  const int a = std::accumulate(send_counts.cbegin(), send_counts.cbegin() + rank, 0);
509  const int b = a + send_counts[rank_index];
510 
511  // Note:
512  // These are copied directly from mat_send, while unpacking permutation applies to indices on
513  // the receiver side. So, we have to "align" the unpacking permutation, by applying the offset
514  // existing between the send and recv side.
515  // This is due to the fact that send and recv buffers might be "unbalanced", e.g. rank1 sends 2
516  // and receive 1 with rank0, so resulting in a shift in indices between the two buffer sides,
517  // following previous example the local part would start at index (0-based) 2 in mat_send and
518  // at index 1 in mat_recv.
519  const int a_r = std::accumulate(recv_counts.cbegin(), recv_counts.cbegin() + rank, 0);
520  const SizeType offset = to_SizeType(a - a_r);
521  std::vector<SizeType> perm_offseted(perm_arr, perm_arr + subm_dist.size().get<C>());
522  std::transform(perm_offseted.begin(), perm_offseted.end(), perm_offseted.begin(),
523  [offset](const SizeType perm) { return perm + offset; });
524

527  mat_out_tiles,
528  [a, b](SizeType i_perm) { return i_perm >= a && i_perm < b; });
529  };

532  whenAllReadOnlyTilesArray(mat_send),
533  whenAllReadWriteTilesArray(i_loc_begin, i_loc_end, mat_out)) |
534  di::transformDetach(di::Policy<DefaultBackend_v<D>>(), std::move(unpack_local_f));

537  all2allData<T, D, C>(sub_task_chain, nranks, sz_loc, send_counts_sender, mat_send, recv_counts_sender,

557  [b](SizeType i_perm) { return i_perm >= b; });
558  };
