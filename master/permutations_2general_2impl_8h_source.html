56// Applies the permutation index `perm_arr` to a portion of the columns/rows(depends on coord) [1] of an
57// input submatrix [2] and saves the result into a subregion [3] of an output submatrix [4].
58//
59// Example column permutations with `perm_arr = [8, 2, 5]`:
60//
61// 2 5 8 out_begin
62// ┌────────────────────┐ ┌─────┬─────────────┐
63// │ in_offset in │ │ │ out │
64// │ │ │ │ └─►┌─┬─┬─┐ │
65// │◄┘ ┌─┐ ┌─┐ ┌─┐ │ │ │c│a│b│ │
66// │ │a│ │b│ │c│ │ │ ┌──►│ │ │ │ │
67// │ │ │ ┌►│ │ │ │◄─┼────┼──┐ │ └─┴─┴─┘ │
68// │ └─┘ │ └─┘ └─┘ │ │ │ │ ▲ │
69// │ │ │ │sz.rows() │ │
70// │ sz.rows() │ │ └─ │
71// │ │ │ sz.cols()│
72// └────────────────────┘ └───────────────────┘
73//
74// Example row permutations with `perm_arr = [3, 1]`:
75//
76// ┌─── in_offset
77// │ out_begin
78// ┌───────▼────────────┐ ┌──┬────────────────┐
79// │ in │ │ │ out │
80// │ ┌─────┐ │ │ └►┌─────┐ │
81// 1 │ │ a │ │ │ │ b │ │
82// │ └─────┘ │ │ ├─────┤ ◄─┐ │
83// │ │ │ │ a │ │ │
84// │ ┌─────┐ │ │ └──▲──┘ │ │
85// 3 │ │ b │ │ │ │ │ │
86// │ └──▲──┘ │ │ sz.cols() │ │
87// │ │ │ │ sz.rows()│
88// └──────────┼─────────┘ └───────────────────┘
89// │
90// sz.cols()
91//
92//
93// [1]: The portion of each input column or row is defined by the interval [in_offset, in_offset +
94// sz.col()) or the interval [in_offset, in_offset + sz.row()) respectively.
95// [2]: The input submatrix is defined by `begin_tiles`, `ld_tiles`, `distr` and `in_tiles`
96// [3]: The subregion is defined by `begin` and `sz`
97// [4]: The output submatrix is defined by `begin_tiles`, `ld_tiles`, `distr` and `out_tiles`
98template <class T, Coord C>
99void applyPermutationOnCPU(
100 const SizeType i_perm, const matrix::Distribution& subm_dist, const SizeType* perm_arr,
101 const std::vector<matrix::internal::TileAsyncRwMutexReadOnlyWrapper<T, Device::CPU>>& in_tiles_fut,
102 const std::vector<matrix::Tile<T, Device::CPU>>& out_tiles) {
103 namespace dist_extra = dlaf::matrix::internal::distribution;
104
105 constexpr auto OC = orthogonal(C);
106
107 const SizeType perm_from = perm_arr[i_perm];
108 const SizeType perm_to = i_perm;
109
110 const SizeType j_in_lc = subm_dist.local_tile_from_local_element<C>(perm_from);
111 const SizeType j_out_lc = subm_dist.local_tile_from_local_element<C>(perm_to);
112
113 const TileElementIndex ij_in_tl(C, subm_dist.tile_element_from_local_element<C>(perm_from), 0);
114 const TileElementIndex ij_out_tl(C, subm_dist.tile_element_from_local_element<C>(perm_to), 0);
115
116 for (SizeType i_lc = 0; i_lc < subm_dist.local_nr_tiles().get<OC>(); ++i_lc) {
117 const LocalTileIndex ij_in(OC, i_lc, j_in_lc);
118 const LocalTileIndex ij_out(OC, i_lc, j_out_lc);
119
120 const SizeType linear_in = dist_extra::local_tile_linear_index(subm_dist, ij_in);
121 const auto& tile_in = in_tiles_fut[to_sizet(linear_in)].get();
122
123 const SizeType linear_out = dist_extra::local_tile_linear_index(subm_dist, ij_out);
124 auto& tile_out = out_tiles[to_sizet(linear_out)];
125
126 DLAF_ASSERT_HEAVY(tile_in.size().template get<OC>() == tile_out.size().template get<OC>(),
127 tile_in.size(), tile_out.size());
128 const TileElementSize region(OC, tile_in.size().template get<OC>(), 1);
129
130 dlaf::tile::lacpy<T>(region, ij_in_tl, tile_in, ij_out_tl, tile_out);
131 }
132}

134template <Backend B, Device D, class T, Coord C>
135void Permutations<B, D, T, C>::call(const SizeType i_begin, const SizeType i_end,
136 Matrix<const SizeType, D>& perms, Matrix<const T, D>& mat_in,
137 Matrix<T, D>& mat_out) {
138 namespace ex = pika::execution::experimental;
139 namespace dist_extra = dlaf::matrix::internal::distribution;
140 using dist_extra::local_element_distance_from_global_tile;
141
142 if (i_begin == i_end)
143 return;
144
145 const matrix::Distribution& dist = mat_in.distribution();
146 const SizeType m = local_element_distance_from_global_tile<Coord::Row>(dist, i_begin, i_end);
147 const SizeType n = local_element_distance_from_global_tile<Coord::Col>(dist, i_begin, i_end);
148 const matrix::Distribution subm_dist(LocalElementSize(m, n), dist.tile_size());
149
150 if (subm_dist.size().isEmpty())
151 return;
152
153 auto next_local_from_global = [&dist](const SizeType i) -> LocalTileIndex {
154 return {dist.next_local_tile_from_global_tile<Coord::Row>(i),
155 dist.next_local_tile_from_global_tile<Coord::Col>(i)};
156 };
157 const LocalTileIndex idx_begin_lc = next_local_from_global(i_begin);
158 const LocalTileIndex idx_end_lc = next_local_from_global(i_end);
159
160 auto perms_range = common::iterate_range2d(LocalTileIndex(idx_begin_lc.get<C>(), 0),
161 LocalTileIndex(idx_end_lc.get<C>(), 1));
162 auto mat_range = common::iterate_range2d(idx_begin_lc, idx_end_lc);
163 auto sender = ex::when_all(ex::when_all_vector(matrix::selectRead(perms, std::move(perms_range))),
164 ex::when_all_vector(matrix::selectRead(mat_in, mat_range)),
165 ex::when_all_vector(matrix::select(mat_out, std::move(mat_range))));
166
167 if constexpr (D == Device::CPU) {
168 const SizeType nperms = subm_dist.size().get<C>();
169
170 auto permute_fn = [subm_dist](const auto i_perm, const auto& index_tile_futs,
171 const auto& mat_in_tiles, const auto& mat_out_tiles) {
172 const SizeType* perm_arr = index_tile_futs[0].get().ptr();
173
174 [[maybe_unused]] const SizeType nperms = subm_dist.size().get<C>();
175 DLAF_ASSERT_HEAVY(i_perm >= 0 && i_perm < nperms, i_perm, nperms);
176 DLAF_ASSERT_HEAVY(perm_arr[i_perm] >= 0 && perm_arr[i_perm] < nperms, perm_arr[i_perm], nperms);
177
178 applyPermutationOnCPU<T, C>(i_perm, subm_dist, perm_arr, mat_in_tiles, mat_out_tiles);
179 };
180
181 ex::start_detached(std::move(sender) | ex::transfer(dlaf::internal::getBackendScheduler<B>()) |
182 ex::bulk(nperms, std::move(permute_fn)));
183 }
184 else {
185#if defined(DLAF_WITH_GPU)
186 auto permute_fn = [subm_dist](const auto& index_tile_futs, const auto& mat_in_tiles,
187 const auto& mat_out_tiles, whip::stream_t stream) {
188 TileElementIndex zero(0, 0);
189 const SizeType* i_ptr = index_tile_futs[0].get().ptr(zero);
190
191 applyPermutationsOnDevice<T, C>(GlobalElementIndex(0, 0), subm_dist.size(), 0, subm_dist, i_ptr,
192 mat_in_tiles, mat_out_tiles, stream);
193 };
194
195 ex::start_detached(std::move(sender) |
196 dlaf::internal::transform(dlaf::internal::Policy<B>(), std::move(permute_fn)));
197#endif
198 }
199}

229template <class T, Device D, Coord C, class SendCountsSender, class RecvCountsSender>
230void all2allData(
231 comm::CommunicatorPipeline<comm::coord_to_communicator_type(orthogonal(C))>& sub_task_chain,
232 int nranks, LocalElementSize sz_loc, SendCountsSender&& send_counts_sender,
233 Matrix<const T, D>& send_mat, RecvCountsSender&& recv_counts_sender, Matrix<T, D>& recv_mat) {
234 namespace ex = pika::execution::experimental;
235
236 using dlaf::common::DataDescriptor;
237
238 const SizeType vec_size = sz_loc.get<orthogonal(C)>();
239 auto sendrecv_f = [vec_size, sub_sub_task_chain = sub_task_chain.sub_pipeline()](
240 std::vector<int> send_counts, std::vector<int> send_displs,
241 const std::vector<matrix::internal::TileAsyncRwMutexReadOnlyWrapper<T, D>>&
242 send_tiles_fut,
243 std::vector<int> recv_counts, std::vector<int> recv_displs,
244 const std::vector<matrix::Tile<T, D>>& recv_tiles) mutable {
245 // Note: both guaranteed to be column-major on allocation
246 const T* send_ptr = send_tiles_fut[0].get().ptr();
247 T* recv_ptr = recv_tiles[0].ptr();
248
249 const SizeType send_ld = send_tiles_fut[0].get().ld();
250 const SizeType recv_ld = recv_tiles[0].ld();
251
252 const SizeType send_perm_stride = (C == Coord::Col) ? send_ld : 1;
253 const SizeType recv_perm_stride = (C == Coord::Col) ? recv_ld : 1;
254
255 // cumulative sum for computing rank data displacements in packed vectors
256 std::exclusive_scan(send_counts.begin(), send_counts.end(), send_displs.begin(), 0);
257 std::exclusive_scan(recv_counts.begin(), recv_counts.end(), recv_displs.begin(), 0);
258
259 std::vector<ex::unique_any_sender<>> all_comms;
260 all_comms.reserve(to_sizet(sub_sub_task_chain.size() - 1) * 2);
261 const comm::IndexT_MPI rank = sub_sub_task_chain.rank();
262 for (comm::IndexT_MPI rank_partner = 0; rank_partner < sub_sub_task_chain.size(); ++rank_partner) {
263 if (rank == rank_partner)
264 continue;
265
266 const auto rank_partner_index = to_sizet(rank_partner);
267
268 if (send_counts[rank_partner_index])
269 all_comms.push_back(
270 sub_sub_task_chain.shared() |
271 dlaf::comm::internal::transformMPI([=](const comm::Communicator& comm, MPI_Request* req) {
272 const SizeType nperms = send_counts[rank_partner_index];
273 auto message = dlaf::comm::make_message(DataDescriptor<const T>(
274 send_ptr + send_displs[rank_partner_index] * send_perm_stride,
275 C == Coord::Col ? nperms : vec_size, C == Coord::Col ? vec_size : nperms, send_ld));
276
277 DLAF_MPI_CHECK_ERROR(MPI_Isend(message.data(), message.count(), message.mpi_type(),
278 rank_partner, 0, comm, req));
279 }));
280 if (recv_counts[rank_partner_index])
281 all_comms.push_back(
282 sub_sub_task_chain.shared() |
283 dlaf::comm::internal::transformMPI([=](const comm::Communicator& comm, MPI_Request* req) {
284 const SizeType nperms = recv_counts[rank_partner_index];
285 auto message = dlaf::comm::make_message(DataDescriptor<T>(
286 recv_ptr + recv_displs[rank_partner_index] * recv_perm_stride,
287 C == Coord::Col ? nperms : vec_size, C == Coord::Col ? vec_size : nperms, recv_ld));
288
289 DLAF_MPI_CHECK_ERROR(MPI_Irecv(message.data(), message.count(), message.mpi_type(),
290 rank_partner, 0, comm, req));
291 }));
292 }
293
294 pika::this_thread::experimental::sync_wait(ex::when_all_vector(std::move(all_comms)));
295 };
296
297 ex::when_all(std::forward<SendCountsSender>(send_counts_sender),
298 ex::just(std::vector<int>(to_sizet(nranks))), whenAllReadOnlyTilesArray(send_mat),
299 std::forward<RecvCountsSender>(recv_counts_sender),
300 ex::just(std::vector<int>(to_sizet(nranks))), whenAllReadWriteTilesArray(recv_mat)) |
301 dlaf::internal::transformDetach(dlaf::internal::Policy<Backend::MC>(), std::move(sendrecv_f));
302}

304// @param nranks number of ranks
305// @param offset_sub where the sub-matrix to permute starts in the global matrix
306// @param loc2sub_index a column matrix that represents a map from local to sub indices
307// @param packing_index a column matrix that represents a map from packed indices to local indices
308//
309// Note: the order of the packed rows or columns on the send side must match the expected order at
310// unpacking on the receive side
311template <Coord C, bool BackwardMapping = false>
312auto initPackingIndex(comm::IndexT_MPI nranks, SizeType offset_sub, const matrix::Distribution& dist,
313 Matrix<const SizeType, Device::CPU>& loc2sub_index,
314 Matrix<SizeType, Device::CPU>& packing_index) {
315 namespace ex = pika::execution::experimental;
316 namespace di = dlaf::internal;
317
318 auto counts_fn = [nranks, offset_sub, dist, nperms = packing_index.size().rows()](
319 const auto& loc2gl_index_tiles, const auto& packing_index_tiles) {
320 const SizeType* loc2sub = loc2gl_index_tiles[0].get().ptr();
321 SizeType* out = packing_index_tiles[0].ptr();
322
323 std::vector<int> counts(to_sizet(nranks));
324
325 for (int rank = 0, rank_displacement = 0; rank < nranks; ++rank) {
326 int& nperms_local = counts[to_sizet(rank)] = 0;
327
328 for (SizeType perm_index_local = 0; perm_index_local < nperms; ++perm_index_local) {
329 const SizeType perm_index_global = offset_sub + loc2sub[perm_index_local];
330 DLAF_ASSERT_HEAVY(perm_index_local >= 0 && perm_index_local < nperms, perm_index_local, nperms);
331 DLAF_ASSERT_HEAVY(perm_index_global >= 0 && perm_index_global < dist.size().get<C>(),
332 perm_index_global, dist.size());
333 if (dist.rankGlobalElement<C>(perm_index_global) == rank) {
334 const SizeType perm_index_packed = rank_displacement + nperms_local;
335
336 if constexpr (BackwardMapping)
337 out[perm_index_packed] = perm_index_local;
338 else
339 out[perm_index_local] = perm_index_packed;
340
341 ++nperms_local;
342 }
343 }
344
345 // Note:
346 // This sorting actually "inverts" the mapping.
347 if constexpr (BackwardMapping)
348 std::sort(out + rank_displacement, out + rank_displacement + nperms_local,
349 [loc2sub](SizeType i1, SizeType i2) { return loc2sub[i1] < loc2sub[i2]; });
350
351 rank_displacement += nperms_local;
352 }
353
354 return counts;
355 };
356
357 return ex::ensure_started(ex::when_all(whenAllReadOnlyTilesArray(loc2sub_index),
358 whenAllReadWriteTilesArray(packing_index)) |
359 di::transform(di::Policy<Backend::MC>{}, std::move(counts_fn)));
360}

377// @param index_map a column matrix that represents a map from local `out` to local `in` indices
378template <class T, Device D, Coord C, class IndexMapSender, class InSender, class OutSender>
379void applyPackingIndex(const matrix::Distribution& subm_dist, IndexMapSender&& index_map, InSender&& in,
380 OutSender&& out) {
381 namespace ex = pika::execution::experimental;
382 namespace di = dlaf::internal;
383
384 auto sender = ex::when_all(std::forward<IndexMapSender>(index_map), std::forward<InSender>(in),
385 std::forward<OutSender>(out));
386
387 if constexpr (D == Device::CPU) {
388 const SizeType nperms = subm_dist.size().get<C>();
389
390 auto permute_fn = [subm_dist](const auto i_perm, const auto& index_tile_futs,
391 const auto& mat_in_tiles, const auto& mat_out_tiles) {
392 const SizeType* perm_arr = index_tile_futs[0].get().ptr();
393
394 [[maybe_unused]] const SizeType nperms = subm_dist.size().get<C>();
395 DLAF_ASSERT_HEAVY(i_perm >= 0 && i_perm < nperms, i_perm, nperms);
396 DLAF_ASSERT_HEAVY(perm_arr[i_perm] >= 0 && perm_arr[i_perm] < nperms, perm_arr[i_perm], nperms);
397
398 applyPermutationOnCPU<T, C>(i_perm, subm_dist, perm_arr, mat_in_tiles, mat_out_tiles);
399 };
400
401 ex::start_detached(std::move(sender) | ex::transfer(di::getBackendScheduler<Backend::MC>()) |
402 ex::bulk(nperms, std::move(permute_fn)));
403 }
404 else {
405#if defined(DLAF_WITH_GPU)
406 auto permute_fn = [subm_dist](const auto& index_tile_futs, const auto& mat_in_tiles,
407 const auto& mat_out_tiles, whip::stream_t stream) {
408 TileElementIndex zero(0, 0);
409 const SizeType* i_ptr = index_tile_futs[0].get().ptr(zero);
410
411 applyPermutationsOnDevice<T, C>(GlobalElementIndex(0, 0), subm_dist.size(), 0, subm_dist, i_ptr,
412 mat_in_tiles, mat_out_tiles, stream);
413 };
414
415 ex::start_detached(std::move(sender) |
416 dlaf::internal::transform(dlaf::internal::Policy<Backend::GPU>(),
417 std::move(permute_fn)));
418#endif
419 }

422template <class T, Coord C, class SendCountsSender, class RecvCountsSender, class UnpackingIndexSender,
423 class MatSendSender, class MatOutSender>
424void unpackLocalOnCPU(const matrix::Distribution& subm_dist, const matrix::Distribution& dist,
425 SendCountsSender&& send_counts, RecvCountsSender&& recv_counts,
426 UnpackingIndexSender&& unpacking_index, MatSendSender&& mat_send,
427 MatOutSender&& mat_out) {
428 namespace ex = pika::execution::experimental;
429 namespace di = dlaf::internal;
430
431 auto setup_unpack_local_f = [subm_dist,
432 rank = dist.rankIndex().get<C>()](auto send_counts, auto recv_counts,
433 auto index_tile_futs, auto mat_in_tiles,
434 auto mat_out_tiles) {
435 const size_t rank_index = to_sizet(rank);
436
437 const SizeType* perm_arr = index_tile_futs[0].get().ptr();
438
439 const int a = std::accumulate(send_counts.cbegin(), send_counts.cbegin() + rank, 0);
440 const int b = a + send_counts[rank_index];
441
442 // Note:
443 // These are copied directly from mat_send, while unpacking permutation applies to indices on
444 // the receiver side. So, we have to "align" the unpacking permutation, by applying the offset
445 // existing between the send and recv side.
446 // This is due to the fact that send and recv buffers might be "unbalanced", e.g. rank1 sends 2
447 // and receive 1 with rank0, so resulting in a shift in indices between the two buffer sides,
448 // following previous example the local part would start at index (0-based) 2 in mat_send and
449 // at index 1 in mat_recv.
450 const int a_r = std::accumulate(recv_counts.cbegin(), recv_counts.cbegin() + rank, 0);
451 const SizeType offset = to_SizeType(a - a_r);
452 std::vector<SizeType> perm_offseted;
453 perm_offseted.reserve(to_sizet(subm_dist.size().get<C>()));
454 std::transform(perm_arr, perm_arr + subm_dist.size().get<C>(), std::back_inserter(perm_offseted),
455 [offset](const SizeType perm) { return perm + offset; });
456
457 return std::tuple(a, b, std::move(perm_offseted), std::move(mat_in_tiles), std::move(mat_out_tiles));
458 };
459
460 auto permutations_unpack_local_f = [subm_dist](const auto i_perm, const auto a, const auto b,
461 const auto& perm_offseted, const auto& mat_in_tiles,
462 const auto& mat_out_tiles) {
463 const SizeType* perm_arr = perm_offseted.data();
464
465 // [a, b)
466 if (a <= perm_arr[i_perm] && perm_arr[i_perm] < b) {
467 applyPermutationOnCPU<T, C>(i_perm, subm_dist, perm_arr, mat_in_tiles, mat_out_tiles);
468 }
469 };
470
471 ex::start_detached(
472 ex::when_all(std::forward<SendCountsSender>(send_counts),
473 std::forward<RecvCountsSender>(recv_counts),
474 std::forward<UnpackingIndexSender>(unpacking_index),
475 std::forward<MatSendSender>(mat_send), std::forward<MatOutSender>(mat_out)) |
476 di::transform(di::Policy<Backend::MC>(), std::move(setup_unpack_local_f)) | ex::unpack() |
477 ex::bulk(subm_dist.size().get<C>(), std::move(permutations_unpack_local_f)));
478}

480template <class T, Coord C, class RecvCountsSender, class UnpackingIndexSender, class MatRecvSender,
481 class MatOutSender>
482void unpackOthersOnCPU(const matrix::Distribution& subm_dist, const matrix::Distribution& dist,
483 RecvCountsSender&& recv_counts, UnpackingIndexSender&& unpacking_index,
484 MatRecvSender&& mat_recv, MatOutSender&& mat_out) {
485 namespace ex = pika::execution::experimental;
486 namespace di = dlaf::internal;
487
488 auto setup_unpack_f = [rank = dist.rankIndex().get<C>()](auto recv_counts, auto index_tile_futs,
489 auto mat_in_tiles, auto mat_out_tiles) {
490 const size_t rank_index = to_sizet(rank);
491 const int a = std::accumulate(recv_counts.cbegin(), recv_counts.cbegin() + rank, 0);
492 const int b = a + recv_counts[rank_index];
493
494 return std::tuple(a, b, std::move(index_tile_futs), std::move(mat_in_tiles),
495 std::move(mat_out_tiles));
496 };
497
498 auto permutations_unpack_f = [subm_dist](const auto i_perm, const auto a, const auto b,
499 const auto& index_tile_futs, const auto& mat_in_tiles,
500 const auto& mat_out_tiles) {
501 const SizeType* perm_arr = index_tile_futs[0].get().ptr();
502
503 // [0, a) and [b, end)
504 if (perm_arr[i_perm] < a || b <= perm_arr[i_perm]) {
505 applyPermutationOnCPU<T, C>(i_perm, subm_dist, perm_arr, mat_in_tiles, mat_out_tiles);
506 }
507 };
508
509 ex::start_detached(ex::when_all(std::forward<RecvCountsSender>(recv_counts),
510 std::forward<UnpackingIndexSender>(unpacking_index),
511 std::forward<MatRecvSender>(mat_recv),
512 std::forward<MatOutSender>(mat_out)) |
513 di::transform(di::Policy<Backend::MC>(), std::move(setup_unpack_f)) | ex::unpack() |
514 ex::bulk(subm_dist.size().get<C>(), std::move(permutations_unpack_f)));
515}

517template <class T, Coord C>
518void permuteOnCPU(
519 comm::CommunicatorPipeline<comm::coord_to_communicator_type(orthogonal(C))>& sub_task_chain,
520 SizeType i_begin, SizeType i_end, Matrix<const SizeType, Device::CPU>& perms,
521 Matrix<const T, Device::CPU>& mat_in, Matrix<T, Device::CPU>& mat_out) {
522 constexpr Device D = Device::CPU;
523
524 using namespace dlaf::matrix;
525
526 namespace ex = pika::execution::experimental;
527 namespace di = dlaf::internal;
528
529 if (i_begin == i_end)
530 return;
531
532 const Distribution& dist = mat_in.distribution();
533 const comm::IndexT_MPI nranks = to_int(dist.commGridSize().get<C>());
534
535 // Local size and index of subproblem [i_begin, i_end)
536 const SizeType offset_sub = dist.globalElementFromGlobalTileAndTileElement<C>(i_begin, 0);
537 const TileElementSize blk = dist.blockSize();
538
539 const LocalTileIndex i_loc_begin{dist.nextLocalTileFromGlobalTile<Coord::Row>(i_begin),
540 dist.nextLocalTileFromGlobalTile<Coord::Col>(i_begin)};
541 const LocalTileIndex i_loc_end{dist.nextLocalTileFromGlobalTile<Coord::Row>(i_end),
542 dist.nextLocalTileFromGlobalTile<Coord::Col>(i_end)};
543 // Note: the local shape of the permutation region may not be square if the process grid is not square
544 const LocalElementSize sz_loc{dist.localElementDistanceFromGlobalTile<Coord::Row>(i_begin, i_end),
545 dist.localElementDistanceFromGlobalTile<Coord::Col>(i_begin, i_end)};
546
547 // If there are no tiles in this rank, nothing to do here
548 if (sz_loc.isEmpty())
549 return;
550
551 // Create a map from send indices to receive indices (inverse of perms)
552 Matrix<SizeType, D> inverse_perms(perms.distribution());
553 eigensolver::internal::invertIndex(i_begin, i_end, perms, inverse_perms);
554
555 // Local distribution used for packing and unpacking
556 const Distribution subm_dist(sz_loc, blk);
557
558 // Local single tile column matrices representing index maps used for packing and unpacking of
559 // communication data
560 const SizeType nvecs = sz_loc.get<C>();
561 const Distribution index_dist(LocalElementSize(nvecs, 1), TileElementSize(blk.get<C>(), 1));
562 Matrix<SizeType, D> local2global_index(index_dist);
563 Matrix<SizeType, D> packing_index(index_dist);
564 Matrix<SizeType, D> unpacking_index(index_dist);
565
566 // Local matrices used for packing data for communication. Both matrices are in column-major order.
567 // The particular constructor is used on purpose to guarantee that columns are stored contiguosly,
568 // such that there is no padding and gaps between them.
569 const LocalElementSize comm_sz = sz_loc;
570 const Distribution comm_dist(comm_sz, blk);
571 const LayoutInfo comm_layout = matrix::colMajorLayout(comm_sz, blk, comm_sz.rows());
572
573 Matrix<T, D> mat_send(comm_dist, comm_layout);
574 Matrix<T, D> mat_recv(comm_dist, comm_layout);
575
576 // Initialize the unpacking index
577 copyLocalPartsFromGlobalIndex<D, C>(i_loc_begin.get<C>(), dist, perms, local2global_index);
578 auto recv_counts_sender =
579 initPackingIndex<C>(nranks, offset_sub, dist, local2global_index, unpacking_index) | ex::split();
580
581 // Initialize the packing index
582 // Here `true` is specified so that the send side matches the order of columns on the receive side
583 copyLocalPartsFromGlobalIndex<D, C>(i_loc_begin.get<C>(), dist, inverse_perms, local2global_index);
584 auto send_counts_sender =
585 initPackingIndex<C, true>(nranks, offset_sub, dist, local2global_index, packing_index) |
586 ex::split();
587
588 // Pack local rows or columns to be sent from this rank
589 applyPackingIndex<T, D, C>(subm_dist, whenAllReadOnlyTilesArray(packing_index),
590 whenAllReadOnlyTilesArray(i_loc_begin, i_loc_end, mat_in),
591 whenAllReadWriteTilesArray(mat_send));
592
593 // Unpacking
594 // separate unpacking:
595 // - locals
596 // - communicated
597 // and then start two different tasks:
598 // - the first depends on mat_send instead of mat_recv (no dependency on comm)
599 // - the last is the same, but it has to skip the part already done for local
600
601 // LOCAL
602 unpackLocalOnCPU<T, C>(subm_dist, dist, send_counts_sender, recv_counts_sender,
603 whenAllReadOnlyTilesArray(unpacking_index), whenAllReadOnlyTilesArray(mat_send),
604 whenAllReadWriteTilesArray(i_loc_begin, i_loc_end, mat_out));
605 // COMMUNICATION-dependent
606 all2allData<T, D, C>(sub_task_chain, nranks, sz_loc, send_counts_sender, mat_send, recv_counts_sender,
607 mat_recv);
608 // OTHERS
609 unpackOthersOnCPU<T, C>(subm_dist, dist, std::move(recv_counts_sender),
610 whenAllReadOnlyTilesArray(unpacking_index),
611 whenAllReadOnlyTilesArray(mat_recv),
612 whenAllReadWriteTilesArray(i_loc_begin, i_loc_end, mat_out));
613}
