50template <Backend B, Device D, class T, Coord coord>
51void permute(SizeType i_begin, SizeType i_end, Matrix<const SizeType, D>& perms,
52 Matrix<const T, D>& mat_in, Matrix<T, D>& mat_out) {
53 DLAF_ASSERT(matrix::local_matrix(perms), perms);
54 DLAF_ASSERT(matrix::local_matrix(mat_in), mat_in);
55 DLAF_ASSERT(matrix::local_matrix(mat_out), mat_out);
56
57 // Note:
58 // These are not implementation constraints, but more logic constraints. Indeed, these ensure that
59 // the range [i_begin, i_end] is square in terms of elements (it would not make sense to have it square
60 // in terms of number of tiles). Moreover, by requiring mat_in and mat_out matrices to have the same
61 // shape, it is ensured that range [i_begin, i_end] is actually the same on both sides.
62 DLAF_ASSERT(square_size(mat_in), mat_in);
63 DLAF_ASSERT(matrix::square_blocksize(mat_in), mat_in);
64 DLAF_ASSERT(matrix::equal_size(mat_in, mat_out), mat_in);
65 DLAF_ASSERT(matrix::equal_blocksize(mat_in, mat_out), mat_in);
66
67 DLAF_ASSERT(matrix::single_tile_per_block(perms), perms);
68 DLAF_ASSERT(matrix::single_tile_per_block(mat_in), mat_in);
69 DLAF_ASSERT(matrix::single_tile_per_block(mat_out), mat_out);
70
71 DLAF_ASSERT(perms.size().rows() == mat_in.size().rows(), perms, mat_in);
72 DLAF_ASSERT(perms.size().cols() == 1, perms);
73 DLAF_ASSERT(perms.blockSize().rows() == mat_in.blockSize().rows(), mat_in, perms);
74
75 DLAF_ASSERT(i_begin >= 0 && i_begin <= i_end, i_begin, i_end);
76 DLAF_ASSERT(i_end <= perms.nrTiles().rows(), i_end, perms);
77
78 internal::Permutations<B, D, T, coord>::call(i_begin, i_end, perms, mat_in, mat_out);
79}
