90  Pipeline sub_pipeline() {
91  namespace ex = pika::execution::experimental;
92 
93  // Move value from pipeline into sub pipeline, then store a sender of the wrapper of the pipeline in
94  // a sender which we will release when the sub pipeline is released. This ensures that all accesses
95  // to the sub pipeline happen after previous accesses and before later accesses to the pipeline.
96  Pipeline sub_pipeline(T{});
97  sub_pipeline.nested_sender =
98  ex::when_all(sub_pipeline.pipeline->readwrite(), this->pipeline->readwrite()) |
99  ex::then([](auto sub_wrapper, auto wrapper) {
100  sub_wrapper.get() = std::move(wrapper.get());
101 
102  return wrapper;
103  }) |
