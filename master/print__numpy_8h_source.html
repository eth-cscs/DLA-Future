48template <class T>
49std::string numpy_value(const T& value) {
50 std::ostringstream os;
51 if (std::isnan(value))
52 os << "np.nan";
53 else
54 os << value;
55 return os.str();
56}

58template <class T>
59std::string numpy_value(const std::complex<T>& value) {
60 std::ostringstream os;
61 const auto imag = std::imag(value);
62 const auto sign = (imag < 0) ? '-' : '+';
63 os << std::real(value) << sign << std::abs(imag) << "j";
64 return os.str();
65}

69template <class T>
70void print(format::numpy, const Tile<const T, Device::CPU>& tile, std::ostream& os = std::cout) {
71 os << "np.array([";
72
73 // Note:
74 // iterate_range2d loops over indices in column-major order, while python default
75 // order is row-major.
76 // For this reason, values are printed in a column-major order, and reordering is deferred
77 // to python by tranposing the resulting array (and shaping it accordingly)
78 for (const auto& index : iterate_range2d(tile.size()))
79 os << internal::numpy_value(tile(index)) << ",";
80
81 os << "]"
82 << ", dtype=np." << internal::numpy_datatype<T>::typestring << ")"
83 << ".reshape" << transposed(tile.size()) << ".T\n";
84}

86template <class T, Device D, template <class, Device> class MatrixLikeT>
87void print(format::numpy, std::string sym, MatrixLikeT<const T, D>& matrix,
88 std::ostream& os = std::cout) {
89 using common::iterate_range2d;
90
91 const auto& distribution = matrix.distribution();
92
93 os << sym << " = np.zeros(" << distribution.size() << ", dtype=np."
94 << internal::numpy_datatype<T>::typestring << ")\n";
95
96 const LocalTileSize local_tiles = distribution.localNrTiles();
97
98 auto getTileTopLeft = [&distribution](const LocalTileIndex& local) -> GlobalElementIndex {
99 return {
100 distribution.template globalElementFromLocalTileAndTileElement<Coord::Row>(local.row(), 0),
101 distribution.template globalElementFromLocalTileAndTileElement<Coord::Col>(local.col(), 0),
102 };
103 };
104
105 for (const auto& ij_local : iterate_range2d(local_tiles)) {
106 const auto tile = pika::this_thread::experimental::sync_wait(matrix.read(ij_local));
107
108 const auto tl = getTileTopLeft(ij_local);
109 const auto br = tl + GlobalElementSize{tile.get().size().rows(), tile.get().size().cols()};
110
111 os << sym << "[" << tl.row() << ":" << br.row() << "," << tl.col() << ":" << br.col() << "] = ";
112 print(format::numpy{}, tile.get(), os);
113 }
114}
115}
