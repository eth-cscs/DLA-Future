26 template <class T>
27 auto groupTausFromBandsToTiles(
28  common::internal::vector<pika::shared_future<common::internal::vector<T>>> taus_band,
29  const SizeType band_size, const SizeType mb) {
30  using common::internal::vector;
31 
32  namespace ex = pika::execution::experimental;
33 
34  auto accumulateVectorSizes = [](const SizeType acc, const vector<T>& v) { return acc + v.size(); };
35 
36  vector<pika::shared_future<vector<T>>> taus;
37 
38  const SizeType group_size = mb / band_size;
39  taus.reserve(dlaf::util::ceilDiv(taus_band.size(), group_size));
40 
41  for (SizeType start = 0; start < taus_band.size(); start += group_size) {
42  const SizeType end = std::min<SizeType>(taus_band.size(), start + group_size);
43  std::vector<pika::shared_future<vector<T>>> block_deps(begin(taus_band) + start,
44  begin(taus_band) + end);
45  taus.emplace_back(ex::when_all_vector(std::move(block_deps)) |
46  ex::then([accumulateVectorSizes](std::vector<vector<T>>&& taus_band_chunks) {
47  const auto nrefls =
48  std::accumulate(cbegin(taus_band_chunks), cend(taus_band_chunks),
49  SizeType(0), accumulateVectorSizes);
50 
51  vector<T> taus_tile;
52  taus_tile.reserve(nrefls);
53 
54  for (auto& taus_band : taus_band_chunks)
55  for (auto& tau : taus_band)
56  taus_tile.emplace_back(std::move(tau));
57 
58  return taus_tile;
59  }) |
60  ex::make_future());
61  }
62  return taus;
63 }
