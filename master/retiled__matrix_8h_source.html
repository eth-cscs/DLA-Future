36 public:

45  RetiledMatrix(Matrix<T, D>& mat, const LocalTileSize& tiles_per_block)
46  : MatrixBase(mat.distribution(), tiles_per_block) {
47  using common::internal::vector;
48  namespace ex = pika::execution::experimental;
49 
50  const auto n = to_sizet(distribution().localNrTiles().linear_size());
51  tile_managers_.reserve(n);
52  for (std::size_t i = 0; i < n; ++i) {
53  tile_managers_.emplace_back(Tile<T, D>());
54  }
55 
56  const auto tile_size = distribution().baseTileSize();
57  vector<SubTileSpec> specs;
58  vector<LocalTileIndex> indices;
59  specs.reserve(tiles_per_block.linear_size());
60  indices.reserve(tiles_per_block.linear_size());
61 
62  for (const auto& orig_tile_index : common::iterate_range2d(mat.distribution().localNrTiles())) {
63  const auto original_tile_size = mat.tileSize(mat.distribution().globalTileIndex(orig_tile_index));
64 
65  for (SizeType j = 0; j < original_tile_size.cols(); j += tile_size.cols())
66  for (SizeType i = 0; i < original_tile_size.rows(); i += tile_size.rows()) {
67  indices.emplace_back(
68  LocalTileIndex{orig_tile_index.row() * tiles_per_block.rows() + i / tile_size.rows(),
69  orig_tile_index.col() * tiles_per_block.cols() + j / tile_size.cols()});
70  specs.emplace_back(
71  SubTileSpec{{i, j}, tileSize(distribution().globalTileIndex(indices.back()))});
72  }
73 
74  auto sub_tiles = splitTileDisjoint(mat.readwrite(orig_tile_index), specs);
75 
76  DLAF_ASSERT_HEAVY(specs.size() == indices.size(), specs.size(), indices.size());
77  for (SizeType j = 0; j < specs.size(); ++j) {
78  const auto i = tileLinearIndex(indices[j]);
79 
80  // Move subtile to be managed by the tile manager of RetiledMatrix. We
81  // use readwrite_with_wrapper to get access to the original tile managed
82  // by the underlying async_rw_mutex.
83  auto s =
84  ex::when_all(tile_managers_[i].readwrite_with_wrapper(), std::move(sub_tiles[to_sizet(j)])) |
85  ex::then([](internal::TileAsyncRwMutexReadWriteWrapper<T, D> empty_tile_wrapper,
86  Tile<T, D> sub_tile) { empty_tile_wrapper.get() = std::move(sub_tile); });
87  ex::start_detached(std::move(s));
88  }
89 
90  specs.clear();
91  indices.clear();
92  }
93  }
