61  template <class VISender, class TSender>
62  static auto gemvColumnT(SizeType first_row_tile, VISender tile_vi,
63  pika::shared_future<common::internal::vector<T>>& taus, TSender&& tile_t) {
64  namespace ex = pika::execution::experimental;
65 
66  auto gemv_func = [first_row_tile](const auto& tile_v, const auto& taus, auto&& tile_t) noexcept {
67  const SizeType k = tile_t.size().cols();
68  DLAF_ASSERT(tile_v.size().cols() == k, tile_v.size().cols(), k);
69  DLAF_ASSERT(taus.size() == k, taus.size(), k);
70 
71  common::internal::SingleThreadedBlasScope single;
72  for (SizeType j = 0; j < k; ++j) {
73  const T tau = taus[j];
74 
75  const TileElementIndex t_start{0, j};
76 
77  // Position of the 1 in the diagonal in the current column.
78  SizeType i_diag = j - first_row_tile;
79  const SizeType first_element_in_col = std::max<SizeType>(0, i_diag);
80 
81  // Break if the reflector starts in the next tile.
82  if (i_diag >= tile_v.size().rows())
83  break;
84 
85  // T(0:j, j) = -tau . V(j:, 0:j)* . V(j:, j)
86  // [j x 1] = [(n-j) x j]* . [(n-j) x 1]
87  TileElementIndex va_start{first_element_in_col, 0};
88  TileElementIndex vb_start{first_element_in_col, j};
89  TileElementSize va_size{tile_v.size().rows() - first_element_in_col, j};
90 
91  if (i_diag >= 0) {
92  tile_t({j, j}) = tau;
93  }
94 
95  blas::gemv(blas::Layout::ColMajor, blas::Op::ConjTrans, va_size.rows(), va_size.cols(), -tau,
96  tile_v.ptr(va_start), tile_v.ld(), tile_v.ptr(vb_start), 1, 1, tile_t.ptr(t_start),
97  1);
98  }
99  return std::move(tile_t);
100  };
101  return dlaf::internal::transform(
102  dlaf::internal::Policy<Backend::MC>(pika::execution::thread_priority::high),
103  std::move(gemv_func),
104  ex::when_all(tile_vi, dlaf::internal::keepFuture(taus), std::forward<TSender>(tile_t)));
105  }

107  template <typename TSender>
108  static auto trmvUpdateColumn(TSender&& tile_t) noexcept {
109  namespace ex = pika::execution::experimental;
110 
111  // Update each column (in order) t = T . t
112  // remember that T is upper triangular, so it is possible to use TRMV
113  auto trmv_func = [](matrix::Tile<T, Device::CPU>&& tile_t) {
114  common::internal::SingleThreadedBlasScope single;
115  for (SizeType j = 0; j < tile_t.size().cols(); ++j) {
116  const TileElementIndex t_start{0, j};
117  const TileElementSize t_size{j, 1};
118 
119  blas::trmv(blas::Layout::ColMajor, blas::Uplo::Upper, blas::Op::NoTrans, blas::Diag::NonUnit,
120  t_size.rows(), tile_t.ptr(), tile_t.ld(), tile_t.ptr(t_start), 1);
121  }
122  // TODO: Why return if the tile is unused?
123  return std::move(tile_t);
124  };
125  return dlaf::internal::transform(
126  dlaf::internal::Policy<Backend::MC>(pika::execution::thread_priority::high),
127  std::move(trmv_func), std::forward<TSender>(tile_t));
128  }

155  const SizeType k = tile_t.size().cols();

183  TileElementIndex va_start{first_element_in_col, 0};
184  TileElementIndex vb_start{first_element_in_col, j};
185  TileElementSize va_size{tile_v.size().rows() - first_element_in_col, j};

190  to_int(tile_v.ld()),

230 template <Backend backend, Device device, class T>
231 void QR_Tfactor<backend, device, T>::call(matrix::Panel<Coord::Col, T, device>& hh_panel,
232  pika::shared_future<common::internal::vector<T>> taus,
233  matrix::ReadWriteTileSender<T, device> t) {
234  namespace ex = pika::execution::experimental;
235 
236  using Helpers = tfactor_l::Helpers<backend, device, T>;
237  // Fast return in case of no reflectors
238  if (hh_panel.getWidth() == 0)
239  return;
240 
241  const auto v_start = hh_panel.offsetElement();
242 
243  matrix::ReadWriteTileSender<T, device> t_local = Helpers::set0(std::move(t));
244 
245  // Note:
246  // T factor is an upper triangular square matrix, built column by column
247  // with taus values on the diagonal
248  //
249  // T(j,j) = tau(j)
250  //
251  // and in the upper triangular part the following formula applies
252  //
253  // T(0:j, j) = T(0:j, 0:j) . -tau(j) . V(j:, 0:j)* . V(j:, j)
254  //
255  //
256  // The result is achieved in two main steps:
257  // 1) t = -tau(j) . V(j:, 0:j)* . V(j:, j)
258  // 2) T(0:j, j) = T(0:j, 0:j) . t
259 
260  // 1st step: compute the column partial result `t`
261  // First we compute the matrix vector multiplication for each column
262  // -tau(j) . V(j:, 0:j)* . V(j:, j)
263  for (const auto& v_i : hh_panel.iteratorLocal()) {
264  const SizeType first_row_tile =
265  std::max<SizeType>(0, v_i.row() * hh_panel.parentDistribution().blockSize().rows() - v_start);
266 
267  // Note:
268  // Since we are writing always on the same t, the gemv are serialized
269  // A possible solution to this would be to have multiple places where to store partial
270  // results, and then locally reduce them just before the reduce over ranks
271  t_local = Helpers::gemvColumnT(first_row_tile, hh_panel.read(v_i), taus, std::move(t_local));
272  }
273 
274  // 2nd step: compute the T factor, by performing the last step on each column
275  // each column depends on the previous part (all reflectors that comes before)
276  // so it is performed sequentially
277  ex::start_detached(Helpers::trmvUpdateColumn(std::move(t_local)));
278 }
