47template <class T>
48struct Helpers<Backend::MC, Device::CPU, T> {
49 static auto prepareT(matrix::ReadOnlyTileSender<T, Device::CPU> taus,
50 matrix::ReadWriteTileSender<T, Device::CPU> tile_t) {
51 namespace ex = pika::execution::experimental;
52 namespace di = dlaf::internal;
53 return ex::when_all(std::move(taus), std::move(tile_t)) |
54 di::transform(di::Policy<Backend::MC>(pika::execution::thread_priority::high),
55 [](const matrix::Tile<const T, Device::CPU>& taus,
56 matrix::Tile<T, Device::CPU>&& tile_t) {
57 tile::internal::set0<T>(tile_t);
58
59 const SizeType k = tile_t.size().cols();
60 lapack::lacpy(blas::Uplo::General, 1, k, taus.ptr(), 1, tile_t.ptr(),
61 tile_t.ld() + 1);
62
63 return std::move(tile_t);
64 });
65 }
66
67 static auto stepGEMV(matrix::ReadOnlyTileSender<T, Device::CPU> tile_vi,
68 matrix::ReadOnlyTileSender<T, Device::CPU> taus,
69 matrix::ReadWriteTileSender<T, Device::CPU> tile_t) {
70 auto gemv_func = [](const matrix::Tile<const T, Device::CPU>& tile_v,
71 const matrix::Tile<const T, Device::CPU>& taus,
72 matrix::Tile<T, Device::CPU> tile_t) noexcept {
73 const SizeType k = tile_t.size().cols();
74
75 DLAF_ASSERT(tile_v.size().cols() == k, tile_v.size().cols(), k);
76 DLAF_ASSERT(taus.size().rows() == k, taus.size().rows(), k);
77
78 common::internal::SingleThreadedBlasScope single;
79 for (SizeType j = 0; j < k; ++j) {
80 // T(0:j, j) = -tau . V(j:, 0:j)* . V(j:, j)
81 // [j x 1] = [(n-j) x j]* . [(n-j) x 1]
82 const TileElementIndex t_start{0, j};
83 const TileElementIndex va_start{0, 0};
84 const TileElementIndex vb_start{0, j};
85 const TileElementSize va_size{tile_v.size().rows(), j};
86 const T tau = tile_t({j, j});
87
88 blas::gemv(blas::Layout::ColMajor, blas::Op::ConjTrans, va_size.rows(), va_size.cols(), -tau,
89 tile_v.ptr(va_start), tile_v.ld(), tile_v.ptr(vb_start), 1, 1, tile_t.ptr(t_start),
90 1);
91 }
92 return tile_t;
93 };
94
95 namespace ex = pika::execution::experimental;
96 namespace di = dlaf::internal;
97
98 return ex::when_all(tile_vi, std::move(taus), std::move(tile_t)) |
99 di::transform(di::Policy<Backend::MC>(pika::execution::thread_priority::high),
100 std::move(gemv_func));
101 }
102
103 static void trmvLoop(const matrix::Tile<T, Device::CPU>& tile_t) {
104 common::internal::SingleThreadedBlasScope single;
105
106 const SizeType k = tile_t.size().cols();
107
108 // Update each column (in order) t = T . t
109 // remember that T is upper triangular, so it is possible to use TRMV
110 for (SizeType j = 0; j < k; ++j) {
111 const TileElementIndex t_start{0, j};
112 const TileElementSize t_size{j, 1};
113
114 blas::trmv(blas::Layout::ColMajor, blas::Uplo::Upper, blas::Op::NoTrans, blas::Diag::NonUnit,
115 t_size.rows(), tile_t.ptr(), tile_t.ld(), tile_t.ptr(t_start), 1);
116 }
117 }
118
119 static auto stepTRMV(matrix::ReadWriteTileSender<T, Device::CPU> tile_t) noexcept {
120 namespace di = dlaf::internal;
121
122 return std::move(tile_t) |
123 di::transform(di::Policy<Backend::MC>(pika::execution::thread_priority::high), trmvLoop);
124 }
125
126 static auto stepCopyDiagAndTRMV(matrix::ReadOnlyTileSender<T, Device::CPU> taus,
127 matrix::ReadWriteTileSender<T, Device::CPU> tile_t) noexcept {
128 auto tausdiag_trmvloop = [](const matrix::Tile<const T, Device::CPU>& taus,
129 matrix::Tile<T, Device::CPU> tile_t) {
130 common::internal::SingleThreadedBlasScope single;
131
132 const SizeType k = tile_t.size().cols();
133 lapack::lacpy(blas::Uplo::General, 1, k, taus.ptr(), 1, tile_t.ptr(), tile_t.ld() + 1);
134
135 trmvLoop(tile_t);
136 };
137
138 namespace di = dlaf::internal;
139 namespace ex = pika::execution::experimental;
140
141 return ex::when_all(std::move(taus), std::move(tile_t)) |
142 di::transform(di::Policy<Backend::MC>(pika::execution::thread_priority::high),
143 std::move(tausdiag_trmvloop));
144 }
145};

196 static void trmvLoop(cublasHandle_t handle, const matrix::Tile<T, Device::GPU>& tile_t) {
197 const SizeType k = tile_t.size().cols();
198
199 // Update each column (in order) t = T . t
200 // remember that T is upper triangular, so it is possible to use TRMV
201 for (SizeType j = 0; j < k; ++j) {
202 const TileElementIndex t_start{0, j};
203 const TileElementSize t_size{j, 1};
204
205 gpublas::internal::Trmv<T>::call(handle, CUBLAS_FILL_MODE_UPPER, CUBLAS_OP_N, CUBLAS_DIAG_NON_UNIT,
206 to_int(t_size.rows()), util::blasToCublasCast(tile_t.ptr()),
207 to_int(tile_t.ld()), util::blasToCublasCast(tile_t.ptr(t_start)),
208 1);
209 }
210 }

247template <Backend backend, Device device, class T>
248void QR_Tfactor<backend, device, T>::call(matrix::Panel<Coord::Col, T, device>& hh_panel,
249 matrix::ReadOnlyTileSender<T, Device::CPU> taus,
250 matrix::ReadWriteTileSender<T, device> tile_t) {
251 namespace ex = pika::execution::experimental;
252
253 using Helpers = tfactor_l::Helpers<backend, device, T>;
254
255 // Fast return in case of no reflectors
256 if (hh_panel.getWidth() == 0)
257 return;
258
259 tile_t = Helpers::prepareT(taus, std::move(tile_t));
260
261 // Note:
262 // T factor is an upper triangular square matrix, built column by column
263 // with taus values on the diagonal
264 //
265 // T(j,j) = tau(j)
266 //
267 // and in the upper triangular part the following formula applies
268 //
269 // T(0:j, j) = T(0:j, 0:j) . -tau(j) . V(j:, 0:j)* . V(j:, j)
270 //
271 //
272 // The result is achieved in two main steps:
273 // 1) t = -tau(j) . V(j:, 0:j)* . V(j:, j)
274 // 2) T(0:j, j) = T(0:j, 0:j) . t
275
276 // 1st step: compute the column partial result `t`
277 // First we compute the matrix vector multiplication for each column
278 // -tau(j) . V(j:, 0:j)* . V(j:, j)
279 for (const auto& i_lc : hh_panel.iteratorLocal()) {
280 // Note:
281 // Since we are writing always on the same t, the gemv are serialized
282 // A possible solution to this would be to have multiple places where to store partial
283 // results, and then locally reduce them just before the reduce over ranks
284 tile_t = Helpers::stepGEMV(hh_panel.read(i_lc), taus, std::move(tile_t));
285 }
286
287 // 2nd step: compute the T factor, by performing the last step on each column
288 // each column depends on the previous part (all reflectors that comes before)
289 // so it is performed sequentially
290 ex::start_detached(Helpers::stepTRMV(std::move(tile_t)));
291}

293template <Backend backend, Device device, class T>
294void QR_Tfactor<backend, device, T>::call(
295 matrix::Panel<Coord::Col, T, device>& hh_panel, matrix::ReadOnlyTileSender<T, Device::CPU> taus,
296 matrix::ReadWriteTileSender<T, device> tile_t,
297 comm::CommunicatorPipeline<comm::CommunicatorType::Col>& mpi_col_task_chain) {
298 namespace ex = pika::execution::experimental;
299
300 using Helpers = tfactor_l::Helpers<backend, device, T>;
301
302 // Fast return in case of no reflectors
303 if (hh_panel.getWidth() == 0)
304 return;
305
306 tile_t = Helpers::prepareT(taus, std::move(tile_t));
307
308 // Note:
309 // T factor is an upper triangular square matrix, built column by column
310 // with taus values on the diagonal
311 //
312 // T(j,j) = tau(j)
313 //
314 // and in the upper triangular part the following formula applies
315 //
316 // T(0:j, j) = T(0:j, 0:j) . -tau(j) . V(j:, 0:j)* . V(j:, j)
317 //
318 //
319 // The result is achieved in two main steps:
320 // 1) t = -tau(j) . V(j:, 0:j)* . V(j:, j)
321 // 2) T(0:j, j) = T(0:j, 0:j) . t
322
323 // 1st step: compute the column partial result `t`
324 // First we compute the matrix vector multiplication for each column
325 // -tau(j) . V(j:, 0:j)* . V(j:, j)
326 for (const auto& i_lc : hh_panel.iteratorLocal()) {
327 // Note:
328 // Since we are writing always on the same t, the gemv are serialized
329 // A possible solution to this would be to have multiple places where to store partial
330 // results, and then locally reduce them just before the reduce over ranks
331 tile_t = Helpers::stepGEMV(hh_panel.read(i_lc), taus, std::move(tile_t));
332 }
333
334 // at this point each rank has its partial result for each column
335 // so, let's reduce the results (on all ranks, so that everyone can independently compute T factor)
336 if (mpi_col_task_chain.size() > 1)
337 tile_t = schedule_all_reduce_in_place(mpi_col_task_chain.exclusive(), MPI_SUM, std::move(tile_t));
338
339 // 2nd step: compute the T factor, by performing the last step on each column
340 // each column depends on the previous part (all reflectors that comes before)
341 // so it is performed sequentially
342 ex::start_detached(Helpers::stepCopyDiagAndTRMV(taus, std::move(tile_t)));
343}
