24class TileCollector {
25 SizeType idx_begin;
26 SizeType idx_end;
27
28 auto iteratorLocal(const matrix::Distribution& dist) const {
29 const bool is_col_matrix = dist.nrTiles().cols() == 1;
30
31 const GlobalTileIndex g_begin(idx_begin, is_col_matrix ? 0 : idx_begin);
32 const GlobalTileIndex g_end(idx_end, is_col_matrix ? 1 : idx_end);
33
34 const LocalTileIndex begin{
35 dist.template nextLocalTileFromGlobalTile<Coord::Row>(g_begin.row()),
36 dist.template nextLocalTileFromGlobalTile<Coord::Col>(g_begin.col()),
37 };
38 const LocalTileIndex end{
39 dist.template nextLocalTileFromGlobalTile<Coord::Row>(g_end.row()),
40 dist.template nextLocalTileFromGlobalTile<Coord::Col>(g_end.col()),
41 };
42 const LocalTileSize size = end - begin;
43
44 return std::make_pair(begin, size);
45 }
46
47public:
48 TileCollector(SizeType i_begin, SizeType i_end) : idx_begin(i_begin), idx_end(i_end) {}
49
50 template <class T, Device D>
51 auto read(Matrix<const T, D>& mat) const {
52 auto [begin, size] = iteratorLocal(mat.distribution());
53 return matrix::selectRead(mat, common::iterate_range2d(begin, size));
54 }
55
56 template <class T, Device D>
57 auto readwrite(Matrix<T, D>& mat) const {
58 auto [begin, size] = iteratorLocal(mat.distribution());
59 return matrix::select(mat, common::iterate_range2d(begin, size));
60 }
61};
