77 template <class T>
78 void add(T alpha, const matrix::Tile<const T, Device::CPU>& tile_b,
79  const matrix::Tile<T, Device::CPU>& tile_a) {
80  DLAF_ASSERT(equal_size(tile_a, tile_b), tile_a, tile_b);
81  if (tile_a.size().isEmpty())
82  return;
83  common::internal::SingleThreadedBlasScope single;
84  for (auto j = 0; j < tile_a.size().cols(); ++j)
85  blas::axpy(tile_a.size().rows(), alpha, tile_b.ptr({0, j}), 1, tile_a.ptr({0, j}), 1);
86 }
