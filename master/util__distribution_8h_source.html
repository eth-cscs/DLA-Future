110  DLAF_ASSERT_HEAVY(0 <= src_rank && src_rank < grid_size, src_rank, grid_size);

112 
113  if (rank == rank_global_tile(global_tile, tiles_per_block, grid_size, src_rank, tile_offset)) {
114  global_tile += tile_offset;
115 
116  SizeType local_block = global_tile / tiles_per_block / grid_size;
117 
118  // tile_offset only affects the source rank
119  bool may_have_partial_first_block = rank == src_rank;

138 inline SizeType next_local_tile_from_global_tile(SizeType global_tile, SizeType tiles_per_block,
139  int grid_size, int rank, int src_rank,
140  SizeType tile_offset) {
141  DLAF_ASSERT_HEAVY(0 <= global_tile, global_tile);
142  DLAF_ASSERT_HEAVY(0 < tiles_per_block, tiles_per_block);
143  DLAF_ASSERT_HEAVY(0 < grid_size, grid_size);
144  DLAF_ASSERT_HEAVY(0 <= rank && rank < grid_size, rank, grid_size);
145  DLAF_ASSERT_HEAVY(0 <= src_rank && src_rank < grid_size, src_rank, grid_size);
146  DLAF_ASSERT_HEAVY(0 <= tile_offset && tile_offset < tiles_per_block, tile_offset, tiles_per_block);
147 
148  int rank_to_src = (rank + grid_size - src_rank) % grid_size;
149  global_tile += tile_offset;
150  SizeType global_block = global_tile / tiles_per_block;
151  SizeType owner_to_src = global_block % grid_size;
152  SizeType local_block = global_block / grid_size;
153 
154  // If there's a tile offset it affects only the source rank block. All other
155  // blocks are whole.
156  bool may_have_partial_first_block = rank == src_rank;
157 
158  if (rank_to_src == owner_to_src)
159  return local_block * tiles_per_block + global_tile % tiles_per_block -
160  (may_have_partial_first_block ? tile_offset : 0);
161 
162  if (rank_to_src < owner_to_src)
163  ++local_block;
164 
165  return local_block * tiles_per_block - (may_have_partial_first_block ? tile_offset : 0);
166 }

185 
186  bool may_have_partial_first_block = rank == src_rank;
187  if (may_have_partial_first_block) {
188  local_tile += tile_offset;

190  // Renumber ranks such that src_rank is 0.

192  SizeType local_block = local_tile / tiles_per_block;
193

195  tile_offset;
