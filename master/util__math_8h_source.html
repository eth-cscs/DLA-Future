26// Interleaves two intervals of length `l` split in blocks of size `b` starting at offsets `o1` and
27// `o2` respectively and returns an array of indices where the splits have occured.
28//
29// o1
30// │
31// └► │ │ │ │ │
32// ─┴───┴───┴───┴───┴─── ◄─┐
33// ▲ │
34// o2 b ─┬──┘ l
35// │ │ │
36// └─► │ │ ▼ │ │ │ │
37// ──┴───┴───┴───┴───┴── ◄─┘
38//
39inline std::vector<SizeType> interleaveSplits(SizeType l, SizeType b, SizeType o1, SizeType o2) {
40 DLAF_ASSERT(l > 0, l);
41 DLAF_ASSERT(b > 0, b);
42 DLAF_ASSERT(o1 >= 0 && o1 <= b, o1, b);
43 DLAF_ASSERT(o2 >= 0 && o2 <= b, o2, b);
44
45 // Set small and big from offsets o1 and o2 s.t small <= big
46 SizeType small = o1;
47 SizeType big = o2;
48 if (small > big)
49 std::swap(small, big);
50
51 // Reserve enough memory for array of splits
52 std::vector<SizeType> splits;
53 splits.reserve(2 * to_sizet(l / b) + 2);
54
55 splits.push_back(0);
56 for (SizeType i = small, j = big; i < l || j < l; i += b, j += b) {
57 if (splits.back() != i && i < l)
58 splits.push_back(i);
59 if (splits.back() != j && j < l)
60 splits.push_back(j);
61 }
62 splits.push_back(l);
63 return splits;
64}

105template <class ArithmeticT, class TA, class TB, class BinaryOp>
106constexpr ArithmeticT generic_integer_op(const TA a, const TB b, BinaryOp op) {
107 return op(integral_cast<ArithmeticT>(a), integral_cast<ArithmeticT>(b));
108}

118template <class ArithmeticT>
119SizeType percent_to_index(SizeType size, ArithmeticT percent) {
120 DLAF_ASSERT(size >= 0, size);
121 DLAF_ASSERT(percent >= 0 && percent <= 100, percent);
122 SizeType idx = std::lround(size * percent / 100.0);
123 DLAF_ASSERT(idx >= 0 && idx <= size, idx, size);
124 return idx;
125}
