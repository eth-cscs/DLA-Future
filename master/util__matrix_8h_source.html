108 template <template <class, Device> class MatrixLikeA, template <class, Device> class MatrixLikeB,
109  template <class, Device> class MatrixLikeC, class T, Device D>
110 bool multipliable(const MatrixLikeA<const T, D>& a, const MatrixLikeB<const T, D>& b,
111  const MatrixLikeC<const T, D>& c, const blas::Op opA, const blas::Op opB) noexcept {
112  const bool isSizeOk = multipliable_sizes(a.size(), b.size(), c.size(), opA, opB);
113 
114  if (a.size().isEmpty() || c.size().isEmpty())
115  return isSizeOk;
116 
117  const bool allSameGrid = same_process_grid(a, b) && same_process_grid(b, c);
118  const bool isTileSizeOk = multipliable_sizes(a.tile_size(), b.tile_size(), c.tile_size(), opA, opB);
119  const bool isOffsetOk = multipliable_sizes(a.tile_size_of({0, 0}), b.tile_size_of({0, 0}),
120  c.tile_size_of({0, 0}), opA, opB);
121 
122  if (local_matrix(c))
123  return allSameGrid && isSizeOk && isTileSizeOk && isOffsetOk;
124 
125  // TODO distributed (fix offset)
126  return allSameGrid && isSizeOk && isTileSizeOk && isOffsetOk;

208 template <class T, class ElementGetter>
209 void set(Matrix<T, Device::CPU>& matrix, ElementGetter el_f) {
210  using pika::execution::thread_stacksize;
211 
212  const Distribution& dist = matrix.distribution();
213  for (auto tile_wrt_local : iterate_range2d(dist.local_nr_tiles())) {
214  GlobalTileIndex tile_wrt_global = dist.global_tile_index(tile_wrt_local);
215  auto tl_index = dist.global_element_index(tile_wrt_global, {0, 0});
216 
217  using TileType = typename std::decay_t<decltype(matrix)>::TileType;
218  auto set_f = [tl_index, el_f = el_f](const TileType& tile) {
219  for (auto el_idx_l : iterate_range2d(tile.size())) {
220  GlobalElementIndex el_idx_g(el_idx_l.row() + tl_index.row(), el_idx_l.col() + tl_index.col());
221  tile(el_idx_l) = el_f(el_idx_g);
222  }
223  };
224 
225  dlaf::internal::transformDetach(dlaf::internal::Policy<Backend::MC>(thread_stacksize::nostack),
226  std::move(set_f), matrix.readwrite(tile_wrt_local));
227  }

282 void set_random(Matrix<T, Device::CPU>& matrix) {
283  using pika::execution::thread_stacksize;
284 
285  const Distribution& dist = matrix.distribution();
286  for (auto tile_wrt_local : iterate_range2d(dist.local_nr_tiles())) {
287  GlobalTileIndex tile_wrt_global = dist.global_tile_index(tile_wrt_local);

306 template <class T>
307 void set_diagonal_tile(const Tile<T, Device::CPU>& tile, internal::getter_random<T>& random_value,
308  SizeType offset_value) {
309  // DIAGONAL
310  // for diagonal tiles get just lower matrix values and set value for both
311  // straight and transposed indices
312  for (SizeType j = 0; j < tile.size().cols(); ++j) {
313  for (SizeType i = 0; i < j; ++i) {
314  auto value = random_value();
315 
316  tile(TileElementIndex{i, j}) = value;
317  tile(TileElementIndex{j, i}) = dlaf::conj(value);
318  }
319  tile(TileElementIndex{j, j}) = std::real(random_value()) + offset_value;
320  }
321 }

323 template <class T>
324 void set_lower_and_upper_tile(const Tile<T, Device::CPU>& tile, internal::getter_random<T>& random_value,
325  TileElementSize full_tile_size, GlobalTileIndex tile_wrt_global,
326  dlaf::matrix::Distribution dist,
327  std::optional<SizeType> band_size = std::nullopt) {
328  auto is_off_band = [](GlobalElementIndex ij, std::optional<SizeType> band_size) {
329  return band_size ? ij.col() < ij.row() - *band_size || ij.col() > ij.row() + *band_size : false;
330  };
331 
332  // LOWER or UPPER (except DIAGONAL)
333  // random values are requested in the same order for both original and transposed
334  for (SizeType j = 0; j < full_tile_size.cols(); ++j) {
335  for (SizeType i = 0; i < full_tile_size.rows(); ++i) {
336  auto value = random_value();
337 
338  // but they are set row-wise in the original tile and col-wise in the
339  // transposed one
340  if (tile_wrt_global.row() > tile_wrt_global.col()) { // LOWER
341  TileElementIndex index{i, j};
342  if (index.isIn(tile.size())) {
343  auto ij = dist.globalElementIndex(tile_wrt_global, index);
344  if (is_off_band(ij, band_size))
345  tile(index) = T(0);
346  else
347  tile(index) = value;
348  }
349  }
350  else { // UPPER
351  TileElementIndex index{j, i};
352  if (index.isIn(tile.size())) {
353  auto ij = dist.globalElementIndex(tile_wrt_global, index);
354  if (is_off_band(ij, band_size))
355  tile(index) = T(0);
356  else
357  tile(index) = dlaf::conj(value);
358  }
359  }
360  }
361  }

380 template <class T>
381 void set_random_hermitian_with_offset(Matrix<T, Device::CPU>& matrix, const SizeType offset_value,
382  std::optional<SizeType> band_size = std::nullopt) {
383  using pika::execution::thread_stacksize;
384 
385  // note:
386  // By assuming square blocksizes, it is easier to locate elements. In fact:
387  // - Elements on the diagonal are stored in the diagonal of the diagonal tiles
388  // - Tiles under the diagonal store elements of the lower triangular matrix
389  // - Tiles over the diagonal store elements of the upper triangular matrix
390 
391  const Distribution& dist = matrix.distribution();
392 
393  DLAF_ASSERT(square_size(matrix), matrix);
394  DLAF_ASSERT(square_blocksize(matrix), matrix);
395 
396  auto full_tile_size = matrix.block_size();
397 
398  for (auto tile_wrt_local : iterate_range2d(dist.local_nr_tiles())) {
399  GlobalTileIndex tile_wrt_global = dist.global_tile_index(tile_wrt_local);
400 
401  auto tl_index = dist.global_element_index(tile_wrt_global, {0, 0});
402 
403  // compute the same seed for original and "transposed" tiles, so transposed ones will know the
404  // values of the original one without the need of accessing real values (nor communication in case
405  // of distributed matrices)
406  SizeType seed;
407  if (tile_wrt_global.row() >= tile_wrt_global.col()) // LOWER or DIAGONAL
408  seed = tl_index.col() + tl_index.row() * matrix.size().cols();
409  else
410  seed = tl_index.row() + tl_index.col() * matrix.size().rows();
411 
412  using TileType = typename std::decay_t<decltype(matrix)>::TileType;
413  auto set_hp_f = [=](const TileType& tile) {
414  internal::getter_random<T> random_value(seed);
415  if (tile_wrt_global.row() == tile_wrt_global.col())
416  internal::set_diagonal_tile(tile, random_value, offset_value);
417  else
418  internal::set_lower_and_upper_tile(tile, random_value, full_tile_size, tile_wrt_global, dist,
419  band_size);
420  };
421 
422  dlaf::internal::transformDetach(dlaf::internal::Policy<Backend::MC>(thread_stacksize::nostack),
423  std::move(set_hp_f), matrix.readwrite(tile_wrt_local));
424  }
425 }
