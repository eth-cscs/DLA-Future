110template <class IndexT, class Tag>
111bool multipliable_sizes(common::Size2D<IndexT, Tag> a, common::Size2D<IndexT, Tag> b,
112 common::Size2D<IndexT, Tag> c, const blas::Op opA, const blas::Op opB) noexcept {
113 if (opA != blas::Op::NoTrans)
114 a.transpose();
115 if (opB != blas::Op::NoTrans)
116 b.transpose();
117
118 return a.rows() == c.rows() && a.cols() == b.rows() && b.cols() == c.cols();
119}

122template <template <class, Device> class MatrixLikeA, template <class, Device> class MatrixLikeB,
123 template <class, Device> class MatrixLikeC, class T, Device D>
124bool multipliable(const MatrixLikeA<const T, D>& a, const MatrixLikeB<const T, D>& b,
125 const MatrixLikeC<const T, D>& c, const blas::Op opA, const blas::Op opB) noexcept {
126 const bool isSizeOk = multipliable_sizes(a.size(), b.size(), c.size(), opA, opB);
127
128 if (a.size().isEmpty() || c.size().isEmpty())
129 return isSizeOk;
130
131 const bool allSameGrid = same_process_grid(a, b) && same_process_grid(b, c);
132 const bool isTileSizeOk = multipliable_sizes(a.tile_size(), b.tile_size(), c.tile_size(), opA, opB);
133 const bool isOffsetOk = multipliable_sizes(a.tile_size_of({0, 0}), b.tile_size_of({0, 0}),
134 c.tile_size_of({0, 0}), opA, opB);
135
136 if (local_matrix(c))
137 return allSameGrid && isSizeOk && isTileSizeOk && isOffsetOk;
138
139 // TODO distributed (fix offset)
140 return allSameGrid && isSizeOk && isTileSizeOk && isOffsetOk;
141}

147template <class T>
148class getter_random {
149 static_assert(std::is_same_v<T, float> || std::is_same_v<T, double>,
150 "T is not compatible with random generator used.");
151
152public:
153 getter_random() : random_engine_(std::minstd_rand::default_seed) {}
154 getter_random(SizeType seed) : random_engine_(static_cast<std::size_t>(seed)) {
155 DLAF_ASSERT(seed >= 0, "");
156 }
157
158 T operator()() {
159 return random_sampler_(random_engine_);
160 }
161
162private:
163 std::mt19937_64 random_engine_;
164 std::uniform_real_distribution<T> random_sampler_{-1, 1};
165};
166
168template <class T>
169class getter_random<std::complex<T>> : private getter_random<T> {
170public:
171 using getter_random<T>::getter_random;
172
173 std::complex<T> operator()() {
174 return std::polar<T>(std::abs(getter_random<T>::operator()()),
175 static_cast<T>(M_PI) * getter_random<T>::operator()());
176 }
177};

222template <class T, class ElementGetter>
223void set(Matrix<T, Device::CPU>& matrix, ElementGetter el_f) {
224 using pika::execution::thread_stacksize;
225
226 const Distribution& dist = matrix.distribution();
227 for (auto tile_wrt_local : iterate_range2d(dist.local_nr_tiles())) {
228 GlobalTileIndex tile_wrt_global = dist.global_tile_index(tile_wrt_local);
229 auto tl_index = dist.global_element_index(tile_wrt_global, {0, 0});
230
231 using TileType = typename std::decay_t<decltype(matrix)>::TileType;
232 auto set_f = [tl_index, el_f = el_f](const TileType& tile) {
233 for (auto el_idx_l : iterate_range2d(tile.size())) {
234 GlobalElementIndex el_idx_g(el_idx_l.row() + tl_index.row(), el_idx_l.col() + tl_index.col());
235 tile(el_idx_l) = el_f(el_idx_g);
236 }
237 };
238
239 dlaf::internal::transformDetach(dlaf::internal::Policy<Backend::MC>(thread_stacksize::nostack),
240 std::move(set_f), matrix.readwrite(tile_wrt_local));
241 }

253template <class T, class ElementGetter>
254void set(Matrix<T, Device::CPU>& matrix, ElementGetter el_f, const blas::Op op) {
255 auto el_op_f = [op, el_f](const GlobalElementIndex& index) -> T {
256 using blas::Op;
257 switch (op) {
258 case Op::NoTrans:
259 return el_f(index);
260 case Op::Trans:
261 return el_f(transposed(index));
262 case Op::ConjTrans:
263 return dlaf::conj(el_f(transposed(index)));
264 default:
265 DLAF_UNIMPLEMENTED(op);
266 return T{};
267 }
268 };
269
270 set(matrix, el_op_f);
271}

295template <class T>
296void set_random(Matrix<T, Device::CPU>& matrix) {
297 using pika::execution::thread_stacksize;
298
299 const Distribution& dist = matrix.distribution();
300 for (auto tile_wrt_local : iterate_range2d(dist.local_nr_tiles())) {
301 GlobalTileIndex tile_wrt_global = dist.global_tile_index(tile_wrt_local);
302 auto tl_index = dist.global_element_index(tile_wrt_global, {0, 0});
303 auto seed = tl_index.col() + tl_index.row() * matrix.size().cols();
304
305 using TileType = typename std::decay_t<decltype(matrix)>::TileType;
306 auto rnd_f = [seed](TileType&& tile) {
307 internal::getter_random<T> random_value(seed);
308 for (auto el_idx : iterate_range2d(tile.size())) {
309 tile(el_idx) = random_value();
310 }
311 };
312
313 dlaf::internal::transformDetach(dlaf::internal::Policy<Backend::MC>(thread_stacksize::nostack),
314 std::move(rnd_f), matrix.readwrite(tile_wrt_local));
315 }

320template <class T>
321void set_diagonal_tile(const Tile<T, Device::CPU>& tile, internal::getter_random<T>& random_value,
322 SizeType offset_value) {
323 // DIAGONAL
324 // for diagonal tiles get just lower matrix values and set value for both
325 // straight and transposed indices
326 for (SizeType j = 0; j < tile.size().cols(); ++j) {
327 for (SizeType i = 0; i < j; ++i) {
328 auto value = random_value();
329
330 tile(TileElementIndex{i, j}) = value;
331 tile(TileElementIndex{j, i}) = dlaf::conj(value);
332 }
333 tile(TileElementIndex{j, j}) = std::real(random_value()) + offset_value;
334 }
335}

337template <class T>
338void set_lower_and_upper_tile(const Tile<T, Device::CPU>& tile, internal::getter_random<T>& random_value,
339 TileElementSize full_tile_size, GlobalTileIndex tile_wrt_global,
340 dlaf::matrix::Distribution dist,
341 std::optional<SizeType> band_size = std::nullopt) {
342 auto is_off_band = [](GlobalElementIndex ij, std::optional<SizeType> band_size) {
343 return band_size ? ij.col() < ij.row() - *band_size || ij.col() > ij.row() + *band_size : false;
344 };
345
346 // LOWER or UPPER (except DIAGONAL)
347 // random values are requested in the same order for both original and transposed
348 for (SizeType j = 0; j < full_tile_size.cols(); ++j) {
349 for (SizeType i = 0; i < full_tile_size.rows(); ++i) {
350 auto value = random_value();
351
352 // but they are set row-wise in the original tile and col-wise in the
353 // transposed one
354 if (tile_wrt_global.row() > tile_wrt_global.col()) { // LOWER
355 TileElementIndex index{i, j};
356 if (index.isIn(tile.size())) {
357 auto ij = dist.globalElementIndex(tile_wrt_global, index);
358 if (is_off_band(ij, band_size))
359 tile(index) = T(0);
360 else
361 tile(index) = value;
362 }
363 }
364 else { // UPPER
365 TileElementIndex index{j, i};
366 if (index.isIn(tile.size())) {
367 auto ij = dist.globalElementIndex(tile_wrt_global, index);
368 if (is_off_band(ij, band_size))
369 tile(index) = T(0);
370 else
371 tile(index) = dlaf::conj(value);
372 }
373 }
374 }
375 }

394template <class T>
395void set_random_hermitian_with_offset(Matrix<T, Device::CPU>& matrix, const SizeType offset_value,
396 std::optional<SizeType> band_size = std::nullopt) {
397 using pika::execution::thread_stacksize;
398
399 // note:
400 // By assuming square blocksizes, it is easier to locate elements. In fact:
401 // - Elements on the diagonal are stored in the diagonal of the diagonal tiles
402 // - Tiles under the diagonal store elements of the lower triangular matrix
403 // - Tiles over the diagonal store elements of the upper triangular matrix
404
405 const Distribution& dist = matrix.distribution();
406
407 DLAF_ASSERT(square_size(matrix), matrix);
408 DLAF_ASSERT(square_blocksize(matrix), matrix);
409
410 auto full_tile_size = matrix.block_size();
411
412 for (auto tile_wrt_local : iterate_range2d(dist.local_nr_tiles())) {
413 GlobalTileIndex tile_wrt_global = dist.global_tile_index(tile_wrt_local);
414
415 auto tl_index = dist.global_element_index(tile_wrt_global, {0, 0});
416
417 // compute the same seed for original and "transposed" tiles, so transposed ones will know the
418 // values of the original one without the need of accessing real values (nor communication in case
419 // of distributed matrices)
420 SizeType seed;
421 if (tile_wrt_global.row() >= tile_wrt_global.col()) // LOWER or DIAGONAL
422 seed = tl_index.col() + tl_index.row() * matrix.size().cols();
423 else
424 seed = tl_index.row() + tl_index.col() * matrix.size().rows();
425
426 using TileType = typename std::decay_t<decltype(matrix)>::TileType;
427 auto set_hp_f = [=](const TileType& tile) {
428 internal::getter_random<T> random_value(seed);
429 if (tile_wrt_global.row() == tile_wrt_global.col())
430 internal::set_diagonal_tile(tile, random_value, offset_value);
431 else
432 internal::set_lower_and_upper_tile(tile, random_value, full_tile_size, tile_wrt_global, dist,
433 band_size);
434 };
435
436 dlaf::internal::transformDetach(dlaf::internal::Policy<Backend::MC>(thread_stacksize::nostack),
437 std::move(set_hp_f), matrix.readwrite(tile_wrt_local));
438 }
439}
