107 matrix::SubTileSpec operator()(const LocalTileIndex& index) const {
108 DLAF_ASSERT_MODERATE(isIndexInRange(index, iteratorLocal()), index, tile_begin_, tile_end_);
109
110 SizeType i_sub = 0, j_sub = 0;
111
112 if (has_subtile_) {
113 const auto ij = dist_.globalTileIndex(index);
114
115 if (ij.row() == offset().row())
116 i_sub += dist_.tileElementFromGlobalElement<Coord::Row>(offset_e_.row());
117 if (ij.col() == offset().col())
118 j_sub += dist_.tileElementFromGlobalElement<Coord::Col>(offset_e_.col());
119 }
120 const TileElementIndex offset_sub(i_sub, j_sub);
121
122 const TileElementSize size_tile = dist_.tileSize(dist_.globalTileIndex(index));
123 const TileElementSize size_sub(size_tile.rows() - offset_sub.row(),
124 size_tile.cols() - offset_sub.col());

126 }
