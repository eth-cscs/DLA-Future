84 template <Device destination_device, CopyToDestination copy_to_destination,
85  CopyFromDestination copy_from_destination, RequireContiguous require_contiguous,
86  typename InSender, typename F>
87 auto withTemporaryTile(InSender&& in_sender, F&& f) {
88  namespace ex = pika::execution::experimental;
89 
90  using dlaf::internal::Policy;
91  using dlaf::internal::transform;
92  using dlaf::internal::whenAllLift;
93  using dlaf::matrix::copy;
94  using dlaf::matrix::Duplicate;
95  using dlaf::matrix::DuplicateNoCopy;
96  using dlaf::matrix::internal::CopyBackend_v;
97  using pika::execution::thread_priority;
98 
99  return std::forward<InSender>(in_sender) |
100  // Start a new asynchronous scope for keeping the input tile alive
101  // until all asynchronous operations are done.
102  ex::let_value(dlaf::common::internal::Unwrapping{[f = std::forward<F>(f)](auto& in) mutable {
103  constexpr Device in_device_type = std::decay_t<decltype(in)>::device;
104  constexpr Backend copy_backend = CopyBackend_v<in_device_type, destination_device>;
105 
106  // In cases that we cannot use the input tile as the temporary tile we need to:
107  // 1. allocate a new tile
108  // 2. optionally copy the input to the temporary tile
109  // 3. call the user-provided callable f and ignore values sent by it
110  // 4. optionally copy the temporary tile back to the input tile
111  // 5. send the input tile to continuations
112  auto helper_withtemp = [&]() mutable {
113  // If the user requested copying the input tile to the temporary
114  // tile we use Duplicate and copy_backend. If the user did not
115  // request copying to the temporary tile we still need to allocate
116  // a new temporary tile with the correct dimensions. In that case
117  // we always use DuplicateNoCopy on the MC backend since allocation
118  // does not require invoking a kernel on a GPU.
119  using duplicate_type =
120  std::conditional_t<bool(copy_to_destination), Duplicate<destination_device>,
121  DuplicateNoCopy<destination_device>>;
122  constexpr auto duplicate_backend = bool(copy_to_destination) ? copy_backend : Backend::MC;
123 
124  return ex::just(std::cref(in)) |
125  // Allocate a new tile and optionally copy the input tile to
126  // the temporary tile.
127  transform(Policy<duplicate_backend>(thread_priority::high), duplicate_type{}) |
128  // Start a new asynchronous scope for keeping the temporary
129  // tile alive until all asynchronous operations are done.
130  ex::let_value([&, f = std::forward<F>(f)](auto& temp) mutable {
131  // Call the user provided callable f and ignore the values
132  // sent by the sender.
133  auto f_sender = f(temp) | ex::drop_value();
134  // If the user requested copying the temporary tile back
135  // to the input tile after the user-provided operation is
136  // done we do so. Otherwise we use the sender f_sender
137  // directly.
138  auto copy_sender = [&]() {
139  if constexpr (bool(copy_from_destination)) {
140  return whenAllLift(std::move(f_sender), std::cref(temp), std::cref(in)) |
141  copy(Policy<copy_backend>(thread_priority::high));
142  }
143  else {
144  return std::move(f_sender);
145  }
146  }();
147  // Send the input tile to continuations if the tile is
148  // non-const.
149  return std::move(copy_sender) | ex::then(moveNonConstTile{in});
150  });
151  };
152 
153  // One of the helpers may be unused depending on which branch is taken
154  dlaf::internal::silenceUnusedWarningFor(helper_withtemp);
155 
156  // If the destination device is the same as the input device we may
157  // be able to avoid allocating a new tile.
158  if constexpr (in_device_type == destination_device) {
159  // In some cases we can directly use the input as a temporary tile.
160  // In that case we simply call the user-provided callable f, ignore
161  // the values sent by the sender returned from f, and finally send
162  // the input tile to continuations if the tile is non-const. This
163  // helper is inside the if constexpr to avoid instantiating it in
164  // cases where it's never needed.
165  auto helper_notemp = [&]() {
166  return f(in) | ex::drop_value() | ex::then(moveNonConstTile{in});
167  };
168 
169  // The destination device is the same as the input device, but if
170  // we require that the temporary tile is contiguous we may have to
171  // allocate a new tile in any case. We have to do a runtime check
172  // to find out. Since this is a runtime check, we wrap the senders
173  // from the different branches in type-erased unique_any_senders.
174  if constexpr (require_contiguous == RequireContiguous::Yes) {
175  // If the input tile is contiguous we can use the input tile as a
176  // temporary tile directly.
177  if (in.is_contiguous()) {
178  return ex::make_unique_any_sender(helper_notemp());
179  }
180  // If the input is not contiguous we have to at least allocate a
181  // new temporary tile.
182  else {
183  return ex::make_unique_any_sender(helper_withtemp());
184  }
185  }
186  // The destination device is the same as the input device, and we
187  // don't require contiguous memory (though we allow it). We can use
188  // the input tile as a temporary tile directly.
189  else {
190  return helper_notemp();
191  }
192  }
193  // If the destination device is different from the input device we
194  // have to at least allocate a new temporary tile on the destination
195  // device.
196  else {
197  return helper_withtemp();
198  }
199  }});
200 }
201 }
