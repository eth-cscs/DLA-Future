87template <Device destination_device, CopyToDestination copy_to_destination,
88 CopyFromDestination copy_from_destination, RequireContiguous require_contiguous,
89 typename InSender, typename F>
90auto withTemporaryTile(InSender&& in_sender, F&& f) {
91 namespace ex = pika::execution::experimental;
92
93 using dlaf::internal::Policy;
94 using dlaf::internal::transform;
95 using dlaf::internal::whenAllLift;
96 using dlaf::matrix::copy;
97 using dlaf::matrix::Duplicate;
98 using dlaf::matrix::DuplicateNoCopy;
99 using dlaf::matrix::internal::CopyBackend_v;
100 using pika::execution::thread_priority;
101
102 return std::forward<InSender>(in_sender) |
103 // Start a new asynchronous scope for keeping the input tile alive
104 // until all asynchronous operations are done.
105 ex::let_value(dlaf::common::internal::Unwrapping{[f = std::forward<F>(f)](auto& in) mutable {
106 constexpr Device in_device_type = std::decay_t<decltype(in)>::device;
107 constexpr Backend copy_backend = CopyBackend_v<in_device_type, destination_device>;
108
109 // In cases that we cannot use the input tile as the temporary tile we need to:
110 // 1. allocate a new tile
111 // 2. optionally copy the input to the temporary tile
112 // 3. call the user-provided callable f and ignore values sent by it
113 // 4. optionally copy the temporary tile back to the input tile
114 // 5. send the input tile to continuations
115 auto helper_withtemp = [&]() mutable {
116 // If the user requested copying the input tile to the temporary
117 // tile we use Duplicate and copy_backend. If the user did not
118 // request copying to the temporary tile we still need to allocate
119 // a new temporary tile with the correct dimensions. In that case
120 // we always use DuplicateNoCopy on the MC backend since allocation
121 // does not require invoking a kernel on a GPU.
122 using duplicate_type =
123 std::conditional_t<bool(copy_to_destination), Duplicate<destination_device>,
124 DuplicateNoCopy<destination_device>>;
125 constexpr auto duplicate_backend = bool(copy_to_destination) ? copy_backend : Backend::MC;
126
127 return ex::just(std::cref(in)) |
128 // Allocate a new tile and optionally copy the input tile to
129 // the temporary tile.
130 transform(Policy<duplicate_backend>(thread_priority::high), duplicate_type{}) |
131 // Start a new asynchronous scope for keeping the temporary
132 // tile alive until all asynchronous operations are done.
133 ex::let_value([&, f = std::forward<F>(f)](auto& temp) mutable {
134 // Call the user provided callable f and ignore the values
135 // sent by the sender.
136 auto f_sender = f(temp) | ex::drop_value();
137 // If the user requested copying the temporary tile back
138 // to the input tile after the user-provided operation is
139 // done we do so. Otherwise we use the sender f_sender
140 // directly.
141 auto copy_sender = [&]() {
142 if constexpr (bool(copy_from_destination)) {
143 return ex::make_unique_any_sender(
144 whenAllLift(std::move(f_sender), std::cref(temp), std::cref(in)) |
145 copy(Policy<copy_backend>(thread_priority::high)));
146 }
147 else {
148 return ex::make_unique_any_sender(std::move(f_sender));
149 }
150 }();
151 // Send the input tile to continuations if the tile is
152 // non-const.
153 return std::move(copy_sender) | ex::then(moveNonConstTile{in});
154 });
155 };
156
157 // One of the helpers may be unused depending on which branch is taken
158 dlaf::internal::silenceUnusedWarningFor(helper_withtemp);
159
160 // If the destination device is the same as the input device we may
161 // be able to avoid allocating a new tile.
162 if constexpr (in_device_type == destination_device) {
163 // In some cases we can directly use the input as a temporary tile.
164 // In that case we simply call the user-provided callable f, ignore
165 // the values sent by the sender returned from f, and finally send
166 // the input tile to continuations if the tile is non-const. This
167 // helper is inside the if constexpr to avoid instantiating it in
168 // cases where it's never needed.
169 auto helper_notemp = [&]() {
170 return f(in) | ex::drop_value() | ex::then(moveNonConstTile{in});
171 };
172
173 // The destination device is the same as the input device, but if
174 // we require that the temporary tile is contiguous we may have to
175 // allocate a new tile in any case. We have to do a runtime check
176 // to find out. Since this is a runtime check, we wrap the senders
177 // from the different branches in type-erased unique_any_senders.
178 if constexpr (require_contiguous == RequireContiguous::Yes) {
179 // If the input tile is contiguous we can use the input tile as a
180 // temporary tile directly.
181 if (in.is_contiguous()) {
182 return ex::make_unique_any_sender(helper_notemp());
183 }
184 // If the input is not contiguous we have to at least allocate a
185 // new temporary tile.
186 else {
187 return ex::make_unique_any_sender(helper_withtemp());
188 }
189 }
190 // The destination device is the same as the input device, and we
191 // don't require contiguous memory (though we allow it). We can use
192 // the input tile as a temporary tile directly.
193 else {
194 return helper_notemp();
195 }
196 }
197 // If the destination device is different from the input device we
198 // have to at least allocate a new temporary tile on the destination
199 // device.
200 else {
201 return helper_withtemp();
202 }
203 }});
204}
