20// Note: the templates of the following helper functions are inverted w.r.t. the Matrix templates

34template <Device D, class T>
35Matrix<T, D> create_matrix_from_col_major(const GlobalElementSize& size,
36 const TileElementSize& tile_size, SizeType ld, T* ptr) {
37 Distribution dist(size, tile_size, {1, 1}, {0, 0}, {0, 0});
38 ColMajorLayout layout{std::move(dist), ld};
39 return Matrix<T, D>(layout, ptr);
40}

55template <Device D, class T>
56Matrix<T, D> create_matrix_from_col_major(const GlobalElementSize& size,
57 const TileElementSize& tile_size, SizeType ld,
58 const comm::CommunicatorGrid& comm,
59 const comm::Index2D& source_rank_index, T* ptr) {
60 Distribution dist(size, tile_size, comm.size(), comm.rank(), source_rank_index);
61 ColMajorLayout layout{std::move(dist), ld};

63 return Matrix<T, D>(layout, ptr);
64}

76template <Device D, class T>
77Matrix<T, D> create_matrix_from_col_major(const GlobalElementSize& size,
78 const TileElementSize& tile_size, SizeType ld,
79 const comm::CommunicatorGrid& comm, T* ptr) {
80 return create_matrix_from_col_major<D>(size, tile_size, ld, comm, {0, 0}, ptr);
