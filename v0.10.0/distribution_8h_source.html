614 template <Coord rc>
615 SizeType global_tile_size_of(SizeType global_tile) const noexcept {
616 DLAF_ASSERT_HEAVY(0 <= global_tile && global_tile < nr_tiles_.get<rc>(), global_tile,
617 nr_tiles_.get<rc>());
618 SizeType n = size_.get<rc>();
619 SizeType nb = tile_size_.get<rc>();
620 if (global_tile == 0) {
621 return std::min(nb - global_tile_element_offset<rc>(), n);
622 }
623 return std::min(nb, n + global_tile_element_offset<rc>() - global_tile * nb);
624 }
