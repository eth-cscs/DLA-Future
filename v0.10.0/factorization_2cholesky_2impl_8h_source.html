149// Local implementation of Lower Cholesky factorization.
150template <Backend backend, Device device, class T>
151void Cholesky<backend, device, T>::call_L(Matrix<T, device>& mat_a) {
152 using namespace cholesky_l;
153 using pika::execution::thread_priority;
154
155 // Number of tile (rows = cols)
156 SizeType nrtile = mat_a.nrTiles().cols();
157
158 for (SizeType k = 0; k < nrtile; ++k) {
159 // Cholesky decomposition on mat_a.readwrite(k,k) r/w potrf (lapack operation)
160 auto kk = LocalTileIndex{k, k};
161
162 potrfDiagTile<backend>(thread_priority::high, mat_a.readwrite(kk));
163
164 for (SizeType i = k + 1; i < nrtile; ++i) {
165 // Update panel mat_a.readwrite(i,k) with trsm (blas operation), using data mat_a.read(k,k)
166 trsmPanelTile<backend>(thread_priority::high, mat_a.read(kk),
167 mat_a.readwrite(LocalTileIndex{i, k}));
168 }
169
170 for (SizeType j = k + 1; j < nrtile; ++j) {
171 // first trailing panel gets high priority (look ahead).
172 const auto trailing_matrix_priority =
173 (j == k + 1) ? thread_priority::high : thread_priority::normal;
174
175 // Update trailing matrix: diagonal element mat_a.readwrite(j,j), reading
176 // mat_a.read(j,k), using herk (blas operation)
177 herkTrailingDiagTile<backend>(trailing_matrix_priority, mat_a.read(LocalTileIndex{j, k}),
178 mat_a.readwrite(LocalTileIndex{j, j}));
179
180 for (SizeType i = j + 1; i < nrtile; ++i) {
181 // Update remaining trailing matrix mat_a.readwrite(i,j), reading
182 // mat_a.read(i,k) and mat_a.read(j,k), using gemm (blas operation)
183 gemmTrailingMatrixTile<backend>(trailing_matrix_priority, mat_a.read(LocalTileIndex{i, k}),
184 mat_a.read(LocalTileIndex{j, k}),
185 mat_a.readwrite(LocalTileIndex{i, j}));
186 }
187 }
188 }

191template <Backend backend, Device device, class T>
192void Cholesky<backend, device, T>::call_L(comm::CommunicatorGrid& grid, Matrix<T, device>& mat_a) {
193 using namespace cholesky_l;
194 using pika::execution::thread_priority;
195
196#ifdef DLAF_WITH_HDF5
197 static std::atomic<size_t> num_cholesky_calls = 0;
198 std::stringstream fname;
199 fname << "cholesky-factorization-" << matrix::internal::TypeToString_v<T> << "-"
200 << std::to_string(num_cholesky_calls) << ".h5";
201 std::optional<matrix::internal::FileHDF5> file;
202
203 if (getTuneParameters().debug_dump_cholesky_factorization_data) {
204 file = matrix::internal::FileHDF5(grid.fullCommunicator(), fname.str());
205 file->write(mat_a, "/input");
206 }
207#endif
208
209 // Set up MPI executor pipelines
210 auto mpi_row_task_chain = grid.row_communicator_pipeline();
211 auto mpi_col_task_chain = grid.col_communicator_pipeline();
212
213 const comm::Index2D this_rank = grid.rank();
214
215 const matrix::Distribution& distr = mat_a.distribution();
216 const SizeType nrtile = mat_a.nrTiles().cols();
217
218 constexpr std::size_t n_workspaces = 2;
219 common::RoundRobin<matrix::Panel<Coord::Col, T, device>> panels(n_workspaces, distr);
220 common::RoundRobin<matrix::Panel<Coord::Row, T, device, matrix::StoreTransposed::Yes>> panelsT(
221 n_workspaces, distr);
222
223 for (SizeType k = 0; k < nrtile; ++k) {
224 const GlobalTileIndex kk_idx(k, k);
225 const comm::Index2D kk_rank = distr.rankGlobalTile(kk_idx);
226
227 // Factorization of diagonal tile and broadcast it along the k-th column
228 if (kk_rank == this_rank)
229 potrfDiagTile<backend>(thread_priority::high, mat_a.readwrite(kk_idx));
230
231 // If there is no trailing matrix
232 const SizeType kt = k + 1;
233 if (kt == nrtile)
234 continue;
235
236 auto& panel = panels.nextResource();
237 auto& panelT = panelsT.nextResource();
238
239 panel.setRangeStart(GlobalTileIndex(kt, kt));
240
241 if (kk_rank.col() == this_rank.col()) {
242 const LocalTileIndex diag_wp_idx{0, distr.localTileFromGlobalTile<Coord::Col>(k)};
243
244 // Note:
245 // panelT shrinked to a single tile for temporarly storing and communicating the diagonal
246 // tile used for the column update
247 panelT.setRange({k, k}, {kt, kt});
248
249 if (kk_rank.row() == this_rank.row())
250 panelT.setTile(diag_wp_idx, mat_a.read(kk_idx));
251 broadcast(kk_rank.row(), panelT, mpi_col_task_chain);
252
253 // COLUMN UPDATE
254 for (SizeType i = distr.nextLocalTileFromGlobalTile<Coord::Row>(kt);
255 i < distr.localNrTiles().rows(); ++i) {
256 const LocalTileIndex local_idx(Coord::Row, i);
257 const LocalTileIndex ik_idx(i, distr.localTileFromGlobalTile<Coord::Col>(k));
258
259 trsmPanelTile<backend>(thread_priority::high, panelT.read(diag_wp_idx), mat_a.readwrite(ik_idx));
260
261 panel.setTile(local_idx, mat_a.read(ik_idx));
262 }
263
264 // row panel has been used for temporary storage of diagonal panel for column update
265 panelT.reset();
266 }
267
268 panelT.setRange({kt, kt}, {nrtile - 1, nrtile - 1});
269
270 broadcast(kk_rank.col(), panel, panelT, mpi_row_task_chain, mpi_col_task_chain);
271
272 // TRAILING MATRIX
273 for (SizeType jt_idx = kt; jt_idx < nrtile; ++jt_idx) {
274 const auto owner = distr.rankGlobalTile({jt_idx, jt_idx});
275
276 if (owner.col() != this_rank.col())
277 continue;
278
279 const auto j = distr.localTileFromGlobalTile<Coord::Col>(jt_idx);
280 const auto trailing_matrix_priority =
281 (jt_idx == kt) ? thread_priority::high : thread_priority::normal;
282 if (this_rank.row() == owner.row()) {
283 const auto i = distr.localTileFromGlobalTile<Coord::Row>(jt_idx);
284
285 herkTrailingDiagTile<backend>(trailing_matrix_priority, panel.read({Coord::Row, i}),
286 mat_a.readwrite(LocalTileIndex{i, j}));
287 }
288
289 for (SizeType i_idx = jt_idx + 1; i_idx < nrtile; ++i_idx) {
290 const auto owner_row = distr.rankGlobalTile<Coord::Row>(i_idx);
291
292 if (owner_row != this_rank.row())
293 continue;
294
295 const auto i = distr.localTileFromGlobalTile<Coord::Row>(i_idx);
296 gemmTrailingMatrixTile<backend>(trailing_matrix_priority, panel.read({Coord::Row, i}),
297 panelT.read({Coord::Col, j}),
298 mat_a.readwrite(LocalTileIndex{i, j}));
299 }
300 }
301
302 panel.reset();
303 panelT.reset();
304 }
305
306#ifdef DLAF_WITH_HDF5
307 if (getTuneParameters().debug_dump_cholesky_factorization_data) {
308 file->write(mat_a, "/cholesky");
309 }
310
311 num_cholesky_calls++;
312#endif
313}
