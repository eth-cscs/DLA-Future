171template <Device D>
172constexpr bool replace_trtri() {
173 if constexpr (D == Device::CPU)
174 return 0;
175#ifdef DLAF_WITH_CUDA
176 return 1;
177#else // DLAF_WITH_HIP
178 return 0;
179#endif

182template <Backend backend, Device D, class T>
183void Triangular<backend, D, T>::call_L(blas::Diag diag, Matrix<T, D>& mat_a) {
184 using namespace triangular_inv_l;
185 using pika::execution::thread_priority;
186
187 // Number of tile (rows = cols)
188 SizeType nrtiles = mat_a.nr_tiles().cols();
189
190 if (nrtiles == 0)
191 return;
192
193 auto tile_size = mat_a.tile_size();
194 std::optional<Matrix<T, D>> ws;
195 if constexpr (replace_trtri<D>())
196 ws.emplace({tile_size.rows(), tile_size.cols()}, tile_size);
197
198 for (SizeType k = nrtiles - 1; k >= 0; --k) {
199 auto kk = LocalTileIndex{k, k};
200
201 for (SizeType i = k + 1; i < nrtiles; ++i) {
202 trsm_col_panel_tile<backend>(thread_priority::high, diag, mat_a.read(kk),
203 mat_a.readwrite(LocalTileIndex{i, k}));
204 for (SizeType j = 0; j < k; ++j) {
205 const auto trailing_matrix_priority =
206 (j == k - 1) ? thread_priority::high : thread_priority::normal;
207
208 gemm_matrix_tile<backend>(trailing_matrix_priority, mat_a.read(LocalTileIndex{i, k}),
209 mat_a.read(LocalTileIndex{k, j}),
210 mat_a.readwrite(LocalTileIndex{i, j}));
211 }
212 }
213
214 for (SizeType j = 0; j < k; ++j) {
215 trsm_row_panel_tile<backend>(thread_priority::high, diag, mat_a.read(kk),
216 mat_a.readwrite(LocalTileIndex{k, j}));
217 }
218
219 if constexpr (replace_trtri<D>()) {
220 LocalTileIndex ws_index{0, 0};
221 inverse_diag_tile_workspace<backend>(thread_priority::high, diag, mat_a.readwrite(kk),
222 ws->readwrite(ws_index));
223 }
224 else {
225 inverse_diag_tile<backend>(thread_priority::high, diag, mat_a.readwrite(kk));
226 }
227 }

230template <Backend backend, Device D, class T>
231void Triangular<backend, D, T>::call_L(comm::CommunicatorGrid& grid, blas::Diag diag,
232 Matrix<T, D>& mat_a) {
233 using namespace triangular_inv_l;
234 using pika::execution::thread_priority;
235
236#ifdef DLAF_WITH_HDF5
237 static std::atomic<size_t> num_triangular_inv_calls = 0;
238 std::stringstream fname;
239 fname << "triangular-inverse-L-" << matrix::internal::TypeToString_v<T> << "-"
240 << std::to_string(num_triangular_inv_calls) << ".h5";
241 std::optional<matrix::internal::FileHDF5> file;
242
243 if (getTuneParameters().debug_dump_triangular_inverse_data) {
244 file = matrix::internal::FileHDF5(grid.fullCommunicator(), fname.str());
245 file->write(mat_a, "/input");
246 }
247#endif
248
249 // Set up MPI executor pipelines
250 auto mpi_row_task_chain = grid.row_communicator_pipeline();
251 auto mpi_col_task_chain = grid.col_communicator_pipeline();
252
253 const comm::Index2D this_rank = grid.rank();
254
255 const matrix::Distribution& dist = mat_a.distribution();
256 const SizeType nrtiles = mat_a.nr_tiles().cols();
257
258 constexpr std::size_t n_workspaces = 2;
259 common::RoundRobin<matrix::Panel<Coord::Col, T, D>> col_panels(n_workspaces, dist);
260 common::RoundRobin<matrix::Panel<Coord::Row, T, D>> row_panels(n_workspaces, dist);
261
262 const SizeType i_local_end = dist.local_nr_tiles().rows();
263
264 for (SizeType k = nrtiles - 1; k >= 0; --k) {
265 const GlobalTileIndex kk(k, k);
266 const comm::Index2D kk_rank = dist.rank_global_tile(kk);
267
268 const SizeType k_local_row = dist.next_local_tile_from_global_tile<Coord::Row>(k);
269 const SizeType k1_local_row = dist.next_local_tile_from_global_tile<Coord::Row>(k + 1);
270 const SizeType k_local_col = dist.next_local_tile_from_global_tile<Coord::Col>(k);
271 const SizeType k1_local_col = dist.next_local_tile_from_global_tile<Coord::Col>(k + 1);
272
273 auto& col_panel = col_panels.nextResource();
274 auto& row_panel = row_panels.nextResource();
275
276 if (k < nrtiles - 1) {
277 row_panel.setRange({0, 0}, {k + 1, k + 1});
278 if (k == 0)
279 row_panel.setHeight(mat_a.tile_size_of(kk).rows());
280
281 if (kk_rank.row() == this_rank.row()) {
282 for (SizeType j_local = 0; j_local < k1_local_col; ++j_local) {
283 const LocalTileIndex kj_panel(Coord::Col, j_local);
284 const LocalTileIndex kj_local(k_local_row, j_local);
285
286 row_panel.setTile(kj_panel, mat_a.read(kj_local));
287 }
288 }
289 broadcast(kk_rank.row(), row_panel, mpi_col_task_chain);
290 }
291
292 if (kk_rank.col() == this_rank.col()) {
293 for (SizeType i_local = k1_local_row; i_local < i_local_end; ++i_local) {
294 const LocalTileIndex kk_local_panel{Coord::Col, k_local_col};
295 const LocalTileIndex ik_local{i_local, k_local_col};
296 trsm_col_panel_tile<backend>(thread_priority::high, diag, row_panel.read(kk_local_panel),
297 mat_a.readwrite(ik_local));
298 }
299 }
300
301 if (k > 0) {
302 col_panel.setRangeStart(kk);
303 if (k == nrtiles - 1)
304 col_panel.setWidth(mat_a.tile_size_of(kk).cols());
305
306 if (kk_rank.col() == this_rank.col()) {
307 for (SizeType i_local = k_local_row; i_local < i_local_end; ++i_local) {
308 const LocalTileIndex ik_panel(Coord::Row, i_local);
309 const LocalTileIndex ik_local(i_local, k_local_col);
310
311 col_panel.setTile(ik_panel, mat_a.read(ik_local));
312 }
313 }
314 broadcast(kk_rank.col(), col_panel, mpi_row_task_chain);
315 }
316
317 for (SizeType i_local = k1_local_row; i_local < i_local_end; ++i_local) {
318 for (SizeType j_local = 0; j_local < k_local_col; ++j_local) {
319 const LocalTileIndex ik_local_panel{Coord::Row, i_local};
320 const LocalTileIndex kj_local_panel{Coord::Col, j_local};
321 const LocalTileIndex ij_local{i_local, j_local};
322
323 const auto trailing_matrix_priority =
324 (j_local == k_local_col - 1) ? thread_priority::high : thread_priority::normal;
325
326 gemm_matrix_tile<backend>(trailing_matrix_priority, col_panel.read(ik_local_panel),
327 row_panel.read(kj_local_panel), mat_a.readwrite(ij_local));
328 }
329 }
330
331 if (kk_rank.row() == this_rank.row()) {
332 for (SizeType j_local = 0; j_local < k_local_col; ++j_local) {
333 const LocalTileIndex kk_local_panel{Coord::Row, k_local_row};
334 const LocalTileIndex kj_local{k_local_row, j_local};
335 trsm_row_panel_tile<backend>(thread_priority::high, diag, col_panel.read(kk_local_panel),
336 mat_a.readwrite(kj_local));
337 }
338 }
339
340 col_panel.reset();
341 row_panel.reset();
342
343 if (kk_rank == this_rank) {
344 if constexpr (replace_trtri<D>()) {
345 row_panel.setRange(kk, {k + 1, k + 1});
346 const LocalTileIndex kk_local_panel{Coord::Col, k_local_col};
347 inverse_diag_tile_workspace<backend>(thread_priority::high, diag, mat_a.readwrite(kk),
348 row_panel.readwrite(kk_local_panel));
349 row_panel.reset();
350 }
351 else {
352 inverse_diag_tile<backend>(thread_priority::high, diag, mat_a.readwrite(kk));
353 }
354 }
355 }
356
357#ifdef DLAF_WITH_HDF5
358 if (getTuneParameters().debug_dump_triangular_inverse_data) {
359 file->write(mat_a, "/triangular_inverse");
360 }
361
362 num_triangular_inv_calls++;
363#endif
364}
