220template <class T, Device D>
221SizeType Matrix<T, D>::compute_ld(SizeType rows, LdSpec ld) noexcept {
222 SizeType min_ld = std::max<SizeType>(1, rows);
223 SizeType ret = 0;
224
225 if (std::holds_alternative<Ld>(ld)) {
226 switch (std::get<Ld>(ld)) {
227 case Ld::Compact:
228 ret = min_ld;
229 break;
230 case Ld::Padded: {
231 SizeType alignment;
232 if constexpr (D == Device::CPU)
233 alignment = std::max<SizeType>(1, 64 / sizeof(T));
234 else
235 alignment = 32;
236
237 ret = (min_ld < alignment ? min_ld : util::ceilDiv(min_ld, alignment) * alignment);
238 break;
239 }
240 default:
241 DLAF_UNIMPLEMENTED("Invalid Ld");
242 }
243 }
244 else
245 ret = std::get<SizeType>(ld);
246
247 DLAF_ASSERT(ret >= min_ld, ret, min_ld);
248 return ret;
249}

251template <class T, Device D>
252void Matrix<T, D>::set_up_non_preallocated_tiles(MatrixAllocation alloc, LdSpec ld) noexcept {
253 using MemView = memory::MemoryView<T, D>;
254 const Distribution& dist = this->distribution();
255
256 const auto& local_nr_tiles = dist.local_nr_tiles();
257 const auto& local_size = dist.local_size();
258
259 if (local_size.isEmpty()) {
260 // Needed to check if ld has a valid value.
261 compute_ld(0, ld);
262 return;
263 }
264
265 DLAF_ASSERT(tile_managers_.empty(), "");
266 tile_managers_.reserve(to_sizet(local_nr_tiles.linear_size()));
267
268 if (alloc == MatrixAllocation::ColMajor) {
269 SizeType mat_ld = compute_ld(local_size.rows(), ld);
270
271 ColMajorLayout layout(dist, mat_ld);
272 MemView mem(layout.min_mem_size());
273
274 for (SizeType j = 0; j < local_nr_tiles.cols(); ++j) {
275 for (SizeType i = 0; i < local_nr_tiles.rows(); ++i) {
276 LocalTileIndex ij(i, j);
277 SizeType offset = layout.tile_offset(ij);
278 TileElementSize tile_size = layout.tile_size_of(ij);
279 SizeType view_size = layout.min_tile_mem_size(ij);
280 DLAF_ASSERT_HEAVY(!tile_size.isEmpty(), tile_size);
281 tile_managers_.emplace_back(TileDataType(tile_size, MemView(mem, offset, view_size), mat_ld));
282 }
283 }
284 }
285 else if (alloc == MatrixAllocation::Blocks) {
286 // To be able to reuse tile methods for blocks we create a distribution with a single tile per block
287 Distribution helper_dist = internal::create_single_tile_per_block_distribution(dist);
288
289 LocalBlockSize local_nr_blocks = helper_dist.local_nr_blocks();
290 DLAF_ASSERT(dist.local_nr_blocks() == local_nr_blocks, dist.local_nr_blocks(), local_nr_blocks);
291
292 common::internal::vector<MemView> mem_views;
293 mem_views.reserve(local_nr_blocks.rows());
294
295 // Notation for the next loops:
296 // size of the current block: mb x nb
297 // size of the current tile: mt x nt
298 // Loop over the columns of blocks and then loop over all the columns of tiles needed to fill the block.
299 for (SizeType j_bl = 0, j = 0; j_bl < local_nr_blocks.cols(); ++j_bl) {
300 SizeType nb = helper_dist.local_tile_size_of<Coord::Col>(j_bl);
301
302 for (SizeType j_el_bl = 0; j_el_bl < nb;) {
303 SizeType nt = dist.local_tile_size_of<Coord::Col>(j);
304 // Loop over the block rows and then loop over all the rows of tiles needed to fill the block.
305 for (SizeType i_bl = 0, i = 0; i_bl < local_nr_blocks.rows(); ++i_bl) {
306 SizeType mb = helper_dist.local_tile_size_of<Coord::Row>(i_bl);
307 SizeType block_ld = compute_ld(mb, ld);
308 if (j_el_bl == 0) {
309 // Create a MemView for each block in the block column.
310 mem_views.emplace_back(block_ld * nb);
311 }
312 for (SizeType i_el_bl = 0; i_el_bl < mb;) {
313 // Create the tile using the correct MemView
314 SizeType mt = dist.local_tile_size_of<Coord::Row>(i);
315 DLAF_ASSERT_HEAVY(!(TileElementSize{mt, nt}.isEmpty()), mt, nt);
316 SizeType offset = i_el_bl + block_ld * j_el_bl;
317 SizeType view_size = (nt - 1) * block_ld + mt;
318 tile_managers_.emplace_back(
319 TileDataType({mt, nt}, MemView(mem_views[i_bl], offset, view_size), block_ld));
320 ++i;
321 i_el_bl += mt;
322 }
323 }
324 ++j;
325 j_el_bl += nt;
326 }
327 mem_views.clear();
328 }
329 }
330 else if (alloc == MatrixAllocation::Tiles) {
331 for (SizeType j = 0; j < local_nr_tiles.cols(); ++j) {
332 for (SizeType i = 0; i < local_nr_tiles.rows(); ++i) {
333 LocalTileIndex ij(i, j);
334 TileElementSize tile_size = dist.tile_size_of(ij);
335 DLAF_ASSERT_HEAVY(!tile_size.isEmpty(), tile_size);
336
337 SizeType tile_ld = compute_ld(tile_size.rows(), ld);
338 SizeType nr_elements = tile_ld * tile_size.cols();
339
340 tile_managers_.emplace_back(TileDataType(tile_size, MemView(nr_elements), tile_ld));
341 }
342 }
343 }
344 else {
345 DLAF_UNREACHABLE_PLAIN;
346 }

495template <class T, Device D>
496void Matrix<const T, D>::wait_local_tiles() noexcept {
497 // Note:
498 // Using a readwrite access to the tile ensures that the access is exclusive and not shared
499 // among multiple tasks.
500
501 const auto range_local = common::iterate_range2d(distribution().local_nr_tiles());
502
503 auto s = pika::execution::experimental::when_all_vector(internal::selectGeneric(
504 [this](const LocalTileIndex& index) {
505 return this->tile_managers_[tileLinearIndex(index)].readwrite();
506 },
507 range_local));
508 [[maybe_unused]] auto tiles = pika::this_thread::experimental::sync_wait(std::move(s));
509}

532template <class T, Device D>
533void Matrix<const T, D>::set_up_sub_pipelines(Matrix<const T, D>& mat) noexcept {
534 namespace ex = pika::execution::experimental;
535
536 // TODO: Optimize read-after-read. This is currently forced to access the base
537 // matrix in readwrite mode so that we can move the tile into the
538 // sub-pipeline. This is semantically not required and should eventually be
539 // optimized.
540 tile_managers_.reserve(mat.tile_managers_.size());
541 for (auto& tm : mat.tile_managers_) {
542 tile_managers_.emplace_back(Tile<T, D>());
543 auto s = ex::when_all(tile_managers_.back().readwrite_with_wrapper(), tm.readwrite()) |
544 ex::then([](internal::TileAsyncRwMutexReadWriteWrapper<T, D> empty_tile_wrapper,
545 Tile<T, D> tile) { empty_tile_wrapper.get() = std::move(tile); });
546 ex::start_detached(std::move(s));
547 }

550template <class T, Device D>
551template <template <class, Device> class MatrixLike>
552void Matrix<const T, D>::set_up_retiled_sub_pipelines(MatrixLike<const T, D>& mat,
553 const LocalTileSize& tiles_per_block) noexcept {
554 DLAF_ASSERT(mat.blockSize() == mat.tile_size(), mat.blockSize(), mat.tile_size());
555
556 using common::internal::vector;
557 namespace ex = pika::execution::experimental;
558
559 const auto n = to_sizet(distribution().local_nr_tiles().linear_size());
560 tile_managers_.reserve(n);
561 for (std::size_t i = 0; i < n; ++i) {
562 tile_managers_.emplace_back(Tile<T, D>());
563 }
564
565 const auto tile_size = distribution().tile_size();
566 vector<SubTileSpec> specs;
567 vector<LocalTileIndex> indices;
568 specs.reserve(tiles_per_block.linear_size());
569 indices.reserve(tiles_per_block.linear_size());
570
571 // TODO: Optimize read-after-read. This is currently forced to access the base matrix in readwrite
572 // mode so that we can move the tile into the sub-pipeline. This is semantically not required and
573 // should eventually be optimized.
574 for (const auto& orig_tile_index : common::iterate_range2d(mat.distribution().local_nr_tiles())) {
575 const auto original_tile_size = mat.tileSize(mat.distribution().global_tile_index(orig_tile_index));
576
577 for (SizeType j = 0; j < original_tile_size.cols(); j += tile_size.cols())
578 for (SizeType i = 0; i < original_tile_size.rows(); i += tile_size.rows()) {
579 indices.emplace_back(
580 LocalTileIndex{orig_tile_index.row() * tiles_per_block.rows() + i / tile_size.rows(),
581 orig_tile_index.col() * tiles_per_block.cols() + j / tile_size.cols()});
582 specs.emplace_back(SubTileSpec{{i, j},
583 tileSize(distribution().global_tile_index(indices.back()))});
584 }
585
586 auto sub_tiles = splitTileDisjoint(mat.readwrite(orig_tile_index), specs);
587
588 DLAF_ASSERT_HEAVY(specs.size() == indices.size(), specs.size(), indices.size());
589 for (SizeType j = 0; j < specs.size(); ++j) {
590 const auto i = tile_linear_index(indices[j]);
591
592 // Move subtile to be managed by the tile manager of RetiledMatrix. We
593 // use readwrite_with_wrapper to get access to the original tile managed
594 // by the underlying async_rw_mutex.
595 auto s =
596 ex::when_all(tile_managers_[i].readwrite_with_wrapper(), std::move(sub_tiles[to_sizet(j)])) |
597 ex::then([](internal::TileAsyncRwMutexReadWriteWrapper<T, D> empty_tile_wrapper,
598 Tile<T, D> sub_tile) { empty_tile_wrapper.get() = std::move(sub_tile); });
599 ex::start_detached(std::move(s));
600 }
601
602 specs.clear();
603 indices.clear();
604 }
