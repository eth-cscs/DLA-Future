51template <Backend B, Device D, class T>
52void generalMatrix(const blas::Op opA, const blas::Op opB, const T alpha, MatrixRef<const T, D>& mat_a,
53 MatrixRef<const T, D>& mat_b, const T beta, MatrixRef<T, D>& mat_c) {
54 DLAF_ASSERT(matrix::local_matrix(mat_a), mat_a);
55 DLAF_ASSERT(matrix::local_matrix(mat_b), mat_b);
56 DLAF_ASSERT(matrix::local_matrix(mat_c), mat_c);
57
58 DLAF_ASSERT_HEAVY(matrix::multipliable(mat_a, mat_b, mat_c, blas::Op::NoTrans, blas::Op::NoTrans),
59 mat_a, mat_b, mat_c);
60
61 if (opA == blas::Op::NoTrans && opB == blas::Op::NoTrans)
62 internal::General<B, D, T>::callNN(alpha, mat_a, mat_b, beta, mat_c);
63 else
64 DLAF_UNIMPLEMENTED(opA, opB);
65}
