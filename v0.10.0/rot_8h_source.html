67template <Device D, class T, class CommSender>
68auto scheduleSendCol(CommSender&& comm, const comm::IndexT_MPI dest, const comm::IndexT_MPI tag,
69 const T* col_data, const SizeType n) {
70 namespace di = dlaf::internal;
71
72 if constexpr (D == Device::CPU) {
73 return di::whenAllLift(std::forward<CommSender>(comm), dest, tag, col_data, n) |
74 dlaf::comm::internal::transformMPI(sendCol<D, T>);
75 }
76#ifdef DLAF_WITH_GPU
77 else if constexpr (D == Device::GPU) {
78 namespace ex = pika::execution::experimental;
79 using dlaf::matrix::internal::CopyBackend_v;
80 using pika::execution::thread_priority;
81 using pika::execution::thread_stacksize;
82
83 return ex::just(memory::MemoryView<T, Device::CPU>{n}) |
84 ex::let_value([comm = std::forward<CommSender>(comm), dest, tag, col_data,
85 n](memory::MemoryView<T, Device::CPU>& mem_view) mutable {
86 auto copy =
87 ex::just(mem_view(), col_data, to_sizet(n) * sizeof(T), whip::memcpy_device_to_host) |
88 di::transform(
89 di::Policy<CopyBackend_v<Device::GPU, Device::CPU>>{thread_priority::high,
90 thread_stacksize::nostack},
91 whip::memcpy_async);
92
93 return di::whenAllLift(std::move(copy), std::forward<CommSender>(comm), dest, tag,
94 mem_view(), n) |
95 dlaf::comm::internal::transformMPI(sendCol<Device::CPU, T>);
96 });
97 }
98#endif
99 else {
100 DLAF_STATIC_UNIMPLEMENTED(T);
101 }
102}

140template <class T>
141struct GivensRotation {
142 SizeType i; // the first column index
143 SizeType j; // the second column index
144 T c; // cosine
145 T s; // sine
146};
147
148// Apply GivenRotations to tiles of the square sub-matrix identified by tile in range [i_begin, i_end).
149//
150// @param i_begin global tile index for both row and column identifying the start of the sub-matrix
151// @param i_end global tile index for both row and column identifying the end of the sub-matrix
152// @param rots GivenRotations to apply (element column indices of rotations are relative to the sub-matrix)
153// @param mat matrix where the sub-matrix is located
154//
155// @pre mat is not distributed
156// @pre memory layout of @p mat is column major.
157template <class T, Device D, class RotsSender>
158void applyGivensRotationsToMatrixColumns(const SizeType i_begin, const SizeType i_end, RotsSender&& rots,
159 Matrix<T, D>& mat) {
160 // Note:
161 // a column index may be paired to more than one other index, this may lead to a race
162 // condition if parallelized trivially. Current implementation is serial.
163
164 namespace ex = pika::execution::experimental;
165 namespace di = dlaf::internal;
166
167 const SizeType n = problemSize(i_begin, i_end, mat.distribution());
168 const SizeType nb = mat.distribution().tile_size().rows();
169
170 auto givens_rots_fn = [n, nb](const auto& rots, const auto& tiles, [[maybe_unused]] auto&&... ts) {
171 // Distribution of the merged subproblems
172 matrix::Distribution distr(LocalElementSize(n, n), TileElementSize(nb, nb));
173
174 common::internal::SingleThreadedBlasScope single;
175
176 for (const GivensRotation<T>& rot : rots) {
177 // Get the index of the tile that has column `rot.i` and the index of the column within the tile.
178 const SizeType i_tile = distr.globalTileLinearIndex(GlobalElementIndex(0, rot.i));
179 const SizeType i_el = distr.tileElementFromGlobalElement<Coord::Col>(rot.i);
180 T* x = tiles[to_sizet(i_tile)].ptr(TileElementIndex(0, i_el));
181
182 // Get the index of the tile that has column `rot.j` and the index of the column within the tile.
183 const SizeType j_tile = distr.globalTileLinearIndex(GlobalElementIndex(0, rot.j));
184 const SizeType j_el = distr.tileElementFromGlobalElement<Coord::Col>(rot.j);
185 T* y = tiles[to_sizet(j_tile)].ptr(TileElementIndex(0, j_el));
186
187 // Apply Givens rotations
188 if constexpr (D == Device::CPU) {
189 blas::rot(n, x, 1, y, 1, rot.c, rot.s);
190 }
191 else {
192 givensRotationOnDevice(n, x, y, rot.c, rot.s, ts...);
193 }
194 }
195 };
196
197 TileCollector tc{i_begin, i_end};
198
199 auto sender = ex::when_all(std::forward<RotsSender>(rots), ex::when_all_vector(tc.readwrite(mat)));
200 di::transformDetach(di::Policy<DefaultBackend_v<D>>(pika::execution::thread_stacksize::nostack),
201 std::move(givens_rots_fn), std::move(sender));
202}

217template <class T, Device D, class GRSender>
218void applyGivensRotationsToMatrixColumns(
219 comm::CommunicatorPipeline<comm::CommunicatorType::Row>& comm_row_chain, const comm::IndexT_MPI tag,
220 const SizeType i_begin, const SizeType i_end, GRSender&& rots_fut, Matrix<T, D>& mat) {
221 // Note:
222 // a column index may be paired to more than one other index, this may lead to a race
223 // condition if parallelized trivially. Current implementation is serial.
224
225 namespace ex = pika::execution::experimental;
226 namespace tt = pika::this_thread::experimental;
227 namespace di = dlaf::internal;
228 using pika::execution::thread_stacksize;
229
230 DLAF_ASSERT_HEAVY(comm_row_chain.size_2d().cols() == mat.commGridSize().cols(),
231 comm_row_chain.size_2d().cols(), mat.commGridSize().cols());
232 DLAF_ASSERT_HEAVY(comm_row_chain.rank_2d().col() == mat.rankIndex().col(),
233 comm_row_chain.rank_2d().col(), mat.rankIndex().col());
234
235 const matrix::Distribution& dist = mat.distribution();
236
237 const SizeType mb = dist.tile_size().rows();
238 const SizeType range_size_limit = std::min(dist.size().rows(), i_end * mb);
239 const SizeType range_size = range_size_limit - i_begin * mb;
240
241 // Note:
242 // Some ranks might not participate to the application of given rotations. This logic checks which
243 // ranks are involved in order to operate in the range [i_begin, i_end).
244 const bool isInRangeRow = [&]() {
245 const SizeType begin = dist.nextLocalTileFromGlobalTile<Coord::Row>(i_begin);
246 const SizeType end = dist.nextLocalTileFromGlobalTile<Coord::Row>(i_end);
247 return end - begin != 0;
248 }();
249
250 const bool isInRangeCol = [&]() {
251 const SizeType begin = dist.nextLocalTileFromGlobalTile<Coord::Col>(i_begin);
252 const SizeType end = dist.nextLocalTileFromGlobalTile<Coord::Col>(i_end);
253 return end - begin != 0;
254 }();
255
256 const bool isInRange = isInRangeRow && isInRangeCol;
257 if (!isInRange)
258 return;
259
260 // Note:
261 // This is the distribution that will be used inside the task. Differently from the original one,
262 // this targets just the range defined by [i_begin, i_end), but keeping the same distribution over
263 // ranks.
264 const matrix::Distribution dist_sub({range_size, range_size}, dist.tile_size(), dist.commGridSize(),
265 dist.rankIndex(), dist.rankGlobalTile({i_begin, i_begin}));
266
267 auto givens_rots_fn = [comm_row_chain = comm_row_chain.sub_pipeline(), tag, dist_sub,
268 mb](std::vector<GivensRotation<T>> rots, std::vector<matrix::Tile<T, D>> tiles,
269 std::vector<matrix::Tile<T, D>> all_ws) mutable {
270 // Note:
271 // It would have been enough to just get the first tile from the beginning, and it would have
272 // worked anyway (thanks to the fact that panel has its own memorychunk and the first tile would
273 // keep alive the entire chunk, so also the part of other tiles, through its memory view).
274 // Anyway, it is more clean to get them all, and then just here use a single one to access the
275 // full column.
276 matrix::Tile<T, D>& tile_ws = all_ws[0];
277
278 // Note:
279 // The entire algorithm relies on a strong assumption about memory layout of all tiles involved,
280 // i.e. both the input tiles selected from the matrix and the tiles for the workspace.
281 // By relying on the fact that all of them are stored in a column-layout, we can work on column
282 // vectors ignoring their distribution over different tiles, and it would be enough just getting
283 // the pointer to the top head of the column and all other data can be easily accessed since it
284 // is stored contiguously after that.
285 // Ignoring tile organization of the memory comes handy also for communication. Indeed, during
286 // column exchange, just a single MPI operation per column is issued, instead of communicating
287 // independently the parts of column from each tile.
288 auto getColPtr = [&dist_sub, &tiles, &tile_ws](const SizeType col_index, const bool hasIt) -> T* {
289 if (hasIt) {
290 const LocalTileIndex tile_col{dist_sub.nextLocalTileFromGlobalElement<Coord::Row>(0),
291 dist_sub.nextLocalTileFromGlobalElement<Coord::Col>(col_index)};
292 const std::size_t linear_tile_col = to_sizet(dist_sub.localTileLinearIndex(tile_col));
293 return tiles[linear_tile_col].ptr(dist_sub.tileElementIndex({0, col_index}));
294 }
295 return tile_ws.ptr({0, 0});
296 };
297
298 const SizeType m = dist_sub.localSize().rows();
299
300 for (const GivensRotation<T>& rot : rots) {
301 const SizeType j_x = rot.i / mb;
302 const SizeType j_y = rot.j / mb;
303
304 const comm::IndexT_MPI rankColX = dist_sub.template rankGlobalTile<Coord::Col>(j_x);
305 const comm::IndexT_MPI rankColY = dist_sub.template rankGlobalTile<Coord::Col>(j_y);
306
307 const bool hasX = dist_sub.rankIndex().col() == rankColX;
308 const bool hasY = dist_sub.rankIndex().col() == rankColY;
309
310 if (!hasX && !hasY)
311 continue;
312
313 T* col_x = getColPtr(rot.i, hasX);
314 T* col_y = getColPtr(rot.j, hasY);
315
316 const bool hasBothXY = hasX && hasY;
317
318 std::vector<ex::unique_any_sender<>> comm_checkpoints;
319 if (!hasBothXY) {
320 const comm::IndexT_MPI rank_partner = hasX ? rankColY : rankColX;
321
322 const T* col_send = hasX ? col_x : col_y;
323 T* col_recv = hasX ? col_y : col_x;
324
325 // Note:
326 // These communications use raw pointers, so correct lifetime management of related tiles
327 // is up to the caller.
328 comm_checkpoints.emplace_back(wrapper::scheduleSendCol<D, T>(comm_row_chain.shared(),
329 rank_partner, tag, col_send, m));
330 comm_checkpoints.emplace_back(wrapper::scheduleRecvCol<D, T>(comm_row_chain.shared(),
331 rank_partner, tag, col_recv, m));
332 }
333
334 // Note:
335 // With a single workspace, just one rotation per time can be done.
336 //
337 // Communications of a step can happen together, since the only available workspace is used
338 // only for receiving the counterpart column. When communications are finished, the rank has
339 // everything needed to compute the rotation, so the actual computation task can start.
340 //
341 // However, communications of multiple steps, even with multiple workspaces, might need to be
342 // kept serialized. Indeed, if rotations are not independent, i.e. if the same column appears
343 // in multiple rotations, the second time the column is communicated, it must be the one rotated
344 // resulting from the previous step, not the original one.
345 // For simplicity, we add the constraint that communications of a step depends on previous step
346 // computation, which might be too tight, but currently it looks like the most straightforward
347 // solution.
348 //
349 // Moreover, even if rotations are independent, scheduling all communications all together
350 // would require a tag ad hoc ensuring that communication between same ranks do not get mixed
351 // (in addition to having a tag ensuring that other calls to this algorithm do not get mixed too).
352 //
353 // Current serialization of the algorithm ensures that communications cannot be scheduled all
354 // together beforehand, and with them, also temporary buffers for the GPU case. If this assumption
355 // will drop, it is relevant to highlight that there is nothing that would stop to schedule and,
356 // more importantly, allocate all of them together.
357 tt::sync_wait(ex::when_all_vector(std::move(comm_checkpoints)) |
358 di::transform(di::Policy<DefaultBackend_v<D>>(thread_stacksize::nostack),
359 [rot, m, col_x, col_y](auto&&... ts) {
360 // Note:
361 // each one computes his own, but just stores either x or y (or both
362 // if on the same rank)
363 if constexpr (D == Device::CPU) {
364 static_assert(sizeof...(ts) == 0,
365 "Parameter pack should be empty for MC.");
366 dlaf::common::internal::SingleThreadedBlasScope single;
367 blas::rot(m, col_x, 1, col_y, 1, rot.c, rot.s);
368 }
369#ifdef DLAF_WITH_GPU
370 else if constexpr (D == Device::GPU) {
371 givensRotationOnDevice(m, col_x, col_y, rot.c, rot.s, ts...);
372 }
373#endif
374 else {
375 DLAF_STATIC_UNIMPLEMENTED(T);
376 }
377 }));
378 }
379
380 comm_row_chain.reset();
381 };
382
383 // Note:
384 // Using a combination of shrinked distribution and an offset given to the panel, the workspace
385 // is allocated just for the part strictly needed by the range [i_begin, i_end)
386 const matrix::Distribution dist_ws({range_size_limit, 1}, dist.tile_size(), dist.commGridSize(),
387 dist.rankIndex(), dist.sourceRankIndex());
388 matrix::Panel<Coord::Col, T, D> workspace(dist_ws, GlobalTileIndex(i_begin, i_begin));
389
390 const TileCollector tc(i_begin, i_end);
391
392 ex::when_all(std::forward<GRSender>(rots_fut), ex::when_all_vector(tc.readwrite(mat)),
393 ex::when_all_vector(select(workspace, workspace.iteratorLocal()))) |
394 di::transformDetach(di::Policy<Backend::MC>(), std::move(givens_rots_fn));
395}
