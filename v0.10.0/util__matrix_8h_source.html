123template <class IndexT, class Tag>
124bool multipliable_sizes(common::Size2D<IndexT, Tag> a, common::Size2D<IndexT, Tag> b,
125 common::Size2D<IndexT, Tag> c, const blas::Op opA, const blas::Op opB) noexcept {
126 if (opA != blas::Op::NoTrans)
127 a.transpose();
128 if (opB != blas::Op::NoTrans)
129 b.transpose();
130
131 return a.rows() == c.rows() && a.cols() == b.rows() && b.cols() == c.cols();
132}

135template <template <class, Device> class MatrixLikeA, template <class, Device> class MatrixLikeB,
136 template <class, Device> class MatrixLikeC, class T, Device D>
137bool multipliable(const MatrixLikeA<const T, D>& a, const MatrixLikeB<const T, D>& b,
138 const MatrixLikeC<const T, D>& c, const blas::Op opA, const blas::Op opB) noexcept {
139 const bool isSizeOk = multipliable_sizes(a.size(), b.size(), c.size(), opA, opB);
140
141 if (a.size().isEmpty() || c.size().isEmpty())
142 return isSizeOk;
143
144 const bool allSameGrid = same_process_grid(a, b) && same_process_grid(b, c);
145 const bool isTileSizeOk = multipliable_sizes(a.tile_size(), b.tile_size(), c.tile_size(), opA, opB);
146 const bool isOffsetOk = multipliable_sizes(a.tile_size_of({0, 0}), b.tile_size_of({0, 0}),
147 c.tile_size_of({0, 0}), opA, opB);
148
149 if (local_matrix(c))
150 return allSameGrid && isSizeOk && isTileSizeOk && isOffsetOk;
151
152 // TODO distributed (fix offset)
153 return allSameGrid && isSizeOk && isTileSizeOk && isOffsetOk;
154}

160template <class T>
161class getter_random {
162 static_assert(std::is_same_v<T, float> || std::is_same_v<T, double>,
163 "T is not compatible with random generator used.");
164
165public:
166 getter_random() : random_engine_(std::minstd_rand::default_seed) {}
167 getter_random(SizeType seed) : random_engine_(static_cast<std::size_t>(seed)) {
168 DLAF_ASSERT(seed >= 0, "");
169 }
170
171 T operator()() {
172 return random_sampler_(random_engine_);
173 }
174
175private:
176 std::mt19937_64 random_engine_;
177 std::uniform_real_distribution<T> random_sampler_{-1, 1};
178};

181template <class T>
182class getter_random<std::complex<T>> : private getter_random<T> {
183public:
184 using getter_random<T>::getter_random;
185
186 std::complex<T> operator()() {
187 return std::polar<T>(std::abs(getter_random<T>::operator()()),
188 static_cast<T>(M_PI) * getter_random<T>::operator()());
189 }

235template <class T, class ElementGetter>
236void set(Matrix<T, Device::CPU>& matrix, ElementGetter el_f) {
237 using pika::execution::thread_stacksize;
238
239 const Distribution& dist = matrix.distribution();
240 for (auto ij_lc : iterate_range2d(dist.local_nr_tiles())) {
241 GlobalTileIndex ij = dist.global_tile_index(ij_lc);
242 auto tile_origin = dist.global_element_index(ij, {0, 0});
243
244 using TileType = typename std::decay_t<decltype(matrix)>::TileType;
245 auto set_f = [tile_origin, el_f = el_f](const TileType& tile) {
246 for (auto ij_el_tl : iterate_range2d(tile.size())) {
247 GlobalElementIndex ij_el_gl(tile_origin.row() + ij_el_tl.row(),
248 tile_origin.col() + ij_el_tl.col());
249 tile(ij_el_tl) = el_f(ij_el_gl);
250 }
251 };
252
253 dlaf::internal::transformDetach(dlaf::internal::Policy<Backend::MC>(thread_stacksize::nostack),
254 std::move(set_f), matrix.readwrite(ij_lc));
255 }

267template <class T, class ElementGetter>
268void set(Matrix<T, Device::CPU>& matrix, ElementGetter el_f, const blas::Op op) {
269 auto el_op_f = [op, el_f](const GlobalElementIndex& index) -> T {
270 using blas::Op;
271 switch (op) {
272 case Op::NoTrans:
273 return el_f(index);
274 case Op::Trans:
275 return el_f(transposed(index));
276 case Op::ConjTrans:
277 return dlaf::conj(el_f(transposed(index)));
278 default:
279 DLAF_UNIMPLEMENTED(op);
280 return T{};
281 }
282 };
283
284 set(matrix, el_op_f);
285}

309template <class T>
310void set_random(Matrix<T, Device::CPU>& matrix) {
311 using pika::execution::thread_stacksize;
312
313 const Distribution& dist = matrix.distribution();
314 for (auto ij_lc : iterate_range2d(dist.local_nr_tiles())) {
315 GlobalTileIndex ij = dist.global_tile_index(ij_lc);
316 auto tile_origin = dist.global_element_index(ij, {0, 0});
317 auto seed = tile_origin.col() + tile_origin.row() * matrix.size().cols();
318
319 using TileType = typename std::decay_t<decltype(matrix)>::TileType;
320 auto rnd_f = [seed](TileType&& tile) {
321 internal::getter_random<T> random_value(seed);
322 for (auto ij_el_tl : iterate_range2d(tile.size())) {
323 tile(ij_el_tl) = random_value();
324 }
325 };
326
327 dlaf::internal::transformDetach(dlaf::internal::Policy<Backend::MC>(thread_stacksize::nostack),
328 std::move(rnd_f), matrix.readwrite(ij_lc));
329 }

334template <class T>
335void set_diagonal_tile(const Tile<T, Device::CPU>& tile, internal::getter_random<T>& random_value,
336 SizeType offset_value) {
337 // DIAGONAL
338 // for diagonal tiles get just lower matrix values and set value for both
339 // straight and transposed indices
340 for (SizeType j = 0; j < tile.size().cols(); ++j) {
341 for (SizeType i = 0; i < j; ++i) {
342 auto value = random_value();
343
344 tile(TileElementIndex{i, j}) = value;
345 tile(TileElementIndex{j, i}) = dlaf::conj(value);
346 }
347 tile(TileElementIndex{j, j}) = std::real(random_value()) + offset_value;
348 }
349}

351template <class T>
352void set_lower_and_upper_tile(const Tile<T, Device::CPU>& tile, internal::getter_random<T>& random_value,
353 TileElementSize full_tile_size, GlobalTileIndex ij,
354 dlaf::matrix::Distribution dist,
355 std::optional<SizeType> band_size = std::nullopt) {
356 auto is_off_band = [](GlobalElementIndex ij_lc, std::optional<SizeType> band_size) {
357 return band_size ? ij_lc.col() < ij_lc.row() - *band_size || ij_lc.col() > ij_lc.row() + *band_size
358 : false;
359 };
360
361 // LOWER or UPPER (except DIAGONAL)
362 // random values are requested in the same order for both original and transposed
363 for (SizeType j = 0; j < full_tile_size.cols(); ++j) {
364 for (SizeType i = 0; i < full_tile_size.rows(); ++i) {
365 auto value = random_value();
366
367 // but they are set row-wise in the original tile and col-wise in the
368 // transposed one
369 if (ij.row() > ij.col()) { // LOWER
370 TileElementIndex index{i, j};
371 if (index.isIn(tile.size())) {
372 auto ij_lc = dist.globalElementIndex(ij, index);
373 if (is_off_band(ij_lc, band_size))
374 tile(index) = T(0);
375 else
376 tile(index) = value;
377 }
378 }
379 else { // UPPER
380 TileElementIndex index{j, i};
381 if (index.isIn(tile.size())) {
382 auto ij_lc = dist.globalElementIndex(ij, index);
383 if (is_off_band(ij_lc, band_size))
384 tile(index) = T(0);
385 else
386 tile(index) = dlaf::conj(value);
387 }
388 }
389 }
390 }

409template <class T>
410void set_random_hermitian_with_offset(Matrix<T, Device::CPU>& matrix, const SizeType offset_value,
411 std::optional<SizeType> band_size = std::nullopt) {
412 using pika::execution::thread_stacksize;
413
414 // note:
415 // By assuming square tile_sizes, it is easier to locate elements. In fact:
416 // - Elements on the diagonal are stored in the diagonal of the diagonal tiles
417 // - Tiles under the diagonal store elements of the lower triangular matrix
418 // - Tiles over the diagonal store elements of the upper triangular matrix
419
420 const Distribution& dist = matrix.distribution();
421
422 DLAF_ASSERT(square_size(matrix), matrix);
423 DLAF_ASSERT(square_tile_size(matrix), matrix);
424
425 auto full_tile_size = matrix.tile_size();
426
427 for (auto ij_lc : iterate_range2d(dist.local_nr_tiles())) {
428 GlobalTileIndex ij = dist.global_tile_index(ij_lc);
429
430 auto tile_origin = dist.global_element_index(ij, {0, 0});
431
432 // compute the same seed for original and "transposed" tiles, so transposed ones will know the
433 // values of the original one without the need of accessing real values (nor communication in case
434 // of distributed matrices)
435 SizeType seed;
436 if (ij.row() >= ij.col()) // LOWER or DIAGONAL
437 seed = tile_origin.col() + tile_origin.row() * matrix.size().cols();
438 else
439 seed = tile_origin.row() + tile_origin.col() * matrix.size().rows();
440
441 using TileType = typename std::decay_t<decltype(matrix)>::TileType;
442 auto set_hp_f = [=](const TileType& tile) {
443 internal::getter_random<T> random_value(seed);
444 if (ij.row() == ij.col())
445 internal::set_diagonal_tile(tile, random_value, offset_value);
446 else
447 internal::set_lower_and_upper_tile(tile, random_value, full_tile_size, ij, dist, band_size);
448 };
449
450 dlaf::internal::transformDetach(dlaf::internal::Policy<Backend::MC>(thread_stacksize::nostack),
451 std::move(set_hp_f), matrix.readwrite(ij_lc));
452 }

549template <class T>
550void set_random_non_zero_diagonal(Matrix<T, Device::CPU>& matrix) {
551 using pika::execution::thread_stacksize;
552
553 // note:
554 // By assuming square tile_sizes, it is easier to locate elements. In fact:
555 // - Elements on the diagonal are stored in the diagonal of the diagonal tiles
556 // - Tiles under the diagonal store elements of the lower triangular matrix
557 // - Tiles over the diagonal store elements of the upper triangular matrix
558
559 const Distribution& dist = matrix.distribution();
560
561 DLAF_ASSERT(square_size(matrix), matrix);
562 DLAF_ASSERT(square_tile_size(matrix), matrix);
563
564 for (auto ij_lc : iterate_range2d(dist.local_nr_tiles())) {
565 GlobalTileIndex ij = dist.global_tile_index(ij_lc);
566
567 const SizeType seed = ij.col() + ij.row() * matrix.size().cols();
568 const BaseType<T> lower_limit = static_cast<BaseType<T>>(0.1);
569
570 using TileType = typename std::decay_t<decltype(matrix)>::TileType;
571 auto set_hp_f = [=](const TileType& tile) {
572 internal::getter_random<T> random_value(seed);
573 if (ij.row() == ij.col()) {
574 for (auto ij_el_tl : iterate_range2d(tile.size())) {
575 auto value = random_value();
576 if (ij_el_tl.row() == ij_el_tl.col() && std::abs(value) < lower_limit) {
577 if (value == T{0})
578 tile(ij_el_tl) = lower_limit;
579 else
580 tile(ij_el_tl) = lower_limit * value / std::abs(value);
581 }
582 else
583 tile(ij_el_tl) = value;
584 }
585 }
586 else
587 for (auto ij_el_tl : iterate_range2d(tile.size())) {
588 tile(ij_el_tl) = random_value();
589 }
590 };
591
592 dlaf::internal::transformDetach(dlaf::internal::Policy<Backend::MC>(thread_stacksize::nostack),
593 std::move(set_hp_f), matrix.readwrite(ij_lc));
594 }
595}
