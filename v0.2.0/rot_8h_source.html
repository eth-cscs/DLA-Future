61 template <Device D, class T, class CommSender>
62 auto scheduleSendCol(CommSender&& comm, const comm::IndexT_MPI dest, const comm::IndexT_MPI tag,
63  const T* col_data, const SizeType n) {
64  namespace di = dlaf::internal;
65 
66  if constexpr (D == Device::CPU) {
67  return di::whenAllLift(std::forward<CommSender>(comm), dest, tag, col_data, n) |
68  dlaf::comm::internal::transformMPI(sendCol<D, T>);
69  }
70 #ifdef DLAF_WITH_GPU
71  else if constexpr (D == Device::GPU) {
72  namespace ex = pika::execution::experimental;
73  using dlaf::matrix::internal::CopyBackend_v;
74  using pika::execution::thread_priority;
75 
76  return ex::just(memory::MemoryView<T, Device::CPU>{n}) |
77  ex::let_value([comm = std::forward<CommSender>(comm), dest, tag, col_data,
78  n](memory::MemoryView<T, Device::CPU>& mem_view) mutable {
79  auto copy =
80  ex::just(mem_view(), col_data, to_sizet(n) * sizeof(T), whip::memcpy_device_to_host) |
81  di::transform(di::Policy<CopyBackend_v<Device::GPU, Device::CPU>>{thread_priority::high},
82  whip::memcpy_async);
83 
84  return di::whenAllLift(std::move(copy), std::forward<CommSender>(comm), dest, tag,
85  mem_view(), n) |
86  dlaf::comm::internal::transformMPI(sendCol<Device::CPU, T>);
87  });
88  }
89 #endif
90  else {
91  DLAF_STATIC_UNIMPLEMENTED(T);
92  }

139 // Apply GivenRotations to tiles of the square sub-matrix identified by tile in range [i_begin, i_end).
140 //
141 // @param i_begin global tile index for both row and column identifying the start of the sub-matrix
142 // @param i_end global tile index for both row and column identifying the end of the sub-matrix
143 // @param rots GivenRotations to apply (element column indices of rotations are relative to the sub-matrix)
144 // @param mat matrix where the sub-matrix is located
145 //
146 // @pre mat is not distributed
147 // @pre memory layout of @p mat is column major.
148 template <class T, Device D, class RotsSender>
149 void applyGivensRotationsToMatrixColumns(const SizeType i_begin, const SizeType i_end, RotsSender&& rots,
150  Matrix<T, D>& mat) {
151  // Note:
152  // a column index may be paired to more than one other index, this may lead to a race
153  // condition if parallelized trivially. Current implementation is serial.
154 
155  namespace ex = pika::execution::experimental;
156  namespace di = dlaf::internal;
157 
158  const SizeType n = problemSize(i_begin, i_end, mat.distribution());
159  const SizeType nb = mat.distribution().blockSize().rows();
160 
161  auto givens_rots_fn = [n, nb](const auto& rots, const auto& tiles, [[maybe_unused]] auto&&... ts) {
162  // Distribution of the merged subproblems
163  matrix::Distribution distr(LocalElementSize(n, n), TileElementSize(nb, nb));
164 
165  common::internal::SingleThreadedBlasScope single;
166 
167  for (const GivensRotation<T>& rot : rots) {
168  // Get the index of the tile that has column `rot.i` and the the index of the column within the tile.
169  const SizeType i_tile = distr.globalTileLinearIndex(GlobalElementIndex(0, rot.i));
170  const SizeType i_el = distr.tileElementFromGlobalElement<Coord::Col>(rot.i);
171  T* x = tiles[to_sizet(i_tile)].ptr(TileElementIndex(0, i_el));
172 
173  // Get the index of the tile that has column `rot.j` and the the index of the column within the tile.
174  const SizeType j_tile = distr.globalTileLinearIndex(GlobalElementIndex(0, rot.j));
175  const SizeType j_el = distr.tileElementFromGlobalElement<Coord::Col>(rot.j);
176  T* y = tiles[to_sizet(j_tile)].ptr(TileElementIndex(0, j_el));
177 
178  // Apply Givens rotations
179  if constexpr (D == Device::CPU) {
180  blas::rot(n, x, 1, y, 1, rot.c, rot.s);
181  }
182  else {
183  givensRotationOnDevice(n, x, y, rot.c, rot.s, ts...);
184  }
185  }
186  };
187 
188  TileCollector tc{i_begin, i_end};
189 
190  auto sender = ex::when_all(std::forward<RotsSender>(rots), ex::when_all_vector(tc.readwrite(mat)));
191  di::transformDetach(di::Policy<DefaultBackend_v<D>>(), std::move(givens_rots_fn), std::move(sender));
192 }

207 template <class T, Device D, class GRSender>
208 void applyGivensRotationsToMatrixColumns(comm::Communicator comm_row, const comm::IndexT_MPI tag,
209  const SizeType i_begin, const SizeType i_end,
210  GRSender&& rots_fut, Matrix<T, D>& mat) {
211  // Note:
212  // a column index may be paired to more than one other index, this may lead to a race
213  // condition if parallelized trivially. Current implementation is serial.
214 
215  namespace ex = pika::execution::experimental;
216  namespace tt = pika::this_thread::experimental;
217  namespace di = dlaf::internal;
218 
219  DLAF_ASSERT_HEAVY(comm_row.size() == mat.commGridSize().cols(), comm_row.size(),
220  mat.commGridSize().cols());
221  DLAF_ASSERT_HEAVY(comm_row.rank() == mat.rankIndex().col(), comm_row.rank(), mat.rankIndex().col());
222 
223  const matrix::Distribution& dist = mat.distribution();
224 
225  const SizeType mb = dist.blockSize().rows();
226  const SizeType range_size_limit = std::min(dist.size().rows(), i_end * mb);
227  const SizeType range_size = range_size_limit - i_begin * mb;
228 
229  // Note:
230  // Some ranks might not participate to the application of given rotations. This logic checks which
231  // ranks are involved in order to operate in the range [i_begin, i_end).
232  const bool isInRangeRow = [&]() {
233  const SizeType begin = dist.nextLocalTileFromGlobalTile<Coord::Row>(i_begin);
234  const SizeType end = dist.nextLocalTileFromGlobalTile<Coord::Row>(i_end);
235  return end - begin != 0;
236  }();
237 
238  const bool isInRangeCol = [&]() {
239  const SizeType begin = dist.nextLocalTileFromGlobalTile<Coord::Col>(i_begin);
240  const SizeType end = dist.nextLocalTileFromGlobalTile<Coord::Col>(i_end);
241  return end - begin != 0;
242  }();
243 
244  const bool isInRange = isInRangeRow && isInRangeCol;
245  if (!isInRange)
246  return;
247 
248  // Note:
249  // This is the distribution that will be used inside the task. Differently from the original one,
250  // this targets just the range defined by [i_begin, i_end), but keeping the same distribution over
251  // ranks.
252  const matrix::Distribution dist_sub({range_size, range_size}, dist.blockSize(), dist.commGridSize(),
253  dist.rankIndex(), dist.rankGlobalTile({i_begin, i_begin}));
254 
255  auto givens_rots_fn = [comm_row, tag, dist_sub, mb](std::vector<GivensRotation<T>> rots,
256  std::vector<matrix::Tile<T, D>> tiles,
257  std::vector<matrix::Tile<T, D>> all_ws) {
258  // Note:
259  // It would have been enough to just get the first tile from the beginning, and it would have
260  // worked anyway (thanks to the fact that panel has its own memorychunk and the first tile would
261  // keep alive the entire chunk, so also the part of other tiles, through its memory view).
262  // Anyway, it is more clean to get them all, and then just here use a single one to access the
263  // full column.
264  matrix::Tile<T, D>& tile_ws = all_ws[0];
265 
266  // Note:
267  // The entire algorithm relies on a strong assumption about memory layout of all tiles involved,
268  // i.e. both the input tiles selected from the matrix and the tiles for the workspace.
269  // By relying on the fact that all of them are stored in a column-layout, we can work on column
270  // vectors ignoring their distribution over different tiles, and it would be enough just getting
271  // the pointer to the top head of the column and all other data can be easily accessed since it
272  // is stored contiguously after that.
273  // Ignoring tile organization of the memory comes handy also for communication. Indeed, during
274  // column exchange, just a single MPI operation per column is issued, instead of communicating
275  // independently the parts of column from each tile.
276  auto getColPtr = [&dist_sub, &tiles, &tile_ws](const SizeType col_index, const bool hasIt) -> T* {
277  if (hasIt) {
278  const LocalTileIndex tile_col{dist_sub.nextLocalTileFromGlobalElement<Coord::Row>(0),
279  dist_sub.nextLocalTileFromGlobalElement<Coord::Col>(col_index)};
280  const std::size_t linear_tile_col = to_sizet(dist_sub.localTileLinearIndex(tile_col));
281  return tiles[linear_tile_col].ptr(dist_sub.tileElementIndex({0, col_index}));
282  }
283  return tile_ws.ptr({0, 0});
284  };
285 
286  const SizeType m = dist_sub.localSize().rows();
287 
288  for (const GivensRotation<T>& rot : rots) {
289  const SizeType j_x = rot.i / mb;
290  const SizeType j_y = rot.j / mb;
291 
292  const comm::IndexT_MPI rankColX = dist_sub.template rankGlobalTile<Coord::Col>(j_x);
293  const comm::IndexT_MPI rankColY = dist_sub.template rankGlobalTile<Coord::Col>(j_y);
294 
295  const bool hasX = dist_sub.rankIndex().col() == rankColX;
296  const bool hasY = dist_sub.rankIndex().col() == rankColY;
297 
298  if (!hasX && !hasY)
299  continue;
300 
301  T* col_x = getColPtr(rot.i, hasX);
302  T* col_y = getColPtr(rot.j, hasY);
303 
304  const bool hasBothXY = hasX && hasY;
305 
306  std::vector<ex::unique_any_sender<>> comm_checkpoints;
307  if (!hasBothXY) {
308  const comm::IndexT_MPI rank_partner = hasX ? rankColY : rankColX;
309 
310  const T* col_send = hasX ? col_x : col_y;
311  T* col_recv = hasX ? col_y : col_x;
312 
313  // Note:
314  // These communications use raw pointers, so correct lifetime management of related tiles
315  // is up to the caller.
316  comm_checkpoints.emplace_back(wrapper::scheduleSendCol<D, T>(comm_row, rank_partner, tag,
317  col_send, m));
318  comm_checkpoints.emplace_back(wrapper::scheduleRecvCol<D, T>(comm_row, rank_partner, tag,
319  col_recv, m));
320  }
321 
322  // Note:
323  // With a single workspace, just one rotation per time can be done.
324  //
325  // Communications of a step can happen together, since the only available workspace is used
326  // only for receiving the counterpart column. When communications are finished, the rank has
327  // everything needed to compute the rotation, so the actual computation task can start.
328  //
329  // However, communications of multiple steps, even with multiple workspaces, might need to be
330  // kept serialized. Indeed, if rotations are not independent, i.e. if the same column appears
331  // in multiple rotations, the second time the column is communicated, it must be the one rotated
332  // resulting from the previous step, not the original one.
333  // For simplicity, we add the constraint that communications of a step depends on previous step
334  // computation, which might be too tight, but currently it looks like the most straightforward
335  // solution.
336  //
337  // Moreover, even if rotations are independent, scheduling all communications all together
338  // would require a tag ad hoc ensuring that communication between same ranks do not get mixed
339  // (in addition to having a tag ensuring that other calls to this algorithm do not get mixed too).
340  //
341  // Current serialization of the algorithm ensures that communications cannot be scheduled all
342  // together beforehand, and with them, also temporary buffers for the GPU case. If this assumption
343  // will drop, it is relevant to highlight that there is nothing that would stop to schedule and,
344  // more importantly, allocate all of them together.
345  tt::sync_wait(ex::when_all_vector(std::move(comm_checkpoints)) |
346  di::transform(di::Policy<DefaultBackend_v<D>>(), [rot, m, col_x,
347  col_y](auto&&... ts) {
348  // Note:
349  // each one computes his own, but just stores either x or y (or both if on the same rank)
350  if constexpr (D == Device::CPU) {
351  static_assert(sizeof...(ts) == 0, "Parameter pack should be empty for MC.");
352  dlaf::common::internal::SingleThreadedBlasScope single;
353  blas::rot(m, col_x, 1, col_y, 1, rot.c, rot.s);
354  }
355 #ifdef DLAF_WITH_GPU
356  else if constexpr (D == Device::GPU) {
357  givensRotationOnDevice(m, col_x, col_y, rot.c, rot.s, ts...);
358  }
359 #endif
360  else {
361  DLAF_STATIC_UNIMPLEMENTED(T);
362  }
363  }));
364  }
365  };
366 
367  // Note:
368  // Using a combination of shrinked distribution and an offset given to the panel, the workspace
369  // is allocated just for the part strictly needed by the range [i_begin, i_end)
370  const matrix::Distribution dist_ws({range_size_limit, 1}, dist.blockSize(), dist.commGridSize(),
371  dist.rankIndex(), dist.sourceRankIndex());
372  matrix::Panel<Coord::Col, T, D> workspace(dist_ws, GlobalTileIndex(i_begin, i_begin));
373 
374  const TileCollector tc(i_begin, i_end);
375 
376  ex::when_all(std::forward<GRSender>(rots_fut), ex::when_all_vector(tc.readwrite(mat)),
377  ex::when_all_vector(select(workspace, workspace.iteratorLocal()))) |
378  di::transformDetach(di::Policy<Backend::MC>(), givens_rots_fn);
379 }
