30 template <class DataIn, class DataOut>
31 void reduceRecv(Communicator& communicator, MPI_Op reduce_op, const DataIn input, const DataOut output) {
32  using T = std::remove_const_t<typename common::data_traits<DataIn>::element_t>;
33 
34  // Wayout for single rank communicator, just copy data
35  if (communicator.size() == 1) {
36  DLAF_ASSERT_MODERATE(input != output, "input and output should not equal (use in-place)");
37  common::copy(input, output);
38  return;
39  }
40 
41  // Buffers not allocated, just placeholders in case we need to allocate them
42  common::Buffer<T> buffer_in, buffer_out;
43 
44  auto message_input = comm::make_message(make_contiguous(input, buffer_in));
45  auto message_output = comm::make_message(make_contiguous(output, buffer_out));
46 
47  DLAF_ASSERT_MODERATE((buffer_in || buffer_out) || (input != output),
48  "input and output should not equal (use in-place)");
49 
50  // if the input buffer has been used, initialize it with input values
51  if (buffer_in)
52  common::copy(input, buffer_in);
53 
54  DLAF_MPI_CHECK_ERROR(MPI_Reduce(message_input.data(), message_output.data(), message_input.count(),
55  message_input.mpi_type(), reduce_op, communicator.rank(),
56  communicator));
57 
58  // if the output buffer has been used, copy-back output values
59  if (buffer_out)
60  common::copy(buffer_out, output);
61 }

133 template <class DataIn, class DataOut>
134 void reduce(const IndexT_MPI rank_root, Communicator& communicator, MPI_Op reduce_op, const DataIn input,
135  const DataOut output) {
136  DLAF_ASSERT(0 <= rank_root && rank_root < communicator.size() && rank_root != MPI_UNDEFINED, rank_root,
137  communicator.size());
138 
139  if (rank_root == communicator.rank())
140  reduceRecv(communicator, reduce_op, input, output);
141  else
142  reduceSend(rank_root, communicator, reduce_op, input);
143 }
