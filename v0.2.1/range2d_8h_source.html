179  IterableRange2D(index2d_t begin_idx, index2d_t end_idx)
180  : begin_idx_(begin_idx), ld_(end_idx.row() - begin_idx.row()),
181  i_max_(ld_ * (end_idx.col() - begin_idx_.col())) {
182  DLAF_ASSERT(begin_idx.row() <= end_idx.row(), begin_idx, end_idx);
183  DLAF_ASSERT(begin_idx.col() <= end_idx.col(), begin_idx, end_idx);
