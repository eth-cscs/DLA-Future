114 template <Coord rc>
115 SizeType global_tile_from_local_tile_on_rank(const Distribution& dist, comm::IndexT_MPI rank,
116  SizeType local_tile) noexcept {
117  using util::matrix::global_tile_from_local_tile;
118  // Assertion 0 <= local_tile is performed by global_tile_from_local_tile
119  const SizeType tiles_per_block = dist.block_size().get<rc>() / dist.tile_size().get<rc>();
120  const SizeType global_tile =
121  global_tile_from_local_tile(local_tile, tiles_per_block, dist.grid_size().get<rc>(), rank,
122  dist.source_rank_index().get<rc>(), global_tile_offset<rc>(dist));
123  // Assert on the result to avoid to compute the local number of tiles on the given rank
124  DLAF_ASSERT_HEAVY(0 <= global_tile && global_tile < dist.nr_tiles().get<rc>(), global_tile,
125  dist.nr_tiles().get<rc>());

168 inline bool is_compatible_sub_distribution(const Distribution& distribution,
169  const GlobalElementIndex& sub_offset,
170  const Distribution& sub_distribution) noexcept {
171  const bool compatible_grid = distribution.block_size() == sub_distribution.block_size() &&
172  distribution.tile_size() == sub_distribution.tile_size() &&
173  distribution.rank_index() == sub_distribution.rank_index() &&
174  distribution.grid_size() == sub_distribution.grid_size();
175  const bool compatible_source_rank_index =
176  distribution.rank_global_tile(distribution.global_tile_index(sub_offset)) ==
177  sub_distribution.source_rank_index();
178  const bool compatible_size =
179  sub_offset.row() + sub_distribution.size().rows() <= distribution.size().rows() &&
180  sub_offset.col() + sub_distribution.size().cols() <= distribution.size().cols();
181  return compatible_grid && compatible_source_rank_index && compatible_size;
182 }

239  SizeType i_end) noexcept {

252 template <Coord rc>
253 SizeType local_element_distance_from_local_tile(const Distribution& dist, SizeType i_loc_begin,
254  SizeType i_loc_end) noexcept {
255  DLAF_ASSERT_HEAVY(i_loc_begin <= i_loc_end, i_loc_begin, i_loc_end);
256  DLAF_ASSERT_HEAVY(0 <= i_loc_begin && i_loc_end <= dist.local_nr_tiles().get<rc>(), i_loc_begin,
257  i_loc_end, dist.local_nr_tiles().get<rc>());
258 
259  SizeType i_loc_last = i_loc_end - 1;
260  if (i_loc_begin > i_loc_last)
261  return 0;
262 
263  SizeType i_el_begin = dist.local_element_from_local_tile_and_tile_element<rc>(i_loc_begin, 0);
264  SizeType i_el_end =
265  dist.local_element_from_local_tile_and_tile_element<rc>(
266  i_loc_last, dist.tile_size_of<rc>(dist.global_tile_from_local_tile<rc>(i_loc_last)) - 1) +
267  1;
268  return i_el_end - i_el_begin;
