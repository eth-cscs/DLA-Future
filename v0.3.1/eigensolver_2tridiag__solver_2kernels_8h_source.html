51 // Cuppen's decomposition
52 //
53 // Substracts the offdiagonal element at the split from the top and bottom diagonal elements and
54 // returns the offdiagonal element. The split is between the last row of the top tile and the first
55 // row of the bottom tile.
56 //
57 template <class T>
58 T cuppensDecomp(const matrix::Tile<T, Device::CPU>& top, const matrix::Tile<T, Device::CPU>& bottom) {
59  TileElementIndex offdiag_idx{top.size().rows() - 1, 1};
60  TileElementIndex top_idx{top.size().rows() - 1, 0};
61  TileElementIndex bottom_idx{0, 0};
62  const T offdiag_val = top(offdiag_idx);
63 
64  // Refence: Lapack working notes: LAWN 69, Serial Cuppen algorithm, Chapter 3
65  //
66  top(top_idx) -= std::abs(offdiag_val);
67  bottom(bottom_idx) -= std::abs(offdiag_val);
68  return offdiag_val;
69 }

98 template <class T>
99 void assembleRank1UpdateVectorTile(bool is_top_tile, T rho,
100  const matrix::Tile<const T, Device::CPU>& evecs_tile,
101  const matrix::Tile<T, Device::CPU>& rank1_tile) {
102  const SizeType m = evecs_tile.size().rows();
103  const SizeType n = evecs_tile.size().cols();
104  DLAF_ASSERT_HEAVY(n == rank1_tile.size().rows(), n, rank1_tile.size().rows());
105 
106  // Copy the bottom row of the top tile or the top row of the bottom tile
107  SizeType row = (is_top_tile) ? m - 1 : 0;
108 
109  // Negate the last row of the top eigenvector subproblem matrix (Q1) if rho < 0
110  //
111  // lapack 3.10.0, dlaed2.f, line 280 and 281
112  int sign = (is_top_tile && rho < 0) ? -1 : 1;
113 
114  for (SizeType i = 0; i < n; ++i) {
115  rank1_tile(TileElementIndex(i, 0)) = sign * evecs_tile(TileElementIndex(row, i)) / T(std::sqrt(2));
116  }
117 }
