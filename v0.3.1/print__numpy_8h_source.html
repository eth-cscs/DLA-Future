67 template <class T>
68 void print(format::numpy, const Tile<const T, Device::CPU>& tile, std::ostream& os = std::cout) {
69  os << "np.array([";
70 
71  // Note:
72  // iterate_range2d loops over indices in column-major order, while python default
73  // order is row-major.
74  // For this reason, values are printed in a column-major order, and reordering is deferred
75  // to python by tranposing the resulting array (and shaping it accordingly)
76  for (const auto& index : iterate_range2d(tile.size()))
77  os << internal::numpy_value(tile(index)) << ",";
78 
79  os << "]"
80  << ", dtype=np." << internal::numpy_datatype<T>::typestring << ")"
81  << ".reshape" << transposed(tile.size()) << ".T\n";
82 }

86  std::ostream& os = std::cout) {

94  const LocalTileSize local_tiles = distribution.localNrTiles();

97  return {
98  distribution.template globalElementFromLocalTileAndTileElement<Coord::Row>(local.row(), 0),
99  distribution.template globalElementFromLocalTileAndTileElement<Coord::Col>(local.col(), 0),
100  };
101  };
