34  const dlaf::matrix::Tile<const T, D>& b,
35  [[maybe_unused]] const dlaf::matrix::Tile<T, D>& c) {
36  SizeType m;
37  SizeType n;
38  SizeType k;
39 
40  if (op_a == blas::Op::NoTrans) {

42  k = a.size().cols();
43  }
44  else {
45  m = a.size().cols();
46  k = a.size().rows();
47  }
48 
49  [[maybe_unused]] SizeType k2;
50  if (op_b == blas::Op::NoTrans) {
51  k2 = b.size().rows();

53  }
54  else {
55  k2 = b.size().cols();
56  n = b.size().rows();
57  }
58 
59  DLAF_ASSERT(m == c.size().rows(), op_a, a, c);
60  DLAF_ASSERT(n == c.size().cols(), op_b, b, c);
61  DLAF_ASSERT(k == k2, op_a, a, op_b, b);
62 
63  return {m, n, k};
