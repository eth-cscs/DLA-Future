91 Pipeline sub_pipeline() {
92 namespace ex = pika::execution::experimental;
93
94 // Move value from pipeline into sub pipeline, then store a sender of the wrapper of the pipeline in
95 // a sender which we will release when the sub pipeline is released. This ensures that all accesses
96 // to the sub pipeline happen after previous accesses and before later accesses to the pipeline.
97 Pipeline sub_pipeline(T{});
98 sub_pipeline.nested_sender =
99 ex::when_all(sub_pipeline.pipeline->readwrite(), this->pipeline->readwrite()) |
100 ex::then([](auto sub_wrapper, auto wrapper) {
101 sub_wrapper.get() = std::move(wrapper.get());
102
103 return wrapper;
104 }) |
105 ex::ensure_started();
106
107 return sub_pipeline;
108 }

114 return pipeline.has_value();
115 }

120 void reset() noexcept {
121 release_parent_pipeline();

125private:
126 void release_parent_pipeline() {
127 namespace ex = pika::execution::experimental;
128
129 if (nested_sender) {
130 DLAF_ASSERT(valid(), "");
131
132 auto s =
133 ex::when_all(pipeline->readwrite(), std::move(nested_sender.value())) |
134 ex::then([](auto sub_wrapper, auto wrapper) { wrapper.get() = std::move(sub_wrapper.get()); });
135 ex::start_detached(std::move(s));
136 nested_sender.reset();
137 }
138 }
