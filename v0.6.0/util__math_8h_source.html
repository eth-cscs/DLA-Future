25// Interleaves two intervals of length `l` split in blocks of size `b` starting at offsets `o1` and
26// `o2` respectively and returns an array of indices where the splits have occured.
27//
28// o1
29// │
30// └► │ │ │ │ │
31// ─┴───┴───┴───┴───┴─── ◄─┐
32// ▲ │
33// o2 b ─┬──┘ l
34// │ │ │
35// └─► │ │ ▼ │ │ │ │
36// ──┴───┴───┴───┴───┴── ◄─┘
37//
38inline std::vector<SizeType> interleaveSplits(SizeType l, SizeType b, SizeType o1, SizeType o2) {
39 DLAF_ASSERT(l > 0, l);
40 DLAF_ASSERT(b > 0, b);
41 DLAF_ASSERT(o1 >= 0 && o1 <= b, o1, b);
42 DLAF_ASSERT(o2 >= 0 && o2 <= b, o2, b);
43
44 // Set small and big from offsets o1 and o2 s.t small <= big
45 SizeType small = o1;
46 SizeType big = o2;
47 if (small > big)
48 std::swap(small, big);
49
50 // Reserve enough memory for array of splits
51 std::vector<SizeType> splits;
52 splits.reserve(2 * to_sizet(l / b) + 2);
53
54 splits.push_back(0);
55 for (SizeType i = small, j = big; i < l || j < l; i += b, j += b) {
56 if (splits.back() != i && i < l)
57 splits.push_back(i);
58 if (splits.back() != j && j < l)
59 splits.push_back(j);
60 }
61 splits.push_back(l);
62 return splits;
63}
