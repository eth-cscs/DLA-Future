61 matrix::Matrix<const T, Device::CPU>& mat_hh) {

63 DLAF_ASSERT(matrix::local_matrix(mat_hh), mat_hh);

70
71 DLAF_ASSERT(matrix::single_tile_per_block(mat_e), mat_e);
72 DLAF_ASSERT(matrix::single_tile_per_block(mat_hh), mat_hh);
73
74 DLAF_ASSERT(band_size >= 2, band_size);
75 DLAF_ASSERT(mat_hh.blockSize().rows() % band_size == 0, mat_hh.blockSize(), band_size);
