44template <Backend backend, Device device, class T>
45void bt_reduction_to_band(const SizeType b, MatrixRef<T, device>& mat_c, Matrix<const T, device>& mat_v,
46 Matrix<const T, Device::CPU>& mat_taus) {
47 DLAF_ASSERT(matrix::local_matrix(mat_c), mat_c);
48 DLAF_ASSERT(matrix::local_matrix(mat_v), mat_v);
49 DLAF_ASSERT(square_size(mat_v), mat_v);
50 DLAF_ASSERT(square_blocksize(mat_v), mat_v);
51 DLAF_ASSERT(mat_c.size().rows() == mat_v.size().rows(), mat_c, mat_v);
52 DLAF_ASSERT(mat_c.blockSize().rows() == mat_v.blockSize().rows(), mat_c, mat_v);
53 DLAF_ASSERT(matrix::single_tile_per_block(mat_c), mat_c);
54 DLAF_ASSERT(matrix::single_tile_per_block(mat_v), mat_v);
55
56 [[maybe_unused]] auto nr_reflectors_blocks = [&b, &mat_v]() {
57 const SizeType m = mat_v.size().rows();
58 const SizeType mb = mat_v.blockSize().rows();
59 return std::max<SizeType>(0, util::ceilDiv(m - b - 1, mb));
60 };
61 DLAF_ASSERT(mat_taus.nrTiles().rows() == nr_reflectors_blocks(), mat_taus.size().rows(), mat_v, b);
62
63 BackTransformationReductionToBand<backend, device, T>::call(b, mat_c, mat_v, mat_taus);
64}

85template <Backend backend, Device device, class T>
86void bt_reduction_to_band(comm::CommunicatorGrid& grid, const SizeType b, MatrixRef<T, device>& mat_c,
87 Matrix<const T, device>& mat_v, Matrix<const T, Device::CPU>& mat_taus) {
88 DLAF_ASSERT(matrix::equal_process_grid(mat_c, grid), mat_c, grid);
89 DLAF_ASSERT(matrix::equal_process_grid(mat_v, grid), mat_v, grid);
90 DLAF_ASSERT(square_size(mat_v), mat_v);
91 DLAF_ASSERT(square_blocksize(mat_v), mat_v);
92 DLAF_ASSERT(mat_c.size().rows() == mat_v.size().rows(), mat_c, mat_v);
93 DLAF_ASSERT(mat_c.blockSize().rows() == mat_v.blockSize().rows(), mat_c, mat_v);
94 DLAF_ASSERT(matrix::single_tile_per_block(mat_c), mat_c);
95 DLAF_ASSERT(matrix::single_tile_per_block(mat_v), mat_v);
96
97 [[maybe_unused]] auto nr_reflectors_blocks = [&b, &mat_v]() {
98 const SizeType m = mat_v.size().rows();
99 const SizeType mb = mat_v.blockSize().rows();
100 return mat_v.distribution().template nextLocalTileFromGlobalTile<Coord::Col>(
101 std::max<SizeType>(0, util::ceilDiv(m - b - 1, mb)));
102 };
103 DLAF_ASSERT(mat_taus.distribution().localNrTiles().rows() == nr_reflectors_blocks(), mat_taus, mat_v,
104 b);
105
106 BackTransformationReductionToBand<backend, device, T>::call(grid, b, mat_c, mat_v, mat_taus);
107}
