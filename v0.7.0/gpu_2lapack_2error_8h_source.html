26 const dlaf::common::internal::source_location& info) noexcept {
27#ifdef DLAF_WITH_CUDA
28 if (st != CUSOLVER_STATUS_SUCCESS) {
29 std::cout << "[CUSOLVER ERROR] " << info << std::endl
30 << cusolver::internal::getErrorString(st) << std::endl;
31 std::terminate();
32 }
33#elif defined(DLAF_WITH_HIP)
34 if (st != rocblas_status_success && st != rocblas_status_size_unchanged &&
35 st != rocblas_status_size_increased && st != rocblas_status_continue) {
36 std::cout << "[ROCSOLVER ERROR] " << info << std::endl
37 << rocsolver::internal::getErrorString(st) << std::endl;
38 std::terminate();
39 }
40#endif
41}
