35// helper function that identifies the owner of a transposed coordinate,
36// it returns both the component of the rank in the transposed dimension and
37// its global cross coordinate (i.e. row == col in the global frame of reference)
38template <Coord dst_coord>
39std::pair<SizeType, comm::IndexT_MPI> transposedOwner(const matrix::Distribution& dist,
40 const LocalTileIndex idx) {
41 const auto idx_cross = dist.template globalTileFromLocalTile<dst_coord>(idx.get(dst_coord));
42 const auto rank_owner = dist.template rankGlobalTile<orthogonal(dst_coord)>(idx_cross);
43 return std::make_pair(idx_cross, rank_owner);
44}

59template <class T, Device D, Coord axis, matrix::StoreTransposed storage,
60 class = std::enable_if_t<!std::is_const_v<T>>>
61void broadcast(comm::IndexT_MPI rank_root, matrix::Panel<axis, T, D, storage>& panel,
62 comm::CommunicatorPipeline<coord_to_communicator_type(orthogonal(axis))>& serial_comm) {
63 constexpr auto comm_coord = axis;
64
65 // do not schedule communication tasks if there is no reason to do so...
66 if (panel.parentDistribution().commGridSize().get(comm_coord) <= 1)
67 return;
68
69 const auto rank = panel.parentDistribution().rankIndex().get(comm_coord);
70
71 namespace ex = pika::execution::experimental;
72 for (const auto& index : panel.iteratorLocal()) {
73 if (rank == rank_root)
74 ex::start_detached(schedule_bcast_send(serial_comm.exclusive(), panel.read(index)));
75 else
76 ex::start_detached(schedule_bcast_recv(serial_comm.exclusive(), rank_root,
77 panel.readwrite(index)));
78 }
79}

123template <class T, Device D, Coord axis, matrix::StoreTransposed storage,
124 matrix::StoreTransposed storageT, class = std::enable_if_t<!std::is_const_v<T>>>
125void broadcast(comm::IndexT_MPI rank_root, matrix::Panel<axis, T, D, storage>& panel,
126 matrix::Panel<orthogonal(axis), T, D, storageT>& panelT,
127 comm::CommunicatorPipeline<comm::CommunicatorType::Row>& row_task_chain,
128 comm::CommunicatorPipeline<comm::CommunicatorType::Col>& col_task_chain) {
129 constexpr Coord axisT = orthogonal(axis);
130
131 constexpr Coord coord = std::decay_t<decltype(panel)>::coord;
132 constexpr Coord coordT = std::decay_t<decltype(panelT)>::coord;
133
134 // Note:
135 // Given a source panel, this communication pattern makes every rank access tiles of both the
136 // source panel and it's tranposed variant (just tile coordinates, data is not transposed).
137 // In particular, it does not give access to all the tiles, but just the ones of interest for
138 // each rank, i.e. the rows and columns of a distributed matrix that the ranks stores locally.
139 //
140 // This happens in two steps (for the sake of example, let's consider a column -> row broadcast,
141 // the opposite is dual):
142 //
143 // 1. broadcast the source panel to panel with the same shape on other ranks
144 // 2. populate the transposed destination panel
145 //
146 // Once the source panel is share by all ranks, the transposed panel can be easily populated,
147 // because each destination tile can be populated with data from the rank owning the diagonal one,
148 // the point of contact between the row and the column (row == col).
149 //
150 // If it is already available locally, the tile is not copied and it just gets "linked", in order
151 // to easily access it via panel coordinates with minimal (to null) overhead.
152 // For this reason, the destination panel will depend on the source panel (as the source panel
153 // may already depend on the matrix).
154
155 DLAF_ASSERT(panel.parentDistribution() == panelT.parentDistribution(),
156 "they must refer to the same matrix");
157
158 const auto& dist = panel.parentDistribution();
159
160 DLAF_ASSERT(square_size(dist), dist.size());
161 DLAF_ASSERT(square_blocksize(dist), dist.blockSize());
162
163 // Note:
164 // This algorithm allow to broadcast panel to panelT using as mirror the parent matrix main diagonal.
165 // This means that it is possible to broadcast panels with different axes just if their global offset
166 // lie on the diagonal.
167 DLAF_ASSERT(panel.rangeStart() == panelT.rangeStart(), panel.rangeStart(), panelT.rangeStart());
168 DLAF_ASSERT_MODERATE(panel.rangeEnd() == panelT.rangeEnd(), panel.rangeEnd(), panelT.rangeEnd());
169
170 // if no panel tiles, just skip it
171 if (panel.rangeStart() == panel.rangeEnd())
172 return;
173
174 // STEP 1
175 constexpr auto comm_dir_step1 = orthogonal(axis);
176 auto& chain_step1 = internal::get_taskchain<comm_dir_step1>(row_task_chain, col_task_chain);
177
178 broadcast(rank_root, panel, chain_step1);
179
180 // STEP 2
181 constexpr auto comm_dir_step2 = orthogonal(axisT);
182 constexpr auto comm_coord_step2 = axisT;
183
184 auto& chain_step2 = internal::get_taskchain<comm_dir_step2>(row_task_chain, col_task_chain);
185
186 const SizeType last_tile = std::max(panelT.rangeStart(), panelT.rangeEnd() - 1);
187 const auto owner = dist.template rankGlobalTile<coordT>(last_tile);
188 const auto range = dist.rankIndex().get(coordT) == owner
189 ? common::iterate_range2d(*panelT.iteratorLocal().begin(),
190 LocalTileIndex(coordT, panelT.rangeEndLocal() - 1, 1))
191 : panelT.iteratorLocal();
192
193 for (const auto& indexT : range) {
194 auto [index_diag, owner_diag] = internal::transposedOwner<coordT>(dist, indexT);
195
196 namespace ex = pika::execution::experimental;
197 if (dist.rankIndex().get(coord) == owner_diag) {
198 const auto index_diag_local = dist.template localTileFromGlobalTile<coord>(index_diag);
199 panelT.setTile(indexT, panel.read({coord, index_diag_local}));
200
201 if (dist.commGridSize().get(comm_coord_step2) > 1)
202 ex::start_detached(schedule_bcast_send(chain_step2.exclusive(), panelT.read(indexT)));
203 }
204 else {
205 if (dist.commGridSize().get(comm_coord_step2) > 1)
206 ex::start_detached(schedule_bcast_recv(chain_step2.exclusive(), owner_diag,
207 panelT.readwrite(indexT)));
208 }
209 }
210}

212}
