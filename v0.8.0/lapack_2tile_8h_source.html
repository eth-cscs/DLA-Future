69template <class T>
70void lacpy(const Tile<const T, Device::CPU>& a, const Tile<T, Device::CPU>& b) {
71 DLAF_ASSERT_MODERATE(a.size() == b.size(), a, b);
72
73 const SizeType m = a.size().rows();
74 const SizeType n = a.size().cols();
75
76 common::internal::SingleThreadedBlasScope single;
77 lapack::lacpy(blas::Uplo::General, m, n, a.ptr(), a.ld(), b.ptr(), b.ld());
78}

85template <class T>
86void lacpy(TileElementSize region, TileElementIndex in_idx, const Tile<const T, Device::CPU>& in,
87 TileElementIndex out_idx, const Tile<T, Device::CPU>& out) {
88 DLAF_ASSERT_MODERATE(in_idx.isIn(in.size() - region + TileElementSize(1, 1)),
89 "Region goes out of bounds for `in`!", region, in_idx, in);
90 DLAF_ASSERT_MODERATE(out_idx.isIn(out.size() - region + TileElementSize(1, 1)),
91 "Region goes out of bounds for `out`!", region, out_idx, out);
92
93 common::internal::SingleThreadedBlasScope single;
94 lapack::lacpy(blas::Uplo::General, region.rows(), region.cols(), in.ptr(in_idx), in.ld(),
95 out.ptr(out_idx), out.ld());
96}

317template <class T>
318dlaf::BaseType<T> lantr(const lapack::Norm norm, const blas::Uplo uplo, const blas::Diag diag,
319 const Tile<T, Device::CPU>& a) noexcept {
320 switch (uplo) {
321 case blas::Uplo::Lower:
322 DLAF_ASSERT(a.size().rows() >= a.size().cols(), a);
323 break;
324 case blas::Uplo::Upper:
325 DLAF_ASSERT(a.size().rows() <= a.size().cols(), a);
326 break;
327 case blas::Uplo::General:
328 DLAF_ASSERT(blas::Uplo::General == uplo, uplo);
329 break;
330 }
331 common::internal::SingleThreadedBlasScope single;
332 return lapack::lantr(norm, uplo, diag, a.size().rows(), a.size().cols(), a.ptr(), a.ld());
333}

349template <class T>
350void hegst(const int itype, const blas::Uplo uplo, const Tile<T, Device::CPU>& a,
351 const Tile<T, Device::CPU>& b) {
352 DLAF_ASSERT(square_size(a), a);
353 DLAF_ASSERT(square_size(b), b);
354 DLAF_ASSERT(a.size() == b.size(), a, b);
355 DLAF_ASSERT(itype >= 1 && itype <= 3, itype);
356
357 common::internal::SingleThreadedBlasScope single;
358 [[maybe_unused]] auto info =
359 lapack::hegst(itype, uplo, a.size().cols(), a.ptr(), a.ld(), b.ptr(), b.ld());
360
361 DLAF_ASSERT(info == 0, info);
362}

382template <class T>
383void stedc(const Tile<T, Device::CPU>& tridiag, const Tile<T, Device::CPU>& evecs) {
384 DLAF_ASSERT(square_size(evecs), evecs);
385 DLAF_ASSERT(tridiag.size().rows() == evecs.size().rows(), tridiag, evecs);
386 DLAF_ASSERT(tridiag.size().cols() == 2, tridiag);
387
388 // In lapackpp see `util.hh` and `job_comp2char()` and `enum class Job`
389 // Note that `lapack::Job::Vec` corresponds to `compz=I` in the LAPACK
390
391 // compz, n, D, E, Z, ldz
392 common::internal::SingleThreadedBlasScope single;
393 lapack::stedc(lapack::Job::Vec, evecs.size().rows(), tridiag.ptr(),
394 tridiag.ptr(TileElementIndex(0, 1)), evecs.ptr(), evecs.ld());
395}

452inline void extendROCSolverWorkspace(cusolverDnHandle_t handle,
453 ::dlaf::memory::MemoryView<std::byte, Device::GPU>&& workspace) {
454 whip::stream_t stream;
455 DLAF_GPUBLAS_CHECK_ERROR(cublasGetStream(handle, &stream));
456 auto f = [workspace = std::move(workspace)](whip::error_t status) { whip::check_error(status); };
457 pika::cuda::experimental::detail::add_event_callback(std::move(f), stream);
458}

489class CusolverInfo {
490#ifdef DLAF_WITH_CUDA
491 memory::MemoryView<T, Device::GPU> workspace_;
492#endif
493 memory::MemoryView<int, Device::GPU> info_;
494
495public:
496#ifdef DLAF_WITH_CUDA

498#endif
499 CusolverInfo() : info_(1) {}
500
501#ifdef DLAF_WITH_CUDA
502 T* workspace() {
503 return workspace_();
504 }
505#endif
506 int* info() {
507 return info_();
508 }

511template <class F, class T>
512void assertExtendInfo(F assertFunc, cusolverDnHandle_t handle, CusolverInfo<T>&& info) {
513 whip::stream_t stream;
514 DLAF_GPULAPACK_CHECK_ERROR(cusolverDnGetStream(handle, &stream));
515 assertFunc(stream, info.info());
516 // Extend info scope to the end of the kernel execution
517 auto extend_info = [info = std::move(info)](whip::error_t status) { whip::check_error(status); };
518 pika::cuda::experimental::detail::add_event_callback(std::move(extend_info), stream);
519}

551template <class T>
552void hegst(cusolverDnHandle_t handle, const int itype, const blas::Uplo uplo,
553 const matrix::Tile<T, Device::GPU>& a, const matrix::Tile<T, Device::GPU>& b) {
554 DLAF_ASSERT(square_size(a), a);
555 DLAF_ASSERT(square_size(b), b);
556 DLAF_ASSERT(a.size() == b.size(), a, b);
557 const auto n = a.size().rows();
558
559#ifdef DLAF_WITH_CUDA
560 int workspace_size;
561 internal::CusolverHegst<T>::callBufferSize(handle, itype, util::blasToCublas(uplo), to_int(n),
562 util::blasToCublasCast(a.ptr()), to_int(a.ld()),
563 util::blasToCublasCast(b.ptr()), to_int(b.ld()),
564 &workspace_size);
565 internal::CusolverInfo<T> info{std::max(1, workspace_size)};
566 internal::CusolverHegst<T>::call(handle, itype, util::blasToCublas(uplo), to_int(n),
567 util::blasToCublasCast(a.ptr()), to_int(a.ld()),
568 util::blasToCublasCast(b.ptr()), to_int(b.ld()),
569 util::blasToCublasCast(info.workspace()), info.info());
570
571 assertExtendInfo(dlaf::gpulapack::internal::assertInfoHegst, handle, std::move(info));
572#elif defined(DLAF_WITH_HIP)
573 internal::CusolverHegst<T>::call(handle, util::blasToRocblas(itype), util::blasToRocblas(uplo),
574 to_int(n), util::blasToRocblasCast(a.ptr()), to_int(a.ld()),
575 util::blasToRocblasCast(b.ptr()), to_int(b.ld()));
576#endif
577}
