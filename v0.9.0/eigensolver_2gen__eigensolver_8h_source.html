68 blas::Uplo uplo, Matrix<T, D>& mat_a, Matrix<T, D>& mat_b, const Factorization factorization,
69 const SizeType eigenvalues_index_begin, const SizeType eigenvalues_index_end) {
70 DLAF_ASSERT(matrix::local_matrix(mat_a), mat_a);
71 DLAF_ASSERT(matrix::local_matrix(mat_b), mat_b);
72 DLAF_ASSERT(matrix::square_size(mat_a), mat_a);

74 DLAF_ASSERT(matrix::square_size(mat_b), mat_b);
75 DLAF_ASSERT(matrix::square_blocksize(mat_b), mat_b);
76 DLAF_ASSERT(mat_a.size() == mat_b.size(), mat_a, mat_b);
77 DLAF_ASSERT(mat_a.blockSize() == mat_b.blockSize(), mat_a, mat_b);
78 DLAF_ASSERT(eigenvalues_index_begin == 0, eigenvalues_index_begin);
79 DLAF_ASSERT(eigenvalues_index_end >= eigenvalues_index_begin, eigenvalues_index_end,
80 eigenvalues_index_begin);
81 DLAF_ASSERT(eigenvalues_index_end <= mat_a.size().rows(), eigenvalues_index_end, mat_a.size().rows());
82
83 const SizeType size = mat_a.size().rows();
84
85 matrix::Matrix<BaseType<T>, D> eigenvalues(LocalElementSize(size, 1),

90 factorization, eigenvalues_index_begin,
