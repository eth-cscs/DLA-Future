49template <Backend backend, Device device, class T>
50void generalized_to_standard(blas::Uplo uplo, Matrix<T, device>& mat_a, Matrix<T, device>& mat_b) {
51 DLAF_ASSERT(matrix::square_size(mat_a), mat_a);
52 DLAF_ASSERT(matrix::square_blocksize(mat_a), mat_a);
53 DLAF_ASSERT(matrix::square_size(mat_b), mat_b);
54 DLAF_ASSERT(matrix::square_blocksize(mat_b), mat_b);
55 DLAF_ASSERT(mat_a.size() == mat_b.size(), mat_a, mat_b);
56 DLAF_ASSERT(mat_a.blockSize() == mat_b.blockSize(), mat_a, mat_b);
57 DLAF_ASSERT(matrix::single_tile_per_block(mat_a), mat_a);
58 DLAF_ASSERT(matrix::single_tile_per_block(mat_b), mat_b);
59 DLAF_ASSERT(matrix::local_matrix(mat_a), mat_a);
60 DLAF_ASSERT(matrix::local_matrix(mat_b), mat_b);
61
62 switch (uplo) {
63 case blas::Uplo::Lower:
64 GenToStd<backend, device, T>::call_L(mat_a, mat_b);
65 break;
66 case blas::Uplo::Upper:
67 GenToStd<backend, device, T>::call_U(mat_a, mat_b);
68 break;
69 case blas::Uplo::General:
70 DLAF_UNIMPLEMENTED(uplo);
71 break;
72 }
73}
