101 ReadOnlySenderType read(const GlobalTileIndex& index) {
102 DLAF_ASSERT(index.isIn(distribution().nrTiles()), index, distribution().nrTiles());
103
104 const auto parent_index(
105 mat_const_.distribution().globalTileIndexFromSubDistribution(origin_, distribution(), index));
106 auto tile_sender = mat_const_.read(parent_index);
107
108 const auto parent_dist = mat_const_.distribution();
109 const auto parent_tile_size = parent_dist.tileSize(parent_index);
110 const auto tile_size = tileSize(index);
111
112 // If the corresponding tile in the parent distribution is exactly the same
113 // size as the tile in the sub-distribution, we don't need to take a subtile
114 // and can return the tile sender directly. This avoids unnecessary wrapping.
115 if (parent_tile_size == tile_size) {
116 return tile_sender;
117 }
118
119 // Otherwise we have to extract a subtile from the tile in the parent
120 // distribution.
121 const auto ij_tile =
122 parent_dist.tileElementOffsetFromSubDistribution(origin_, distribution(), index);
123 return splitTile(std::move(tile_sender), SubTileSpec{ij_tile, tile_size});
124 }
