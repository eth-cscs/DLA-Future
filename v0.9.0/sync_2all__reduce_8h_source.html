31template <class DataIn, class DataOut>
32void allReduce(Communicator& communicator, MPI_Op reduce_operation, const DataIn input,
33 const DataOut output) {
34 using common::make_contiguous;
35
36 using T = std::remove_const_t<typename common::data_traits<DataIn>::element_t>;
37
38 // Wayout for single rank communicator, just copy data
39 if (communicator.size() == 1) {
40 DLAF_ASSERT_MODERATE(input != output, "input and output should not equal (use in-place)");
41 common::copy(input, output);
42 return;
43 }
44
45 // Buffers not allocated, just placeholders in case we need to allocate them
46 common::Buffer<T> buffer_in, buffer_out;
47
48 auto message_input = comm::make_message(make_contiguous(input, buffer_in));
49 auto message_output = comm::make_message(make_contiguous(output, buffer_out));
50
51 DLAF_ASSERT_MODERATE((buffer_in || buffer_out) || (input != output),
52 "input and output should not equal (use in-place)");
53
54 // if the input buffer has been used, initialize it with input values
55 if (buffer_in)
56 common::copy(input, buffer_in);
57
58 DLAF_MPI_CHECK_ERROR(MPI_Allreduce(message_input.data(), message_output.data(), message_input.count(),
59 message_input.mpi_type(), reduce_operation, communicator));
60
61 // if the output buffer has been used, copy-back output values
62 if (buffer_out)
63 common::copy(buffer_out, output);
64}
