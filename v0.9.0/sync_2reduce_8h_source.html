31template <class DataIn, class DataOut>
32void reduceRecv(Communicator& communicator, MPI_Op reduce_op, const DataIn input, const DataOut output) {
33 using T = std::remove_const_t<typename common::data_traits<DataIn>::element_t>;
34
35 // Wayout for single rank communicator, just copy data
36 if (communicator.size() == 1) {
37 DLAF_ASSERT_MODERATE(input != output, "input and output should not equal (use in-place)");
38 common::copy(input, output);
39 return;
40 }
41
42 // Buffers not allocated, just placeholders in case we need to allocate them
43 common::Buffer<T> buffer_in, buffer_out;
44
45 auto message_input = comm::make_message(make_contiguous(input, buffer_in));
46 auto message_output = comm::make_message(make_contiguous(output, buffer_out));
47
48 DLAF_ASSERT_MODERATE((buffer_in || buffer_out) || (input != output),
49 "input and output should not equal (use in-place)");
50
51 // if the input buffer has been used, initialize it with input values
52 if (buffer_in)
53 common::copy(input, buffer_in);
54
55 DLAF_MPI_CHECK_ERROR(MPI_Reduce(message_input.data(), message_output.data(), message_input.count(),
56 message_input.mpi_type(), reduce_op, communicator.rank(),
57 communicator));
58
59 // if the output buffer has been used, copy-back output values
60 if (buffer_out)
61 common::copy(buffer_out, output);
62}

134template <class DataIn, class DataOut>
135void reduce(const IndexT_MPI rank_root, Communicator& communicator, MPI_Op reduce_op, const DataIn input,
136 const DataOut output) {
137 DLAF_ASSERT(0 <= rank_root && rank_root < communicator.size() && rank_root != MPI_UNDEFINED, rank_root,
138 communicator.size());
139
140 if (rank_root == communicator.rank())
141 reduceRecv(communicator, reduce_op, input, output);
142 else
143 reduceSend(rank_root, communicator, reduce_op, input);
144}
