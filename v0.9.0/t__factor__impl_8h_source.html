55template <Backend B>
56auto split_tfactor_work(const std::size_t nrtiles) {
57 const std::size_t min_workers = 1;
58 const std::size_t available_workers = get_tfactor_num_workers<B>();
59 const std::size_t ideal_workers = util::ceilDiv(to_sizet(nrtiles), to_sizet(1));
60
61 struct {
62 std::size_t nworkers;
63 std::size_t batch_size;
64 } params;
65
66 params.batch_size = util::ceilDiv(nrtiles, std::clamp(ideal_workers, min_workers, available_workers));
67 DLAF_ASSERT_MODERATE(params.batch_size > 0, params.batch_size, nrtiles);
68 params.nworkers = std::max<std::size_t>(1, util::ceilDiv(nrtiles, params.batch_size));
69
70 return params;
71}

88 static matrix::ReadWriteTileSender<T, Device::CPU> step_gemv(
89 matrix::Panel<Coord::Col, T, Device::CPU>& hh_panel,
90 matrix::ReadOnlyTileSender<T, Device::CPU> taus,
91 matrix::ReadWriteTileSender<T, Device::CPU> tile_t,
92 matrix::Panel<Coord::Col, T, Device::CPU>& workspaces) {
93 namespace ex = pika::execution::experimental;
94 namespace di = dlaf::internal;
95
96 using pika::execution::thread_priority;
97
98 std::vector<matrix::ReadOnlyTileSender<T, Device::CPU>> hh_tiles =
99 selectRead(hh_panel, hh_panel.iteratorLocal());
100
101 const SizeType k = hh_panel.getWidth();
102
103 if (hh_tiles.size() == 0)
104 return tile_t;
105
106 const auto workers_params = split_tfactor_work<Backend::MC>(hh_tiles.size());
107 const std::size_t nworkers = workers_params.nworkers;
108 const std::size_t batch_size = workers_params.batch_size;
109
110 const auto range_workspaces = common::iterate_range2d(LocalTileSize{to_SizeType(nworkers - 1), 1});
111 DLAF_ASSERT(to_sizet(std::distance(range_workspaces.begin(), range_workspaces.end())) >=
112 nworkers - 1,
113 std::distance(range_workspaces.begin(), range_workspaces.end()), nworkers - 1);
114
115 const auto hp_scheduler = di::getBackendScheduler<Backend::MC>(thread_priority::high);
116 return ex::when_all(ex::when_all_vector(std::move(hh_tiles)), std::move(taus), std::move(tile_t),
117 ex::when_all_vector(select(workspaces, range_workspaces))) |
118 ex::continues_on(hp_scheduler) |
119 ex::let_value([hp_scheduler, k, nworkers, batch_size](auto& hh_tiles, auto& taus,
120 auto& tile_t, auto& workspaces) {
121 return ex::just(std::make_unique<pika::barrier<>>(nworkers)) |
122 ex::continues_on(hp_scheduler) |
123 ex::bulk(
124 nworkers,
125 [=, &hh_tiles, &taus, &tile_t, &workspaces](const std::size_t worker_id,
126 auto& barrier_ptr) mutable {
127 const auto barrier_busy_wait = getTFactorBarrierBusyWait();
128 DLAF_ASSERT_HEAVY(k == taus.get().size().rows(), k, taus.get().size().rows());
129
130 const std::size_t begin = worker_id * batch_size;
131 const std::size_t end =
132 std::min(worker_id * batch_size + batch_size, hh_tiles.size());
133
134 const matrix::Tile<T, Device::CPU> ws_worker =
135 (worker_id == 0 ? tile_t : workspaces[worker_id - 1])
136 .subTileReference({{0, 0}, tile_t.size()});
137
138 tile::internal::set0<T>(ws_worker);
139 lapack::lacpy(blas::Uplo::General, 1, k, taus.get().ptr(), 1, ws_worker.ptr(),
140 ws_worker.ld() + 1);
141
142 // make it work on worker_id section of tiles
143 for (std::size_t index = begin; index < end; ++index) {
144 const matrix::Tile<const T, Device::CPU>& tile_v = hh_tiles[index].get();
145 loop_gemv(tile_v, taus.get(), ws_worker);
146 }
147
148 barrier_ptr->arrive_and_wait(barrier_busy_wait);
149
150 // reduce ws_T in tile_t
151 if (worker_id == 0) {
152 for (std::size_t other_worker = 1; other_worker < nworkers; ++other_worker) {
153 matrix::Tile<T, Device::CPU> ws =
154 workspaces[other_worker - 1].subTileReference({{0, 0}, tile_t.size()});
155 tile::internal::add(T(1), ws, tile_t);
156 }
157 }
158 }) |
159 // Note: ignore the barrier sent by the bulk and just return tile_t
160 ex::then([&tile_t](auto) mutable { return std::move(tile_t); });
161 });
162 }

164 static auto step_copy_diag_and_trmv(matrix::ReadOnlyTileSender<T, Device::CPU> taus,
165 matrix::ReadWriteTileSender<T, Device::CPU> tile_t) noexcept {
166 auto tausdiag_trmvloop = [](const matrix::Tile<const T, Device::CPU>& taus,
167 matrix::Tile<T, Device::CPU> tile_t) {
168 common::internal::SingleThreadedBlasScope single;
169
170 const SizeType k = tile_t.size().cols();
171 lapack::lacpy(blas::Uplo::General, 1, k, taus.ptr(), 1, tile_t.ptr(), tile_t.ld() + 1);
172
173 loop_trmv(tile_t);
174 };
175
176 namespace di = dlaf::internal;
177 namespace ex = pika::execution::experimental;
178
179 return ex::when_all(std::move(taus), std::move(tile_t)) |
180 di::transform(di::Policy<Backend::MC>(pika::execution::thread_priority::high),
181 std::move(tausdiag_trmvloop));
182 }
183
184private:
185 static void loop_gemv(const matrix::Tile<const T, Device::CPU>& tile_v,
186 const matrix::Tile<const T, Device::CPU>& taus,
187 const matrix::Tile<T, Device::CPU>& tile_t) noexcept {
188 const SizeType k = tile_t.size().cols();
189
190 DLAF_ASSERT(tile_v.size().cols() == k, tile_v.size().cols(), k);
191 DLAF_ASSERT(taus.size().rows() == k, taus.size().rows(), k);
192
193 common::internal::SingleThreadedBlasScope single;
194 for (SizeType j = 0; j < k; ++j) {
195 // T(0:j, j) = -tau . V(j:, 0:j)* . V(j:, j)
196 // [j x 1] = [(n-j) x j]* . [(n-j) x 1]
197 const TileElementIndex t_start{0, j};
198 const TileElementIndex va_start{0, 0};
199 const TileElementIndex vb_start{0, j};
200 const TileElementSize va_size{tile_v.size().rows(), j};
201 const T tau = tile_t({j, j});
202
203 blas::gemv(blas::Layout::ColMajor, blas::Op::ConjTrans, va_size.rows(), va_size.cols(), -tau,
204 tile_v.ptr(va_start), tile_v.ld(), tile_v.ptr(vb_start), 1, 1, tile_t.ptr(t_start), 1);
205 }
206 }
207
208 static void loop_trmv(const matrix::Tile<T, Device::CPU>& tile_t) {
209 common::internal::SingleThreadedBlasScope single;
210
211 const SizeType k = tile_t.size().cols();
212
213 // Update each column (in order) t = T . t
214 // remember that T is upper triangular, so it is possible to use TRMV
215 for (SizeType j = 0; j < k; ++j) {
216 const TileElementIndex t_start{0, j};
217 const TileElementSize t_size{j, 1};
218
219 blas::trmv(blas::Layout::ColMajor, blas::Uplo::Upper, blas::Op::NoTrans, blas::Diag::NonUnit,
220 t_size.rows(), tile_t.ptr(), tile_t.ld(), tile_t.ptr(t_start), 1);
221 }
222 }

237 static matrix::ReadWriteTileSender<T, Device::GPU> step_gemv(
238 matrix::Panel<Coord::Col, const T, Device::GPU>& hh_panel,
239 matrix::ReadOnlyTileSender<T, Device::GPU> taus,
240 matrix::ReadWriteTileSender<T, Device::GPU> tile_t,
241 matrix::Panel<Coord::Col, T, Device::GPU>& workspaces) {
242 namespace ex = pika::execution::experimental;
243 namespace di = dlaf::internal;
244
245 using pika::execution::thread_priority;
246
247 std::vector<matrix::ReadOnlyTileSender<T, Device::GPU>> hh_tiles =
248 selectRead(hh_panel, hh_panel.iteratorLocal());
249
250 const SizeType k = hh_panel.getWidth();
251
252 if (hh_tiles.size() == 0)
253 return tile_t;
254
255 const auto workers_params = split_tfactor_work<Backend::GPU>(hh_tiles.size());
256 const std::size_t nworkers = workers_params.nworkers;
257 const std::size_t batch_size = workers_params.batch_size;
258
259 const auto range_workspaces = common::iterate_range2d(LocalTileSize{to_SizeType(nworkers - 1), 1});
260 DLAF_ASSERT(to_sizet(std::distance(range_workspaces.begin(), range_workspaces.end())) >=
261 nworkers - 1,
262 std::distance(range_workspaces.begin(), range_workspaces.end()), nworkers - 1);
263
264 for (std::size_t worker_id = 0; worker_id < nworkers; ++worker_id) {
265 const std::size_t begin = worker_id * batch_size;
266 const std::size_t end = std::min(hh_tiles.size(), (worker_id + 1) * batch_size);
267
268 DLAF_ASSERT(end >= begin, begin, end);
269
270 if (end == begin)
271 continue;
272
273 std::vector<matrix::ReadOnlyTileSender<T, Device::GPU>> input_tiles;
274 input_tiles.reserve(end - begin);
275 for (std::size_t sub = begin; sub < end; ++sub)
276 input_tiles.emplace_back(std::move(hh_tiles[sub]));
277
278 matrix::ReadWriteTileSender<T, Device::GPU> workspace =
279 worker_id == 0 ? std::move(tile_t)
280 : workspaces.readwrite(LocalTileIndex{to_SizeType(worker_id - 1), 0});
281
282 workspace =
283 di::whenAllLift(ex::when_all_vector(std::move(input_tiles)), taus, std::move(workspace)) |
284 di::transform<dlaf::internal::TransformDispatchType::Blas>(
285 di::Policy<Backend::GPU>(thread_priority::high),
286 [k](cublasHandle_t handle, auto&& hh_tiles, auto&& taus,
287 matrix::Tile<T, Device::GPU>& tile_t_full) {
288 DLAF_ASSERT_MODERATE(k == taus.size().rows(), k, taus.size().rows());
289 DLAF_ASSERT(taus.size().cols() == 1, taus.size().cols());
290
291 matrix::Tile<T, Device::GPU> tile_t = tile_t_full.subTileReference({{0, 0}, {k, k}});
292
293 // Note:
294 // set tile_t to zero (next loop_gemv cumulates results in it)
295 whip::stream_t stream;
296 DLAF_GPUBLAS_CHECK_ERROR(cublasGetStream(handle, &stream));
297
298 whip::memset_2d_async(tile_t.ptr(), sizeof(T) * to_sizet(tile_t.ld()), 0,
299 sizeof(T) * to_sizet(k), to_sizet(k), stream);
300
301 // Note:
302 // - call one gemv per tile
303 // - being on the same stream, they are already serialised on GPU
304 for (std::size_t index = 0; index < hh_tiles.size(); ++index) {
305 const matrix::Tile<const T, Device::GPU>& tile_v = hh_tiles[index].get();
306 Helpers::loop_gemv(handle, tile_v, tile_t);
307 }
308
309 return std::move(tile_t_full);
310 });
311
312 if (worker_id == 0)
313 tile_t = std::move(workspace);
314 else
315 ex::start_detached(std::move(workspace));
316 }
317
318 if (nworkers > 1 and k > 1) {
319 tile_t =
320 di::whenAllLift(std::move(tile_t),
321 ex::when_all_vector(selectRead(workspaces, range_workspaces))) |
322 di::transform<dlaf::internal::TransformDispatchType::Plain>(
323 di::Policy<Backend::GPU>(thread_priority::high),
324 [nworkers](auto&& tile_t, auto&& workspaces, whip::stream_t stream) {
325 for (std::size_t index = 0; index < nworkers - 1; ++index) {
326 matrix::Tile<const T, Device::GPU> ws =
327 workspaces[index].get().subTileReference({{0, 0}, tile_t.size()});
328 DLAF_ASSERT(equal_size(ws, tile_t), ws, tile_t);
329 gpulapack::add(blas::Uplo::Upper, tile_t.size().rows() - 1, tile_t.size().cols() - 1,
330 T(1), ws.ptr({0, 1}), ws.ld(), tile_t.ptr({0, 1}), tile_t.ld(), stream);
331 }
332 return std::move(tile_t);
333 });
334 }
335
336 return tile_t;
337 }

339 static auto step_copy_diag_and_trmv(matrix::ReadOnlyTileSender<T, Device::GPU> taus,
340 matrix::ReadWriteTileSender<T, Device::GPU> tile_t) noexcept {
341 // Update each column (in order) t = T . t
342 // remember that T is upper triangular, so it is possible to use TRMV
343 auto trmv_func = [](cublasHandle_t handle, const matrix::Tile<const T, Device::GPU>& taus,
344 matrix::Tile<T, Device::GPU>& tile_t) {
345 whip::stream_t stream;
346 DLAF_GPUBLAS_CHECK_ERROR(cublasGetStream(handle, &stream));
347
348 const SizeType k = tile_t.size().cols();
349 gpulapack::larft_gemv1_fixtau(k, taus.ptr(), 1, tile_t.ptr(), tile_t.ld(), stream);
350
351 loop_trmv(handle, tile_t);
352 };
353
354 namespace ex = pika::execution::experimental;
355 namespace di = dlaf::internal;
356
357 return ex::when_all(std::move(taus), std::move(tile_t)) |
358 di::transform<di::TransformDispatchType::Blas>(
359 di::Policy<Backend::GPU>(pika::execution::thread_priority::high), std::move(trmv_func));
360 }

362private:
363 static void loop_gemv(cublasHandle_t handle, const matrix::Tile<const T, Device::GPU>& tile_v,
364 const matrix::Tile<T, Device::GPU>& tile_t) noexcept {
365 const SizeType m = tile_v.size().rows();
366 const SizeType k = tile_t.size().cols();
367 DLAF_ASSERT(tile_v.size().cols() == k, tile_v.size().cols(), k);
368 gpulapack::larft_gemv1_notau(handle, m, k, tile_v.ptr(), tile_v.ld(), tile_t.ptr(), tile_t.ld());
369 }
370
371 static void loop_trmv(cublasHandle_t handle, const matrix::Tile<T, Device::GPU>& tile_t) {
372 const SizeType k = tile_t.size().cols();
373
374 // Update each column (in order) t = T . t
375 // remember that T is upper triangular, so it is possible to use TRMV
376 for (SizeType j = 0; j < k; ++j) {
377 const TileElementIndex t_start{0, j};
378 const TileElementSize t_size{j, 1};
379
380 gpublas::internal::Trmv<T>::call(handle, CUBLAS_FILL_MODE_UPPER, CUBLAS_OP_N, CUBLAS_DIAG_NON_UNIT,
381 to_int(t_size.rows()), util::blasToCublasCast(tile_t.ptr()),
382 to_int(tile_t.ld()), util::blasToCublasCast(tile_t.ptr(t_start)),
383 1);
384 }
385 }

390template <Backend backend, Device device, class T>
391void QR_Tfactor<backend, device, T>::call(matrix::Panel<Coord::Col, T, device>& hh_panel,
392 matrix::ReadOnlyTileSender<T, device> taus,
393 matrix::ReadWriteTileSender<T, device> tile_t,
394 matrix::Panel<Coord::Col, T, device>& workspaces) {
395 namespace ex = pika::execution::experimental;
396
397 using Helpers = tfactor_l::Helpers<backend, device, T>;
398
399 // Fast return in case of no reflectors
400 if (hh_panel.getWidth() == 0)
401 return;
402
403 // Note:
404 // T factor is an upper triangular square matrix, built column by column
405 // with taus values on the diagonal
406 //
407 // T(j,j) = tau(j)
408 //
409 // and in the upper triangular part the following formula applies
410 //
411 // T(0:j, j) = T(0:j, 0:j) . -tau(j) . V(j:, 0:j)* . V(j:, j)
412 //
413 // The result is achieved in two main steps:
414 // 1) "GEMV" t = -tau(j) . V(j:, 0:j)* . V(j:, j)
415 // 2) "TRMV" T(0:j, j) = T(0:j, 0:j) . t
416
417 tile_t = Helpers::step_gemv(hh_panel, taus, std::move(tile_t), workspaces);
418 ex::start_detached(Helpers::step_copy_diag_and_trmv(taus, std::move(tile_t)));
419}

421template <Backend backend, Device device, class T>
422void QR_Tfactor<backend, device, T>::call(
423 matrix::Panel<Coord::Col, T, device>& hh_panel, matrix::ReadOnlyTileSender<T, device> taus,
424 matrix::ReadWriteTileSender<T, device> tile_t, matrix::Panel<Coord::Col, T, device>& workspaces,
425 comm::CommunicatorPipeline<comm::CommunicatorType::Col>& mpi_col_task_chain) {
426 namespace ex = pika::execution::experimental;
427
428 using Helpers = tfactor_l::Helpers<backend, device, T>;
429
430 // Fast return in case of no reflectors
431 if (hh_panel.getWidth() == 0)
432 return;
433
434 // Note:
435 // T factor is an upper triangular square matrix, built column by column
436 // with taus values on the diagonal
437 //
438 // T(j,j) = tau(j)
439 //
440 // and in the upper triangular part the following formula applies
441 //
442 // T(0:j, j) = T(0:j, 0:j) . -tau(j) . V(j:, 0:j)* . V(j:, j)
443 //
444 // The result is achieved in two main steps:
445 // 1) "GEMV" t = -tau(j) . V(j:, 0:j)* . V(j:, j)
446 // 2) "TRMV" T(0:j, j) = T(0:j, 0:j) . t
447
448 // Note:
449 // reset is needed because not all ranks might have computations to do, but they will participate
450 // to mpi reduction anyway.
451 tile_t = Helpers::set0_and_return(std::move(tile_t));
452 tile_t = Helpers::step_gemv(hh_panel, taus, std::move(tile_t), workspaces);
453
454 // Note: at this point each rank has its partial result for each column
455 // so, let's reduce the results (on all ranks, so that everyone can independently compute T factor)
456 if (mpi_col_task_chain.size() > 1)
457 tile_t = schedule_all_reduce_in_place(mpi_col_task_chain.exclusive(), MPI_SUM, std::move(tile_t));
458
459 ex::start_detached(Helpers::step_copy_diag_and_trmv(taus, std::move(tile_t)));
460}
