53template <typename F>

57 template <typename... Ts>
58 auto operator()(Ts&&... ts) -> decltype(std::move(f)(dlaf::common::internal::unwrap(ts)...)) {
59 using result_type = decltype(std::move(f)(dlaf::common::internal::unwrap(ts)...));

78[[nodiscard]] decltype(auto) transformMPI(F&& f, Sender&& sender) {
79 using dlaf::common::internal::ConsumeRvalues;
